Option Explicit

' macro_inject_qualified.txt
' Injecteert gekwalificeerde code uit .txt-bestanden in het huidige project.
' - Verwijdert en hermaakt StdModules en ClassModules.
' - Update code van ThisWorkbook (Document module).
' - Update code van bestaande UserForms (geen creatie van nieuwe forms uit .txt).

Public Sub InjectQualifiedModules()
    On Error GoTo ErrHandler

    ' === Vereisten ===
    ' 1) Trust center: "Trust access to the VBA project object model" moet aan staan.
    ' 2) Alle .txt-bestanden staan in dezelfde map als het workbook.

    Dim vbProj As Object          ' VBIDE.VBProject (late-bound)
    Dim fso As Object             ' Scripting.FileSystemObject
    Dim folder As String

    Set vbProj = ThisWorkbook.VBProject
    Set fso = CreateObject("Scripting.FileSystemObject")
    folder = ThisWorkbook.Path

    ' === Injecteer standaard modules ===
    Dim stdModules As Variant
    stdModules = Array("Utils", "Reporting", "MonthlySheet", "FormLayout", "FormData", "FormEvents", "FormBackground", "DataCategories", "CallForm", "Navigation", "TableSetup")
    Dim i As Long
    For i = LBound(stdModules) To UBound(stdModules)
        ReplaceStdOrClassModule vbProj, fso, folder, stdModules(i), 1 ' vbext_ct_StdModule
    Next i

    ' === Injecteer classes ===
    ReplaceStdOrClassModule vbProj, fso, folder, "CBtnEvents", 2 ' vbext_ct_ClassModule
    ReplaceStdOrClassModule vbProj, fso, folder, "CCmbEvents", 2

    ' === Update ThisWorkbook ===
    UpdateDocumentCode vbProj, fso, folder, "ThisWorkbook"

    ' === Update bestaande UserForms ===
    UpdateFormCodeIfExists vbProj, fso, folder, "frmInvoer"

    MsgBox "Injectie gereed. Compileer nu: Debug → Compile VBAProject.", vbInformation
    Exit Sub

ErrHandler:
    MsgBox "InjectQualifiedModules: " & Err.Description, vbCritical
End Sub

Private Sub ReplaceStdOrClassModule(ByVal vbProj As Object, ByVal fso As Object, ByVal folder As String, ByVal compName As String, ByVal compType As Long)
    On Error GoTo ErrHandler
    Dim filePath As String, vbComp As Object, ts As Object, codeText As String
    filePath = folder & "\" & compName & ".txt"
    If Not fso.FileExists(filePath) Then Exit Sub
    Set ts = fso.OpenTextFile(filePath, 1): codeText = ts.ReadAll: ts.Close
    Set vbComp = GetVBComponentSafe(vbProj, compName)
    If Not vbComp Is Nothing Then vbProj.VBComponents.Remove vbComp
    Set vbComp = vbProj.VBComponents.Add(compType): vbComp.Name = compName
    vbComp.CodeModule.AddFromString codeText
    Exit Sub
ErrHandler:
    MsgBox "ReplaceStdOrClassModule(" & compName & "): " & Err.Description, vbCritical
End Sub

Private Sub UpdateDocumentCode(ByVal vbProj As Object, ByVal fso As Object, ByVal folder As String, ByVal docName As String)
    On Error GoTo ErrHandler
    Dim filePath As String, vbComp As Object, ts As Object, codeText As String
    filePath = folder & "\" & docName & ".txt"
    If Not fso.FileExists(filePath) Then Exit Sub
    Set ts = fso.OpenTextFile(filePath, 1): codeText = ts.ReadAll: ts.Close
    Set vbComp = GetVBComponentSafe(vbProj, docName)
    If vbComp Is Nothing Then Exit Sub
    With vbComp.CodeModule
        .DeleteLines 1, .CountOfLines
        .AddFromString codeText
    End With
    Exit Sub
ErrHandler:
    MsgBox "UpdateDocumentCode(" & docName & "): " & Err.Description, vbCritical
End Sub

Private Sub UpdateFormCodeIfExists(ByVal vbProj As Object, ByVal fso As Object, ByVal folder As String, ByVal formName As String)
    On Error GoTo ErrHandler
    Dim filePath As String, vbComp As Object, ts As Object, codeText As String
    filePath = folder & "\" & formName & ".txt"
    If Not fso.FileExists(filePath) Then Exit Sub
    Set vbComp = GetVBComponentSafe(vbProj, formName)
    If vbComp Is Nothing Or vbComp.Type <> 3 Then Exit Sub
    Set ts = fso.OpenTextFile(filePath, 1): codeText = ts.ReadAll: ts.Close
    With vbComp.CodeModule
        .DeleteLines 1, .CountOfLines
        .AddFromString codeText
    End With
    Exit Sub
ErrHandler:
    MsgBox "UpdateFormCodeIfExists(" & formName & "): " & Err.Description, vbCritical
End Sub

Private Function GetVBComponentSafe(ByVal vbProj As Object, ByVal name As String) As Object
    On Error Resume Next
    Set GetVBComponentSafe = vbProj.VBComponents(name)
End Function
