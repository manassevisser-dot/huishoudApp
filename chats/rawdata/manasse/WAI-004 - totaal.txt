```ts
// WAI-004A — NumericParser & NL-formatters (non-negative)
// -------------------------------------------------------

/** Strip emoji (surrogate pairs) en beperk lengte optioneel */
export const stripEmojiAndLimit = (s: string, max = 100): string => {
  if (!s) return '';
  const noEmoji = s.replace(/[\uD800-\uDBFF][\uDC00-\uDFFF]/g, '');
  return noEmoji.slice(0, max);
};

/** Tijdens typen: sta alleen cijfers, komma en punt toe; strip whitespace/emoji; VERWIJDER '-' */
export function formatDutchValue(raw: string): string {
  const cleaned = stripEmojiAndLimit(String(raw ?? ''), 256)
    .replace(/\s+/g, '')
    .replace(/-/g, '')           // ⬅️ minus altijd weg op parser-niveau (WAI-004A policy)
    .replace(/[^0-9,.]/g, '');   // alleen [0-9], ',' en '.'
  return cleaned;
}

/** NL-parser: string/number → cents (integer, non-negative). */
export function parseToCents(input: string | number): number {
  // Numbers: behandel als euro-waarde → centen (altijd ≥ 0)
  if (typeof input === 'number') {
    const cents = Math.round(Math.abs(input) * 100);
    return cents;
  }
  if (!input) return 0;

  // 1) Sanitisatie (verwijdert '-' en emoji/whitespace)
  let s = formatDutchValue(input);

  // 2) Alle duizendtallen '.' verwijderen
  s = s.replace(/\./g, '');

  // 3) Laatste ',' naar '.' als decimaal
  const lastComma = s.lastIndexOf(',');
  if (lastComma >= 0) {
    s = s.slice(0, lastComma).replace(/,/g, '') + '.' + s.slice(lastComma + 1).replace(/,/g, '');
  } else {
    s = s.replace(/,/g, '');
  }

  // 4) Parse en rond naar centen, dwing non-negative af
  const val = parseFloat(s);
  if (isNaN(val)) return 0;
  return Math.max(0, Math.round(val * 100));
}

/** Formatter: cents → "1.250,50" (zonder €), altijd 2 decimalen */
export function formatCentsToDutch(cents: number): string {
  const euros = (Number.isFinite(cents) ? cents : 0) / 100;
  return new Intl.NumberFormat('nl-NL', {
    minimumFractionDigits: 2,
    maximumFractionDigits: 2,
  }).format(euros);
}

/** Read-only currency (met €) — niet gebruiken in input-context */
export function formatCurrency(amount: number): string {
  return new Intl.NumberFormat('nl-NL', {
    style: 'currency',
    currency: 'EUR',
  }).format(amount);
}
```

# \[WAI‑004B] — **MoneyInput** (blijft ongewijzigd t.o.v. vorige oplevering)

```tsx
import * as React from 'react';
import { View, Text, TextInput } from 'react-native';
import { useAppStyles } from '../styles/useAppStyles';
import { parseToCents, formatCentsToDutch, formatDutchValue } from '../utils/numbers';

type MoneyInputProps = {
  value: number; // cents, non-negative
  onValueChange: (cents: number) => void;
  disabled?: boolean;
  placeholder?: string;
  accessibilityLabel?: string; // default "Bedrag"
};

const MoneyInput: React.FC<MoneyInputProps> = ({
  value,
  onValueChange,
  disabled = false,
  placeholder = '',
  accessibilityLabel = 'Bedrag',
}) => {
  const { styles } = useAppStyles();
  const [localValue, setLocalValue] = React.useState<string>(formatCentsToDutch(value));

  React.useEffect(() => {
    setLocalValue(formatCentsToDutch(value));
  }, [value]);

  const onFocus = () => setLocalValue(formatDutchValue(localValue));
  const onChangeText = (text: string) => setLocalValue(formatDutchValue(text)); // '-'/+ al verwijderd in formatDutchValue
  const onBlur = () => {
    const cents = parseToCents(localValue);
    onValueChange(cents); // UI emit non-negative cents
    setLocalValue(formatCentsToDutch(cents));
  };

  return (
    <View style={styles.inputMoneyRow}>
      <Text style={styles.moneyPrefix} accessibilityLabel="Euro-teken">€</Text>
      <TextInput
        value={localValue}
        onChangeText={onChangeText}
        onFocus={onFocus}
        onBlur={onBlur}
        placeholder={placeholder}
        editable={!disabled}
        accessibilityLabel={accessibilityLabel}
        keyboardType="decimal-pad"
        style={styles.moneyTextInput}
      />
    </View>
  );
};

export default MoneyInput;
```

# \[WAI‑004C] — **Finance kernel** (contextuele tekenlogica; non‑negative inputs)

### 1) Kernidee

*   **Alle inputs** naar de kernel zijn **non‑negative centen** (UI/compat‑loader normaliseren).
*   **Netto:** `net = income - expenses`. Hierdoor wordt het teken **contextueel** toegepast: expenses verminderen het totaal.

### 2) 📄 `src/logic/finance.ts` (kleine aanpassing aan compat‑loader)

```ts
import { parseToCents } from '../utils/numbers';

export type IncomeItem = { id: string; amount: number /* cents, non-negative */ };
export type ExpenseItem = { id: string; amount: number /* cents, non-negative */ };

/** Compat: normalise legacy value to non-negative cents */
export function normalizeToCents(value: unknown): number {
  if (typeof value === 'number') {
    // Treat as euros if fractional; enforce non-negative
    const abs = Math.abs(value);
    if (!Number.isInteger(value)) return Math.round(abs * 100);
    if (abs < 100 && abs !== 0) return Math.round(abs * 100); // small-euro heuristic
    return Math.round(abs); // already cents
  }
  if (typeof value === 'string') {
    // Parser is non-negative; minus in legacy strings wordt genegeerd conform WAI-004A
    return parseToCents(value);
  }
  return 0;
}

export function normalizeIncome(items: Array<any>): IncomeItem[] {
  return (items ?? []).map((it) => ({
    id: String(it.id ?? ''),
    amount: normalizeToCents(it.amount ?? it.value ?? 0),
  }));
}
export function normalizeExpenses(items: Array<any>): ExpenseItem[] {
  return (items ?? []).map((it) => ({
    id: String(it.id ?? ''),
    amount: normalizeToCents(it.amount ?? it.value ?? 0),
  }));
}

export function sumIncomeCents(items: IncomeItem[]): number {
  return items.reduce((acc, it) => acc + (it.amount || 0), 0);
}
export function sumExpensesCents(items: ExpenseItem[]): number {
  return items.reduce((acc, it) => acc + (it.amount || 0), 0);
}

export function computeNetCents(income: IncomeItem[], expenses: ExpenseItem[]): number {
  return sumIncomeCents(income) - sumExpensesCents(expenses);
}

export function computeSummary(incomeRaw: Array<any>, expenseRaw: Array<any>) {
  const income = normalizeIncome(incomeRaw);
  const expenses = normalizeExpenses(expenseRaw);
  return {
    totalIncome: sumIncomeCents(income),
    totalExpenses: sumExpensesCents(expenses),
    net: computeNetCents(income, expenses),
  };
}
```

## 1️⃣ Strategische Audit van de Artefacten

### ✅ WAI-004A: De Parser (`numbers.ts`)
De implementatie van `parseToCents` is nu robuust voor de Nederlandse markt. De tweestaps-verwerking (eerst duizendtallen verwijderen, dan de *laatste* komma als decimaal behandelen) elimineert de beruchte "1.250,50 -> 1.25" bug. De toevoeging van de `Math.max(0, ...)` guard is de ultieme veiligheid voor de "non-negative" eis.

### ✅ WAI-004B: De UI (`MoneyInput.tsx`)
Conform **ADR-13** is de component volledig dom. De scheiding tussen de `localValue` (string voor UX) en de `onValueChange` (integer voor de State) is de goudstandaard voor financiële invoer in 2025. Het gebruik van `keyboardType="decimal-pad"` op mobile zal bovendien de invoer van letters fysiek beperken.

### ✅ WAI-004C: De Kernel (`finance.ts`)
De `computeNetCents` functie hanteert nu de juiste contextuele logica:
`netto = som(inkomsten) - som(uitgaven)`.
Omdat beide bronnen nu gegarandeerd positief zijn, kan er geen verwarring meer ontstaan over dubbele ontkenningen (bijv. het aftrekken van een negatieve uitgave).
