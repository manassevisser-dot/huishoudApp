Option Explicit
Private Sub Workbook_Open()
 On Error GoTo ErrHandler
 Utils.EnsureDBHeaders
 ' Alleen vorige week samenvatten
 Dim dPrev As Date: dPrev = DateAdd("d", -7, Date)
 Dim w As Long: w = Utils.IsoWeek(dPrev)
 Dim y As Long: y = Year(dPrev)
 If Weekday(Date, vbMonday) = 1 Then
   If Not Reporting.WeekTotaalBestaat(y, w) Then Reporting.BerekenWeekTotaal
 Else
   If Not Reporting.WeekTotaalBestaat(y, w) Then
     Dim antwoord As VbMsgBoxResult
     antwoord = MsgBox("Weekoverzicht van vorige week ontbreekt. Nu berekenen?", vbYesNo + vbQuestion, "Weekoverzicht")
     If antwoord = vbYes Then Reporting.BerekenWeekTotaal
   End If
 End If
 ' Maandoverzicht: alleen als vorige maand niet aanwezig
 If Format(Date, "YYYY-MM") <> Reporting.LaatsteMaandInMaandTabel() Then Reporting.BerekenMaandTotaal
 ' Huidige maandblad: alleen aanmaken als nog niet bestaat
 MonthlySheet.MaakHuidigeMaandSheet
 Exit Sub
ErrHandler:
 Utils.LogError4 "ThisWorkbook", "Workbook_Open", Err.Number, Err.Description
 MsgBox "Fout in Workbook_Open: " & Err.Description, vbCritical
End Sub

Public Sub LogError4(ByVal moduleName As String, ByVal funcName As String, ByVal errNum As Long, ByVal errDesc As String)
 On Error Resume Next
 Dim ws As Worksheet
 Set ws = ThisWorkbook.Worksheets("ErrorLog")
 If ws Is Nothing Then
   Set ws = ThisWorkbook.Worksheets.Add
   ws.Name = "ErrorLog"
   ws.Range("A1:D1").Value = Array("Datum", "Module", "Functie", "Foutmelding")
 End If
 Dim r As Long: r = ws.Cells(ws.Rows.Count, 1).End(xlUp).Row + 1
 ws.Cells(r, 1).Value = Now
 ws.Cells(r, 2).Value = moduleName
 ws.Cells(r, 3).Value = funcName
 ws.Cells(r, 4).Value = "[" & errNum & "] " & errDesc
End Sub

Public Function LastRow(ByVal ws As Worksheet, Optional ByVal col As Long = 1) As Long
 With ws
  If Application.WorksheetFunction.CountA(.Columns(col)) = 0 Then
    LastRow = 1
  Else
    LastRow = .Cells(.Rows.Count, col).End(xlUp).Row
  End If
 End With
End Function
Public Function IsoWeek(ByVal d As Date) As Long
 IsoWeek = DatePart("ww", d, vbMonday, vbFirstFourDays)
End Function
Public Function ParseAmount(ByVal s As String) As Double
 Dim decSep As String: decSep = Application.International(xlDecimalSeparator)
 s = Trim$(s)
 If s = "" Then ParseAmount = 0: Exit Function
 s = Replace(Replace(s, ".", decSep), ",", decSep)
 ParseAmount = CDbl(s)
End Function

Public Function DictAddOrIncrement(ByRef dict As Object, ByVal key As Variant, Optional ByVal inc As Double = 1) As Double
 If Not dict.Exists(key) Then dict(key) = inc Else dict(key) = dict(key) + inc
 DictAddOrIncrement = dict(key)
End Function
Public Function ToMonthlyAmount(ByVal periode As String, ByVal bedrag As Double) As Double
 periode = LCase$(Trim$(periode))
 Select Case periode
  Case "jaar", "per jaar": ToMonthlyAmount = bedrag / 12
  Case "kwartaal", "per kwartaal": ToMonthlyAmount = bedrag / 3
  Case "maand", "per maand": ToMonthlyAmount = bedrag
  Case "4 weken": ToMonthlyAmount = bedrag * 13 / 12
  Case "week", "per week": ToMonthlyAmount = bedrag * 52 / 12
  Case "dag", "per dag": ToMonthlyAmount = bedrag * 365 / 12
  Case Else: ToMonthlyAmount = bedrag
 End Select
End Function

Public Function MonthNameNL(ByVal m As Long) As String
 Dim arr
 arr = Array("Januari", "Februari", "Maart", "April", "Mei", "Juni", "Juli", "Augustus", "September", "Oktober", "November", "December")
 If m >= 1 And m <= 12 Then MonthNameNL = arr(m - 1) Else MonthNameNL = "Onbekend"
End Function
Public Function GetTempFile(ByVal baseName As String, ByVal ext As String) As String
 On Error GoTo ErrHandler
 Dim tmp As String
 tmp = Environ$("TEMP")
 If Len(tmp) = 0 Then tmp = Environ$("TMP")
 If Len(tmp) = 0 Then tmp = ThisWorkbook.Path
 If Right$(tmp, 1) <> "" Then tmp = tmp & ""
 GetTempFile = tmp & baseName & "_" & Format(Now, "yyyymmdd_hhnnss") & ext
 Exit Function
ErrHandler:
 GetTempFile = ThisWorkbook.Path & "" & baseName & "_" & Format(Now, "yyyymmdd_hhnnss") & ext
 LogError4 "Utils", "GetTempFile", Err.Number, Err.Description
End Function

Public Const DYN_PREFIX As String = "dyn_"
Public Sub FrmLayoutSafeAreaDyn(frm As Object)
 On Error GoTo ErrHandler
 ' Config uit Configuratie!tblFrmInvoerConfig
 MonthlySheet.EnsureConfigSheet
 Dim BG_MARGIN As Double: BG_MARGIN = MonthlySheet.ConfigGet("BG_MARGIN", 6)
 Dim ROW_SPACING As Double: ROW_SPACING = MonthlySheet.ConfigGet("ROW_SPACING", 34)
 Dim LABEL_WIDTH As Double: LABEL_WIDTH = MonthlySheet.ConfigGet("LABEL_WIDTH", 120)
 Dim GAP_LABEL_TO_INPUT As Double: GAP_LABEL_TO_INPUT = MonthlySheet.ConfigGet("GAP_LABEL_TO_INPUT", 12)
 Dim RIGHT_PADDING As Double: RIGHT_PADDING = MonthlySheet.ConfigGet("RIGHT_PADDING", 20)
 Dim INP_MAX As Double: INP_MAX = MonthlySheet.ConfigGet("INP_MAX", 360)
 Dim SAFE_LEFT_PCT As Double: SAFE_LEFT_PCT = MonthlySheet.ConfigGet("SAFE_LEFT_PCT", 0.12)
 Dim SAFE_RIGHT_PCT As Double: SAFE_RIGHT_PCT = MonthlySheet.ConfigGet("SAFE_RIGHT_PCT", 0.88)
 Dim SAFE_TOP_PCT As Double: SAFE_TOP_PCT = MonthlySheet.ConfigGet("SAFE_TOP_PCT", 0.2)
 Dim SAFE_BOTTOM_PCT As Double: SAFE_BOTTOM_PCT = MonthlySheet.ConfigGet("SAFE_BOTTOM_PCT", 0.8)

 Dim bgLeft As Single, bgTop As Single, bgW As Single, bgH As Single
 Dim bg As MSForms.Image: Set bg = TryGetImageDyn(frm, FormLayout.DYN_PREFIX & "imgBackground")
 If bg Is Nothing Then
  bgLeft = BG_MARGIN
  bgTop = BG_MARGIN
  bgW = frm.InsideWidth - 2 * BG_MARGIN
  bgH = frm.InsideHeight - 2 * BG_MARGIN
 Else
  bgLeft = bg.Left: bgTop = bg.Top: bgW = bg.Width: bgH = bg.Height
 End If
 Dim safeLeft As Single, safeTop As Single, safeWidth As Single, safeHeight As Single
 safeLeft = bgLeft + (SAFE_LEFT_PCT * bgW)
 safeTop = bgTop + (SAFE_TOP_PCT * bgH)
 safeWidth = (SAFE_RIGHT_PCT - SAFE_LEFT_PCT) * bgW
 safeHeight = (SAFE_BOTTOM_PCT - SAFE_TOP_PCT) * bgH
 Dim colLbl As Single, colInp As Single, inpWidth As Single, t As Single
 colLbl = safeLeft
 colInp = safeLeft + LABEL_WIDTH + GAP_LABEL_TO_INPUT
 inpWidth = safeWidth - (colInp - safeLeft) - RIGHT_PADDING
 If inpWidth > INP_MAX Then
  Dim extraSpace As Single
  extraSpace = (inpWidth - INP_MAX) / 2
  colInp = colInp + extraSpace
  inpWidth = INP_MAX
 End If
 t = safeTop
 Dim ctrl As Control
 Set ctrl = TryGetControl(frm, FormLayout.DYN_PREFIX & "lblDatumVandaag"): If Not ctrl Is Nothing Then ctrl.Left = colLbl: ctrl.Top = t
 t = t + ROW_SPACING
 Set ctrl = TryGetControl(frm, FormLayout.DYN_PREFIX & "lblCat"): If Not ctrl Is Nothing Then ctrl.Left = colLbl: ctrl.Top = t
 Set ctrl = TryGetControl(frm, FormLayout.DYN_PREFIX & "cmbCategorie"): If Not ctrl Is Nothing Then ctrl.Left = colInp: ctrl.Top = t: ctrl.Width = inpWidth
 t = t + ROW_SPACING
 Set ctrl = TryGetControl(frm, FormLayout.DYN_PREFIX & "lblSub"): If Not ctrl Is Nothing Then ctrl.Left = colLbl: ctrl.Top = t
 Set ctrl = TryGetControl(frm, FormLayout.DYN_PREFIX & "cmbSubcategorie"): If Not ctrl Is Nothing Then ctrl.Left = colInp: ctrl.Top = t: ctrl.Width = inpWidth - 36
 Set ctrl = TryGetControl(frm, FormLayout.DYN_PREFIX & "btnNieuwSub"): If Not ctrl Is Nothing Then ctrl.Left = colInp + (inpWidth - 36) + 8: ctrl.Top = t: ctrl.Width = 28: ctrl.Height = 22
 t = t + ROW_SPACING
 Set ctrl = TryGetControl(frm, FormLayout.DYN_PREFIX & "lblBedrag"): If Not ctrl Is Nothing Then ctrl.Left = colLbl: ctrl.Top = t
 Set ctrl = TryGetControl(frm, FormLayout.DYN_PREFIX & "txtBedrag"): If Not ctrl Is Nothing Then ctrl.Left = colInp: ctrl.Top = t: ctrl.Width = inpWidth
 t = t + ROW_SPACING
 Set ctrl = TryGetControl(frm, FormLayout.DYN_PREFIX & "lblBetaal"): If Not ctrl Is Nothing Then ctrl.Left = colLbl: ctrl.Top = t
 Set ctrl = TryGetControl(frm, FormLayout.DYN_PREFIX & "cmbBetaalwijze"): If Not ctrl Is Nothing Then ctrl.Left = colInp: ctrl.Top = t: ctrl.Width = inpWidth
 Dim btnOps As Control, btnAnn As Control
 Set btnOps = TryGetControl(frm, FormLayout.DYN_PREFIX & "imgBtnOpslaan")
 Set btnAnn = TryGetControl(frm, FormLayout.DYN_PREFIX & "imgBtnAnnuleren")
 If Not btnOps Is Nothing And Not btnAnn Is Nothing Then
  Dim btnTotalWidth As Single: btnTotalWidth = btnOps.Width + 24 + btnAnn.Width
  Dim btnLeft As Single: btnLeft = safeLeft + (safeWidth - btnTotalWidth) / 2
  Dim btnY As Single: btnY = safeTop + safeHeight - 60
  btnOps.Left = btnLeft: btnOps.Top = btnY
  btnAnn.Left = btnLeft + btnOps.Width + 24: btnAnn.Top = btnY
 End If
 Exit Sub
ErrHandler:
 Utils.LogError4 "FormLayout", "FrmLayoutSafeAreaDyn", Err.Number, Err.Description
End Sub

End Sub

 Dim i As Long, maxLen As Long

  If Len(CStr(cmb.List(i))) > maxLen Then maxLen = Len(CStr(cmb.List(i)))
 Next i

Public Function GetLayoutSpec(ByVal keyName As String, ByVal fieldName As String) As Variant
 On Error GoTo ErrHandler
 Dim wsRef As Worksheet: Set wsRef = ThisWorkbook.Worksheets("Referenties")
 Dim lo As ListObject: Set lo = wsRef.ListObjects("tblLayoutSpecs")
 Dim hdr As Range, data As Range, r As Range, colIndex As Long
 Set hdr = lo.HeaderRowRange
 Set data = lo.DataBodyRange
 If data Is Nothing Then Exit Function
 colIndex = Application.Match(fieldName, Application.Transpose(hdr.Value), 0)
 If IsError(colIndex) Or colIndex <= 0 Then Exit Function
 For Each r In data.Rows
  If StrComp(CStr(r.Cells(1, 1).Value), keyName, vbTextCompare) = 0 Then
   GetLayoutSpec = r.Cells(1, colIndex).Value
   Exit Function
  End If
 Next r
 Exit Function
ErrHandler:
 Utils.LogError4 "LayoutSpecsHelpers", "GetLayoutSpec", Err.Number, Err.Description
End Function

Private Function ParseTag(ByVal tagText As String, ByVal key As String) As String
 On Error GoTo ErrHandler
 Dim parts() As String: parts = Split(tagText, vbLf)
 Dim i As Long, kv()
 For i = LBound(parts) To UBound(parts)
  kv = Split(parts(i), "=")
  If UBound(kv) = 1 Then If StrComp(kv(0), key, vbTextCompare) = 0 Then ParseTag = kv(1): Exit Function
 Next i
 Exit Function
ErrHandler:
 Utils.LogError4 "LayoutSpecsHelpers", "ParseTag", Err.Number, Err.Description
End Function

Public Sub BerekenWeekTotaal()
 On Error GoTo ErrHandler
 Dim ws As Worksheet: Set ws = ThisWorkbook.Worksheets("Database")
 Utils.EnsureDBHeaders
 Dim lr As Long: lr = Utils.LastRow(ws, DBCols.dbDate)
 If lr < 2 Then Exit Sub
 Dim dPrev As Date: dPrev = DateAdd("d", -7, Date)
 Dim tgtWeek As Long: tgtWeek = Utils.IsoWeek(dPrev)
 Dim tgtYear As Long: tgtYear = Year(dPrev)
 Dim i As Long, total As Double
 Dim catDict As Object: Set catDict = CreateObject("Scripting.Dictionary")
 For i = 2 To lr
  If CLng(Val(ws.Cells(i, DBCols.dbYear).Value)) = tgtYear And _
     CLng(Val(ws.Cells(i, DBCols.dbWeek).Value)) = tgtWeek And _
     CStr(ws.Cells(i, DBCols.dbType).Value) = "Uitgave" Then
     Dim amount As Double: amount = CDbl(Val(ws.Cells(i, DBCols.dbAmount).Value))
     total = total + amount
     Dim cat As String: cat = CStr(ws.Cells(i, DBCols.dbCategory).Value)
     If Len(Trim$(cat)) > 0 Then Utils.DictAddOrIncrement catDict, cat, 1
  End If
 Next i
 Dim mostCat As String, maxCount As Long, k As Variant
 For Each k In catDict.Keys
  If catDict(k) > maxCount Then maxCount = catDict(k): mostCat = CStr(k)
 Next k
 If ws.Cells(1, SummaryCols.colWeekSummaryStart).Value = "" Then
  ws.Cells(1, SummaryCols.colWeekSummaryStart).Resize(1, 5).Value = Array("Jaar" & vbLf & "Week", "Jaar", "Weeknr", "Totaal", "Meest gebruikte categorie")
 End If
 Dim outRow As Long: outRow = Utils.LastRow(ws, SummaryCols.colWeekSummaryStart) + 1
 ws.Cells(outRow, SummaryCols.colWeekSummaryStart).Value = CStr(tgtYear) & vbLf & CStr(tgtWeek)
 ws.Cells(outRow, SummaryCols.colWeekSummaryStart + 1).Value = tgtYear
 ws.Cells(outRow, SummaryCols.colWeekSummaryStart + 2).Value = tgtWeek
 ws.Cells(outRow, SummaryCols.colWeekSummaryStart + 3).Value = total
 ws.Cells(outRow, SummaryCols.colWeekSummaryStart + 4).Value = mostCat
 Exit Sub
ErrHandler:
 Utils.LogError4 "Reporting", "BerekenWeekTotaal", Err.Number, Err.Description
End Sub
Public Sub BerekenMaandTotaal()
 On Error GoTo ErrHandler
 Dim ws As Worksheet: Set ws = ThisWorkbook.Worksheets("Database")
 Utils.EnsureDBHeaders
 Dim lr As Long: lr = Utils.LastRow(ws, DBCols.dbDate)
 If lr < 2 Then Exit Sub
 Dim prevMonthKey As String: prevMonthKey = Format(DateAdd("m", -1, Date), "YYYY-MM")
 Dim i As Long
 Dim totaal As Double, dagenCount As Long
 Dim weekDict As Object: Set weekDict = CreateObject("Scripting.Dictionary")
 Dim catDict As Object: Set catDict = CreateObject("Scripting.Dictionary")
 Dim catCountDict As Object: Set catCountDict = CreateObject("Scripting.Dictionary")
 Dim duursteDag As Double, duursteWeek As Double
 Dim maxCatValue As Double, minCatValue As Double
 Dim maxCatName As String, minCatName As String
 maxCatValue = 0: minCatValue = 1E+20
 For i = 2 To lr
  If CStr(ws.Cells(i, DBCols.dbMonthKey).Value) = prevMonthKey And CStr(ws.Cells(i, DBCols.dbType).Value) = "Uitgave" Then
   Dim bedrag As Double: bedrag = CDbl(Val(ws.Cells(i, DBCols.dbAmount).Value))
   totaal = totaal + bedrag
   dagenCount = dagenCount + 1
   If bedrag > duursteDag Then duursteDag = bedrag
   Dim weekKey As String: weekKey = CStr(ws.Cells(i, DBCols.dbYear).Value) & "-W" & CStr(ws.Cells(i, DBCols.dbWeek).Value)
   Utils.DictAddOrIncrement weekDict, weekKey, bedrag
   Dim categorie As String: categorie = CStr(ws.Cells(i, DBCols.dbCategory).Value)
   Utils.DictAddOrIncrement catDict, categorie, bedrag
   Utils.DictAddOrIncrement catCountDict, categorie, 1
  End If
 Next i
 Dim key As Variant
 For Each key In weekDict.Keys: If weekDict(key) > duursteWeek Then duursteWeek = weekDict(key)
 Next key
 For Each key In catDict.Keys
  If catDict(key) > maxCatValue Then maxCatValue = catDict(key): maxCatName = CStr(key)
  If catDict(key) < minCatValue Then minCatValue = catDict(key): minCatName = CStr(key)
 Next key
 Dim overzichtRow As Long: overzichtRow = Utils.LastRow(ws, SummaryCols.colMonthSummaryStart) + 1
 If ws.Cells(1, SummaryCols.colMonthSummaryStart).Value = "" Then
  ws.Cells(1, SummaryCols.colMonthSummaryStart).Resize(1, 7).Value = Array("mnd", "ttl", "gpd", "gpw", "dudg", "duwk", "m_u_a")
 End If
 ws.Cells(overzichtRow, SummaryCols.colMonthSummaryStart).Value = prevMonthKey
 ws.Cells(overzichtRow, SummaryCols.colMonthSummaryStart + 1).Value = totaal
 ws.Cells(overzichtRow, SummaryCols.colMonthSummaryStart + 2).Value = IIf(dagenCount > 0, totaal / dagenCount, 0)
 ws.Cells(overzichtRow, SummaryCols.colMonthSummaryStart + 3).Value = IIf(weekDict.Count > 0, totaal / weekDict.Count, 0)
 ws.Cells(overzichtRow, SummaryCols.colMonthSummaryStart + 4).Value = duursteDag
 ws.Cells(overzichtRow, SummaryCols.colMonthSummaryStart + 5).Value = duursteWeek
 ws.Cells(overzichtRow, SummaryCols.colMonthSummaryStart + 6).Value = maxCatName
 ' Categorie-overzicht append met maandkolom
 Dim catStartCol As Long: catStartCol = SummaryCols.colMonthSummaryStart + 8 ' X
 If ws.Cells(1, catStartCol - 1).Value = "" Then
  ws.Cells(1, catStartCol - 1).Resize(1, 4).Value = Array("Maand", "Categorie", "Aantal", "Totaal")
 End If
 Dim r As Long: r = Utils.LastRow(ws, catStartCol - 1) + 1
 For Each key In catDict.Keys
  ws.Cells(r, catStartCol - 1).Value = prevMonthKey
  ws.Cells(r, catStartCol).Value = key
  ws.Cells(r, catStartCol + 1).Value = catCountDict(key)
  ws.Cells(r, catStartCol + 2).Value = catDict(key)
  r = r + 1
 Next key
 Exit Sub
ErrHandler:
 Utils.LogError4 "Reporting", "BerekenMaandTotaal", Err.Number, Err.Description
End Sub

 MsgBox "Tabellen ingesteld: tblVastInkomsten/tblVastUitgaven (indien aanwezig), tblCategorieen, tblSubcategorieen.", vbInformation
 Exit Sub
ErrHandler:
 Utils.LogError4 "TableSetup", "SetupVastEnCategorieTabellen", Err.Number, Err.Description
End Sub
Private Sub SeedCategorieen(ByVal loCat As ListObject, ByVal loSub As ListObject)
 On Error GoTo ErrHandler
 Dim cats As Variant: cats = Array("Eten", "Vervoer", "Hapje/Drankje", "Cadeau", "Lisa", "Kyra", "Kleding e.d", "Overig")
 Dim i As Long
 For i = LBound(cats) To UBound(cats)
  If Not CategoryExists(loCat, CStr(cats(i))) Then AddCategory loCat, CStr(cats(i))
 Next i
 AddSub loSub, "Lisa", "Zakgeld": AddSub loSub, "Lisa", "Kleding": AddSub loSub, "Lisa", "Onzin"
 AddSub loSub, "Kyra", "Zakgeld": AddSub loSub, "Kyra", "Kleding": AddSub loSub, "Kyra", "Onzin"
 Exit Sub
ErrHandler:
 Utils.LogError4 "TableSetup", "SeedCategorieen", Err.Number, Err.Description
End Sub
Public Function CategoryExists(ByVal loCat As ListObject, ByVal cat As String) As Boolean
 On Error Go To ErrHandler
 Dim r As Range
 For Each r In loCat.ListColumns(1).DataBodyRange
  If LCase$(Trim$(r.Value)) = LCase$(Trim$(cat)) Then CategoryExists = True: Exit Function
 Next r
 Exit Function
ErrHandler:
 Utils.LogError4 "TableSetup", "CategoryExists", Err.Number, Err.Description
End Function
Public Sub AddCategory(ByVal loCat As ListObject, ByVal cat As String)
 On Error GoTo ErrHandler
 Dim lr As ListRow: Set lr = loCat.ListRows.Add
 lr.Range.Cells(1, 1).Value = cat
 Exit Sub
ErrHandler:
 Utils.LogError4 "TableSetup", "AddCategory", Err.Number, Err.Description
End Sub
Public Sub AddSub(ByVal loSub As ListObject, ByVal cat As String, ByVal subcat As String)
 On Error GoTo ErrHandler
 Dim r As Range
 If Not loSub.DataBodyRange Is Nothing Then
  For Each r In loSub.DataBodyRange.Rows
   If LCase$(Trim$(r.Cells(1, 1).Value)) = LCase$(Trim$(cat)) And LCase$(Trim$(r.Cells(1, 2).Value)) = LCase$(Trim$(subcat)) Then Exit Sub
  Next r
 End If
 Dim lr As ListRow: Set lr = loSub.ListRows.Add
 lr.Range.Cells(1, 1).Value = cat
 lr.Range.Cells(1, 2).Value = subcat
 Exit Sub
ErrHandler:
 Utils.LogError4 "TableSetup", "AddSub", Err.Number, Err.Description
End Sub

' Class module name: CTxtEvents
Option Explicit
Public WithEvents Txt As MSForms.TextBox
Private Sub Txt_KeyPress(ByVal KeyAscii As MSForms.ReturnInteger)
 Dim dec As String: dec = Application.International(xlDecimalSeparator)
 Select Case KeyAscii
  Case 48 To 57 ' 0–9
  Case Asc(dec) ' één decimaal
   If InStr(Txt.Text, dec) > 0 Then KeyAscii = 0
  Case 8 ' Backspace
  Case Else: KeyAscii = 0
 End Select
End Sub

 Txt.Text = Format(CDbl(Txt.Text), "0.00")
End Sub
