
// App.jsx
import React, { useState } from 'react';
import { View, Text, TextInput, TouchableOpacity, StyleSheet, ActivityIndicator } from 'react-native';
import FinancieelBeheerApp from './FinancieelBeheerApp'; // pad aanpassen indien nodig

const authenticateWithN8N = async ({ email, password }) => {
  // Vervang door jouw echte n8n login endpoint
  const loginEndpoint = 'https://uw.n8n.server/webhook/login';
  const payload = { email, password };
  const resp = await fetch(loginEndpoint, {
    method: 'POST', headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify(payload),
  });
  if (!resp.ok) {
    const err = await resp.text();
    throw new Error(`Login fout (${resp.status}): ${err}`);
  }
  const data = await resp.json();
  // Verwacht response zoals: { userId: '...' }
  if (!data?.userId) throw new Error('Login ok, maar userId ontbreekt.');
  return data.userId;
};

function LoginScreen({ onSuccess }) {
  const [email, setEmail] = useState('');
  const [password, setPassword] = useState('');
  const [busy, setBusy] = useState(false);
  const [msg, setMsg] = useState('');

  const handleLogin = async () => {
    setBusy(true); setMsg('');
    try {
      const uid = await authenticateWithN8N({ email, password });
      onSuccess(uid);
    } catch (e) {
      setMsg(e.message || 'Login mislukt.');
    } finally {
      setBusy(false);
    }
  };

  return (
    <View style={styles.container}>
      <Text style={styles.title}>Inloggen</Text>
      <View style={styles.inputGroup}>
        <Text>E-mail</Text>
        <TextInput style={styles.input} value={email} onChangeText={setEmail} keyboardType="email-address" />
      </View>
      <View style={styles.inputGroup}>
        <Text>Wachtwoord</Text>
        <TextInput style={styles.input} value={password} onChangeText={setPassword} secureTextEntry />
      </View>
      {msg ? <Text style={styles.error}>{msg}</Text> : null}
      {busy ? (
        <View style={styles.row}><ActivityIndicator /><Text style={{ marginLeft: 8 }}>Bezig met inloggen...</Text></View>
      ) : (
        <TouchableOpacity onPress={handleLogin} style={styles.btn}>
          <Text style={styles.btnText}>Login</Text>
        </TouchableOpacity>
      )}
    </View>
  );
}

export default function App() {
  const [userId, setUserId] = useState(null);
  return userId ? <FinancieelBeheerApp userId={userId} /> : <LoginScreen onSuccess={setUserId} />;
}

const styles = StyleSheet.create({
  container: { padding: 16, gap: 12 },
  title: { fontSize: 20, fontWeight: '700' },
  inputGroup: { marginTop: 8 },
  input: { borderWidth: 1, borderColor: '#ccc', borderRadius: 8, paddingHorizontal: 12, paddingVertical: 10 },
  row: { flexDirection: 'row', alignItems: 'center' },
  btn: { backgroundColor: '#0f766e', paddingVertical: 10, paddingHorizontal: 14, borderRadius: 8, marginTop: 12 },
  btnText: { color: '#fff', fontWeight: '600' },
  error: { color: '#A80000' },
});
