#!/bin/bash
# WAI-007: Finance Kernel Isolation & ADR Compliance

audit_wai_007() {
    log_section "WAI-007: Finance Kernel Isolation"
    
    # ADR-03: Domain logic isolation
    log_check "ADR-03: Finance logic ONLY in domain/"
    if grep -r "calculate.*Income\|calculate.*Tax\|calculate.*Net" src/components/ src/screens/ 2>/dev/null; then
        fail "ADR-03 VIOLATION: Finance logic found outside domain/"
        ((violations++))
    else
        pass "ADR-03 COMPLIANT: Finance logic isolated"
    fi
    
    # ADR-05: Minor-unit enforcement (no floats in finance)
    log_check "ADR-05: No float operations in finance domain"
    if grep -r "parseFloat\|Number(\|toFixed" src/domain/finance/ 2>/dev/null | grep -v "// ADR-05 exception"; then
        fail "ADR-05 VIOLATION: Float operations in finance kernel"
        ((violations++))
    else
        pass "ADR-05 COMPLIANT: Only integer (minor-unit) operations"
    fi
    
    # ADR-06: Input hardening (central parser)
    log_check "ADR-06: Input parsing centralized"
    if grep -r "Number(\|parseInt(\|parseFloat(" src/components/ src/screens/ 2>/dev/null | \
       grep -v "parseMoneyInput\|// ADR-06 approved"; then
        fail "ADR-06 VIOLATION: Direct parsing in UI layer"
        ((violations++))
    else
        pass "ADR-06 COMPLIANT: All parsing via parseMoneyInput()"
    fi
    
    # ADR-11: Test coverage requirement
    log_check "ADR-11: Finance kernel test coverage > 80%"
    coverage=$(npm test -- --coverage --silent 2>/dev/null | grep "src/domain/finance" | awk '{print $4}' | tr -d '%')
    if [ -z "$coverage" ] || [ "$coverage" -lt 80 ]; then
        fail "ADR-11 VIOLATION: Finance coverage ${coverage}% (required: 80%+)"
        ((violations++))
    else
        pass "ADR-11 COMPLIANT: Finance coverage ${coverage}%"
    fi
    
    # ADR-04: UI components are dumb (no business logic)
    log_check "ADR-04: UI components have no calculation logic"
    if grep -r "function calculate\|const calculate\|export.*calculate" src/components/ 2>/dev/null; then
        fail "ADR-04 VIOLATION: Calculation functions in UI components"
        ((violations++))
    else
        pass "ADR-04 COMPLIANT: UI is dumb projector"
    fi
    
    # ADR-02: Type safety (no 'any' in new code)
    log_check "ADR-02: No 'any' types in finance domain"
    if grep -r ": any\|as any" src/domain/finance/ 2>/dev/null; then
        fail "ADR-02 VIOLATION: 'any' type found in finance domain"
        ((violations++))
    else
        pass "ADR-02 COMPLIANT: Strict typing enforced"
    fi
    
    # ADR-12: Traceability (changelog references WAI)
    log_check "ADR-12: Changelog contains WAI-007 reference"
    if grep -q "WAI-007" CHANGELOG.md 2>/dev/null; then
        pass "ADR-12 COMPLIANT: WAI-007 documented in changelog"
    else
        fail "ADR-12 VIOLATION: WAI-007 not in changelog"
        ((violations++))
    fi
}
