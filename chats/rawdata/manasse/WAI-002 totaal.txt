// WAI-002: Refactor (dom atom + StyleSheet + guards)
// Target: src/components/InputCounter.tsx
import * as React from 'react';
import { View, Text, TouchableOpacity, StyleSheet } from 'react-native';
import { useAppStyles } from '../styles/useAppStyles';

export type InputCounterProps = {
  value: number;
  min?: number;
  max?: number;
  onValueChange: (value: number) => void;
  accessibilityLabel?: string;
};

const InputCounter: React.FC<InputCounterProps> = ({
  value,
  min = 0,
  max,
  onValueChange,
  accessibilityLabel,
}) => {
  const { colors } = useAppStyles();

  // Defensieve guard: UI blijft stabiel zelfs als parent contract breekt
  const safeValue = typeof value === 'number' && !Number.isNaN(value) ? value : min;

  const canDec = safeValue > min;
  const canInc = max === undefined || safeValue < max;

  const dec = () => {
    const next = safeValue - 1;
    if (next >= min) onValueChange(next);
  };
  const inc = () => {
    const next = safeValue + 1;
    if (max === undefined || next <= max) onValueChange(next);
  };

  return (
    <View style={styles.row}>
      <TouchableOpacity
        onPress={dec}
        disabled={!canDec}
        style={[styles.btn, { backgroundColor: canDec ? colors.primary : colors.secondary }, !canDec && styles.disabled]}
        accessibilityRole="button"
        accessibilityLabel={`${accessibilityLabel ?? 'Waarde'} verlagen`}
      >
        <Text style={[styles.sign, { color: canDec ? colors.primaryText : colors.textSecondary }]}>−</Text>
      </TouchableOpacity>

      <Text style={[styles.value, { color: colors.textPrimary }]} accessibilityLabel={accessibilityLabel}>
        {safeValue}
      </Text>

      <TouchableOpacity
        onPress={inc}
        disabled={!canInc}
        style={[styles.btn, { backgroundColor: canInc ? colors.primary : colors.secondary }, !canInc && styles.disabled]}
        accessibilityRole="button"
        accessibilityLabel={`${accessibilityLabel ?? 'Waarde'} verhogen`}
      >
        <Text style={[styles.sign, { color: canInc ? colors.primaryText : colors.textSecondary }]}>+</Text>
      </TouchableOpacity>
    </View>
  );
};

const styles = StyleSheet.create({
  row: { flexDirection: 'row', alignItems: 'center' },
  btn: { paddingHorizontal: 20, paddingVertical: 10, borderRadius: 8 },
  sign: { fontSize: 20, fontWeight: 'bold' },
  value: { marginHorizontal: 20, fontSize: 24, fontWeight: 'bold', minWidth: 40, textAlign: 'center' },
  disabled: { opacity: 0.5 },
});

export default InputCounter;

import * as React from 'react';
import renderer from 'react-test-renderer';
import { TouchableOpacity } from 'react-native';
import InputCounter from '../InputCounter';

// Mock styling hook to avoid theme coupling in tests
jest.mock('../../styles/useAppStyles', () => ({
  useAppStyles: () => ({
    colors: {
      primary: '#0070f3',
      secondary: '#eaeaea',
      primaryText: '#ffffff',
      textSecondary: '#999999',
      textPrimary: '#111111',
    },
  }),
}));

describe('InputCounter', () => {
  it('renders stable (snapshot)', () => {
    const tree = renderer.create(
      <InputCounter value={3} min={1} max={5} onValueChange={() => {}} accessibilityLabel="Aantal" />
    ).toJSON();
    expect(tree).toMatchSnapshot();
  });

  it('increments and decrements within bounds', () => {
    const onChange = jest.fn();
    const testRenderer = renderer.create(
      <InputCounter value={2} min={1} max={3} onValueChange={onChange} accessibilityLabel="Adults" />
    );
    const root = testRenderer.root;
    const buttons = root.findAllByType(TouchableOpacity);
    const decBtn = buttons[0];
    const incBtn = buttons[1];

    // Decrement to 1
    decBtn.props.onPress();
    expect(onChange).toHaveBeenLastCalledWith(1);

    // Decrement again (blocked at min)
    onChange.mockClear();
    decBtn.props.onPress();
    expect(onChange).not.toHaveBeenCalled();

    // Increment to 2 and then 3
    incBtn.props.onPress();
    expect(onChange).toHaveBeenLastCalledWith(2);
    incBtn.props.onPress();
    expect(onChange).toHaveBeenLastCalledWith(3);

    // Increment again (blocked at max)
    onChange.mockClear();
    incBtn.props.onPress();
    expect(onChange).not.toHaveBeenCalled();
  });

  it('guards against invalid value by falling back to min', () => {
    const onChange = jest.fn();
    const testRenderer = renderer.create(
      // @ts-expect-error: testing defensive guard
      <InputCounter value={NaN as unknown as number} min={4} max={10} onValueChange={onChange} />
    );
    const root = testRenderer.root;
    const buttons = root.findAllByType(TouchableOpacity);
    const decBtn = buttons[0];

    // Since safeValue becomes min (4), decrement is blocked
    decBtn.props.onPress();
    expect(onChange).not.toHaveBeenCalled();
  });
});

# WAI-002 — InputCounter “Dom” hardening & StyleSheet guards

## Why
- Align with ADR-13 (UI is dumb): atom has no domain/state logic.
- Reduce coupling: no imports from context/domain.
- Visual integrity: consolidate inline styles into StyleSheet + tokens.

## What
- Refactor `src/components/InputCounter.tsx` to a pure atom with guards (`canDec/canInc`).
- Extract inline styles to a local `StyleSheet`; consume `useAppStyles()` tokens.
- No behavior changes; props contract preserved (`value/min/max/onValueChange`).

## Tests
- Unit: increment/decrement within bounds; defensive guard for invalid values.
- Snapshot: verifies stable render and token usage does not regress visuals.

## Scope & Non-goals
- No domain logic, no parsing, no FormContext usage.
- Money parsing deferred to WAI-004 (MoneyInput + NumericParser).

UPDATE VAN CTO:
Hier is de analyse en de volledige implementatie voor **WAI-002: InputCounter "Dom" hardening & isolation**. Dit pakket volgt strikt de ADR-13 richtlijn (UI is dom) en bereidt de component voor op integratie met de nieuwe domeinregels.

### 1. 🔍 Analyse van de Refactor

Op basis van de documentatie en de broncode zijn de volgende verbeteringen doorgevoerd om de component "Phoenix-ready" te maken:

* 
**Isolatie:** Alle imports vanuit de context of het domein zijn verwijderd; de component is nu een pure "atom".


* 
**Defensieve Logica:** Er is een `safeValue` guard toegevoegd die terugvalt op de minimumwaarde als de parent een ongeldige waarde (zoals `NaN`) meestuurt.


* 
**StyleSheet Consolidatie:** Inline styles zijn vervangen door een lokale `StyleSheet` gecombineerd met design tokens uit `useAppStyles`.


* 
**Expliciete Guards:** De knop-logica gebruikt nu expliciete `canDec` en `canInc` checks voor betere leesbaarheid en betrouwbaarheid.
