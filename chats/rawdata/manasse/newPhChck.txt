#!/bin/bash
# phoenix-check.sh - Project Phoenix Audit Entry Point
set -euo pipefail

# --- Basis Omgevingsvariabelen ---
BOLD='\033[1m'; DIM='\033[2m'; RED='\033[0;31m'; GREEN='\033[0;32m'; 
YELLOW='\033[0;33m'; BLUE='\033[0;34m'; NC='\033[0m';
SCRIPT_VERSION="1.3.0-modular"
PROJECT_ROOT=$(pwd)
CACHE_DIR="/tmp/phoenix-audit-cache"

# --- Module Loader ---
MODULE_DIR="./.phoenix"
for module in core.sh checkers.sh extractors.sh audits.sh reports.sh; do
    if [[ -f "$MODULE_DIR/$module" ]]; then
        # shellcheck source=/dev/null
        source "$MODULE_DIR/$module"
    else
        echo -e "${RED}❌ Kritieke fout: Module $module niet gevonden in $MODULE_DIR${NC}"
        exit 1
    fi
done

# --- Main Functie ---
main() {
    local PARALLEL=false VERBOSE=false DRY_RUN=false
    local INSTALL_HOOK=false UNINSTALL_HOOK=false

    # CLI Parsing
    while [[ $# -gt 0 ]]; do
        case "$1" in
            -v|--verbose) VERBOSE=true ;;
            -p|--parallel) PARALLEL=true ;;
            --install-hook) INSTALL_HOOK=true ;;
            --uninstall-hook) UNINSTALL_HOOK=true ;;
            *) echo "Onbekende optie: $1"; exit 1 ;;
        esac
        shift
    done

    load_config
    init_counters

    if [[ "$INSTALL_HOOK" == "true" ]]; then install_git_hook; exit 0; fi
    if [[ "$UNINSTALL_HOOK" == "true" ]]; then uninstall_git_hook; exit 0; fi

    echo -e "${BLUE}${BOLD}🚀 Phoenix Audit v${SCRIPT_VERSION}${NC}"
    
    # Uitvoering (Sequentieel of Parallel)
    if [[ "$PARALLEL" == "true" ]]; then
        audit_wai_001_003 & audit_wai_004 & audit_wai_006a & audit_wai_006e & wait
    else
        audit_wai_001_003
        audit_wai_004
        audit_wai_006a
        audit_wai_006e
    fi

    mock_generate_report
    show_summary
    local _exit_code=$?

    # Cleanup
    [[ "$VERBOSE" != "true" ]] && rm -rf "$CACHE_DIR"
    exit $_exit_code
}

main "$@"
