Option Explicit

Private Sub Workbook_Open()
    On Error GoTo ErrHandler
    EnsureDBHeaders

    ' Alleen vorige week samenvatten
    Dim dPrev As Date: dPrev = DateAdd("d", -7, Date)
    Dim w As Long: w = IsoWeek(dPrev)
    Dim y As Long: y = Year(dPrev)

    If Weekday(Date, vbMonday) = 1 Then
        If Not WeekTotaalBestaat(y, w) Then BerekenWeekTotaal
    Else
        If Not WeekTotaalBestaat(y, w) Then
            Dim antwoord As VbMsgBoxResult
            antwoord = MsgBox("Weekoverzicht van vorige week ontbreekt. Nu berekenen?", vbYesNo + vbQuestion, "Weekoverzicht")
            If antwoord = vbYes Then BerekenWeekTotaal
        End If
    End If

    ' Maandoverzicht: alleen als vorige maand niet aanwezig
    If Format(Date, "YYYY-MM") <> LaatsteMaandInMaandTabel() Then BerekenMaandTotaal

    ' Huidige maandblad: alleen aanmaken als nog niet bestaat
    MaakHuidigeMaandSheet
    Exit Sub

ErrHandler:
    MsgBox "Fout tijdens opstarten: " & Err.Description, vbExclamation
End Sub

Private Function FrmBgFileExists(ByVal p As String) As Boolean
    On Error Resume Next
    FrmBgFileExists = (Dir$(p) <> "")
    On Error GoTo 0
End Function

Private Function FrmGetTempFile(ByVal baseName As String, ByVal ext As String) As String
    Dim tmp As String
    tmp = Environ$("TEMP")
    If Len(tmp) = 0 Then tmp = Environ$("TMP")
    If Len(tmp) = 0 Then tmp = ThisWorkbook.Path
    If Right$(tmp, 1) <> "\" Then tmp = tmp & "\"
    FrmGetTempFile = tmp & baseName & "_" & Format(Now, "yyyymmdd_hhnnss") & ext
End Function

Private Function FrmGetReferentiesSheet() As Worksheet

End Function



Private Function FrmExportShapeToJpg(ByVal ws As Worksheet, ByVal shapeName As String) As String
    ' Exporteer een shape naar JPG (via tijdelijke Chart) en onthoud aspect-ratio
    On Error GoTo ErrHandler
    Dim shp As Shape
    Set shp = ws.Shapes(shapeName)

    ' Aspect ratio opslaan (breedte / hoogte) op moment van export
    If shp.Width > 0 And shp.Height > 0 Then
        mBgAspect = shp.Width / shp.Height
    Else
        mBgAspect = 0
    End If

    Dim ch As ChartObject
    Set ch = ws.ChartObjects.Add(Left:=shp.Left, Top:=shp.Top, Width:=shp.Width, Height:=shp.Height)
    shp.Copy
    ch.Activate
    ch.Chart.Paste

    Dim outPath As String
    outPath = FrmGetTempFile("bg_frmInvoer", ".jpg")
    ch.Chart.Export Filename:=outPath, FilterName:="JPG"
    ch.Delete

    FrmExportShapeToJpg = outPath
    Exit Function
ErrHandler:
    FrmExportShapeToJpg = vbNullString
End Function



Private Function FrmPickBackgroundPath() As String
    ' Probeert eerst: los JPG (landscape), dan shapes op 'Referenties'
    Dim fileLand As String
    fileLand = ThisWorkbook.Path & "\bg_frmInvoer_landscape.jpg"
    If FrmBgFileExists(fileLand) Then
        FrmPickBackgroundPath = fileLand
        Exit Function
    End If

    Dim wsRef As Worksheet: Set wsRef = FrmGetReferentiesSheet()
    If Not wsRef Is Nothing Then
        ' Eerst expliciet de landscape-shape (jouw naam)
        FrmPickBackgroundPath = FrmExportShapeToJpg(wsRef, "bg_frmInvoer_landscape")
        If Len(FrmPickBackgroundPath) > 0 Then Exit Function

        ' Zo niet, val terug op portrait
        FrmPickBackgroundPath = FrmExportShapeToJpg(wsRef, "bg_frmInvoer")
    End If
End Function

Private Sub FrmBuildBackground(ByVal imgPath As String)
    On Error GoTo ErrHandler

    ' Verwijder eerdere background
    Dim c As Control
    For Each c In Me.Controls
        If c.name = "imgBackground" Then
            Me.Controls.Remove "imgBackground"
            Exit For
        End If
    Next c

    ' Nieuwe Image-control
    Dim bg As MSForms.Image
    Set bg = Me.Controls.Add("Forms.Image.1", "imgBackground")

    ' We vullen NIET het hele venster; we berekenen een "contain & center" frame
    ' 1) Binnenruimte (zonder buitenmarges)
    Dim availW As Single, availH As Single
    availW = Me.InsideWidth - 2 * BG_MARGIN
    availH = Me.InsideHeight - 2 * BG_MARGIN

    ' 2) Aspect-ratio van achtergrond; fallback als onbekend
    Dim ar As Double
    ar = IIf(mBgAspect > 0, mBgAspect, 16# / 9#) ' default 16:9

    ' 3) Fitting: maximaal passend binnen availW x availH
    Dim fitW As Single, fitH As Single
    fitW = availW
    fitH = fitW / ar
    If fitH > availH Then
        fitH = availH
        fitW = fitH * ar
    End If

    ' 4) Centreren
    With bg
        .Left = BG_MARGIN + (availW - fitW) / 2
        .Top = BG_MARGIN + (availH - fitH) / 2
        .Width = fitW
        .Height = fitH
        .Picture = LoadPicture(imgPath)
        .PictureSizeMode = fmPictureSizeModeClip   ' geen automatische zoom/stretch
        .PictureAlignment = fmPictureAlignmentTopLeft
        .ZOrder 1   ' achter alle content
    End With
    Exit Sub
ErrHandler:
    MsgBox "Kon achtergrond niet laden: " & Err.Description, vbExclamation, "Background"
End Sub

Private Sub FrmLayoutSafeArea()
    On Error Resume Next

    ' ---- Pak de achtergrond-image ----
    Dim bg As MSForms.Image
    Set bg = Me.Controls("imgBackground")
    If bg Is Nothing Then Exit Sub

    ' ---- Bereken "contain & center" frame voor de achtergrond ----
    Dim availW As Single, availH As Single
    availW = Me.InsideWidth - 2 * BG_MARGIN
    availH = Me.InsideHeight - 2 * BG_MARGIN

    Dim ar As Double
    ar = IIf(mBgAspect > 0, mBgAspect, 16# / 9#) ' fallback 16:9

    Dim fitW As Single, fitH As Single
    fitW = availW
    fitH = fitW / ar
    If fitH > availH Then
        fitH = availH
        fitW = fitH * ar
    End If

    With bg
        .Left = BG_MARGIN + (availW - fitW) / 2
        .Top = BG_MARGIN + (availH - fitH) / 2
        .Width = fitW
        .Height = fitH
        .ZOrder 1
    End With

    ' ---- Safe area binnen de getoonde achtergrond ----

    Const L_SAFE_LEFT_PCT As Double = 0.12   ' was 0.10
    Const L_SAFE_RIGHT_PCT As Double = 0.88  ' was 0.90
    Const L_SAFE_TOP_PCT As Double = 0.15    ' was 0.12
    Const L_SAFE_BOTTOM_PCT As Double = 0.85 ' was 0.88


    Dim safeLeft As Single, safeTop As Single
    Dim safeWidth As Single, safeHeight As Single

    safeLeft = bg.Left + (L_SAFE_LEFT_PCT * bg.Width)
    safeTop = bg.Top + (L_SAFE_TOP_PCT * bg.Height)
    safeWidth = (L_SAFE_RIGHT_PCT - L_SAFE_LEFT_PCT) * bg.Width
    safeHeight = (L_SAFE_BOTTOM_PCT - L_SAFE_TOP_PCT) * bg.Height

    ' ---- Kolommen & afmetingen binnen safe area ----
    Dim colLbl As Single, colInp As Single, inpWidth As Single, t As Single

    colLbl = safeLeft
    colInp = safeLeft + COL_LABEL_TO_INPUT

    inpWidth = safeWidth - (colInp - safeLeft) - RIGHT_PADDING
    If inpWidth < 140 Then inpWidth = 140

' Maak invoervelden duidelijk kleiner en gecentreerd
Const INP_MAX As Single = 360   ' kies 300–380 naar smaak
If inpWidth > INP_MAX Then
    Dim extraSpace As Single
    extraSpace = (inpWidth - INP_MAX) / 2
    colInp = colInp + extraSpace
    inpWidth = INP_MAX
End If

    ' ---- Plaats controls rij-voor-rij ----
    t = safeTop

    ' Type
    Me.Controls("lblType").Left = colLbl
    Me.Controls("lblType").Top = t
    Me.Controls("optUitgave").Left = colInp
    Me.Controls("optUitgave").Top = t
    Me.Controls("optInkomen").Left = Me.Controls("optUitgave").Left + 160
    Me.Controls("optInkomen").Top = t

    ' Datum
    t = t + ROW_SPACING
    Me.Controls("lblDatum").Left = colLbl
    Me.Controls("lblDatum").Top = t
    Me.Controls("txtDatum").Left = colInp
    Me.Controls("txtDatum").Top = t
    Me.Controls("txtDatum").Width = inpWidth

    ' Categorie
    t = t + ROW_SPACING
    Me.Controls("lblCat").Left = colLbl
    Me.Controls("lblCat").Top = t
    Me.Controls("cmbCategorie").Left = colInp
    Me.Controls("cmbCategorie").Top = t
    Me.Controls("cmbCategorie").Width = inpWidth

    ' Subcategorie + "+"
    t = t + ROW_SPACING
    Me.Controls("lblSub").Left = colLbl
    Me.Controls("lblSub").Top = t
    Me.Controls("cmbSubcategorie").Left = colInp
    Me.Controls("cmbSubcategorie").Top = t
    Me.Controls("cmbSubcategorie").Width = inpWidth - (Me.Controls("btnNieuwSub").Width + 12)
    Me.Controls("btnNieuwSub").Left = colInp + Me.Controls("cmbSubcategorie").Width + 12
    Me.Controls("btnNieuwSub").Top = t

    ' Bedrag
    t = t + ROW_SPACING
    Me.Controls("lblBedrag").Left = colLbl
    Me.Controls("lblBedrag").Top = t
    Me.Controls("txtBedrag").Left = colInp
    Me.Controls("txtBedrag").Top = t
    Me.Controls("txtBedrag").Width = inpWidth

    ' Betaalwijze
    t = t + ROW_SPACING
    Me.Controls("lblBetaal").Left = colLbl
    Me.Controls("lblBetaal").Top = t
    Me.Controls("cmbBetaalwijze").Left = colInp
    Me.Controls("cmbBetaalwijze").Top = t
    Me.Controls("cmbBetaalwijze").Width = inpWidth

    ' Omschrijving
    t = t + ROW_SPACING
    Me.Controls("lblOms").Left = colLbl
    Me.Controls("lblOms").Top = t
    Me.Controls("txtOmschrijving").Left = colInp
    Me.Controls("txtOmschrijving").Top = t
    Me.Controls("txtOmschrijving").Width = inpWidth

    ' Buttons onderin safe area, gecentreerd
    ' --- Knoppen binnen het frame en gecentreerd ---
Dim btnY As Single
btnY = safeTop + safeHeight - 40          ' iets hoger (was 56); binnen het venster

Dim btnOps As MSForms.CommandButton, btnAnn As MSForms.CommandButton
Set btnOps = Me.Controls("btnOpslaan")
Set btnAnn = Me.Controls("btnAnnuleren")

' Houd ook hier een maximum totale breedte aan (esthetisch)
Const BTN_MAX_TOTAL As Single = 280       ' 120 + 24 + 120 = 264; 280 geeft wat marge

Dim btnTotalWidth As Single
btnTotalWidth = btnOps.Width + BTN_SPACING + btnAnn.Width
If btnTotalWidth > BTN_MAX_TOTAL Then btnTotalWidth = BTN_MAX_TOTAL

Dim btnLeft As Single
btnLeft = safeLeft + (safeWidth - btnTotalWidth) / 2

btnOps.Top = btnY
btnAnn.Top = btnY
btnOps.Left = btnLeft
btnAnn.Left = btnLeft + (btnTotalWidth - btnAnn.Width)

    ' ---- ScrollHeight dynamisch ----
    Dim contentBottom As Single, formInnerHeight As Single
    contentBottom = Me.Controls("btnOpslaan").Top + Me.Controls("btnOpslaan").Height + 24
    formInnerHeight = Me.InsideHeight

    If contentBottom > formInnerHeight Then
        Me.ScrollHeight = contentBottom
    Else
        Me.ScrollHeight = formInnerHeight
    End If

    On Error GoTo 0
End Sub

Private Sub btnNieuwSub_Click()
    Dim cat As String: cat = CStr(Me.Controls("cmbCategorie").Value)
    If Len(Trim$(cat)) = 0 Then
        MsgBox "Kies eerst een categorie.", vbExclamation
        Exit Sub
    End If

    Dim nm As String
    nm = InputBox("Nieuwe subcategorie voor " & cat & ":", "Subcategorie toevoegen")
    If Len(Trim$(nm)) = 0 Then Exit Sub

    Dim wsRef As Worksheet: Set wsRef = FrmGetReferentiesSheet()
    If wsRef Is Nothing Then
        MsgBox "Blad 'Referenties' ontbreekt.", vbCritical
        Exit Sub
    End If

    Dim loSub As ListObject
    On Error Resume Next
    Set loSub = wsRef.ListObjects("tblSubcategorieen")
    On Error GoTo 0
    If loSub Is Nothing Then
        MsgBox "Tabel 'tblSubcategorieen' ontbreekt.", vbCritical
        Exit Sub
    End If

    ' AddSub staat in jouw TableSetup-module
    AddSub loSub, cat, nm

    ' Herladen lijst
    cmbCategorie_Change
    Me.Controls("cmbSubcategorie").Value = nm
End Sub

Private Sub cmbCategorie_Change()
    Dim wsRef As Worksheet: Set wsRef = FrmGetReferentiesSheet()
    If wsRef Is Nothing Then Exit Sub

    Dim loSub As ListObject
    On Error Resume Next
    Set loSub = wsRef.ListObjects("tblSubcategorieen")
    On Error GoTo 0

    Dim cmbSub As MSForms.ComboBox: Set cmbSub = Me.Controls("cmbSubcategorie")
    cmbSub.Clear

    If loSub Is Nothing Or loSub.DataBodyRange Is Nothing Then
        cmbSub.Enabled = False
        Exit Sub
    End If

    Dim r As Range, cat As String: cat = CStr(Me.Controls("cmbCategorie").Value)
    For Each r In loSub.DataBodyRange.Rows
        If LCase$(Trim$(r.Cells(1, 1).Value)) = LCase$(Trim$(cat)) Then
            cmbSub.AddItem CStr(r.Cells(1, 2).Value)
        End If
    Next r

    cmbSub.Enabled = (cmbSub.ListCount > 0)
End Sub

Private Sub btnOpslaan_Click()
    On Error GoTo ErrHandler

    If Len(Trim$(Me.Controls("cmbCategorie").Value)) = 0 Then MsgBox "Kies een categorie.", vbExclamation: Exit Sub
    If Len(Trim$(Me.Controls("txtBedrag").Value)) = 0 Then MsgBox "Voer een bedrag in.", vbExclamation: Exit Sub

    Dim d As Date: d = DateValue(Me.Controls("txtDatum").Value)
    Dim bedrag As Double: bedrag = ParseAmount(CStr(Me.Controls("txtBedrag").Value)) ' uit Utils
    If bedrag <= 0 Then MsgBox "Bedrag moet > 0 zijn.", vbExclamation: Exit Sub

    Dim soort As String: soort = IIf(Me.Controls("optUitgave").Value, "Uitgave", "Inkomen")
    Dim beta As String: beta = CStr(Me.Controls("cmbBetaalwijze").Value)
    Dim cat As String: cat = CStr(Me.Controls("cmbCategorie").Value)
    Dim subcat As String: subcat = CStr(Me.Controls("cmbSubcategorie").Value)
    Dim oms As String: oms = CStr(Me.Controls("txtOmschrijving").Value)

    Dim wsDB As Worksheet: Set wsDB = ThisWorkbook.Worksheets("Database")
    EnsureDBHeaders ' uit Utils

    Dim lo As ListObject
    On Error Resume Next
    Set lo = wsDB.ListObjects("tblDagelijkseTransacties")
    On Error GoTo 0

    If lo Is Nothing Then
        Dim r As Long: r = wsDB.Cells(wsDB.Rows.Count, 1).End(xlUp).Row + 1
        wsDB.Cells(r, 1).Value = d
        wsDB.Cells(r, 2).Value = Year(d)
        wsDB.Cells(r, 3).Value = Format(d, "YYYY-MM")
        wsDB.Cells(r, 4).Value = DatePart("ww", d, vbMonday, vbFirstFourDays)
        wsDB.Cells(r, 5).Value = Format(d, "dddd")
        wsDB.Cells(r, 6).Value = cat
        wsDB.Cells(r, 7).Value = subcat
        wsDB.Cells(r, 8).Value = oms
        wsDB.Cells(r, 9).Value = bedrag
        wsDB.Cells(r, 10).Value = beta
        wsDB.Cells(r, 11).Value = soort
    Else
        Dim newRow As ListRow: Set newRow = lo.ListRows.Add
        With newRow.Range
            .Cells(1, 1).Value = d
            .Cells(1, 2).Value = Year(d)
            .Cells(1, 3).Value = Format(d, "YYYY-MM")
            .Cells(1, 4).Value = DatePart("ww", d, vbMonday, vbFirstFourDays)

' Maak alleen een sheet aan voor de HUIDIGE maand als die nog niet bestaat
Public Sub MaakHuidigeMaandSheet()
    On Error GoTo ErrHandler
    Application.ScreenUpdating = False
    Application.Calculation = xlCalculationManual

    Dim mName As String: mName = MonthNameNL(Month(Date))
    If SheetBestaat(mName) Then GoTo CleanExit

    Dim wsV As Worksheet: Set wsV = ThisWorkbook.Worksheets("Vast")
    Dim wsDoel As Worksheet: Set wsDoel = ThisWorkbook.Worksheets.Add
    wsDoel.name = mName

    Dim loInk As ListObject: Set loInk = TryGetListObject(wsV, "tblVastInkomsten")
    Dim loUit As ListObject: Set loUit = TryGetListObject(wsV, "tblVastUitgaven")

    Dim r As Long: r = 1
    With wsDoel
        .Cells(r, 1).Value = "Overzicht voor " & mName
        .Cells(r, 1).Font.Bold = True
        r = r + 2

        ' Inkomsten
        .Cells(r, 1).Value = "Inkomsten": .Cells(r, 1).Font.Bold = True
        r = r + 1: .Cells(r, 1).Value = "Bron": .Cells(r, 2).Value = "Bedrag"
        r = r + 1
        Dim startInk As Long: startInk = r

        If Not loInk Is Nothing Then
            Dim rw As ListRow
            For Each rw In loInk.DataBodyRange.Rows
                If Nz(rw.Range.Cells(1, 3).Value, False) Then
                    .Cells(r, 1).Value = rw.Range.Cells(1, 1).Value
                   
.Cells(r, 2).Value = Round(ToMonthlyAmount( _
    CStr(rw.Range.Cells(1, 2).Value), _
    CDbl(Val(rw.Range.Cells(1, 4).Value)) _
), 2)

                    r = r + 1
                End If
            Next rw
        End If

        .Cells(r, 1).Value = "Totaal inkomsten"
        If r > startInk Then .Cells(r, 2).Formula = "=SUM(B" & startInk & ":B" & (r - 1) & ")" Else .Cells(r, 2).Value = 0
        Dim inkTotalRow As Long: inkTotalRow = r
        r = r + 2

        ' Uitgaven
        .Cells(r, 1).Value = "Uitgaven": .Cells(r, 1).Font.Bold = True
        r = r + 1: .Cells(r, 1).Value = "Categorie": .Cells(r, 2).Value = "Bedrag"
        r = r + 1
        Dim startUit As Long: startUit = r

        If Not loUit Is Nothing Then
            For Each rw In loUit.DataBodyRange.Rows
                If Nz(rw.Range.Cells(1, 3).Value, False) Then
                    .Cells(r, 1).Value = rw.Range.Cells(1, 1).Value
                    .Cells(r, 2).Value = Round(ToMonthlyAmount(CStr(rw.Range.Cells(1, 2).Value), CDbl(Val(rw.Range.Cells(1, 4).Value))), 2)
                    r = r + 1
                End If
            Next rw
        End If

        .Cells(r, 1).Value = "Totaal uitgaven"
        If r > startUit Then .Cells(r, 2).Formula = "=SUM(B" & startUit & ":B" & (r - 1) & ")" Else .Cells(r, 2).Value = 0
        Dim uitTotalRow As Long: uitTotalRow = r
        r = r + 2

        ' Saldo
        .Cells(r, 1).Value = "Saldo"
        .Cells(r, 2).Formula = "=B" & inkTotalRow & "-B" & uitTotalRow
        .Columns(1).AutoFit
        .Columns(2).NumberFormat = "#,##0.00"
    End With

CleanExit:
    Application.Calculation = xlCalculationAutomatic
    Application.ScreenUpdating = True
    Exit Sub

ErrHandler:
    MsgBox "Fout in MaakHuidigeMaandSheet: " & Err.Description, vbCritical
    Resume CleanExit
End Sub

Private Function Nz(ByVal v As Variant, Optional ByVal defaultValue As Variant) As Variant
    If IsError(v) Or IsEmpty(v) Or v = "" Then
        If IsMissing(defaultValue) Then Nz = 0 Else Nz = defaultValue
    Else
        Nz = v
    End If
End Function

Public Sub BerekenWeekTotaal()
    On Error GoTo ErrHandler
    Dim ws As Worksheet: Set ws = ThisWorkbook.Worksheets("Database")
    EnsureDBHeaders

    Dim lr As Long: lr = LastRow(ws, dbDate)
    If lr < 2 Then Exit Sub

    ' Vorige week
    Dim dPrev As Date: dPrev = DateAdd("d", -7, Date)
    Dim tgtWeek As Long: tgtWeek = IsoWeek(dPrev)
    Dim tgtYear As Long: tgtYear = Year(dPrev)

    Dim i As Long, total As Double
    Dim catDict As Object: Set catDict = CreateObject("Scripting.Dictionary")

    For i = 2 To lr
        If CLng(Val(ws.Cells(i, dbYear).Value)) = tgtYear And _
           CLng(Val(ws.Cells(i, dbWeek).Value)) = tgtWeek And _
           CStr(ws.Cells(i, dbType).Value) = "Uitgave" Then
            Dim amount As Double: amount = CDbl(Val(ws.Cells(i, dbAmount).Value))
            total = total + amount
            Dim cat As String: cat = CStr(ws.Cells(i, dbCategory).Value)
            If Len(Trim$(cat)) > 0 Then DictAddOrIncrement catDict, cat, 1
        End If
    Next i

    Dim mostCat As String, maxCount As Long, k As Variant
    For Each k In catDict.Keys
        If catDict(k) > maxCount Then maxCount = catDict(k): mostCat = CStr(k)
    Next k

    If ws.Cells(1, WEEK_SUMMARY_COL).Value = "" Then
        ws.Cells(1, WEEK_SUMMARY_COL).Resize(1, 5).Value = Array("Jaar|Week", "Jaar", "Weeknr", "Totaal", "Meest gebruikte categorie")
    End If
    Dim outRow As Long: outRow = LastRow(ws, WEEK_SUMMARY_COL) + 1
    ws.Cells(outRow, WEEK_SUMMARY_COL).Value = CStr(tgtYear) & "|" & CStr(tgtWeek)
    ws.Cells(outRow, WEEK_SUMMARY_COL + 1).Value = tgtYear
    ws.Cells(outRow, WEEK_SUMMARY_COL + 2).Value = tgtWeek
    ws.Cells(outRow, WEEK_SUMMARY_COL + 3).Value = total
    ws.Cells(outRow, WEEK_SUMMARY_COL + 4).Value = mostCat
    Exit Sub

ErrHandler:
    MsgBox "Fout in BerekenWeekTotaal: " & Err.Description, vbCritical
End Sub

Public Sub BerekenMaandTotaal()
    On Error GoTo ErrHandler
    Dim ws As Worksheet: Set ws = ThisWorkbook.Worksheets("Database")
    EnsureDBHeaders

    Dim lr As Long: lr = LastRow(ws, dbDate)
    If lr < 2 Then Exit Sub

    Dim prevMonthKey As String: prevMonthKey = Format(DateAdd("m", -1, Date), "YYYY-MM")

    Dim i As Long
    Dim totaal As Double, dagenCount As Long
    Dim weekDict As Object: Set weekDict = CreateObject("Scripting.Dictionary")
    Dim catDict As Object: Set catDict = CreateObject("Scripting.Dictionary")
    Dim catCountDict As Object: Set catCountDict = CreateObject("Scripting.Dictionary")
    Dim duursteDag As Double, duursteWeek As Double
    Dim maxCatValue As Double, minCatValue As Double
    Dim maxCatName As String, minCatName As String
    maxCatValue = 0: minCatValue = 1E+20

    For i = 2 To lr
        If CStr(ws.Cells(i, dbMonthKey).Value) = prevMonthKey And CStr(ws.Cells(i, dbType).Value) = "Uitgave" Then
            Dim bedrag As Double: bedrag = CDbl(Val(ws.Cells(i, dbAmount).Value))
            totaal = totaal + bedrag
            dagenCount = dagenCount + 1
            If bedrag > duursteDag Then duursteDag = bedrag

            Dim weekKey As String: weekKey = CStr(ws.Cells(i, dbYear).Value) & "-W" & CStr(ws.Cells(i, dbWeek).Value)
            DictAddOrIncrement weekDict, weekKey, bedrag

            Dim categorie As String: categorie = CStr(ws.Cells(i, dbCategory).Value)
            DictAddOrIncrement catDict, categorie, bedrag
            DictAddOrIncrement catCountDict, categorie, 1
        End If
    Next i

    Dim key As Variant
    For Each key In weekDict.Keys
        If weekDict(key) > duursteWeek Then duursteWeek = weekDict(key)
    Next key

    For Each key In catDict.Keys
        If catDict(key) > maxCatValue Then maxCatValue = catDict(key): maxCatName = CStr(key)
        If catDict(key) < minCatValue Then minCatValue = catDict(key): minCatName = CStr(key)
    Next key

    Dim overzichtRow As Long: overzichtRow = LastRow(ws, MONTH_SUMMARY_COL) + 1
    If ws.Cells(1, MONTH_SUMMARY_COL).Value = "" Then
        ws.Cells(1, MONTH_SUMMARY_COL).Resize(1, 7).Value = Array("mnd", "ttl", "gpd", "gpw", "dudg", "duwk", "m_u_a")
    End If

    ws.Cells(overzichtRow, MONTH_SUMMARY_COL).Value = prevMonthKey
    ws.Cells(overzichtRow, MONTH_SUMMARY_COL + 1).Value = totaal
    ws.Cells(overzichtRow, MONTH_SUMMARY_COL + 2).Value = IIf(dagenCount > 0, totaal / dagenCount, 0)
    ws.Cells(overzichtRow, MONTH_SUMMARY_COL + 3).Value = IIf(weekDict.Count > 0, totaal / weekDict.Count, 0)
    ws.Cells(overzichtRow, MONTH_SUMMARY_COL + 4).Value = duursteDag
    ws.Cells(overzichtRow, MONTH_SUMMARY_COL + 5).Value = duursteWeek
    ws.Cells(overzichtRow, MONTH_SUMMARY_COL + 6).Value = maxCatName

    ' Categorie-overzicht append met maandkolom
    Dim catStartCol As Long: catStartCol = MONTH_SUMMARY_COL + 8 'X
    If ws.Cells(1, catStartCol - 1).Value = "" Then
        ws.Cells(1, catStartCol - 1).Resize(1, 4).Value = Array("Maand", "Categorie", "Aantal", "Totaal")
    End If
    Dim r As Long: r = LastRow(ws, catStartCol - 1) + 1
    For Each key In catDict.Keys
        ws.Cells(r, catStartCol - 1).Value = prevMonthKey
        ws.Cells(r, catStartCol).Value = key
        ws.Cells(r, catStartCol + 1).Value = catCountDict(key)
        ws.Cells(r, catStartCol + 2).Value = catDict(key)
        r = r + 1
    Next key
    Exit Sub

ErrHandler:
    MsgBox "Fout in BerekenMaandTotaal: " & Err.Description, vbCritical
End Sub

Public Sub AddSub(ByVal loSub As ListObject, ByVal cat As String, ByVal subcat As String)

    Dim r As Range
    If loSub.DataBodyRange Is Nothing Then GoTo AddNew
    For Each r In loSub.DataBodyRange.Rows
        If LCase$(Trim$(r.Cells(1, 1).Value)) = LCase$(Trim$(cat)) And _
           LCase$(Trim$(r.Cells(1, 2).Value)) = LCase$(Trim$(subcat)) Then Exit Sub 'bestaat al

    Dim lr As ListRow: Set lr = loSub.ListRows.Add
    lr.Range.Cells(1, 1).Value = cat
    lr.Range.Cells(1, 2).Value = subcat
End Sub

Public Function ParseAmount(ByVal s As String) As Double
    Dim decSep As String: decSep = Application.International(xlDecimalSeparator)
    s = Trim$(s)
    If s = "" Then ParseAmount = 0: Exit Function
    s = Replace(Replace(s, ".", decSep), ",", decSep)
    ParseAmount = CDbl(s)
End Function

Public Function DictAddOrIncrement(ByRef dict As Object, ByVal key As Variant, Optional ByVal inc As Double = 1) As Double
    If Not dict.Exists(key) Then dict(key) = inc Else dict(key) = dict(key) + inc
    DictAddOrIncrement = dict(key)
End Function

Public Function ToMonthlyAmount(ByVal periode As String, ByVal bedrag As Double) As Double
    periode = LCase$(Trim$(periode))
    Select Case periode
        Case "jaar", "per jaar":         ToMonthlyAmount = bedrag / 12
        Case "kwartaal", "per kwartaal": ToMonthlyAmount = bedrag / 3
        Case "maand", "per maand":       ToMonthlyAmount = bedrag
        Case "4 weken":                   ToMonthlyAmount = bedrag * 13 / 12
        Case "week", "per week":         ToMonthlyAmount = bedrag * 52 / 12
        Case "dag", "per dag":           ToMonthlyAmount = bedrag * 365 / 12
        Case Else:                         ToMonthlyAmount = bedrag
    End Select
End Function

Public Function EnsureTable(ByVal ws As Worksheet, ByVal tableName As String, ByVal headerRange As Range) As ListObject
    Dim lo As ListObject
    Set lo = TryGetListObject(ws, tableName)
    If lo Is Nothing Then
        Set lo = ws.ListObjects.Add(xlSrcRange, headerRange.CurrentRegion, , xlYes)
        lo.name = tableName
    End If
    Set EnsureTable = lo
End Function

' --- Helpers: temp-pad en chart-export ---
Private Function GetTempFile(ByVal baseName As String, ByVal ext As String) As String
    ' ext vb. ".jpg" of ".png"
    Dim tmp As String
    tmp = Environ$("TEMP")
    If Len(tmp) = 0 Then tmp = Environ$("TMP")
    If Len(tmp) = 0 Then tmp = ThisWorkbook.Path
    If Right$(tmp, 1) <> "\" Then tmp = tmp & "\"
    GetTempFile = tmp & baseName & "_" & Format(Now, "yyyymmdd_hhnnss") & ext
End Function

Private Function ExportShapeToJpg(ByVal ws As Worksheet, ByVal shapeName As String) As String
    ' Exporteer een shape naar JPG via een tijdelijke Chart
    On Error GoTo ErrHandler
    Dim shp As Shape
    Set shp = ws.Shapes(shapeName)

    ' Maak een tijdelijke Chart-object (inline), kopieer shape erin en exporteer
    Dim ch As ChartObject
    Set ch = ws.ChartObjects.Add(Left:=shp.Left, Top:=shp.Top, Width:=shp.Width, Height:=shp.Height)

    ' Chart heeft een Chart-area; we plakken de shape als afbeelding in de chart
    shp.Copy
    ch.Activate
    ch.Chart.Paste

    ' Export
    Dim outPath As String
    outPath = GetTempFile("bg_frmInvoer", ".jpg")
    ch.Chart.Export Filename:=outPath, FilterName:="JPG"

    ' Opruimen
    ch.Delete

    ExportShapeToJpg = outPath
    Exit Function

ErrHandler:
    ExportShapeToJpg = vbNullString
End Function
Public Function ParseAmount(ByVal s As String) As Double
    Dim decSep As String: decSep = Application.International(xlDecimalSeparator) ' NL: ","
    s = Trim$(s)
    ' Normaliseer beide scheidingstekens naar lokale separator
    s = Replace(s, ".", decSep)
    s = Replace(s, ",", decSep)
    If Len(s) = 0 Then
        ParseAmount = 0
    Else
        ParseAmount = CDbl(s)
    End If
End Function
