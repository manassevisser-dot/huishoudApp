// App.tsx (login fallback-ready, TypeScript)
import React, { useState } from 'react';
import { View, Text, TextInput, TouchableOpacity, StyleSheet, ActivityIndicator } from 'react-native';
import FinancieelBeheerApp from './FinancieelBeheerApp.tsx'; // expliciet .tsx voor web/Expo resolutie

// Zet hier jouw n8n login URL; laat '' (leeg) voor mock-only
const N8N_LOGIN_URL: string = '';

type LoginPayload = { email: string; password: string };

async function authenticateWithFallback({ email, password }: LoginPayload): Promise<string> {
  // 1) Mock-only: geen URL → direct mock
  if (!N8N_LOGIN_URL || N8N_LOGIN_URL.trim() === '') {
    await new Promise(r => setTimeout(r, 300));
    return 'mock-user-id-123';
  }
  // 2) Probeer n8n, anders fallback → mock
  try {
    const resp = await fetch(N8N_LOGIN_URL, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ email, password }),
    });
    if (!resp.ok) {
      console.warn('n8n login niet OK:', resp.status);
      await new Promise(r => setTimeout(r, 200));
      return 'mock-user-id-123';
    }
    const data = await resp.json();
    if (!data?.userId) {
      console.warn('n8n login: userId ontbreekt');
      await new Promise(r => setTimeout(r, 200));
      return 'mock-user-id-123';
    }
    return data.userId as string;
  } catch (e: any) {
    console.warn('n8n login error, fallback naar mock:', e?.message);
    await new Promise(r => setTimeout(r, 200));
    return 'mock-user-id-123';
  }
}

type LoginScreenProps = { onSuccess: (uid: string) => void };
function LoginScreen({ onSuccess }: LoginScreenProps) {
  const [email, setEmail] = useState<string>('');
  const [password, setPassword] = useState<string>('');
  const [busy, setBusy] = useState<boolean>(false);
  const [msg, setMsg] = useState<string>('');

  const handleLogin = async () => {
    setBusy(true); setMsg('');
    try {
      const uid = await authenticateWithFallback({ email, password });
      onSuccess(uid);
    } catch (e: any) {
      setMsg(e?.message ?? 'Login mislukt (fallback).');
    } finally {
      setBusy(false);
    }
  };

  return (
    <View style={styles.container}>
      <Text style={styles.title}>Inloggen</Text>
      <View style={styles.inputGroup}>
        <Text>E-mail</Text>
        <TextInput style={styles.input} value={email} onChangeText={setEmail} keyboardType="email-address" />
      </View>
      <View style={styles.inputGroup}>
        <Text>Wachtwoord</Text>
        <TextInput style={styles.input} value={password} onChangeText={setPassword} secureTextEntry />
      </View>
      {msg ? <Text style={styles.error}>{msg}</Text> : null}
      {busy ? (
        <View style={styles.row}><ActivityIndicator /><Text style={{ marginLeft: 8 }}>Bezig met inloggen...</Text></View>
      ) : (
        <TouchableOpacity onPress={handleLogin} style={styles.btn}>
          <Text style={styles.btnText}>Login</Text>
        </TouchableOpacity>
      )}
    </View>
  );
}

export default function App() {
  const [userId, setUserId] = useState<string | null>(null);
  return userId ? <FinancieelBeheerApp userId={userId} /> : <LoginScreen onSuccess={setUserId} />;
}

const styles = StyleSheet.create({
  container: { padding: 16, gap: 12 },
  title: { fontSize: 20, fontWeight: '700' },
  inputGroup: { marginTop: 8 },
  input: { borderWidth: 1, borderColor: '#ccc', borderRadius: 8, paddingHorizontal: 12, paddingVertical: 10 },
  row: { flexDirection: 'row', alignItems: 'center', marginTop: 8 },
  btn: { backgroundColor: '#0f766e', paddingVertical: 10, paddingHorizontal: 14, borderRadius: 8, marginTop: 12 },
  btnText: { color: '#fff', fontWeight: '600' },
  error: { color: '#A80000' },
});
