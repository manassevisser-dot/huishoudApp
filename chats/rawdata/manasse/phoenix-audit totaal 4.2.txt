#!/bin/bash
# ===================================================================
# PROJECT PHOENIX: ULTIMATE AUDIT SCRIPT v3.0
# ===================================================================
# Features: Progress bars, parallel buffering, reports, git hooks,
#           fix suggestions, configurable thresholds
# ===================================================================

set -euo pipefail
IFS=$'\n\t'

readonly SCRIPT_VERSION="3.0.0"
readonly PROJECT_ROOT="${PROJECT_ROOT:-$(git rev-parse --show-toplevel 2>/dev/null || echo "$PWD")}"
readonly CACHE_DIR="/tmp/phoenix-audit-$(id -u)"
readonly CONFIG_FILE="${PROJECT_ROOT}/.phoenix-audit.conf"

# Colors
if [[ "${NO_COLOR:-}" == "1" ]] || [[ ! -t 1 ]]; then
    readonly RED='' GREEN='' BLUE='' YELLOW='' BOLD='' NC=''
    readonly ICON_OK="[OK]" ICON_FAIL="[FAIL]" ICON_WARN="[WARN]"
else
    readonly RED='\033[0;31m' GREEN='\033[0;32m' BLUE='\033[0;34m'
    readonly YELLOW='\033[1;33m' BOLD='\033[1m' NC='\033[0m'
    readonly ICON_OK="✅" ICON_FAIL="❌" ICON_WARN="⚠️"
fi

# Flags
VERBOSE="${VERBOSE:-false}"
PARALLEL="${PARALLEL:-false}"
DRY_RUN="${DRY_RUN:-false}"
PROGRESS="${PROGRESS:-true}"
GENERATE_REPORT="${GENERATE_REPORT:-false}"
REPORT_FORMAT="${REPORT_FORMAT:-html}"
INSTALL_HOOK="${INSTALL_HOOK:-false}"

# Default thresholds
declare -A THRESHOLDS=(
    ["min_ssot"]=1 ["min_reducer"]=3 ["min_kernel"]=2
    ["min_decimal"]=3 ["min_parsing"]=2 ["max_deep"]=0
    ["max_console"]=0 ["fsm_states"]=5
)

load_config() {
    [[ -f "$CONFIG_FILE" ]] && source "$CONFIG_FILE" 2>/dev/null || true
}

init_audit() {
    rm -rf "$CACHE_DIR"
    mkdir -p "$CACHE_DIR/locks" "$CACHE_DIR/sections"
    echo "0" > "$CACHE_DIR/total"
    echo "0" > "$CACHE_DIR/passed"
    echo "0" > "$CACHE_DIR/failed"
    : > "$CACHE_DIR/failed_items.txt"
}

atomic_inc() {
    local file="$CACHE_DIR/$1" lockfile="${file}.lock"
    [[ ! -f "$file" ]] && echo "0" > "$file"
    
    do_inc() {
        local v=$(cat "$file" | tr -dc '0-9'); echo $((${v:-0} + 1)) > "$file"
    }
    
    if command -v flock >/dev/null 2>&1; then
        ( flock -x 200; do_inc ) 200>"$lockfile"
    else
        do_inc
    fi
}

read_counter() {
    local v=$(cat "$CACHE_DIR/$1" 2>/dev/null | tr -dc '0-9'); echo "${v:-0}"
}

log_section() { echo; echo -e "${BOLD}${BLUE}━━━━━━━━━━ $* ━━━━━━━━━━${NC}"; }
log_verbose() { [[ "$VERBOSE" == "true" ]] && echo -e "  ${BLUE}→${NC} $*"; return 0; }

show_progress() {
    [[ "$PROGRESS" != "true" || ! -t 1 ]] && return
    local cur=$1 tot=$2 msg="${3:-Processing}"
    local pct=$((cur * 100 / tot)) filled=$((pct / 2)) empty=$((50 - filled))
    local bar=""; for ((i=0;i<filled;i++)); do bar+="█"; done
    for ((i=0;i<empty;i++)); do bar+="░"; done
    printf "\r${BLUE}${msg}:${NC} [${bar}] ${BOLD}${pct}%%${NC} (${cur}/${tot})"
    [[ "$cur" -eq "$tot" ]] && echo
}

safe_grep() {
    local pat="$1" tgt="$2" excl="${3:-}" res=""
    if command -v rg &>/dev/null; then
        res=$(rg -n --type-add 'ts:*.{ts,tsx}' --type ts -e "$pat" "$tgt" 2>/dev/null | \
              { [[ -n "$excl" ]] && grep -v "$excl" || cat; } || true)
    else
        local opts=(-rn --include="*.ts" --include="*.tsx" --include="*.js" --include="*.jsx")
        res=$(grep "${opts[@]}" -E "$pat" "$tgt" 2>/dev/null | \
              { [[ -n "$excl" ]] && grep -v "$excl" || cat; } || true)
    fi
    echo "$res"
}

count_lines() {
    local t="$1"; [[ -z "$t" ]] && { echo "0"; return; }
    local c=$(echo "$t" | grep -c '^' 2>/dev/null || echo "0")
    echo "$(($(echo "$c" | tr -dc '0-9')))"
}

suggest_fix() {
    local id="$1"
    case "$id" in
        deep_imports)
            echo -e "${YELLOW}💡 Add path aliases to tsconfig.json:${NC}"
            echo '   "paths": { "@/*": ["./src/*"] }'
            echo -e "   ${GREEN}import { x } from '@/utils'${NC}"
            ;;
        legacy_parsing)
            echo -e "${YELLOW}💡 Create centralized parser in src/utils/numbers.ts${NC}"
            echo '   export const parseNumber = (input: string) => Math.max(0, parseFloat(input.replace(",", ".")) || 0);'
            ;;
        console_statements)
            echo -e "${YELLOW}💡 Remove console.log or use proper logger${NC}"
            ;;
        reducer_actions)
            echo -e "${YELLOW}💡 Add declarative action types: SET_, TOGGLE_, UPDATE_, RESET_${NC}"
            ;;
        fsm_incomplete)
            echo -e "${YELLOW}💡 Define all 5 FSM states: INITIALIZING, HYDRATING, UNBOARDING, READY, ERROR${NC}"
            ;;
    esac
}

record_failure() {
    echo "$2" >> "$CACHE_DIR/failed_items.txt"
}

check_pattern() {
    local pat="$1" tgt="$2" desc="$3" min="${4:-1}" excl="${5:-}" id="${6:-}"
    atomic_inc "total"
    log_verbose "Check: $desc"
    
    [[ ! -e "$tgt" ]] && {
        echo -e "${RED}${ICON_FAIL} $desc ${BOLD}(not found)${NC}"
        atomic_inc "failed"; record_failure "$id" "$desc"; return 1
    }
    
    local m=$(safe_grep "$pat" "$tgt" "$excl") c=$(count_lines "$m")
    
    if [[ "$c" -ge "$min" ]]; then
        echo -e "${GREEN}${ICON_OK} $desc ${BOLD}($c)${NC}"
        atomic_inc "passed"
        [[ "$VERBOSE" == "true" && -n "$m" ]] && echo "$m" | head -3 | sed 's/^/    /'
    else
        echo -e "${RED}${ICON_FAIL} $desc ${BOLD}($c < $min)${NC}"
        atomic_inc "failed"; record_failure "$id" "$desc"
        [[ -n "$id" ]] && suggest_fix "$id"
        return 1
    fi
}

check_antipattern() {
    local pat="$1" tgt="$2" desc="$3" excl="${4:-}" id="${5:-}"
    atomic_inc "total"
    
    [[ ! -e "$tgt" ]] && {
        echo -e "${YELLOW}${ICON_WARN} $desc ${BOLD}(skip)${NC}"
        atomic_inc "passed"; return 0
    }
    
    local m=$(safe_grep "$pat" "$tgt" "$excl") c=$(count_lines "$m")
    
    if [[ "$c" -eq 0 ]]; then
        echo -e "${GREEN}${ICON_OK} $desc ${BOLD}(clean)${NC}"
        atomic_inc "passed"
    else
        echo -e "${RED}${ICON_FAIL} $desc ${BOLD}($c)${NC}"
        [[ -n "$m" ]] && echo "$m" | head -5 | sed 's/^/    /'
        atomic_inc "failed"; record_failure "$id" "$desc"
        [[ -n "$id" ]] && suggest_fix "$id"
        return 1
    fi
}

check_file_exists() {
    local fp="$1" desc="$2" mins="${3:-1}"
    atomic_inc "total"
    
    [[ ! -f "$fp" ]] && {
        echo -e "${RED}${ICON_FAIL} $desc (not found)${NC}"
        atomic_inc "failed"; record_failure "file" "$desc"; return 1
    }
    
    local sz=$(stat -f%z "$fp" 2>/dev/null || stat -c%s "$fp" 2>/dev/null || echo "0")
    
    if [[ "$sz" -lt "$mins" ]]; then
        echo -e "${RED}${ICON_FAIL} $desc (${sz}B < ${mins}B)${NC}"
        atomic_inc "failed"; record_failure "file" "$desc"; return 1
    fi
    
    echo -e "${GREEN}${ICON_OK} $desc (${sz}B)${NC}"
    atomic_inc "passed"
}

run_cmd() {
    local cmd="$1" desc="$2" allow="${3:-false}"
    atomic_inc "total"
    log_verbose "Run: $cmd"
    
    [[ "$DRY_RUN" == "true" ]] && {
        echo -e "${YELLOW}${ICON_WARN} $desc [DRY RUN]${NC}"
        atomic_inc "passed"; return 0
    }
    
    local out; out=$(eval "$cmd" 2>&1) && {
        echo -e "${GREEN}${ICON_OK} $desc${NC}"
        atomic_inc "passed"; return 0
    } || {
        [[ "$allow" == "true" ]] && {
            echo -e "${YELLOW}${ICON_WARN} $desc${NC}"
            atomic_inc "passed"; return 0
        } || {
            echo -e "${RED}${ICON_FAIL} $desc${NC}"
            atomic_inc "failed"; record_failure "cmd" "$desc"
            [[ "$VERBOSE" == "true" ]] && echo "$out" | tail -10 | sed 's/^/    /'
            return 1
        }
    }
}

audit_wai_001_003() {
    log_section "1. CORE & HOUSEHOLD (WAI-001-003)"
    run_cmd "npm run typecheck" "TypeScript"
    check_file_exists "golden_master_scenarios.json" "Golden Master" 50
    check_pattern "as const" "src/selectors/householdSelectors.ts" "SSOT" "${THRESHOLDS[min_ssot]}" "" "ssot"
    check_pattern "(SET_|TOGGLE_|UPDATE_|RESET_)" "src/context/FormContext.tsx" "Actions" "${THRESHOLDS[min_reducer]}" "" "reducer_actions"
    check_antipattern "\\b(calculate|compute|derive)\\(" "src/context/FormContext.tsx" "No logic in reducer" "^[[:space:]]*//|/\\*" "reducer"
}

audit_wai_004() {
    log_section "2. EURO-GARANTIE (WAI-004)"
    check_pattern "Math\\.max\\(0," "src/utils/numbers.ts" "Non-negative parser" 1 "" "parser"
    check_pattern "\\b(computeNetCents|toCents|fromCents)" "src/logic/finance.ts" "Kernel" "${THRESHOLDS[min_kernel]}" "" "kernel"
    check_antipattern "\\bparseFloat\\(|\\bparseInt\\(" "src/logic/finance.ts" "No unsafe parse" "" "unsafe"
    check_pattern "\\.toFixed\\(2\\)" "src/logic" "Decimal precision" "${THRESHOLDS[min_decimal]}" "" "decimal"
}

audit_wai_006a() {
    log_section "3. APP STATUS FSM (WAI-006A)"
    
    local fsm="src/app/hooks/useAppOrchestration.ts"
    atomic_inc "total"
    
    if [[ -f "$fsm" ]]; then
        local found=0
        for s in "INITIALIZING" "HYDRATING" "UNBOARDING" "READY" "ERROR"; do
            grep -q "'$s'" "$fsm" 2>/dev/null && ((found++)) || true
        done
        
        if [[ "$found" -eq "${THRESHOLDS[fsm_states]}" ]]; then
            echo -e "${GREEN}${ICON_OK} FSM complete (${found}/${THRESHOLDS[fsm_states]})${NC}"
            atomic_inc "passed"
        else
            echo -e "${RED}${ICON_FAIL} FSM incomplete (${found}/${THRESHOLDS[fsm_states]})${NC}"
            atomic_inc "failed"; record_failure "fsm_incomplete" "FSM"
            suggest_fix "fsm_incomplete"
        fi
    else
        echo -e "${RED}${ICON_FAIL} FSM not found${NC}"
        atomic_inc "failed"; record_failure "fsm" "FSM file"
    fi
    
    check_pattern "switch.*\\(status\\)" "src/App.tsx" "Pure projector" 1 "" "projector"
    check_antipattern "useEffect\\(.*\\[\\]|fetch\\(|axios\\." "src/App.tsx" "No side effects" "^[[:space:]]*//|/\\*" "effects"
}

audit_wai_006d() {
    log_section "4. INPUT HARDENING (WAI-006D)"
    check_antipattern "parseFloat\\(|Number\\([^)]*\\)" "src/ui" "No legacy parse" "MoneyField\\.tsx" "legacy_parsing"
    check_pattern "export.*function.*(parse|sanitize|validate)" "src/utils/numbers.ts" "Centralized parse" "${THRESHOLDS[min_parsing]}" "" "centralized"
}

audit_wai_006e() {
    log_section "5. CLEANUP (WAI-006E)"
    
    atomic_inc "total"
    local deep=$(grep -r "\\.\\./\\.\\./\\.\\." src --include="*.ts" --include="*.tsx" 2>/dev/null || true)
    local dc=$(count_lines "$deep")
    
    if [[ "$dc" -le "${THRESHOLDS[max_deep]}" ]]; then
        echo -e "${GREEN}${ICON_OK} No deep imports${NC}"
        atomic_inc "passed"
    else
        echo -e "${RED}${ICON_FAIL} Deep imports ($dc)${NC}"
        [[ "$VERBOSE" == "true" ]] && echo "$deep" | head -5 | sed 's/^/    /'
        atomic_inc "failed"; record_failure "deep_imports" "Deep imports"
        suggest_fix "deep_imports"
    fi
    
    check_pattern "\"@/\\*\":" "tsconfig.json" "Path aliases" 1 "" "aliases"
    check_antipattern "console\\.(log|debug|info)" "src" "No console" "(\\.test\\.|dev-tools|^[[:space:]]*//)" "console_statements"
}

generate_report() {
    [[ "$GENERATE_REPORT" != "true" ]] && return
    
    local t=$(read_counter "total") p=$(read_counter "passed") f=$(read_counter "failed")
    local r=$([[ "$t" -gt 0 ]] && echo $((p * 100 / t)) || echo 0)
    local g="F" gc="fail"
    [[ "$r" -ge 90 ]] && { g="A"; gc="pass"; }
    [[ "$r" -ge 80 && "$r" -lt 90 ]] && { g="B"; gc="warn"; }
    
    local file="${PROJECT_ROOT}/phoenix-audit-report.${REPORT_FORMAT}"
    
    if [[ "$REPORT_FORMAT" == "html" ]]; then
        cat > "$file" <<EOF
<!DOCTYPE html>
<html><head><meta charset="UTF-8"><title>Phoenix Audit</title>
<style>body{font-family:sans-serif;max-width:800px;margin:40px auto;padding:20px}
.pass{color:#28a745}.fail{color:#dc3545}.grade{font-size:4rem;text-align:center}
h1{color:#667eea}pre{background:#f5f5f5;padding:10px;overflow-x:auto}</style>
</head><body><h1>🚀 Phoenix Audit Report</h1>
<div class="grade $gc">Grade: $g ($r%)</div>
<h2>Statistics</h2><p>Total: $t | Passed: <span class="pass">$p</span> | Failed: <span class="fail">$f</span></p>
<h2>Failed Checks</h2><pre>$(sort -u "$CACHE_DIR/failed_items.txt" 2>/dev/null || echo "All checks passed!")</pre>
<p><small>Generated: $(date)</small></p></body></html>
EOF
    elif [[ "$REPORT_FORMAT" == "markdown" ]]; then
        cat > "$file" <<EOF
# 🚀 Phoenix Audit Report

**Grade:** $g ($r%)

## Statistics
- **Total:** $t
- **Passed:** $p
- **Failed:** $f

## Failed Checks
\`\`\`
$(sort -u "$CACHE_DIR/failed_items.txt" 2>/dev/null || echo "All checks passed!")
\`\`\`

---
*Generated: $(date)*
EOF
    elif [[ "$REPORT_FORMAT" == "json" ]]; then
        local fails=$(sort -u "$CACHE_DIR/failed_items.txt" 2>/dev/null | jq -R . | jq -s . || echo "[]")
        cat > "$file" <<EOF
{
  "version": "$SCRIPT_VERSION",
  "timestamp": "$(date -u +%Y-%m-%dT%H:%M:%SZ)",
  "stats": {"total": $t, "passed": $p, "failed": $f, "rate": $r},
  "grade": "$g",
  "failures": $fails
}
EOF
    fi
    
    echo -e "${GREEN}${ICON_OK} Report: $file${NC}"
}

install_git_hook() {
    local hook_file="${PROJECT_ROOT}/.git/hooks/pre-commit"
    
    cat > "$hook_file" <<'HOOK'
#!/bin/bash
echo "🔍 Running Phoenix audit..."
if bash phoenix-check.sh; then
    echo "✅ Audit passed"
    exit 0
else
    echo "❌ Audit failed - commit blocked"
    echo "Run: bash phoenix-check.sh -v"
    exit 1
fi
HOOK
    
    chmod +x "$hook_file"
    echo -e "${GREEN}${ICON_OK} Git hook installed: $hook_file${NC}"
}

show_summary() {
    local t=$(read_counter "total") p=$(read_counter "passed") f=$(read_counter "failed")
    local r=$([[ "$t" -gt 0 ]] && echo $((p * 100 / t)) || echo 0)
    local g="F"; [[ "$r" -ge 95 ]] && g="A+" || [[ "$r" -ge 90 ]] && g="A" || [[ "$r" -ge 85 ]] && g="B" || [[ "$r" -ge 70 ]] && g="C"
    
    log_section "AUDIT RESULTS"
    echo -e "Total:     ${BOLD}$t${NC}"
    echo -e "Passed:    ${GREEN}${BOLD}$p${NC}"
    echo -e "Failed:    ${RED}${BOLD}$f${NC}"
    echo -e "Rate:      ${BOLD}$r%${NC}"
    echo -e "Grade:     ${BOLD}$g${NC}"
    echo
    
    if [[ "$f" -gt 0 ]]; then
        echo -e "${RED}${BOLD}FAILED CHECKS:${NC}"
        sort -u "$CACHE_DIR/failed_items.txt" | sed 's/^/  • /'
        echo
        echo -e "${RED}${BOLD}⚠️  AUDIT FAILED${NC}"
        return 1
    else
        echo -e "${GREEN}${BOLD}🎉 AUDIT PASSED!${NC}"
        return 0
    fi
}

show_help() {
    cat <<EOF
${BOLD}Phoenix Audit v${SCRIPT_VERSION}${NC}

USAGE: $0 [OPTIONS]

OPTIONS:
  -v, --verbose       Verbose output
  -p, --parallel      Parallel execution
  --dry-run           Simulate
  --no-progress       Disable progress bar
  --report FORMAT     Generate report (html/markdown/json)
  --install-hook      Install git pre-commit hook
  -h, --help          Show help

EXAMPLES:
  $0                              # Run audit
  $0 -v --report html             # Verbose with HTML report
  $0 --parallel --report json     # Fast parallel with JSON
  $0 --install-hook               # Install git hook

CONFIG FILE: .phoenix-audit.conf
  THRESHOLDS[min_ssot]=2          # Override thresholds
  VERBOSE=true                    # Set defaults

EOF
}

main() {
    while [[ $# -gt 0 ]]; do
        case "$1" in
            -v|--verbose) VERBOSE=true ;;
            -p|--parallel) PARALLEL=true ;;
            --dry-run) DRY_RUN=true ;;
            --no-progress) PROGRESS=false ;;
            --report) GENERATE_REPORT=true; REPORT_FORMAT="$2"; shift ;;
            --install-hook) INSTALL_HOOK=true ;;
            -h|--help) show_help; exit 0 ;;
            *) echo "Unknown: $1"; show_help; exit 2 ;;
        esac
        shift
    done
    
    load_config
    init_audit
    
    [[ "$INSTALL_HOOK" == "true" ]] && { install_git_hook; exit 0; }
    
    echo -e "${BLUE}${BOLD}🚀 Phoenix Audit v${SCRIPT_VERSION}${NC}"
    echo -e "${BLUE}📁 $PROJECT_ROOT${NC}"
    echo -e "${BLUE}📅 $(date '+%Y-%m-%d %H:%M:%S')${NC}"
    [[ "$DRY_RUN" == "true" ]] && echo -e "${YELLOW}⚠️  DRY RUN${NC}"
    echo
    
    cd "$PROJECT_ROOT" || exit 1
    
    if [[ "$PARALLEL" == "true" ]]; then
        echo -e "${BLUE}Parallel mode...${NC}"; echo
        ( audit_wai_001_003 & audit_wai_004 & audit_wai_006a & audit_wai_006d & audit_wai_006e & wait )
    else
        local total=5 cur=0
        for audit in audit_wai_001_003 audit_wai_004 audit_wai_006a audit_wai_006d audit_wai_006e; do
            $audit; ((cur++)); show_progress "$cur" "$total" "Progress"
        done
    fi
    
    generate_report
    show_summary
    exit $?
}

main "$@"
