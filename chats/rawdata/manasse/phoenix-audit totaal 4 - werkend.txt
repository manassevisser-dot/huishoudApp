# ==================== CONFIGURATION ====================

readonly SCRIPT_VERSION="2.3.0"
readonly PROJECT_ROOT="${PROJECT_ROOT:-$(git rev-parse --show-toplevel 2>/dev/null || echo "$PWD")}"

# Colors & Icons (NO_COLOR compliant)
if [[ "${NO_COLOR:-}" == "1" ]] || [[ ! -t 1 ]]; then
    readonly RED='' GREEN='' BLUE='' YELLOW='' BOLD='' NC=''
    readonly ICON_OK="[OK]" ICON_FAIL="[FAIL]" ICON_WARN="[WARN]"
else

    readonly YELLOW='\033[1;33m' BOLD='\033[1m' NC='\033[0m'

# ==================== CORE UTILITIES ====================

init_audit() {
    rm -rf "$CACHE_DIR"
    mkdir -p "$CACHE_DIR/locks"
    echo "0" > "$CACHE_DIR/total"
    echo "0" > "$CACHE_DIR/passed"
    echo "0" > "$CACHE_DIR/failed"
    : > "$CACHE_DIR/failed_items.txt"
}

# Fixed atomic increment with proper number handling
atomic_inc() {
    local counter="$1"
    local file="$CACHE_DIR/$counter"
    local lockfile="${file}.lock"
    
    # Ensure file exists
    [[ ! -f "$file" ]] && echo "0" > "$file"
    
    # Read-increment-write atomically
    increment_operation() {
        # Read current value, strip all non-digits, default to 0
        local current
        current=$(cat "$file" 2>/dev/null | tr -dc '0-9')
        current="${current:-0}"
        
        # Increment and write back
        echo $((current + 1)) > "$file"
    }
    
    # Use flock if available, otherwise direct (not parallel-safe but works)
    if command -v flock >/dev/null 2>&1; then
        ( flock -x 200; increment_operation ) 200>"$lockfile"
    else
        increment_operation
    fi
}

# Read counter value safely
read_counter() {
    local counter="$1"
    local file="$CACHE_DIR/$counter"
    
    if [[ -f "$file" ]]; then
        local value
        value=$(cat "$file" | tr -dc '0-9')
        echo "${value:-0}"
    else
        echo "0"
    fi
}

log_section() {
    echo
    echo -e "${BOLD}${BLUE}━━━━━━━━━━ $* ━━━━━━━━━━${NC}"
}

log_verbose() {
    [[ "$VERBOSE" == "true" ]] && echo -e "  ${BLUE}→${NC} $*"
    return 0
}

# ==================== SMART GREP ENGINE ====================

# This is the KEY FIX: proper output handling
safe_grep() {
    local pattern="$1"
    local target="$2"
    local exclude_pattern="${3:-}"
    
    local result=""
    
    # Build grep command based on available tools
    if command -v rg &>/dev/null; then
        # Ripgrep (fastest)
        if [[ -n "$exclude_pattern" ]]; then
            result=$(rg -n --type-add 'ts:*.{ts,tsx}' --type ts \
                     -e "$pattern" "$target" 2>/dev/null | \
                     grep -v "$exclude_pattern" 2>/dev/null || true)
        else
            result=$(rg -n --type-add 'ts:*.{ts,tsx}' --type ts \
                     -e "$pattern" "$target" 2>/dev/null || true)
        fi
    else
        # Standard grep fallback
        local grep_opts=(-rn --include="*.ts" --include="*.tsx" --include="*.js" --include="*.jsx")
        
        if [[ -n "$exclude_pattern" ]]; then
            result=$(grep "${grep_opts[@]}" -E "$pattern" "$target" 2>/dev/null | \
                     grep -v "$exclude_pattern" 2>/dev/null || true)
        else
            result=$(grep "${grep_opts[@]}" -E "$pattern" "$target" 2>/dev/null || true)
        fi
    fi
    
    # Output the result (may be empty)
    echo "$result"
}

# FIXED: Proper counting without newline issues
count_lines() {
    local text="$1"
    
    # If empty, return 0
    if [[ -z "$text" ]]; then
        echo "0"
        return 0
    fi
    
    # Count non-empty lines
    local count
    count=$(echo "$text" | grep -c '^' 2>/dev/null || echo "0")
    
    # Ensure it's a valid number
    count=$(echo "$count" | tr -dc '0-9')
    echo "${count:-0}"
}

# ==================== CHECK FUNCTIONS ====================

check_pattern() {
    local pattern="$1"
    local target="$2"
    local desc="$3"
    local min_count="${4:-1}"
    local exclude="${5:-}"
    
    atomic_inc "total"
    
    log_verbose "Checking pattern: $pattern in $target"
    
    # Check if target exists
    if [[ ! -e "$target" ]]; then
        echo -e "${RED}${ICON_FAIL} $desc ${BOLD}(target not found: $target)${NC}"
        atomic_inc "failed"
        echo "$desc" >> "$CACHE_DIR/failed_items.txt"
        return 1
    fi
    
    # Perform search
    local matches
    matches=$(safe_grep "$pattern" "$target" "$exclude")
    
    # Count results - THIS IS THE FIX
    local count
    count=$(count_lines "$matches")
    
    # Evaluate
    if [[ "$count" -ge "$min_count" ]]; then
        echo -e "${GREEN}${ICON_OK} $desc ${BOLD}($count)${NC}"
        atomic_inc "passed"
        
        if [[ "$VERBOSE" == "true" && -n "$matches" ]]; then
            echo "$matches" | head -3 | sed 's/^/    /'
        fi
        return 0
    else
        echo -e "${RED}${ICON_FAIL} $desc ${BOLD}(found $count, expected ≥$min_count)${NC}"
        atomic_inc "failed"
        echo "$desc" >> "$CACHE_DIR/failed_items.txt"
        return 1
    fi
}

check_antipattern() {
    local pattern="$1"
    local target="$2"
    local desc="$3"
    local exclude="${4:-}"
    
    atomic_inc "total"
    
    log_verbose "Checking anti-pattern: $pattern in $target"
    
    # Check if target exists
    if [[ ! -e "$target" ]]; then
        echo -e "${YELLOW}${ICON_WARN} $desc ${BOLD}(target not found, skipping)${NC}"
        atomic_inc "passed"
        return 0
    fi
    
    # Perform search
    local matches
    matches=$(safe_grep "$pattern" "$target" "$exclude")
    
    # Count results
    local count
    count=$(count_lines "$matches")
    
    # Evaluate (anti-pattern should be 0)
    if [[ "$count" -eq 0 ]]; then
        echo -e "${GREEN}${ICON_OK} $desc ${BOLD}(clean)${NC}"
        atomic_inc "passed"
        return 0
    else
        echo -e "${RED}${ICON_FAIL} $desc ${BOLD}($count violation(s))${NC}"
        atomic_inc "failed"
        echo "$desc" >> "$CACHE_DIR/failed_items.txt"
        
        # Show violations
        if [[ -n "$matches" ]]; then
            echo "$matches" | head -5 | sed 's/^/    /'
            local remaining=$((count - 5))
            [[ "$remaining" -gt 0 ]] && echo "    ... and $remaining more"
        fi
        return 1
    fi
}

check_file_exists() {
    local filepath="$1"
    local desc="$2"
    local min_size="${3:-1}"
    
    atomic_inc "total"
    
    if [[ ! -f "$filepath" ]]; then
        echo -e "${RED}${ICON_FAIL} $desc ${BOLD}(not found)${NC}"
        atomic_inc "failed"
        echo "$desc" >> "$CACHE_DIR/failed_items.txt"
        return 1
    fi
    
    local size
    size=$(stat -f%z "$filepath" 2>/dev/null || stat -c%s "$filepath" 2>/dev/null || echo "0")
    
    if [[ "$size" -lt "$min_size" ]]; then
        echo -e "${RED}${ICON_FAIL} $desc ${BOLD}(too small: ${size}B)${NC}"
        atomic_inc "failed"
        echo "$desc" >> "$CACHE_DIR/failed_items.txt"
        return 1
    fi
    
    echo -e "${GREEN}${ICON_OK} $desc ${BOLD}(${size}B)${NC}"
    atomic_inc "passed"
    return 0
}

run_cmd() {
    local cmd="$1"
    local desc="$2"
    local allow_fail="${3:-false}"
    
    atomic_inc "total"
    
    log_verbose "Running: $cmd"
    
    if [[ "$DRY_RUN" == "true" ]]; then
        echo -e "${YELLOW}${ICON_WARN} $desc ${BOLD}[DRY RUN]${NC}"
        atomic_inc "passed"
        return 0
    fi
    
    local output
    local exit_code
    
    output=$(eval "$cmd" 2>&1) && exit_code=0 || exit_code=$?
    
    if [[ "$exit_code" -eq 0 ]]; then
        echo -e "${GREEN}${ICON_OK} $desc${NC}"
        atomic_inc "passed"
        return 0
    else
        if [[ "$allow_fail" == "true" ]]; then
            echo -e "${YELLOW}${ICON_WARN} $desc ${BOLD}(exit $exit_code)${NC}"
            atomic_inc "passed"
            return 0
        else
            echo -e "${RED}${ICON_FAIL} $desc ${BOLD}(exit $exit_code)${NC}"
            atomic_inc "failed"
            echo "$desc" >> "$CACHE_DIR/failed_items.txt"
            [[ "$VERBOSE" == "true" ]] && echo "$output" | tail -10 | sed 's/^/    /'
            return 1
        fi
    fi
}

}

# ==================== REPORTING ====================

generate_report() {
    # Read final counters
    local total passed failed
    total=$(read_counter "total")
    passed=$(read_counter "passed")
    failed=$(read_counter "failed")
    
    # Calculate pass rate
    local rate=0
    [[ "$total" -gt 0 ]] && rate=$((passed * 100 / total))
    
    log_section "AUDIT RESULTS"
    
    echo -e "Total checks:    ${BOLD}$total${NC}"
    echo -e "Passed:          ${GREEN}${BOLD}$passed${NC}"
    echo -e "Failed:          ${RED}${BOLD}$failed${NC}"
    echo -e "Pass rate:       ${BOLD}$rate%${NC}"
    echo
    
    # Determine grade
    local grade
    if [[ "$rate" -ge 95 ]]; then grade="A+"
    elif [[ "$rate" -ge 90 ]]; then grade="A"
    elif [[ "$rate" -ge 85 ]]; then grade="B"
    elif [[ "$rate" -ge 70 ]]; then grade="C"
    else grade="F"
    fi
    echo -e "Grade:           ${BOLD}$grade${NC}"
    echo
    
    # Show failures if any
    if [[ "$failed" -gt 0 ]]; then
        echo -e "${RED}${BOLD}FAILED CHECKS:${NC}"
        if [[ -f "$CACHE_DIR/failed_items.txt" ]]; then
            sort -u "$CACHE_DIR/failed_items.txt" | sed 's/^/  • /'
        fi
        echo
        echo -e "${RED}${BOLD}⚠️  AUDIT FAILED - Please fix the issues above${NC}"
        return 1
    else
        echo -e "${GREEN}${BOLD}🎉 PHOENIX AUDIT PASSED!${NC}"
        echo -e "${GREEN}All WAI-001 through WAI-006 requirements satisfied.${NC}"
        return 0
    fi
}

# ==================== MAIN ====================

show_help() {
    cat <<EOF
${BOLD}Phoenix Audit Script v${SCRIPT_VERSION}${NC}

USAGE:
  $0 [OPTIONS]

OPTIONS:
  -v, --verbose     Enable verbose output
  -p, --parallel    Run sections in parallel
  --dry-run         Simulate without executing commands
  -h, --help        Show this help

ENVIRONMENT VARIABLES:
  VERBOSE=true      Enable verbose mode
  PARALLEL=true     Enable parallel execution
  DRY_RUN=true      Enable dry-run mode
  NO_COLOR=1        Disable colored output

EXAMPLES:
  $0                    # Run audit
  $0 -v                 # Verbose mode
  $0 --parallel         # Parallel execution
  $0 --dry-run          # Test run without execution
  VERBOSE=true $0       # Via environment

EXIT CODES:
  0 - All checks passed
  1 - One or more checks failed
  2 - Invalid arguments

EOF
}

main() {
    # Parse arguments
    while [[ $# -gt 0 ]]; do
        case "$1" in
            -v|--verbose) VERBOSE=true ;;
            -p|--parallel) PARALLEL=true ;;
            --dry-run) DRY_RUN=true ;;
            -h|--help) show_help; exit 0 ;;
            *) echo "Unknown option: $1"; show_help; exit 2 ;;
        esac
        shift
    done
    
    # Initialize
    init_audit
    
    echo -e "${BLUE}${BOLD}🚀 Phoenix Audit v${SCRIPT_VERSION}${NC}"
    echo -e "${BLUE}📁 Project: $PROJECT_ROOT${NC}"
    echo -e "${BLUE}📅 Started: $(date '+%Y-%m-%d %H:%M:%S')${NC}"
    [[ "$DRY_RUN" == "true" ]] && echo -e "${YELLOW}${BOLD}⚠️  DRY RUN MODE${NC}"
    echo
    
    # Change to project directory
    cd "$PROJECT_ROOT" || exit 1
    
    # Run audit sections
    if [[ "$PARALLEL" == "true" ]]; then
        echo -e "${BLUE}Running in parallel mode...${NC}"
        echo
        (
            audit_wai_001_003 &
            audit_wai_004 &
            audit_wai_006a &
            audit_wai_006d &
            audit_wai_006e &
            wait
        )
    else
        audit_wai_001_003
        audit_wai_004
        audit_wai_006a
        audit_wai_006d
        audit_wai_006e
    fi
    
    # Generate and display report
    generate_report
    local exit_code=$?
    
    exit "$exit_code"
}

# Execute main function
main "$@"
