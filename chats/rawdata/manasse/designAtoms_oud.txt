import React from 'react';

export const COLORS = {
  primary: '#00A98F',
  secondary: '#FF6B6B',
  background: '#F9F9F9',
  text: '#333333',
  income: '#90EE90',
  lightText: '#666666',
  inputBackground: '#FFFFFF',
  border: '#DDDDDD',
  disabled: '#AAAAAA',
};

interface PrimaryButtonProps {
  text: string;
  onPress: () => void;
  disabled?: boolean;
  fullWidth?: boolean;
  icon?: React.ReactNode;
}

export const PrimaryButton: React.FC<PrimaryButtonProps> = ({ text, onPress, disabled = false, fullWidth = true, icon }): JSX.Element => (
  <button
    onClick={onPress}
    disabled={disabled}
    style={{
      backgroundColor: disabled ? COLORS.disabled : COLORS.primary,
      color: 'white',
      padding: '14px 24px',
      width: fullWidth ? '100%' : 'auto',
      opacity: disabled ? 0.5 : 1,
      cursor: disabled ? 'not-allowed' : 'pointer',
      border: 'none',
      fontSize: '16px',
      display: 'flex',
      alignItems: 'center',
      justifyContent: 'center',
      gap: '8px',
    }}
  >
    {text}
    {icon && icon}
  </button>
);

interface CurrencyInputProps {
  label: string;
  value: number | '';
  onChange: (value: number) => void;
  placeholder?: string;
  required?: boolean;
  type?: 'income' | 'expense';
  helperText?: string;
}

export const CurrencyInput: React.FC<CurrencyInputProps> = ({ label, value, onChange, placeholder = '0.00', required = false, type = 'expense', helperText }): JSX.Element => {
  const borderColor = type === 'income' ? COLORS.income : COLORS.secondary;
  const safeValue = value === '' ? '' : Number.isFinite(value) ? value : 0;

  return (
    <div style={{ marginBottom: '16px' }}>
      <label style={{ display: 'block', fontSize: '14px', fontWeight: 600, color: COLORS.text, marginBottom: '8px' }}>
        {label} {required && <span style={{ color: COLORS.secondary }}>*</span>}
      </label>
      <div style={{ position: 'relative' }}>
        <span style={{ position: 'absolute', left: '16px', top: '50%', transform: 'translateY(-50%)', fontSize: '18px', fontWeight: 600, color: COLORS.text }}>€</span>
        <input
          type="number"
          step="0.01"
          min="0"
          value={safeValue}
          onChange={(e: React.ChangeEvent<HTMLInputElement>) => {
            const num = parseFloat(e.target.value);
            onChange(Number.isFinite(num) ? num : 0);
          }}
          placeholder={placeholder}
          style={{ width: '100%', padding: '14px 16px 14px 40px', fontSize: '18px', fontWeight: 600, color: COLORS.text, backgroundColor: COLORS.inputBackground, border: `2px solid ${borderColor}`, borderRadius: '10px', outline: 'none' }}
          onFocus={(e: React.FocusEvent<HTMLInputElement>) => { e.currentTarget.style.borderWidth = '3px'; e.currentTarget.style.borderColor = borderColor; }}
          onBlur={(e: React.FocusEvent<HTMLInputElement>) => { e.currentTarget.style.borderWidth = '2px'; }}
        />
      </div>
      {helperText && <p style={{ fontSize: '12px', color: COLORS.lightText, marginTop: '6px', marginLeft: '4px' }}>{helperText}</p>}
    </div>
  );
};

interface SegmentedControlProps {
  options: { label: string; value: string | boolean }[];
  selectedValue: string | boolean | null;
  onChange: (value: string | boolean) => void;
  fullWidth?: boolean;
}

export const SegmentedControl: React.FC<SegmentedControlProps> = ({ options, selectedValue, onChange, fullWidth = true }): JSX.Element => (
  <div style={{ display: 'flex', gap: '8px', width: fullWidth ? '100%' : 'auto' }}>
    {options.map(option => {
      const isSelected = selectedValue === option.value;
      return (
        <button
          key={String(option.value)}
          onClick={() => onChange(option.value)}
          style={{ flex: 1, padding: '12px 16px', fontSize: '15px', fontWeight: 600, color: isSelected ? 'white' : COLORS.text, backgroundColor: isSelected ? COLORS.primary : COLORS.inputBackground, border: `2px solid ${isSelected ? COLORS.primary : COLORS.border}`, borderRadius: '10px', cursor: 'pointer' }}
        >
          {option.label}
        </button>
      );
    })}
  </div>
);

interface BodyTextProps {
  children: React.ReactNode;
  style?: React.CSSProperties;
  size?: 'small' | 'regular' | 'large';
  color?: string;
}

export const BodyText: React.FC<BodyTextProps> = ({ children, style = {}, size = 'regular', color }): JSX.Element => {
  const sizeMap = { small: '13px', regular: '15px', large: '17px' };
  return <p style={{ fontSize: sizeMap[size], fontWeight: 400, color: color ?? COLORS.text, lineHeight: '1.5', margin: 0, ...style }}>{children}</p>;
};

interface Heading2Props {
  children: React.ReactNode;
  style?: React.CSSProperties;
  color?: string;
}

export const Heading2: React.FC<Heading2Props> = ({ children, style = {}, color }): JSX.Element => (
  <h2 style={{ fontSize: '20px', fontWeight: 700, color: color ?? COLORS.primary, lineHeight: '1.3', margin: 0, marginBottom: '16px', ...style }}>{children}</h2>
);
