' Beschrijving: Workbook_Open - automatisch gegenereerde omschrijving.
Private Sub Workbook_Open()
    On Error GoTo ErrHandler
    EnsureDBHeaders

    ' Alleen vorige week samenvatten
    Dim dPrev As Date: dPrev = DateAdd("d", -7, Date)
    Dim w As Long: w = IsoWeek(dPrev)
    Dim y As Long: y = Year(dPrev)

    If Weekday(Date, vbMonday) = 1 Then
        If Not WeekTotaalBestaat(y, w) Then BerekenWeekTotaal
    Else
        If Not WeekTotaalBestaat(y, w) Then
            Dim antwoord As VbMsgBoxResult
            antwoord = MsgBox("Weekoverzicht van vorige week ontbreekt. Nu berekenen?", vbYesNo + vbQuestion, "Weekoverzicht")
            If antwoord = vbYes Then BerekenWeekTotaal
        End If
    End If

    ' Maandoverzicht: alleen als vorige maand niet aanwezig
    If Format(Date, "YYYY-MM") <> LaatsteMaandInMaandTabel() Then BerekenMaandTotaal

    ' Huidige maandblad: alleen aanmaken als nog niet bestaat
    MaakHuidigeMaandSheet
    Exit Sub

ErrHandler:
       Utils.LogError4 "ThisWorkbook", "Workbook_Open", Err.Number, Err.Description
    MsgBox "Fout in Workbook_Open: " & Err.Description, vbCritical
End Sub

Public Function GetSubcategories(ByVal cat As String) As Collection
    On Error GoTo ErrHandler
    Dim wsRef As Worksheet: Set wsRef = ThisWorkbook.Worksheets("Referenties")
    Dim lo As ListObject
    On Error Resume Next: Set lo = wsRef.ListObjects("tblSubcategorieen"): On Error GoTo 0
    If lo Is Nothing Then Exit Function

    Dim res As New Collection
    Dim rng As Range, r As Range
    Set rng = lo.DataBodyRange
    If rng Is Nothing Then
        Set GetSubcategories = res
        Exit Function
    End If

    For Each r In rng.Rows
        If LCase$(Trim$(CStr(r.Cells(1, 1).Value))) = LCase$(Trim$(cat)) Then
            On Error Resume Next
            res.Add CStr(r.Cells(1, 2).Value), CStr(r.Cells(1, 2).Value)
            On Error GoTo 0
        End If
    Next r

    Set GetSubcategories = res
Exit Function

ErrHandler:
    Utils.LogError4 "DataCategories", "GetSubcategories", Err.Number, Err.Description
    MsgBox "Fout in GetSubcategories: " & Err.Description, vbCritical

' Beschrijving: EnsureBackgroundImage - automatisch gegenereerde omschrijving.
Public Sub EnsureBackgroundImage(frm As Object)
    On Error GoTo ErrHandler
    frm.Controls.Remove "dyn_imgBackground"

    Dim imgPath As String
    imgPath = GetBackgroundPath()
    If Len(imgPath) = 0 Then Exit Sub

    Dim img As MSForms.Image
    Set img = frm.Controls.Add("Forms.Image.1", "dyn_imgBackground", True)

    Dim pic As StdPicture
    Set pic = LoadPicture(imgPath)
    If pic Is Nothing Then Exit Sub

    Dim BG_MARGIN As Single: BG_MARGIN = 6
    Dim availW As Single: availW = frm.InsideWidth - 2 * BG_MARGIN
    Dim availH As Single: availH = frm.InsideHeight - 2 * BG_MARGIN

    Dim ar As Double: ar = pic.Width / pic.Height
    If ar <= 0 Then ar = 16 / 9

    Dim fitW As Single: fitW = availW
    Dim fitH As Single: fitH = fitW / ar
    If fitH > availH Then
        fitH = availH
        fitW = fitH * ar
    End If

    With img
        .Left = BG_MARGIN + (availW - fitW) / 2
        .Top = BG_MARGIN + (availH - fitH) / 2
        .Width = fitW
        .Height = fitH
        .Picture = pic
        .PictureSizeMode = fmPictureSizeModeStretch
        .ZOrder 1
    End With
Exit Sub

ErrHandler:
    Utils.LogError4 "FormBackground", "EnsureBackgroundImage", Err.Number, Err.Description
    MsgBox "Fout in EnsureBackgroundImage: " & Err.Description, vbCritical
End Sub

Public Function GetBackgroundPath() As String

    Dim wsRef As Worksheet: Set wsRef = ThisWorkbook.Worksheets("Referenties")
    If wsRef Is Nothing Then Exit Function

    Dim shp As Shape

Public Sub HandleOpslaan_dyn(frm As Object)
    On Error GoTo ErrHandler
    
    Dim wsDB As Worksheet: Set wsDB = ThisWorkbook.Worksheets("Database")
    Utils.EnsureDBHeaders
    
    Dim outR As Long: outR = Utils.LastRow(wsDB, dbDate) + 1
    Dim d As Date: d = Date
    
    Dim cat As String: cat = frm.Controls(FormLayout.DYN_PREFIX & "cmbCategorie").Value
    Dim subcat As String: subcat = frm.Controls(FormLayout.DYN_PREFIX & "cmbSubcategorie").Value
    Dim bedrag As Double: bedrag = Utils.ParseAmount(frm.Controls(FormLayout.DYN_PREFIX & "txtBedrag").Text)
    Dim betaal As String: betaal = frm.Controls(FormLayout.DYN_PREFIX & "cmbBetaalwijze").Value
    
    If Len(cat) = 0 Then
        MsgBox "Kies een categorie", vbExclamation
        Exit Sub
    End If
    
    With wsDB
        .Cells(outR, dbDate).Value = d
        .Cells(outR, dbYear).Value = Year(d)
        .Cells(outR, dbMonthKey).Value = Format(d, "yyyy-mm")
        .Cells(outR, dbWeek).Value = Utils.IsoWeek(d)
        .Cells(outR, dbDay).Value = Format(d, "dddd")
        .Cells(outR, dbCategory).Value = cat
        .Cells(outR, dbSubcategory).Value = subcat
        .Cells(outR, dbAmount).Value = Round(bedrag, 2)
        .Cells(outR, dbPayment).Value = betaal
        .Cells(outR, dbType).Value = "Uitgave"
    End With
    
    MsgBox "Opgeslagen!", vbInformation
    Unload frm
    Exit Sub

ErrHandler:
    Utils.LogError4 "FormEvents", "HandleOpslaan_dyn", Err.Number, Err.Description
    MsgBox "Fout: " & Err.Description, vbCritical
End Sub

    Else

    Dim safeLeft As Single, safeTop As Single, safeWidth As Single, safeHeight As Single
    safeLeft = bgLeft + (L_SAFE_LEFT_PCT * bgW)
    safeTop = bgTop + (L_SAFE_TOP_PCT * bgH)
    safeWidth = (L_SAFE_RIGHT_PCT - L_SAFE_LEFT_PCT) * bgW
    safeHeight = (L_SAFE_BOTTOM_PCT - L_SAFE_TOP_PCT) * bgH

    If inpWidth > INP_MAX Then
        Dim extraSpace As Single
        extraSpace = (inpWidth - INP_MAX) / 2
        colInp = colInp + extraSpace
        inpWidth = INP_MAX
    End If

    Dim ctrl As Control

    Set ctrl = TryGetControl(frm, FormLayout.DYN_PREFIX & "lblDatumVandaag")

    For i = 0 To cmb.ListCount - 1
        s = CStr(cmb.List(i))
        If Len(s) > maxLen Then maxLen = Len(s)
    Next i

    Const MINW As Long = 120
    Const MAXW As Long = 360
    Dim newW As Long
    If maxLen = 0 Then
        newW = MINW
    Else
        newW = Application.Max(MINW, maxLen * 7)
        If newW > MAXW Then newW = MAXW
    End If
    cmb.Width = newW
Exit Sub

ErrHandler:
    Utils.LogError4 "FormLayout", "AutoFitDropdownDyn", Err.Number, Err.Description
    MsgBox "Fout in AutoFitDropdownDyn: " & Err.Description, vbCritical
End Sub

' Returns a field value from Referenties!tblLayoutSpecs for a given Key
Public Function GetLayoutSpec(ByVal keyName As String, ByVal fieldName As String) As Variant
    On Error GoTo ErrHandler
    Dim wsRef As Worksheet: Set wsRef = ThisWorkbook.Worksheets("Referenties")
    Dim lo As ListObject: Set lo = wsRef.ListObjects("tblLayoutSpecs")
    Dim hdr As Range, data As Range, r As Range, colIndex As Long
    Set hdr = lo.HeaderRowRange
    Set data = lo.DataBodyRange
    If data Is Nothing Then Exit Function
    colIndex = Application.Match(fieldName, Application.Transpose(hdr.Value), 0)
    If IsError(colIndex) Or colIndex <= 0 Then Exit Function
    For Each r In data.Rows
        If StrComp(CStr(r.Cells(1, 1).Value), keyName, vbTextCompare) = 0 Then
            GetLayoutSpec = r.Cells(1, colIndex).Value
            Exit Function
        End If
    Next r
    Exit Function
ErrHandler:
    Utils.LogError4 "LayoutSpecsHelpers", "GetLayoutSpec", Err.Number, Err.Description
    MsgBox "Fout in GetLayoutSpec: " & Err.Description, vbCritical
End Function

' Exports a shape from Referenties sheet to a temporary JPG path and returns the path
Public Function GetShapePath(ByVal shapeName As String, Optional ByVal suffix As String = "") As String
    On Error GoTo ErrHandler
    Dim wsRef As Worksheet: Set wsRef = ThisWorkbook.Worksheets("Referenties")
    Dim shp As Shape: Set shp = wsRef.Shapes(shapeName)
    If shp Is Nothing Then Exit Function
    Dim tmp As String: tmp = Utils.GetTempFile(shapeName & suffix, ".jpg")
    Dim ch As ChartObject
    Set ch = wsRef.ChartObjects.Add(Left:=shp.Left, Top:=shp.Top, Width:=shp.Width, Height:=shp.Height)
    shp.Copy: ch.Activate: ch.Chart.Paste
    ch.Chart.Export tmp, "JPG": ch.Delete
    If Dir(tmp) <> "" Then GetShapePath = tmp
    Exit Function
ErrHandler:
    Utils.LogError4 "LayoutSpecsHelpers", "GetLayoutSpec", Err.Number, Err.Description
    MsgBox "Fout in GetShapePath: " & Err.Description, vbCritical

End Function

' Creates/places an Image button on a form using specs for a logical key (Prev/Next/Save/Cancel/RowAdd/RowDel)
Public Sub PlaceButtonFromSpec(frm As Object, ByVal controlName As String, ByVal keyName As String)
    On Error GoTo ErrHandler
    Dim baseShape As String, hoverShape As String
    baseShape = CStr(GetLayoutSpec(keyName, "ShapeName"))
    hoverShape = baseShape & "_hover" ' naming convention
    Dim x As Single, y As Single, w As Single, h As Single
    x = CSng(GetLayoutSpec(keyName, "X"))
    y = CSng(GetLayoutSpec(keyName, "Y"))
    w = CSng(GetLayoutSpec(keyName, "Width"))
    h = CSng(GetLayoutSpec(keyName, "Height"))
    ' Create image control (or re-use if exists)
    Dim img As MSForms.Image
    Set img = FormLayout.TryGetImageDyn(frm, controlName)
    If img Is Nothing Then
        Set img = frm.Controls.Add("Forms.Image.1", controlName, True)
    End If
    ' Set default picture
    Dim picPath As String: picPath = GetShapePath(baseShape)
    If Len(picPath) > 0 Then img.Picture = LoadPicture(picPath)
    ' Position and size
    img.Left = x: img.Top = y: img.Width = w: img.Height = h
    ' Store hover base in Tag for event handlers
    img.Tag = "base=" & baseShape & "|hover=" & hoverShape
    Exit Sub
ErrHandler:
    Utils.LogError4 "LayoutSpecsHelpers", "PlaceButtonFromSpec", Err.Number, Err.Description
    MsgBox "Fout in PlaceButtonFromSpec: " & Err.Description, vbCritical
End Sub

' Parses "key=value" pairs in Image.Tag; format: "base=btnShapeLarge|hover=btnShapeLarge_hover"
Private Function ParseTag(ByVal tagText As String, ByVal key As String) As String
    On Error GoTo ErrHandler
    Dim parts() As String: parts = Split(tagText, "|")
    Dim i As Long, kv()
    For i = LBound(parts) To UBound(parts)
        kv = Split(parts(i), "=")
        If UBound(kv) = 1 Then
            If StrComp(kv(0), key, vbTextCompare) = 0 Then
                ParseTag = kv(1)
                Exit Function
            End If
        End If
    Next i
    Exit Function
ErrHandler:
    Utils.LogError4 "LayoutSpecsHelpers", "ParseTag", Err.Number, Err.Description
    MsgBox "Fout in ParseTag: " & Err.Description, vbCritical

End Function

                    ), 2)

            .Cells(r, 2).Formula = "=SUM(B" & startInk & ":B" & (r - 1) & ")"

        End If

        r = r + 1: .Cells(r, 1).Value = "Categorie": .Cells(r, 2).Value = "Bedrag"

' Beschrijving: BerekenWeekTotaal - automatisch gegenereerde omschrijving.
Public Sub BerekenWeekTotaal()
    On Error GoTo ErrHandler
    Dim ws As Worksheet: Set ws = ThisWorkbook.Worksheets("Database")
    EnsureDBHeaders

    Dim lr As Long: lr = LastRow(ws, dbDate)
    If lr < 2 Then Exit Sub

    ' Vorige week
    Dim dPrev As Date: dPrev = DateAdd("d", -7, Date)
    Dim tgtWeek As Long: tgtWeek = IsoWeek(dPrev)
    Dim tgtYear As Long: tgtYear = Year(dPrev)

    Dim i As Long, total As Double
    Dim catDict As Object: Set catDict = CreateObject("Scripting.Dictionary")

    For i = 2 To lr
        If CLng(Val(ws.Cells(i, dbYear).Value)) = tgtYear And _
           CLng(Val(ws.Cells(i, dbWeek).Value)) = tgtWeek And _
           CStr(ws.Cells(i, dbType).Value) = "Uitgave" Then
            Dim amount As Double: amount = CDbl(Val(ws.Cells(i, dbAmount).Value))
            total = total + amount
            Dim cat As String: cat = CStr(ws.Cells(i, dbCategory).Value)
            If Len(Trim$(cat)) > 0 Then DictAddOrIncrement catDict, cat, 1
        End If
    Next i

    Dim mostCat As String, maxCount As Long, k As Variant
    For Each k In catDict.keys
        If catDict(k) > maxCount Then maxCount = catDict(k): mostCat = CStr(k)
    Next k

    If ws.Cells(1, SummaryCols.colWeekSummaryStart).Value = "" Then
        ws.Cells(1, SummaryCols.colWeekSummaryStart).Resize(1, 5).Value = Array("Jaar|Week", "Jaar", "Weeknr", "Totaal", "Meest gebruikte categorie")
    End If
    Dim outRow As Long: outRow = LastRow(ws, SummaryCols.colWeekSummaryStart) + 1
    ws.Cells(outRow, SummaryCols.colWeekSummaryStart).Value = CStr(tgtYear) & "|" & CStr(tgtWeek)
    ws.Cells(outRow, SummaryCols.colWeekSummaryStart + 1).Value = tgtYear
    ws.Cells(outRow, SummaryCols.colWeekSummaryStart + 2).Value = tgtWeek
    ws.Cells(outRow, SummaryCols.colWeekSummaryStart + 3).Value = total
    ws.Cells(outRow, SummaryCols.colWeekSummaryStart + 4).Value = mostCat
    Exit Sub
    
ErrHandler:
    Utils.LogError4 "Reporting", "BerekenWeekTotaal", Err.Number, Err.Description
    MsgBox "Fout in BerekenWeekTotaal: " & Err.Description, vbCritical
End Sub

' Beschrijving: BerekenMaandTotaal - automatisch gegenereerde omschrijving.
Public Sub BerekenMaandTotaal()
    On Error GoTo ErrHandler
    Dim ws As Worksheet: Set ws = ThisWorkbook.Worksheets("Database")
    EnsureDBHeaders

    Dim lr As Long: lr = LastRow(ws, dbDate)
    If lr < 2 Then Exit Sub

    Dim prevMonthKey As String: prevMonthKey = Format(DateAdd("m", -1, Date), "YYYY-MM")

    Dim i As Long
    Dim totaal As Double, dagenCount As Long
    Dim weekDict As Object: Set weekDict = CreateObject("Scripting.Dictionary")
    Dim catDict As Object: Set catDict = CreateObject("Scripting.Dictionary")
    Dim catCountDict As Object: Set catCountDict = CreateObject("Scripting.Dictionary")
    Dim duursteDag As Double, duursteWeek As Double
    Dim maxCatValue As Double, minCatValue As Double
    Dim maxCatName As String, minCatName As String
    maxCatValue = 0: minCatValue = 1E+20

    For i = 2 To lr
        If CStr(ws.Cells(i, dbMonthKey).Value) = prevMonthKey And CStr(ws.Cells(i, dbType).Value) = "Uitgave" Then
            Dim bedrag As Double: bedrag = CDbl(Val(ws.Cells(i, dbAmount).Value))
            totaal = totaal + bedrag
            dagenCount = dagenCount + 1
            If bedrag > duursteDag Then duursteDag = bedrag

            Dim weekKey As String: weekKey = CStr(ws.Cells(i, dbYear).Value) & "-W" & CStr(ws.Cells(i, dbWeek).Value)
            DictAddOrIncrement weekDict, weekKey, bedrag

            Dim categorie As String: categorie = CStr(ws.Cells(i, dbCategory).Value)
            DictAddOrIncrement catDict, categorie, bedrag
            DictAddOrIncrement catCountDict, categorie, 1
        End If
    Next i

    Dim key As Variant
    For Each key In weekDict.keys
        If weekDict(key) > duursteWeek Then duursteWeek = weekDict(key)
    Next key

    For Each key In catDict.keys
        If catDict(key) > maxCatValue Then maxCatValue = catDict(key): maxCatName = CStr(key)
        If catDict(key) < minCatValue Then minCatValue = catDict(key): minCatName = CStr(key)
    Next key

    Dim overzichtRow As Long: overzichtRow = LastRow(ws, SummaryCols.colMonthSummaryStart) + 1
    If ws.Cells(1, SummaryCols.colMonthSummaryStart).Value = "" Then
        ws.Cells(1, SummaryCols.colMonthSummaryStart).Resize(1, 7).Value = Array("mnd", "ttl", "gpd", "gpw", "dudg", "duwk", "m_u_a")
    End If

    ws.Cells(overzichtRow, SummaryCols.colMonthSummaryStart).Value = prevMonthKey
    ws.Cells(overzichtRow, SummaryCols.colMonthSummaryStart + 1).Value = totaal
    ws.Cells(overzichtRow, SummaryCols.colMonthSummaryStart + 2).Value = IIf(dagenCount > 0, totaal / dagenCount, 0)
    ws.Cells(overzichtRow, SummaryCols.colMonthSummaryStart + 3).Value = IIf(weekDict.Count > 0, totaal / weekDict.Count, 0)
    ws.Cells(overzichtRow, SummaryCols.colMonthSummaryStart + 4).Value = duursteDag
    ws.Cells(overzichtRow, SummaryCols.colMonthSummaryStart + 5).Value = duursteWeek
    ws.Cells(overzichtRow, SummaryCols.colMonthSummaryStart + 6).Value = maxCatName

    ' Categorie-overzicht append met maandkolom
    Dim catStartCol As Long: catStartCol = SummaryCols.colMonthSummaryStart + 8 'X
    If ws.Cells(1, catStartCol - 1).Value = "" Then
        ws.Cells(1, catStartCol - 1).Resize(1, 4).Value = Array("Maand", "Categorie", "Aantal", "Totaal")
    End If
    Dim r As Long: r = LastRow(ws, catStartCol - 1) + 1
    For Each key In catDict.keys
        ws.Cells(r, catStartCol - 1).Value = prevMonthKey

' Converteer de huidige "Vast"-ranges naar tabellen en maak categorie/subcategorie-tabellen
Public Sub SetupVastEnCategorieTabellen()
    On Error GoTo ErrHandler
    
    Dim wsV As Worksheet
    Dim wsRef As Worksheet
    Dim rngInk As Range, rngUit As Range
    Dim loCat As ListObject, loSub As ListObject
    
    ' 1) Werkblad Vast ophalen
    Set wsV = ThisWorkbook.Worksheets("Vast")
    
    ' 2) Vaste Inkomsten: Named Range controleren
    On Error Resume Next
    Set rngInk = ThisWorkbook.names("rngVastInkomsten").RefersToRange
    On Error GoTo ErrHandler
    If Not rngInk Is Nothing Then
        Call Utils.EnsureTable(wsV, "tblVastInkomsten", rngInk)
    End If
    
    ' 3) Vaste Uitgaven: Named Range controleren
    On Error Resume Next
    Set rngUit = ThisWorkbook.names("rngVastUitgaven").RefersToRange
    On Error GoTo ErrHandler
    If Not rngUit Is Nothing Then
        Call Utils.EnsureTable(wsV, "tblVastUitgaven", rngUit)
    End If
    
    ' 4) Werkblad Referenties aanmaken indien niet aanwezig
    If Not Utils.SheetBestaat("Referenties") Then
        Set wsRef = ThisWorkbook.Worksheets.Add
        wsRef.name = "Referenties"
    Else
        Set wsRef = ThisWorkbook.Worksheets("Referenties")
    End If
    
    ' 5) Tabel tblCategorieen [Categorie]
    If wsRef.Cells(1, 1).Value = "" Then wsRef.Range("A1").Value = "Categorie"
    Set loCat = Utils.EnsureTable(wsRef, "tblCategorieen", wsRef.Range("A1"))
    
    ' 6) Tabel tblSubcategorieen [Categorie, Subcategorie]
    If wsRef.Cells(1, 3).Value = "" Then
        wsRef.Range("C1").Value = "Categorie"
        wsRef.Range("D1").Value = "Subcategorie"
    End If
    Set loSub = Utils.EnsureTable(wsRef, "tblSubcategorieen", wsRef.Range("C1:D1"))
    
    ' 7) Seed-data toevoegen
    Call SeedCategorieen(loCat, loSub)
    
    ' 8) Succesmelding
    MsgBox "Tabellen ingesteld: tblVastInkomsten/tblVastUitgaven (indien aanwezig), tblCategorieen, tblSubcategorieen.", vbInformation
    Exit Sub

ErrHandler:
    Utils.LogError4 "TableSetup", "SetupVastEnCategorieTabellen", Err.Number, Err.Description
    MsgBox "Fout in SetupVastEnCategorieTabellen: " & Err.Description, vbCritical
End Sub


' Beschrijving: SeedCategorieen - automatisch gegenereerde omschrijving.
Private Sub SeedCategorieen(ByVal loCat As ListObject, ByVal loSub As ListObject)
On Error GoTo ErrHandler
    Dim cats As Variant: cats = Array("Eten", "Vervoer", "Hapje/Drankje", "Cadeau", "Lisa", "Kyra", "Kleding e.d", "Overig")
    Dim i As Long
    For i = LBound(cats) To UBound(cats)
        If Not CategoryExists(loCat, CStr(cats(i))) Then AddCategory loCat, CStr(cats(i))
    Next i

    ' Lisa/Kyra subcategorieen
    AddSub loSub, "Lisa", "Zakgeld"
    AddSub loSub, "Lisa", "Kleding"
    AddSub loSub, "Lisa", "Onzin"
    AddSub loSub, "Kyra", "Zakgeld"
    AddSub loSub, "Kyra", "Kleding"
    AddSub loSub, "Kyra", "Onzin"
Exit Sub
ErrHandler:
    Utils.LogError4 "TableSetup", "SeedCategorieen", Err.Number, Err.Description
    MsgBox "Fout in SeedCategorieen: " & Err.Description, vbCritical
End Sub

' Beschrijving: AddSub - automatisch gegenereerde omschrijving.
Public Sub AddSub(ByVal loSub As ListObject, ByVal cat As String, ByVal subcat As String)
On Error GoTo ErrHandler
    ' Voeg subcategorie toe als niet bestaat
    Dim r As Range
    If loSub.DataBodyRange Is Nothing Then GoTo AddNew

        If LCase$(Trim$(r.Cells(1, 1).Value)) = LCase$(Trim$(cat)) And _
           LCase$(Trim$(r.Cells(1, 2).Value)) = LCase$(Trim$(subcat)) Then Exit Sub 'bestaat al
    Next r
AddNew:
    Dim lr As ListRow: Set lr = loSub.ListRows.Add
    lr.Range.Cells(1, 1).Value = cat
    lr.Range.Cells(1, 2).Value = subcat
Exit Sub

ErrHandler:
    Utils.LogError4 "TableSetup", "AddSub", Err.Number, Err.Description
    MsgBox "Fout in AddSub: " & Err.Description, vbCritical
End Sub

' Beschrijving: ParseAmount - automatisch gegenereerde omschrijving.
Public Function ParseAmount(ByVal s As String) As Double
    Dim decSep As String: decSep = Application.International(xlDecimalSeparator)
    s = Trim$(s)
    If s = "" Then ParseAmount = 0: Exit Function
    s = Replace(Replace(s, ".", decSep), ",", decSep)
    ParseAmount = CDbl(s)
End Function

' Beschrijving: ToMonthlyAmount - automatisch gegenereerde omschrijving.
Public Function ToMonthlyAmount(ByVal periode As String, ByVal bedrag As Double) As Double
    periode = LCase$(Trim$(periode))
    Select Case periode
        Case "jaar", "per jaar":         ToMonthlyAmount = bedrag / 12
        Case "kwartaal", "per kwartaal": ToMonthlyAmount = bedrag / 3
        Case "maand", "per maand":       ToMonthlyAmount = bedrag
        Case "4 weken":                   ToMonthlyAmount = bedrag * 13 / 12
        Case "week", "per week":         ToMonthlyAmount = bedrag * 52 / 12
        Case "dag", "per dag":           ToMonthlyAmount = bedrag * 365 / 12
        Case Else:                         ToMonthlyAmount = bedrag
    End Select
End Function

' --- Helpers: temp-pad en chart-export ---
' Beschrijving: GetTempFile - automatisch gegenereerde omschrijving.
Public Function GetTempFile(ByVal baseName As String, ByVal ext As String) As String
    ' ext vb. ".jpg" of ".png"
    On Error GoTo ErrHandler
    Dim tmp As String
    tmp = Environ$("TEMP")
    If Len(tmp) = 0 Then tmp = Environ$("TMP")
    If Len(tmp) = 0 Then tmp = ThisWorkbook.Path
    If Right$(tmp, 1) <> "\" Then tmp = tmp & "\"
    GetTempFile = tmp & baseName & "_" & Format(Now, "yyyymmdd_hhnnss") & ext
Exit Function

ErrHandler:
    Utils.LogError4 "Utils", "GetTempFile", Err.Number, Err.Description
    MsgBox "Fout in GetTempFile: " & Err.Description, vbCritical
End Function

' Beschrijving: ExportShapeToJpg - automatisch gegenereerde omschrijving.
Private Function ExportShapeToJpg(ByVal ws As Worksheet, ByVal shapeName As String) As String

    On Error GoTo ErrHandler
    Dim shp As Shape
    Set shp = ws.Shapes(shapeName)

    ' Maak een tijdelijke Chart-object (inline), kopieer shape erin en exporteer
    Dim ch As ChartObject
    Set ch = ws.ChartObjects.Add(Left:=shp.Left, Top:=shp.Top, Width:=shp.Width, Height:=shp.Height)

    ' Chart heeft een Chart-area; we plakken de shape als afbeelding in de chart
    shp.Copy
    ch.Activate
    ch.Chart.Paste

    ' Export
    Dim outPath As String
    outPath = GetTempFile("bg_frmInvoer", ".jpg")
    ch.Chart.Export Filename:=outPath, FilterName:="JPG"

    ' Opruimen
    ch.Delete

    ExportShapeToJpg = outPath
    Exit Function

ErrHandler:
    ExportShapeToJpg = vbNullString
    Utils.LogError4 "Utils", "ExportShapeToJpg", Err.Number, Err.Description
    MsgBox "Fout in ExportShapeToJpg: " & Err.Description, vbCritical
End Function

' Beschrijving: Txt_KeyPress - automatisch gegenereerde omschrijving.
Private Sub Txt_KeyPress(ByVal KeyAscii As MSForms.ReturnInteger)
    Dim dec As String
    dec = Application.International(xlDecimalSeparator)

    Select Case KeyAscii
        Case 48 To 57       ' 0–9 OK
        Case Asc(dec)      ' decimaal OK maar slechts 1
            If InStr(Txt.Text, dec) > 0 Then KeyAscii = 0
        Case 8             ' Backspace OK
        Case Else
            KeyAscii = 0   ' alles anders blokkeren
    End Select
End Sub

End Sub
