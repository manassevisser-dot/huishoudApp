Option Explicit
Private Sub Workbook_Open()
 On Error GoTo ErrHandler
 Utils.EnsureDBHeaders
 ' Alleen vorige week samenvatten
 Dim dPrev As Date: dPrev = DateAdd("d", -7, Date)
 Dim w As Long: w = Utils.IsoWeek(dPrev)
 Dim Y As Long: Y = Year(dPrev)
 If Weekday(Date, vbMonday) = 1 Then
   If Not Reporting.WeekTotaalBestaat(Y, w) Then Reporting.BerekenWeekTotaal
 Else
   If Not Reporting.WeekTotaalBestaat(Y, w) Then
     Dim antwoord As VbMsgBoxResult
     antwoord = MsgBox("Weekoverzicht van vorige week ontbreekt. Nu berekenen?", vbYesNo + vbQuestion, "Weekoverzicht")
     If antwoord = vbYes Then Reporting.BerekenWeekTotaal
   End If
 End If
 ' Maandoverzicht: alleen als vorige maand niet aanwezig
 If Format(Date, "YYYY-MM") <> Reporting.LaatsteMaandInMaandTabel() Then Reporting.BerekenMaandTotaal
 ' Huidige maandblad: alleen aanmaken als nog niet bestaat
 MonthlySheet.MaakHuidigeMaandSheet
 Exit Sub
ErrHandler:
 Utils.LogError4 "ThisWorkbook", "Workbook_Open", Err.Number, Err.Description
 MsgBox "Fout in Workbook_Open: " & Err.Description, vbCritical
End Sub

' ===== MODULE: FormBackground (v5.4 FIXED) =====
Option Explicit

Public Function GetBackgroundPath() As String
    On Error GoTo ErrHandler
    Dim wsRef As Worksheet: Set wsRef = ThisWorkbook.Worksheets("Referenties")
    If wsRef Is Nothing Then Exit Function

    Dim bgShapeName As String
    bgShapeName = LayoutSpecsHelpers.GetBackgroundShapeNameFromSpecs()
    If Len(bgShapeName) = 0 Then bgShapeName = "bg_frmInvoer_landscape"

    Dim shp As Shape
    On Error Resume Next
    Set shp = wsRef.Shapes(bgShapeName)
    On Error GoTo ErrHandler
    If shp Is Nothing Then
        Utils.LogError4 "FormBackground", "GetBackgroundPath", 0, "Shape niet gevonden: " & bgShapeName
        Exit Function
    End If

    Dim tmp As String
    tmp = Utils.GetTempFile_FIXED("frmInvoer_bg", ".jpg")

    ' 1) Shape.Export met retry en size-validatie
    Dim attempt As Long, exportSuccess As Boolean
    For attempt = 1 To 3
        On Error Resume Next
        shp.Export tmp, "JPG"
        ' wacht max 500ms en valideer file size
        Dim t0 As Double: t0 = Timer
        Do While Timer - t0 < 0.5
            DoEvents
            If Dir$(tmp) <> "" Then If FileLen(tmp) > 100 Then exportSuccess = True: Exit Do
        Loop
        On Error GoTo ErrHandler
        If exportSuccess Then Exit For Else If Dir$(tmp) <> "" Then Kill tmp
    Next attempt

    If exportSuccess Then
        GetBackgroundPath = tmp
        Exit Function
    End If

    ' 2) Fallback via ChartObject met bounds guards
    Dim ch As ChartObject
    Dim safeLeft As Double, safeTop As Double, safeWidth As Double, safeHeight As Double
    safeLeft = IIf(shp.Left < 0, 0, shp.Left)
    safeTop = IIf(shp.Top < 0, 0, shp.Top)
    safeWidth = IIf(shp.Width < 10, 10, shp.Width)
    safeHeight = IIf(shp.Height < 10, 10, shp.Height)

    On Error Resume Next
    Set ch = wsRef.ChartObjects.Add(safeLeft, safeTop, safeWidth, safeHeight)
    If ch Is Nothing Then
        Utils.LogError4 "FormBackground", "GetBackgroundPath", 0, "ChartObject.Add gefaald"
        Exit Function
    End If
    shp.CopyPicture Appearance:=xlScreen, Format:=xlPicture
    ch.Chart.Paste
    ch.Chart.Export tmp, "JPG"
    ch.Delete
    On Error GoTo ErrHandler

    If Dir$(tmp) <> "" And FileLen(tmp) > 100 Then
        GetBackgroundPath = tmp
    Else
        Utils.LogError4 "FormBackground", "GetBackgroundPath", 0, "Export volledig gefaald"
    End If
    Exit Function

ErrHandler:
    Utils.LogError4 "FormBackground", "GetBackgroundPath", Err.Number, Err.Description
    On Error Resume Next
    If Not ch Is Nothing Then ch.Delete
End Function

Option Explicit
Public Sub LoadCategoriesIntoCombo(frm As Object)
 On Error GoTo ErrHandler
 Dim ws As Worksheet: Set ws = ThisWorkbook.Worksheets("Referenties")
 Dim lo As ListObject: Set lo = ws.ListObjects("tblCategorieen")
 Dim cmb As MSForms.ComboBox: Set cmb = frm.Controls(FormLayout.DYN_PREFIX & "cmbCategorie")
 cmb.Clear
 Dim r As Range
 If lo.DataBodyRange Is Nothing Then Exit Sub
 For Each r In lo.DataBodyRange.Columns(1).Cells
  If Len(r.Value) > 0 Then cmb.AddItem r.Value
 Next r
 cmb.ListIndex = -1
 Exit Sub
ErrHandler:
 Utils.LogError4 "FormData", "LoadCategoriesIntoCombo", Err.Number, Err.Description
End Sub
Public Sub LoadSubcategoriesIntoCombo(frm As Object, ByVal cat As String)
 On Error GoTo ErrHandler
 Dim ws As Worksheet: Set ws = ThisWorkbook.Worksheets("Referenties")
 Dim lo As ListObject: Set lo = ws.ListObjects("tblSubcategorieen")
 Dim cmb As MSForms.ComboBox: Set cmb = frm.Controls(FormLayout.DYN_PREFIX & "cmbSubcategorie")
 cmb.Clear
 Dim r As Range
 If lo.DataBodyRange Is Nothing Then Exit Sub
 For Each r In lo.DataBodyRange.Rows
  If LCase$(Trim$(CStr(r.Cells(1, 1).Value))) = LCase$(Trim$(cat)) Then cmb.AddItem r.Cells(1, 2).Value
 Next r
 cmb.ListIndex = -1
 Exit Sub
ErrHandler:
 Utils.LogError4 "FormData", "LoadSubcategoriesIntoCombo", Err.Number, Err.Description
End Sub
Public Sub LoadPaymentMethodsIntoCombo(frm As Object)
 On Error GoTo ErrHandler
 Dim ws As Worksheet: Set ws = ThisWorkbook.Worksheets("Referenties")
 Dim cmb As MSForms.ComboBox: Set cmb = frm.Controls(FormLayout.DYN_PREFIX & "cmbBetaalwijze")
 cmb.Clear
 If WorksheetHasTable(ws, "tbl_payment") Then
  Dim lo As ListObject: Set lo = ws.ListObjects("tbl_payment")
  Dim r As Range: For Each r In lo.DataBodyRange.Columns(1).Cells: cmb.AddItem r.Value: Next r
 Else
  Dim defaults As Variant
  defaults = Array("Contant", "Pin", "Creditcard", "Tikkie", "iDEAL", "Automatische Incasso")
  Dim i As Long: For i = LBound(defaults) To UBound(defaults): cmb.AddItem defaults(i): Next i
 End If
 cmb.ListIndex = 0
 Exit Sub
ErrHandler:
 Utils.LogError4 "FormData", "LoadPaymentMethodsIntoCombo", Err.Number, Err.Description
End Sub
Public Function WorksheetHasTable(ws As Worksheet, tblName As String) As Boolean
 On Error Resume Next
 WorksheetHasTable = Not ws.ListObjects(tblName) Is Nothing
 On Error GoTo 0
End Function

Public Sub HookEventsDynamic(frm As Object)
 On Error GoTo ErrHandler
 ' Hover events voor Images
 Dim c As Control
 For Each c In frm.Controls
  If TypeOf c Is MSForms.Image Then
   Dim ie As CImgEvents
   Set ie = New CImgEvents
   Set ie.Img = c
   Set ie.ParentForm = frm
  End If
 Next c
 ' Combo events voor categorie
 Dim cmbCat As MSForms.ComboBox: Set cmbCat = FormLayout.TryGetControl(frm, FormLayout.DYN_PREFIX & "cmbCategorie")
 If Not cmbCat Is Nothing Then
  Dim ce As CCmbEvents
  Set ce = New CCmbEvents
  Set ce.cmb = cmbCat
  Set ce.ParentForm = frm
 End If
 ' Text events voor bedrag
 Dim txtB As MSForms.TextBox: Set txtB = FormLayout.TryGetControl(frm, FormLayout.DYN_PREFIX & "txtBedrag")
 If Not txtB Is Nothing Then
  Dim te As CTxtEvents
  Set te = New CTxtEvents
  Set te.Txt = txtB
 End If
 Exit Sub
ErrHandler:
 Utils.LogError4 "FormEvents", "HookEventsDynamic", Err.Number, Err.Description
End Sub

Public Sub HandleOpslaan_dyn(frm As Object)
 On Error GoTo ErrHandler
 Dim wsDB As Worksheet: Set wsDB = ThisWorkbook.Worksheets("Database")
 Utils.EnsureDBHeaders
 Dim outR As Long: outR = Utils.LastRow(wsDB, DBCols.dbDate) + 1
 Dim d As Date: d = Date
 Dim cat As String: cat = frm.Controls(FormLayout.DYN_PREFIX & "cmbCategorie").Value
 Dim subcat As String: subcat = frm.Controls(FormLayout.DYN_PREFIX & "cmbSubcategorie").Value
 Dim bedrag As Double: bedrag = Utils.ParseAmount_FIXED(frm.Controls(FormLayout.DYN_PREFIX & "txtBedrag").Text)
 Dim betaal As String: betaal = frm.Controls(FormLayout.DYN_PREFIX & "cmbBetaalwijze").Value
 If Len(cat) = 0 Then MsgBox "Kies een categorie", vbExclamation: Exit Sub
 With wsDB
  .Cells(outR, DBCols.dbDate).Value = d
  .Cells(outR, DBCols.dbYear).Value = Year(d)
  .Cells(outR, DBCols.dbMonthKey).Value = Format(d, "yyyy-mm")
  .Cells(outR, DBCols.dbWeek).Value = Utils.IsoWeek(d)
  .Cells(outR, DBCols.dbDay).Value = Format(d, "dddd")
  .Cells(outR, DBCols.dbCategory).Value = cat
  .Cells(outR, DBCols.dbSubcategory).Value = subcat
  .Cells(outR, DBCols.dbAmount).Value = Round(bedrag, 2)
  .Cells(outR, DBCols.dbPayment).Value = betaal
  .Cells(outR, DBCols.dbType).Value = "Uitgave"
 End With
 MsgBox "Opgeslagen!", vbInformation
 Unload frm
 Exit Sub
ErrHandler:
 Utils.LogError4 "FormEvents", "HandleOpslaan_dyn", Err.Number, Err.Description
 MsgBox "Fout: " & Err.Description, vbCritical

Option Explicit

' -------------------------------------------------------
' Win32 DPI via GetDC/GetDeviceCaps (stabiel op alle Win)
' -------------------------------------------------------
#If VBA7 Then
    Private Declare PtrSafe Function GetDC Lib "user32" (ByVal hwnd As LongPtr) As LongPtr
    Private Declare PtrSafe Function ReleaseDC Lib "user32" (ByVal hwnd As LongPtr, ByVal hdc As LongPtr) As Long
    Private Declare PtrSafe Function GetDeviceCaps Lib "gdi32" (ByVal hdc As LongPtr, ByVal nIndex As Long) As Long
#Else
    Private Declare Function GetDC Lib "user32" (ByVal hwnd As Long) As Long
    Private Declare Function ReleaseDC Lib "user32" (ByVal hwnd As Long, ByVal hdc As Long) As Long
    Private Declare Function GetDeviceCaps Lib "gdi32" (ByVal hdc As Long, ByVal nIndex As Long) As Long
#End If

Private Const LOGPIXELSX As Long = 88   ' horizontale DPI (Windows)

' Publiek: prefix voor dynamische controls
Public Const DYN_PREFIX As String = "dyn_"

' -------------------------------------------------------
' DPI schaalfactor (96 DPI = 100% ? factor = dpi / 96)
' -------------------------------------------------------
Private Function DpiScale() As Double
    On Error Resume Next
    #If VBA7 Then
        Dim hdc As LongPtr
    #Else
        Dim hdc As Long
    #End If
    Dim dpi As Long
    hdc = GetDC(0)
    If hdc <> 0 Then
        dpi = GetDeviceCaps(hdc, LOGPIXELSX)
        Call ReleaseDC(0, hdc)
    End If
    If dpi <= 0 Then dpi = 96
    DpiScale = dpi / 96#
End Function

' -------------------------------------------------------
' Safe-area layout (DPI-aware) voor dynamische forms
' -------------------------------------------------------
Public Sub FrmLayoutSafeAreaDyn(frm As Object)
    On Error GoTo ErrHandler

    ' Zorg dat configuratie-tabel bestaat (keys met defaults)
    MonthlySheet.EnsureConfigSheet

    Dim SCALE As Double: SCALE = DpiScale()

    ' Pixelwaarden (schalen met DPI)
    Dim BG_MARGIN As Double:           BG_MARGIN = MonthlySheet.ConfigGet("BG_MARGIN", 6) * SCALE
    Dim ROW_SPACING As Double:         ROW_SPACING = MonthlySheet.ConfigGet("ROW_SPACING", 34) * SCALE
    Dim LABEL_WIDTH As Double:         LABEL_WIDTH = MonthlySheet.ConfigGet("LABEL_WIDTH", 120) * SCALE
    Dim GAP_LABEL_TO_INPUT As Double:  GAP_LABEL_TO_INPUT = MonthlySheet.ConfigGet("GAP_LABEL_TO_INPUT", 12) * SCALE
    Dim RIGHT_PADDING As Double:       RIGHT_PADDING = MonthlySheet.ConfigGet("RIGHT_PADDING", 20) * SCALE
    Dim INP_MAX As Double:             INP_MAX = MonthlySheet.ConfigGet("INP_MAX", 360) * SCALE

    ' Percentages (blijven gelijk)
    Dim SAFE_LEFT_PCT As Double:       SAFE_LEFT_PCT = MonthlySheet.ConfigGet("SAFE_LEFT_PCT", 0.12)
    Dim SAFE_RIGHT_PCT As Double:      SAFE_RIGHT_PCT = MonthlySheet.ConfigGet("SAFE_RIGHT_PCT", 0.88)
    Dim SAFE_TOP_PCT As Double:        SAFE_TOP_PCT = MonthlySheet.ConfigGet("SAFE_TOP_PCT", 0.2)
    Dim SAFE_BOTTOM_PCT As Double:     SAFE_BOTTOM_PCT = MonthlySheet.ConfigGet("SAFE_BOTTOM_PCT", 0.8)

    ' Achtergrond image (dynamisch of berekend)
    Dim bgLeft As Single, bgTop As Single, bgW As Single, bgH As Single
    Dim bg As Object: Set bg = TryGetImageDyn(frm, DYN_PREFIX & "imgBackground")
    If bg Is Nothing Then
        bgLeft = BG_MARGIN: bgTop = BG_MARGIN
        bgW = frm.InsideWidth - 2 * BG_MARGIN
        bgH = frm.InsideHeight - 2 * BG_MARGIN
    Else
        bgLeft = bg.Left: bgTop = bg.Top: bgW = bg.Width: bgH = bg.Height
    End If
    If bgW < 0 Then bgW = 0
    If bgH < 0 Then bgH = 0

    ' Safe-area binnen achtergrond
    Dim safeLeft As Single, safeTop As Single, safeWidth As Single, safeHeight As Single
    safeLeft = bgLeft + (SAFE_LEFT_PCT * bgW)
    safeTop = bgTop + (SAFE_TOP_PCT * bgH)
    safeWidth = (SAFE_RIGHT_PCT - SAFE_LEFT_PCT) * bgW
    safeHeight = (SAFE_BOTTOM_PCT - SAFE_TOP_PCT) * bgH
    If safeWidth < 0 Then safeWidth = 0
    If safeHeight < 0 Then safeHeight = 0

    ' Kolommen en inputbreedte
    Dim colLbl As Single, colInp As Single, inpWidth As Single, t As Single
    colLbl = safeLeft
    colInp = safeLeft + LABEL_WIDTH + GAP_LABEL_TO_INPUT
    inpWidth = safeWidth - (colInp - safeLeft) - RIGHT_PADDING
    If inpWidth < 0 Then inpWidth = 0
    If inpWidth > INP_MAX Then
        Dim extraSpace As Single: extraSpace = (inpWidth - INP_MAX) / 2
        colInp = colInp + extraSpace
        inpWidth = INP_MAX
    End If

    ' Rij-plaatsing (labels/inputs)
    t = safeTop
    Dim ctrl As Object

    ' Datum label
    Set ctrl = TryGetControl(frm, DYN_PREFIX & "lblDatumVandaag"): If Not ctrl Is Nothing Then ctrl.Left = colLbl: ctrl.Top = t

    ' Categorie
    t = t + ROW_SPACING
    Set ctrl = TryGetControl(frm, DYN_PREFIX & "lblCat"): If Not ctrl Is Nothing Then ctrl.Left = colLbl: ctrl.Top = t
    Set ctrl = TryGetControl(frm, DYN_PREFIX & "cmbCategorie"): If Not ctrl Is Nothing Then ctrl.Left = colInp: ctrl.Top = t: ctrl.Width = inpWidth

    ' Subcategorie (+ nieuw sub)
    t = t + ROW_SPACING
    Set ctrl = TryGetControl(frm, DYN_PREFIX & "lblSub"): If Not ctrl Is Nothing Then ctrl.Left = colLbl: ctrl.Top = t
    Dim wSub As Single: wSub = Application.Max(0, inpWidth - 36 * SCALE)
    Set ctrl = TryGetControl(frm, DYN_PREFIX & "cmbSubcategorie"): If Not ctrl Is Nothing Then ctrl.Left = colInp: ctrl.Top = t: ctrl.Width = wSub
    Set ctrl = TryGetControl(frm, DYN_PREFIX & "btnNieuwSub"): If Not ctrl Is Nothing Then ctrl.Left = colInp + wSub + 8 * SCALE: ctrl.Top = t: ctrl.Width = 28 * SCALE: ctrl.Height = 22 * SCALE

    ' Bedrag
    t = t + ROW_SPACING
    Set ctrl = TryGetControl(frm, DYN_PREFIX & "lblBedrag"): If Not ctrl Is Nothing Then ctrl.Left = colLbl: ctrl.Top = t
    Set ctrl = TryGetControl(frm, DYN_PREFIX & "txtBedrag"): If Not ctrl Is Nothing Then ctrl.Left = colInp: ctrl.Top = t: ctrl.Width = inpWidth

    ' Betaalwijze
    t = t + ROW_SPACING
    Set ctrl = TryGetControl(frm, DYN_PREFIX & "lblBetaal"): If Not ctrl Is Nothing Then ctrl.Left = colLbl: ctrl.Top = t
    Set ctrl = TryGetControl(frm, DYN_PREFIX & "cmbBetaalwijze"): If Not ctrl Is Nothing Then ctrl.Left = colInp: ctrl.Top = t: ctrl.Width = inpWidth

    ' Knoppen onderaan (gecentreerd binnen safe-area)

Option Explicit

Public Function GetLayoutSpec(ByVal keyName As String, ByVal fieldName As String) As Variant
  On Error GoTo ErrHandler
  Dim wsRef As Worksheet: Set wsRef = ThisWorkbook.Worksheets("Referenties")
  Dim lo As ListObject: Set lo = wsRef.ListObjects("tblLayoutSpecs")
  If lo Is Nothing Or lo.DataBodyRange Is Nothing Then Exit Function

  Dim hdr As Range: Set hdr = lo.HeaderRowRange
  Dim data As Range: Set data = lo.DataBodyRange
  Dim i As Long, colIndex As Long: colIndex = 0

  ' Zoek kolom case-insensitief, met Trim
  For i = 1 To hdr.Columns.Count
    If StrComp(Trim$(CStr(hdr.Cells(1, i).Value)), Trim$(fieldName), vbTextCompare) = 0 Then
      colIndex = i: Exit For
    End If
  Next i
  If colIndex = 0 Then
    Utils.LogError4 "LayoutSpecsHelpers", "GetLayoutSpec", 1001, "Kolom '" & fieldName & "' niet gevonden"
    Exit Function
  End If

  Dim r As Range
  For Each r In data.Rows
    If StrComp(Trim$(CStr(r.Cells(1, 1).Value)), Trim$(keyName), vbTextCompare) = 0 Then
      GetLayoutSpec = r.Cells(1, colIndex).Value
      Exit Function
    End If
  Next r
  ' Niet gevonden
  Utils.LogError4 "LayoutSpecsHelpers", "GetLayoutSpec", 1002, "Key '" & keyName & "' niet gevonden in specs"
  Exit Function

ErrHandler:
  Utils.LogError4 "LayoutSpecsHelpers", "GetLayoutSpec", Err.Number, Err.Description
End Function

Private Function NzDbl(v As Variant, ByVal def As Double) As Double
  If IsError(v) Then NzDbl = def: Exit Function
  If IsMissing(v) Then NzDbl = def: Exit Function
  If IsNull(v) Then NzDbl = def: Exit Function
  If VarType(v) = vbString And Trim$(CStr(v)) = "" Then NzDbl = def: Exit Function
  If Not IsNumeric(v) Then NzDbl = def Else NzDbl = CDbl(v)
End Function

Public Sub PlaceButtonFromSpec(frm As Object, ByVal controlName As String, ByVal keyName As String)
  On Error GoTo ErrHandler
  Dim baseShape As String
  baseShape = CStr(GetLayoutSpec(keyName, "ShapeName"))
  If Len(Trim$(baseShape)) = 0 Then
    Utils.LogError4 "LayoutSpecsHelpers", "PlaceButtonFromSpec", 1003, "Lege ShapeName voor key '" & keyName & "'"
    Exit Sub
  End If

  Dim X As Single, Y As Single, w As Single, h As Single
  X = CSng(NzDbl(GetLayoutSpec(keyName, "X"), 0))
  Y = CSng(NzDbl(GetLayoutSpec(keyName, "Y"), 0))
  w = CSng(NzDbl(GetLayoutSpec(keyName, "Width"), 80))   ' veilige default
  h = CSng(NzDbl(GetLayoutSpec(keyName, "Height"), 28))  ' veilige default
  If w < 0 Then w = 0
  If h < 0 Then h = 0

  Dim Img As MSForms.Image
  Set Img = FormLayout.TryGetImageDyn(frm, controlName)
  If Img Is Nothing Then Set Img = frm.Controls.Add("Forms.Image.1", controlName, True)

  Dim picPath As String: picPath = GetShapePath(baseShape)
  If Len(picPath) > 0 Then Img.Picture = LoadPicture(picPath)
  Img.Left = X: Img.Top = Y: Img.Width = w: Img.Height = h

  ' Hover alleen voor knoppen (achtergrond heeft geen hover)
  Dim hoverShape As String
  hoverShape = baseShape & "_hover"
  If InStr(1, LCase$(baseShape), "bg_", vbTextCompare) > 0 Then
    Img.Tag = "base=" & baseShape          ' geen hover
  Else
    Img.Tag = "base=" & baseShape & vbLf & "hover=" & hoverShape
  End If
  Exit Sub

ErrHandler:
  Utils.LogError4 "LayoutSpecsHelpers", "PlaceButtonFromSpec", Err.Number, Err.Description
End Sub

Private Function ParseTag(ByVal tagText As String, ByVal key As String) As String
  On Error GoTo ErrHandler
  Dim parts() As String: parts = Split(tagText, vbLf)
  Dim i As Long, kv()
  For i = LBound(parts) To UBound(parts)
    kv = Split(parts(i), "=")
    If UBound(kv) = 1 Then If StrComp(kv(0), key, vbTextCompare) = 0 Then ParseTag = kv(1): Exit Function
  Next i
  Exit Function
ErrHandler:
  Utils.LogError4 "LayoutSpecsHelpers", "ParseTag", Err.Number, Err.Description
End Function

Public Function GetShapePath(ByVal shapeName As String, Optional ByVal suffix As String = "") As String
  On Error GoTo ErrHandler
  Dim wsRef As Worksheet: Set wsRef = ThisWorkbook.Worksheets("Referenties")
  If wsRef Is Nothing Then Exit Function
  If wsRef.Visible = xlSheetVeryHidden Then wsRef.Visible = xlSheetVisible
  wsRef.Activate: DoEvents

  Dim shp As Shape: Set shp = wsRef.Shapes(shapeName)
  If shp Is Nothing Then
    Utils.LogError4 "LayoutSpecsHelpers", "GetShapePath", 1004, "Shape '" & shapeName & "' niet gevonden"
    Exit Function
  End If

  Dim w As Single, h As Single
  w = Application.Max(8, shp.Width)
  h = Application.Max(8, shp.Height)

  Dim tmp As String: tmp = Utils.GetTempFile_FIXED(shapeName & suffix, ".jpg")
  Dim ch As ChartObject: Set ch = wsRef.ChartObjects.Add(Left:=shp.Left, Top:=shp.Top, Width:=w, Height:=h)

  Dim i As Long, pasted As Boolean
  For i = 1 To 3
    On Error Resume Next
    shp.CopyPicture Appearance:=xlScreen, Format:=xlPicture
    ch.Activate
    ch.Chart.Paste
    On Error GoTo ErrHandler
    If ch.Chart.Shapes.Count > 0 Then pasted = True: Exit For
    DoEvents: Application.CutCopyMode = False
  Next i

  If pasted Then ch.Chart.Export tmp, "JPG"
  If Dir$(tmp) <> "" Then GetShapePath = tmp

CleanUp:
  On Error Resume Next
  ch.Delete: Application.CutCopyMode = False
  On Error GoTo 0
  Exit Function

ErrHandler:
  Utils.LogError4 "LayoutSpecsHelpers", "GetShapePath", Err.Number, Err.Description
  Resume CleanUp
End Function

  On Error Resume Next: Set lo = wsRef.ListObjects("tblLayoutSpecs"): On Error GoTo ErrHandler

Private Function FormImageTagCheck() As String
    On Error GoTo ErrHandler
    Dim frm As Object: Set frm = GetActiveFormSafe()
    If frm Is Nothing Then
        FormImageTagCheck = "LET OP (Form): geen actief form gevonden; open een UserForm en run opnieuw."
        Exit Function
    End If
    Dim c As Control, issues As String
    For Each c In frm.Controls
        If TypeOf c Is MSForms.Image Then
            Dim tagText As String: tagText = CStr(c.Tag)
            Dim baseName As String: baseName = ParseTagLocal(tagText, "base")
            Dim hoverName As String: hoverName = ParseTagLocal(tagText, "hover")
            If Len(baseName) = 0 Then
                issues = issues & "- Image '" & c.Name & "': Tag mist 'base'." & vbCrLf
            ElseIf Not ShapeExistsOn(ThisWorkbook.Worksheets("Referenties"), baseName) Then
                issues = issues & "- Image '" & c.Name & "': base-shape niet gevonden: " & baseName & vbCrLf
            End If
            If InStr(1, LCase$(c.Name), "imgbtn", vbTextCompare) > 0 Or _
               InStr(1, LCase$(c.Name), "imgaddrow", vbTextCompare) > 0 Or _
               InStr(1, LCase$(c.Name), "imgdelrow", vbTextCompare) > 0 Then
                If Len(hoverName) = 0 Then
                    issues = issues & "LET OP: Image '" & c.Name & "': Tag mist 'hover'." & vbCrLf
                ElseIf Not ShapeExistsOn(ThisWorkbook.Worksheets("Referenties"), hoverName) Then
                    issues = issues & "LET OP: Image '" & c.Name & "': hover-shape niet gevonden: " & hoverName & vbCrLf
                End If
            End If
        End If
    Next c
    If Len(issues) = 0 Then FormImageTagCheck = "OK (Form): alle Image-tags valide (base/hover) en shapes gevonden." Else FormImageTagCheck = "LET OP (Form):" & vbCrLf & issues
    Exit Function
ErrHandler:
    FormImageTagCheck = "FOUT in FormImageTagCheck: [" & Err.Number & "] " & Err.Description
End Function

    Dim lo As ListObject: Set lo = wsRef.ListObjects("tblLayoutSpecs")
    If lo Is Nothing Or lo.DataBodyRange Is Nothing Then
        ShapesExistenceCheck = "LET OP: 'tblLayoutSpecs' ontbreekt of heeft geen rijen; skip shapes-check."
        Exit Function
    End If
    Dim colShape As Long: colShape = FindColumnIndex(lo, "ShapeName")
    Dim colKey As Long: colKey = FindColumnIndex(lo, "Key")
    Dim r As Range, shapeName As String, keyName As String, report As String
    For Each r In lo.DataBodyRange.Rows
        keyName = Trim$(CStr(r.Cells(1, colKey).Value))
        shapeName = Trim$(CStr(r.Cells(1, colShape).Value))
        If Len(shapeName) > 0 Then
            If Not ShapeExistsOn(wsRef, shapeName) Then report = report & "- Key '" & keyName & "': base-shape ontbreekt: " & shapeName & vbCrLf
            If LCase$(keyName) <> "background" Then
                Dim hoverName As String: hoverName = shapeName & "_hover"
                If Not ShapeExistsOn(wsRef, hoverName) Then report = report & "LET OP: Key '" & keyName & "': hover-shape ontbreekt: " & hoverName & vbCrLf
            End If
        Else
            report = report & "- Key '" & keyName & "': ShapeName is leeg." & vbCrLf
        End If
    Next r
    If Len(report) = 0 Then ShapesExistenceCheck = "OK (Shapes): base/hover bestaan (hover niet vereist voor Background)." Else ShapesExistenceCheck = "LET OP (Shapes):" & vbCrLf & report
    Exit Function
ErrHandler:
    ShapesExistenceCheck = "FOUT in ShapesExistenceCheck: [" & Err.Number & "] " & Err.Description
End Function

Private Function ParseTagLocal(ByVal tagText As String, ByVal key As String) As String
    On Error GoTo ErrHandler
    If Len(tagText) = 0 Then Exit Function
    Dim parts() As String: parts = Split(tagText, vbLf)
    Dim i As Long, kv() As String, rawValue As String
    For i = LBound(parts) To UBound(parts)
        kv = Split(parts(i), "=")
        If UBound(kv) >= 1 Then
            If StrComp(Trim$(kv(0)), key, vbTextCompare) = 0 Then
                rawValue = Mid$(parts(i), InStr(parts(i), "=") + 1)
                rawValue = Trim$(rawValue)
                If Len(rawValue) > 0 And InStr(rawValue, vbNullChar) = 0 Then ParseTagLocal = rawValue: Exit Function
            End If
        End If
    Next i
    Exit Function
ErrHandler:
    Utils.LogError4 "modLayoutValidator", "ParseTagLocal", Err.Number, "Tag parse error voor key '" & key & "'"
    ParseTagLocal = ""
End Function

    On Error GoTo ErrHandler
    If Not m_IsInitialized Then Initialize
    Dim prevName As String: prevName = GetFormName(GetPrevSeq(TypeName(currentForm)))
    If Len(prevName) = 0 Then Exit Sub
    currentForm.Hide
    Select Case prevName
        Case "frmHuishouden": frmHuishouden.Show
        Case "frmVast": frmVast.Show
        Case "frmInvoer": frmInvoer.Show

Private Function GetNextSeq(ByVal formName As String) As FormSequence
    Dim seq As FormSequence: seq = GetSeq(formName)
    If seq = fsInvoer Then GetNextSeq = fsHuishouden Else GetNextSeq = seq + 1
End Function

Private Function GetPrevSeq(ByVal formName As String) As FormSequence
    Dim seq As FormSequence: seq = GetSeq(formName)
    If seq = fsHuishouden Then GetPrevSeq = fsInvoer Else GetPrevSeq = seq - 1
End Function

Private Function GetSeq(ByVal formName As String) As FormSequence
    Dim i As Long
    For i = LBound(m_Forms) To UBound(m_Forms)
        If StrComp(m_Forms(i).FormName, formName, vbTextCompare) = 0 Then GetSeq = m_Forms(i).SequenceIndex: Exit Function
    Next i
    GetSeq = 0
End Function

Private Function GetFormName(ByVal seq As FormSequence) As String
    Dim i As Long
    For i = LBound(m_Forms) To UBound(m_Forms)
        If m_Forms(i).SequenceIndex = seq Then GetFormName = m_Forms(i).FormName: Exit Function
    Next i
    GetFormName = ""
End Function

' Rapporttype voor leesbare status
Private Type tCheckReport
    OkCount As Long
    WarnCount As Long
    ErrCount As Long
    Details As String
End Type

' Publieke entry-point: voert alle sanity checks uit en geeft samenvatting terug
Public Function RunAssetSanity(Optional ByVal includeSpecsCheck As Boolean = True) As String
    On Error GoTo ErrHandler
    
    Dim rep As tCheckReport
    rep.Details = ""
    
    ' 1) Check generieke button shapes + hover
    AppendReport rep, CheckGenericButtonShapes()
    
    ' 2) Check background shape (statisch)
    AppendReport rep, CheckBackgroundShape("bg_frmInvoer_landscape")
    
    ' 3) Optioneel: check tblLayoutSpecs keys & ShapeName koppeling
    If includeSpecsCheck Then
        AppendReport rep, CheckLayoutSpecsKeys()
    End If
    
    Dim header As String
    header = "Sanity-rapport" & vbCrLf & String(40, "-") & vbCrLf & _
             "OK: " & rep.OkCount & " | Waarschuwingen: " & rep.WarnCount & " | Fouten: " & rep.ErrCount & vbCrLf & vbCrLf
    
    RunAssetSanity = header & rep.Details
    Exit Function
    
ErrHandler:
    RunAssetSanity = "FOUT in RunAssetSanity: [" & Err.Number & "] " & Err.Description
End Function

' ---- Helpers: per check produceer tCheckReport ----

Private Function CheckGenericButtonShapes() As tCheckReport
    On Error GoTo ErrHandler
    Dim rc As tCheckReport
    Dim baseNames As Variant
    baseNames = Array("btnShapeSmall", "btnShapeMedium", "btnShapeLarge")
    
    Dim wsRef As Worksheet: Set wsRef = ThisWorkbook.Worksheets("Referenties")
    Dim i As Long, nm As String, nmHover As String
    
    For i = LBound(baseNames) To UBound(baseNames)
        nm = CStr(baseNames(i))
        nmHover = nm & "_hover"
        
        If ShapeExistsOn(wsRef, nm) Then
            rc.OkCount = rc.OkCount + 1
            rc.Details = rc.Details & "OK  : Shape '" & nm & "'" & vbCrLf
        Else
            rc.ErrCount = rc.ErrCount + 1
            rc.Details = rc.Details & "FOUT: Shape '" & nm & "' ontbreekt op Referenties" & vbCrLf
        End If
        
        If ShapeExistsOn(wsRef, nmHover) Then
            rc.OkCount = rc.OkCount + 1
            rc.Details = rc.Details & "OK  : Shape '" & nmHover & "'" & vbCrLf
        Else
            rc.WarnCount = rc.WarnCount + 1
            rc.Details = rc.Details & "LET OP: Hover-shape '" & nmHover & "' ontbreekt (hover werkt dan niet)" & vbCrLf
        End If
    Next i
    
    CheckGenericButtonShapes = rc
    Exit Function
    
ErrHandler:
    CheckGenericButtonShapes.ErrCount = CheckGenericButtonShapes.ErrCount + 1
    CheckGenericButtonShapes.Details = CheckGenericButtonShapes.Details & "FOUT in CheckGenericButtonShapes: [" & Err.Number & "] " & Err.Description & vbCrLf
End Function

Private Function CheckBackgroundShape(ByVal bgShapeName As String) As tCheckReport
    On Error GoTo ErrHandler
    Dim rc As tCheckReport
    Dim wsRef As Worksheet: Set wsRef = ThisWorkbook.Worksheets("Referenties")
    
    If ShapeExistsOn(wsRef, bgShapeName) Then
        rc.OkCount = rc.OkCount + 1
        rc.Details = rc.Details & "OK  : Background-shape '" & bgShapeName & "'" & vbCrLf
    Else
        rc.WarnCount = rc.WarnCount + 1
        rc.Details = rc.Details & "LET OP: Background-shape '" & bgShapeName & "' niet gevonden (form laad zonder achtergrond)" & vbCrLf
    End If
    
    CheckBackgroundShape = rc
    Exit Function
    
ErrHandler:
    CheckBackgroundShape.ErrCount = CheckBackgroundShape.ErrCount + 1
    CheckBackgroundShape.Details = CheckBackgroundShape.Details & "FOUT in CheckBackgroundShape: [" & Err.Number & "] " & Err.Description & vbCrLf
End Function

Private Function CheckLayoutSpecsKeys() As tCheckReport
    On Error GoTo ErrHandler
    Dim rc As tCheckReport
    
    Dim wsRef As Worksheet: Set wsRef = ThisWorkbook.Worksheets("Referenties")
    Dim lo As ListObject
    On Error Resume Next
    Set lo = wsRef.ListObjects("tblLayoutSpecs")
    On Error GoTo ErrHandler
    
    If lo Is Nothing Then
        rc.WarnCount = rc.WarnCount + 1
        rc.Details = rc.Details & "LET OP: Tabel 'tblLayoutSpecs' niet gevonden op Referenties" & vbCrLf
        CheckLayoutSpecsKeys = rc
        Exit Function
    End If
    
    If lo.DataBodyRange Is Nothing Then
        rc.WarnCount = rc.WarnCount + 1
        rc.Details = rc.Details & "LET OP: 'tblLayoutSpecs' heeft geen rijen" & vbCrLf
        CheckLayoutSpecsKeys = rc
        Exit Function
    End If
    
    ' Zoek kolommen Key en ShapeName
    Dim colKey As Long, colShape As Long
    colKey = FindColumnIndex(lo, "Key")
    colShape = FindColumnIndex(lo, "ShapeName")
    
    If colKey = 0 Or colShape = 0 Then
        rc.ErrCount = rc.ErrCount + 1
        rc.Details = rc.Details & "FOUT: Kolom 'Key' of 'ShapeName' ontbreekt in tblLayoutSpecs" & vbCrLf
        CheckLayoutSpecsKeys = rc
        Exit Function
    End If
    
    ' Loop door rijen; check dat ShapeName bestaat als shape
    Dim r As Range, keyName As String, shapeName As String
    For Each r In lo.DataBodyRange.Rows
        keyName = CStr(Trim$(r.Cells(1, colKey).Value))
        shapeName = CStr(Trim$(r.Cells(1, colShape).Value))
        
        If Len(keyName) = 0 Then
            rc.WarnCount = rc.WarnCount + 1
            rc.Details = rc.Details & "LET OP: Lege Key in tblLayoutSpecs (rij " & r.Row & ")" & vbCrLf
        End If
        
        If Len(shapeName) = 0 Then
            rc.ErrCount = rc.ErrCount + 1
            rc.Details = rc.Details & "FOUT: Key '" & keyName & "' heeft lege ShapeName" & vbCrLf
        ElseIf Not ShapeExistsOn(wsRef, shapeName) Then
            rc.ErrCount = rc.ErrCount + 1
            rc.Details = rc.Details & "FOUT: Key '" & keyName & "' verwijst naar ontbrekende shape '" & shapeName & "'" & vbCrLf
        Else
            rc.OkCount = rc.OkCount + 1
            rc.Details = rc.Details & "OK  : Key '" & keyName & "' ? Shape '" & shapeName & "'" & vbCrLf
        End If
    Next r
    
    CheckLayoutSpecsKeys = rc
    Exit Function
    
ErrHandler:
    CheckLayoutSpecsKeys.ErrCount = CheckLayoutSpecsKeys.ErrCount + 1
    CheckLayoutSpecsKeys.Details = CheckLayoutSpecsKeys.Details & "FOUT in CheckLayoutSpecsKeys: [" & Err.Number & "] " & Err.Description & vbCrLf
End Function

Private Sub AppendReport(ByRef total As tCheckReport, ByRef part As tCheckReport)
    total.OkCount = total.OkCount + part.OkCount
    total.WarnCount = total.WarnCount + part.WarnCount
    total.ErrCount = total.ErrCount + part.ErrCount
    If Len(part.Details) > 0 Then
        total.Details = total.Details & part.Details
        If Right$(total.Details, 2) <> vbCrLf Then total.Details = total.Details & vbCrLf
    End If
End Sub

Option Explicit

Public Sub NormalizeSpecsHeaders()
  On Error GoTo ErrHandler

  ' 1) Sheet 'Referenties' ophalen of melden
  Dim ws As Worksheet
  Set ws = GetWorksheetSafe("Referenties")
  If ws Is Nothing Then
    MsgBox "FOUT: Werkblad 'Referenties' niet gevonden.", vbCritical
    Exit Sub
  End If

  ' 2) Probeer bestaande tabel op te halen
  Dim lo As ListObject
  Set lo = TryGetListObjectSafe(ws, "tblLayoutSpecs")

  ' 3) Als tabel niet bestaat: laat gebruiker header-rij selecteren en maak de tabel
  If lo Is Nothing Then
    Dim hdr As Range
    MsgBox "Selecteer nu de HEADER-RIJ van tblLayoutSpecs (exact één rij met alle kolomnamen) en klik OK in de volgende prompt.", vbInformation
    On Error Resume Next
    Set hdr = Application.InputBox(Prompt:="Selecteer de header-rij (bijv. K1:S1)", Type:=8) ' Type 8 = Range
    On Error GoTo ErrHandler

    If hdr Is Nothing Then
      MsgBox "Actie geannuleerd. Geen header-rij geselecteerd.", vbExclamation
      Exit Sub
    End If
    If hdr.Rows.Count <> 1 Then
      MsgBox "Selecteer exact één rij met de kopteksten.", vbExclamation
      Exit Sub
    End If

    ' Maak ListObject vanaf de geselecteerde header-rij + CurrentRegion
    Set lo = ws.ListObjects.Add(xlSrcRange, hdr.CurrentRegion, , xlYes)
    lo.name = "tblLayoutSpecs"
  End If

  ' 4) Zet de exacte headers (9 kolommen, volgorde vast)
  Dim exactHeaders As Variant
  exactHeaders = Array("Key", "ShapeName", "Width", "Height", "X", "Y", "DefaultColor", "HoverColor", "Notes")

  Dim i As Long
  ' Controleren of aantal kolommen voldoende is, anders breidt de header-rij uit
  If lo.HeaderRowRange.Columns.Count < UBound(exactHeaders) - LBound(exactHeaders) + 1 Then
    ' Vergroot de header-range door naastliggende lege cellen op te nemen
    Dim needCols As Long
    needCols = (UBound(exactHeaders) - LBound(exactHeaders) + 1) - lo.HeaderRowRange.Columns.Count
    lo.HeaderRowRange.Resize(1, lo.HeaderRowRange.Columns.Count + needCols).Cells(1, 1).Select
    ' (Excel past de ListObject range zelf meestal aan bij CurrentRegion)
  End If

  For i = LBound(exactHeaders) To UBound(exactHeaders)
    lo.HeaderRowRange.Cells(1, i + 1).Value = exactHeaders(i)
  Next i

  ' 5) Trim spaties in DataBodyRange (als er rijen zijn)
  If Not lo.DataBodyRange Is Nothing Then
    Dim c As Range
    For Each c In lo.DataBodyRange
      If VarType(c.Value) = vbString Then c.Value = Trim$(CStr(c.Value))
    Next c
  End If

  ' 6) Feedback
  MsgBox "tblLayoutSpecs genormaliseerd:" & vbCrLf & _
         " - Sheet: " & ws.name & vbCrLf & _
         " - Tabelnaam: " & lo.name & vbCrLf & _
         " - Headers gezet (Key, ShapeName, Width, Height, X, Y, DefaultColor, HoverColor, Notes)", vbInformation
  Exit Sub

ErrHandler:
  MsgBox "FOUT in NormalizeSpecsHeaders: [" & Err.Number & "] " & Err.Description, vbCritical
End Sub

Option Explicit
Public Function WeekTotaalBestaat(ByVal jaar As Long, ByVal weeknr As Long) As Boolean
 On Error GoTo ErrHandler
 Dim ws As Worksheet: Set ws = ThisWorkbook.Worksheets("Database")
 Dim lr As Long: lr = Utils.LastRow(ws, SummaryCols.colWeekSummaryStart)
 If lr < 2 Then WeekTotaalBestaat = False: Exit Function
 Dim rng As Range
 Set rng = ws.Range(ws.Cells(2, SummaryCols.colWeekSummaryStart), ws.Cells(lr, SummaryCols.colWeekSummaryStart)).Find( _
  what:=CStr(jaar) & vbLf & CStr(weeknr), LookIn:=xlValues, LookAt:=xlWhole)
 WeekTotaalBestaat = Not rng Is Nothing
 Exit Function
ErrHandler:
 Utils.LogError4 "Reporting", "WeekTotaalBestaat", Err.Number, Err.Description
End Function

Public Sub BerekenWeekTotaal()
 On Error GoTo ErrHandler
 Dim ws As Worksheet: Set ws = ThisWorkbook.Worksheets("Database")
 Utils.EnsureDBHeaders
 Dim lr As Long: lr = Utils.LastRow(ws, DBCols.dbDate)
 If lr < 2 Then Exit Sub
 Dim dPrev As Date: dPrev = DateAdd("d", -7, Date)
 Dim tgtWeek As Long: tgtWeek = Utils.IsoWeek(dPrev)
 Dim tgtYear As Long: tgtYear = Year(dPrev)
 Dim i As Long, total As Double
 Dim catDict As Object: Set catDict = CreateObject("Scripting.Dictionary")
 For i = 2 To lr
  If CLng(Val(ws.Cells(i, DBCols.dbYear).Value)) = tgtYear And _
     CLng(Val(ws.Cells(i, DBCols.dbWeek).Value)) = tgtWeek And _
     CStr(ws.Cells(i, DBCols.dbType).Value) = "Uitgave" Then
     Dim amount As Double: amount = CDbl(Val(ws.Cells(i, DBCols.dbAmount).Value))
     total = total + amount
     Dim cat As String: cat = CStr(ws.Cells(i, DBCols.dbCategory).Value)
     If Len(Trim$(cat)) > 0 Then Utils.DictAddOrIncrement catDict, cat, 1
  End If
 Next i
 Dim mostCat As String, maxCount As Long, k As Variant
 For Each k In catDict.keys
  If catDict(k) > maxCount Then maxCount = catDict(k): mostCat = CStr(k)
 Next k
 If ws.Cells(1, SummaryCols.colWeekSummaryStart).Value = "" Then
  ws.Cells(1, SummaryCols.colWeekSummaryStart).Resize(1, 5).Value = Array("Jaar" & vbLf & "Week", "Jaar", "Weeknr", "Totaal", "Meest gebruikte categorie")
 End If
 Dim outRow As Long: outRow = Utils.LastRow(ws, SummaryCols.colWeekSummaryStart) + 1
 ws.Cells(outRow, SummaryCols.colWeekSummaryStart).Value = CStr(tgtYear) & vbLf & CStr(tgtWeek)
 ws.Cells(outRow, SummaryCols.colWeekSummaryStart + 1).Value = tgtYear
 ws.Cells(outRow, SummaryCols.colWeekSummaryStart + 2).Value = tgtWeek
 ws.Cells(outRow, SummaryCols.colWeekSummaryStart + 3).Value = total
 ws.Cells(outRow, SummaryCols.colWeekSummaryStart + 4).Value = mostCat
 Exit Sub
ErrHandler:
 Utils.LogError4 "Reporting", "BerekenWeekTotaal", Err.Number, Err.Description
End Sub
Public Sub BerekenMaandTotaal()
 On Error GoTo ErrHandler
 Dim ws As Worksheet: Set ws = ThisWorkbook.Worksheets("Database")
 Utils.EnsureDBHeaders
 Dim lr As Long: lr = Utils.LastRow(ws, DBCols.dbDate)
 If lr < 2 Then Exit Sub
 Dim prevMonthKey As String: prevMonthKey = Format(DateAdd("m", -1, Date), "YYYY-MM")
 Dim i As Long
 Dim totaal As Double, dagenCount As Long
 Dim weekDict As Object: Set weekDict = CreateObject("Scripting.Dictionary")
 Dim catDict As Object: Set catDict = CreateObject("Scripting.Dictionary")
 Dim catCountDict As Object: Set catCountDict = CreateObject("Scripting.Dictionary")
 Dim duursteDag As Double, duursteWeek As Double
 Dim maxCatValue As Double, minCatValue As Double
 Dim maxCatName As String, minCatName As String
 maxCatValue = 0: minCatValue = 1E+20
 For i = 2 To lr
  If CStr(ws.Cells(i, DBCols.dbMonthKey).Value) = prevMonthKey And CStr(ws.Cells(i, DBCols.dbType).Value) = "Uitgave" Then
   Dim bedrag As Double: bedrag = CDbl(Val(ws.Cells(i, DBCols.dbAmount).Value))
   totaal = totaal + bedrag
   dagenCount = dagenCount + 1
   If bedrag > duursteDag Then duursteDag = bedrag
   Dim weekKey As String: weekKey = CStr(ws.Cells(i, DBCols.dbYear).Value) & "-W" & CStr(ws.Cells(i, DBCols.dbWeek).Value)
   Utils.DictAddOrIncrement weekDict, weekKey, bedrag
   Dim categorie As String: categorie = CStr(ws.Cells(i, DBCols.dbCategory).Value)
   Utils.DictAddOrIncrement catDict, categorie, bedrag
   Utils.DictAddOrIncrement catCountDict, categorie, 1
  End If
 Next i
 Dim key As Variant
 For Each key In weekDict.keys: If weekDict(key) > duursteWeek Then duursteWeek = weekDict(key)
 Next key
 For Each key In catDict.keys
  If catDict(key) > maxCatValue Then maxCatValue = catDict(key): maxCatName = CStr(key)
  If catDict(key) < minCatValue Then minCatValue = catDict(key): minCatName = CStr(key)
 Next key
 Dim overzichtRow As Long: overzichtRow = Utils.LastRow(ws, SummaryCols.colMonthSummaryStart) + 1
 If ws.Cells(1, SummaryCols.colMonthSummaryStart).Value = "" Then
  ws.Cells(1, SummaryCols.colMonthSummaryStart).Resize(1, 7).Value = Array("mnd", "ttl", "gpd", "gpw", "dudg", "duwk", "m_u_a")
 End If
 ws.Cells(overzichtRow, SummaryCols.colMonthSummaryStart).Value = prevMonthKey
 ws.Cells(overzichtRow, SummaryCols.colMonthSummaryStart + 1).Value = totaal
 ws.Cells(overzichtRow, SummaryCols.colMonthSummaryStart + 2).Value = IIf(dagenCount > 0, totaal / dagenCount, 0)
 ws.Cells(overzichtRow, SummaryCols.colMonthSummaryStart + 3).Value = IIf(weekDict.Count > 0, totaal / weekDict.Count, 0)
 ws.Cells(overzichtRow, SummaryCols.colMonthSummaryStart + 4).Value = duursteDag
 ws.Cells(overzichtRow, SummaryCols.colMonthSummaryStart + 5).Value = duursteWeek
 ws.Cells(overzichtRow, SummaryCols.colMonthSummaryStart + 6).Value = maxCatName
 ' Categorie-overzicht append met maandkolom
 Dim catStartCol As Long: catStartCol = SummaryCols.colMonthSummaryStart + 8 ' X
 If ws.Cells(1, catStartCol - 1).Value = "" Then
  ws.Cells(1, catStartCol - 1).Resize(1, 4).Value = Array("Maand", "Categorie", "Aantal", "Totaal")
 End If
 Dim r As Long: r = Utils.LastRow(ws, catStartCol - 1) + 1
 For Each key In catDict.keys
  ws.Cells(r, catStartCol - 1).Value = prevMonthKey
  ws.Cells(r, catStartCol).Value = key
  ws.Cells(r, catStartCol + 1).Value = catCountDict(key)
  ws.Cells(r, catStartCol + 2).Value = catDict(key)
  r = r + 1
 Next key
 Exit Sub
ErrHandler:
 Utils.LogError4 "Reporting", "BerekenMaandTotaal", Err.Number, Err.Description
End Sub

Option Explicit
Public Sub SetupVastEnCategorieTabellen()
 On Error GoTo ErrHandler
 Dim wsV As Worksheet, wsRef As Worksheet
 Dim rngInk As Range, rngUit As Range
 Set wsV = ThisWorkbook.Worksheets("Vast")
 On Error Resume Next: Set rngInk = ThisWorkbook.names("rngVastInkomsten").RefersToRange: On Error GoTo ErrHandler
 If Not rngInk Is Nothing Then Call Utils.EnsureTable(wsV, "tblVastInkomsten", rngInk)
 On Error Resume Next: Set rngUit = ThisWorkbook.names("rngVastUitgaven").RefersToRange: On Error GoTo ErrHandler
 If Not rngUit Is Nothing Then Call Utils.EnsureTable(wsV, "tblVastUitgaven", rngUit)
 If Not Utils.SheetBestaat("Referenties") Then Set wsRef = ThisWorkbook.Worksheets.Add: wsRef.name = "Referenties" Else Set wsRef = ThisWorkbook.Worksheets("Referenties")
 If wsRef.Cells(1, 1).Value = "" Then wsRef.Range("A1").Value = "Categorie"
 Dim loCat As ListObject: Set loCat = Utils.EnsureTable(wsRef, "tblCategorieen", wsRef.Range("A1"))
 If wsRef.Cells(1, 3).Value = "" Then wsRef.Range("C1").Value = "Categorie": wsRef.Range("D1").Value = "Subcategorie"
 Dim loSub As ListObject: Set loSub = Utils.EnsureTable(wsRef, "tblSubcategorieen", wsRef.Range("C1:D1"))
 SeedCategorieen loCat, loSub
 MsgBox "Tabellen ingesteld: tblVastInkomsten/tblVastUitgaven (indien aanwezig), tblCategorieen, tblSubcategorieen.", vbInformation
 Exit Sub
ErrHandler:
 Utils.LogError4 "TableSetup", "SetupVastEnCategorieTabellen", Err.Number, Err.Description
End Sub
Private Sub SeedCategorieen(ByVal loCat As ListObject, ByVal loSub As ListObject)
 On Error GoTo ErrHandler
 Dim cats As Variant: cats = Array("Eten", "Vervoer", "Hapje/Drankje", "Cadeau", "Lisa", "Kyra", "Kleding e.d", "Overig")
 Dim i As Long
 For i = LBound(cats) To UBound(cats)
  If Not CategoryExists(loCat, CStr(cats(i))) Then AddCategory loCat, CStr(cats(i))
 Next i
 AddSub loSub, "Lisa", "Zakgeld": AddSub loSub, "Lisa", "Kleding": AddSub loSub, "Lisa", "Onzin"
 AddSub loSub, "Kyra", "Zakgeld": AddSub loSub, "Kyra", "Kleding": AddSub loSub, "Kyra", "Onzin"
 Exit Sub
ErrHandler:
 Utils.LogError4 "TableSetup", "SeedCategorieen", Err.Number, Err.Description
End Sub
Public Function CategoryExists(ByVal loCat As ListObject, ByVal cat As String) As Boolean
 On Error GoTo ErrHandler
 Dim r As Range
 For Each r In loCat.ListColumns(1).DataBodyRange
  If LCase$(Trim$(r.Value)) = LCase$(Trim$(cat)) Then CategoryExists = True: Exit Function
 Next r
 Exit Function
ErrHandler:
 Utils.LogError4 "TableSetup", "CategoryExists", Err.Number, Err.Description
End Function
Public Sub AddCategory(ByVal loCat As ListObject, ByVal cat As String)
 On Error GoTo ErrHandler
 Dim lr As ListRow: Set lr = loCat.ListRows.Add
 lr.Range.Cells(1, 1).Value = cat
 Exit Sub
ErrHandler:
 Utils.LogError4 "TableSetup", "AddCategory", Err.Number, Err.Description
End Sub
Public Sub AddSub(ByVal loSub As ListObject, ByVal cat As String, ByVal subcat As String)
 On Error GoTo ErrHandler
 Dim r As Range
 If Not loSub.DataBodyRange Is Nothing Then
  For Each r In loSub.DataBodyRange.Rows
   If LCase$(Trim$(r.Cells(1, 1).Value)) = LCase$(Trim$(cat)) And LCase$(Trim$(r.Cells(1, 2).Value)) = LCase$(Trim$(subcat)) Then Exit Sub
  Next r
 End If
 Dim lr As ListRow: Set lr = loSub.ListRows.Add
 lr.Range.Cells(1, 1).Value = cat
 lr.Range.Cells(1, 2).Value = subcat
 Exit Sub
ErrHandler:
 Utils.LogError4 "TableSetup", "AddSub", Err.Number, Err.Description
End Sub

Option Explicit

Public Sub LogError4(ByVal moduleName As String, ByVal funcName As String, ByVal errNum As Long, ByVal errDesc As String)
    On Error Resume Next
    Dim ws As Worksheet: Set ws = ThisWorkbook.Worksheets("ErrorLog")
    If ws Is Nothing Then
        Set ws = ThisWorkbook.Worksheets.Add
        ws.Name = "ErrorLog"
        ws.Range("A1:D1").Value = Array("Datum", "Module", "Functie", "Foutmelding")
    End If
    Dim r As Long: r = ws.Cells(ws.Rows.Count, 1).End(xlUp).Row + 1
    ws.Cells(r, 1).Value = Now
    ws.Cells(r, 2).Value = moduleName
    ws.Cells(r, 3).Value = funcName
    ws.Cells(r, 4).Value = "[" & errNum & "] " & errDesc
End Sub

Public Sub LogError(ByVal source As String, ByVal errDesc As String)
 On Error Resume Next
 Call LogError4(source, "", Err.Number, errDesc)
End Sub

Public Function LastRow(ByVal ws As Worksheet, Optional ByVal col As Long = 1) As Long
 With ws
  If Application.WorksheetFunction.CountA(.Columns(col)) = 0 Then
    LastRow = 1
  Else
    LastRow = .Cells(.Rows.Count, col).End(xlUp).Row
  End If
 End With
End Function
Public Function IsoWeek(ByVal d As Date) As Long
 IsoWeek = DatePart("ww", d, vbMonday, vbFirstFourDays)
End Function
Public Function ParseAmount_FIXED(ByVal s As String) As Double
 Dim decSep As String: decSep = Application.International(xlDecimalSeparator)
 s = Trim$(s)
 If s = "" Then ParseAmount = 0: Exit Function
 s = Replace(Replace(s, ".", decSep), ",", decSep)
 ParseAmount = CDbl(s)
End Function
Public Sub EnsureDBHeaders()
 Dim ws As Worksheet: Set ws = ThisWorkbook.Worksheets("Database")
 If ws Is Nothing Then
   Set ws = ThisWorkbook.Worksheets.Add
   ws.name = "Database"
 End If
 If ws.Cells(1, DBCols.dbDate).Value = "" Then
   ws.Cells(1, DBCols.dbDate).Resize(1, 11).Value = Array( _
    "Datum", "Jaar", "MaandKey", "WeekISO", "DagNaam", "Categorie", "Subcategorie", "Omschrijving", "Bedrag", "Betaalwijze", "Soort")
 End If
End Sub
Public Function DictAddOrIncrement(ByRef dict As Object, ByVal key As Variant, Optional ByVal inc As Double = 1) As Double
 If Not dict.exists(key) Then dict(key) = inc Else dict(key) = dict(key) + inc
 DictAddOrIncrement = dict(key)
End Function
Public Function ToMonthlyAmount(ByVal periode As String, ByVal bedrag As Double) As Double
 periode = LCase$(Trim$(periode))
 Select Case periode
  Case "jaar", "per jaar": ToMonthlyAmount = bedrag / 12
  Case "kwartaal", "per kwartaal": ToMonthlyAmount = bedrag / 3
  Case "maand", "per maand": ToMonthlyAmount = bedrag
  Case "4 weken": ToMonthlyAmount = bedrag * 13 / 12
  Case "week", "per week": ToMonthlyAmount = bedrag * 52 / 12
  Case "dag", "per dag": ToMonthlyAmount = bedrag * 365 / 12
  Case Else: ToMonthlyAmount = bedrag
 End Select
End Function

Public Function MonthNameNL(ByVal m As Long) As String
 Dim arr
 arr = Array("Januari", "Februari", "Maart", "April", "Mei", "Juni", "Juli", "Augustus", "September", "Oktober", "November", "December")
 If m >= 1 And m <= 12 Then MonthNameNL = arr(m - 1) Else MonthNameNL = "Onbekend"
End Function



Public Function GetTempFile_FIXED(ByVal baseName As String, ByVal ext As String) As String
    On Error GoTo ErrHandler
    Dim tmp As String
    tmp = Environ$("TEMP")
    If Len(tmp) = 0 Then tmp = Environ$("TMP")
    If Len(tmp) = 0 Then tmp = ThisWorkbook.Path
    If Right$(tmp, 1) <> Application.PathSeparator Then tmp = tmp & Application.PathSeparator
    Dim safeName As String: safeName = baseName
	safeName = Replace(safeName, "\\", "_")
	safeName = Replace(safeName, "/",  "_")
	safeName = Replace(safeName, ":",  "_")
	safeName = Replace(safeName, "*",  "_")
	safeName = Replace(safeName, "?",  "_")
	safeName = Replace(safeName, """", "_")
	safeName = Replace(safeName, "<",  "_")
	safeName = Replace(safeName, ">",  "_")
	safeName = Replace(safeName, "\n", "_") ' of vbLf

    GetTempFile_FIXED = tmp & safeName & "_" & Format(Now, "yyyymmdd_hhnnss") & ext
    Exit Function
ErrHandler:
    GetTempFile_FIXED = ThisWorkbook.Path & Application.PathSeparator & baseName & "_" & Format(Now, "yyyymmdd_hhnnss") & ext
    LogError4 "Utils", "GetTempFile_FIXED", Err.Number, Err.Description
End Function

' ===== CLASS: CImgEvents (hover debounce + navigation) =====
' Class name must be: CImgEvents
Option Explicit

Public WithEvents Img As MSForms.Image
Public ParentForm As Object
Private m_lastHover As Double

Private Sub Img_MouseMove(ByVal Button As Integer, ByVal Shift As Integer, ByVal X As Single, ByVal Y As Single)
    On Error Resume Next
    ' Debounce: maximaal 1 update per 100 ms
    If Timer - m_lastHover >= 0.1 Then
        LayoutSpecsHelpers.ApplyHoverPicture ParentForm, Img.Name
        m_lastHover = Timer
    End If
End Sub

Private Sub Img_MouseDown(ByVal Button As Integer, ByVal Shift As Integer, ByVal X As Single, ByVal Y As Single)
    On Error GoTo ErrHandler
    Select Case LCase$(Img.Name)
        Case "dyn_imgbtnopslaan":   FormEvents.HandleOpslaan_dyn ParentForm
        Case "dyn_imgbtnannuleren": FormEvents.HandleAnnuleren_dyn ParentForm
        Case "dyn_imgbtnvolgende":  FormNavigator.NavigateNext ParentForm
        Case "dyn_imgbtnvorige":    FormNavigator.NavigatePrevious ParentForm
    End Select
    Exit Sub
ErrHandler:
    Utils.LogError4 "CImgEvents", "Img_MouseDown", Err.Number, Err.Description
End Sub



'=========================
' CLASS: CTxtEvents
'=========================
' Class module name: CTxtEvents
Option Explicit
Public WithEvents Txt As MSForms.TextBox
Private Sub Txt_KeyPress(ByVal KeyAscii As MSForms.ReturnInteger)
 Dim dec As String: dec = Application.International(xlDecimalSeparator)
 Select Case KeyAscii
  Case 48 To 57 ' 0–9
  Case Asc(dec) ' één decimaal
   If InStr(Txt.Text, dec) > 0 Then KeyAscii = 0
  Case 8 ' Backspace
  Case Else: KeyAscii = 0
 End Select
End Sub
Private Sub Txt_Exit(ByVal Cancel As MSForms.ReturnBoolean)
 On Error Resume Next
 If Len(Txt.Text) = 0 Then Exit Sub
 Txt.Text = Format(CDbl(Txt.Text), "0.00")
End Sub
