Option Explicit
Private Sub Workbook_Open()
 On Error GoTo ErrHandler
 Utils.EnsureDBHeaders
 ' Alleen vorige week samenvatten
 Dim dPrev As Date: dPrev = DateAdd("d", -7, Date)
 Dim w As Long: w = Utils.IsoWeek(dPrev)
 Dim y As Long: y = Year(dPrev)
 If Weekday(Date, vbMonday) = 1 Then
   If Not Reporting.WeekTotaalBestaat(y, w) Then Reporting.BerekenWeekTotaal
 Else
   If Not Reporting.WeekTotaalBestaat(y, w) Then
     Dim antwoord As VbMsgBoxResult
     antwoord = MsgBox("Weekoverzicht van vorige week ontbreekt. Nu berekenen?", vbYesNo + vbQuestion, "Weekoverzicht")
     If antwoord = vbYes Then Reporting.BerekenWeekTotaal
   End If
 End If
 ' Maandoverzicht: alleen als vorige maand niet aanwezig
 If Format(Date, "YYYY-MM") <> Reporting.LaatsteMaandInMaandTabel() Then Reporting.BerekenMaandTotaal
 ' Huidige maandblad: alleen aanmaken als nog niet bestaat
 MonthlySheet.MaakHuidigeMaandSheet
 Exit Sub
ErrHandler:
 Utils.LogError4 "ThisWorkbook", "Workbook_Open", Err.Number, Err.Description
 MsgBox "Fout in Workbook_Open: " & Err.Description, vbCritical
End Sub

Public Sub LoadCategoriesIntoCombo(frm As Object)
 On Error GoTo ErrHandler

Public Sub HookEventsDynamic(frm As Object)
 On Error GoTo ErrHandler
 ' Hover events voor Images
 Dim c As Control
 For Each c In frm.Controls
  If TypeOf c Is MSForms.Image Then
   Dim ie As CImgEvents
   Set ie = New CImgEvents
   Set ie.Img = c
   Set ie.ParentForm = frm
  End If
 Next c
 ' Combo events voor categorie
 Dim cmbCat As MSForms.ComboBox: Set cmbCat = FormLayout.TryGetControl(frm, FormLayout.DYN_PREFIX & "cmbCategorie")
 If Not cmbCat Is Nothing Then
  Dim ce As CCmbEvents
  Set ce = New CCmbEvents
  Set ce.cmb = cmbCat
  Set ce.ParentForm = frm
 End If
 ' Text events voor bedrag
 Dim txtB As MSForms.TextBox: Set txtB = FormLayout.TryGetControl(frm, FormLayout.DYN_PREFIX & "txtBedrag")
 If Not txtB Is Nothing Then
  Dim te As CTxtEvents
  Set te = New CTxtEvents
  Set te.Txt = txtB
 End If
 Exit Sub
ErrHandler:
 Utils.LogError4 "FormEvents", "HookEventsDynamic", Err.Number, Err.Description
End Sub

Public Sub HandleOpslaan_dyn(frm As Object)
 On Error GoTo ErrHandler
 Dim wsDB As Worksheet: Set wsDB = ThisWorkbook.Worksheets("Database")
 Utils.EnsureDBHeaders
 Dim outR As Long: outR = Utils.LastRow(wsDB, DBCols.dbDate) + 1
 Dim d As Date: d = Date
 Dim cat As String: cat = frm.Controls(FormLayout.DYN_PREFIX & "cmbCategorie").Value
 Dim subcat As String: subcat = frm.Controls(FormLayout.DYN_PREFIX & "cmbSubcategorie").Value
 Dim bedrag As Double: bedrag = Utils.ParseAmount(frm.Controls(FormLayout.DYN_PREFIX & "txtBedrag").Text)
 Dim betaal As String: betaal = frm.Controls(FormLayout.DYN_PREFIX & "cmbBetaalwijze").Value
 If Len(cat) = 0 Then MsgBox "Kies een categorie", vbExclamation: Exit Sub
 With wsDB
  .Cells(outR, DBCols.dbDate).Value = d
  .Cells(outR, DBCols.dbYear).Value = Year(d)
  .Cells(outR, DBCols.dbMonthKey).Value = Format(d, "yyyy-mm")
  .Cells(outR, DBCols.dbWeek).Value = Utils.IsoWeek(d)
  .Cells(outR, DBCols.dbDay).Value = Format(d, "dddd")
  .Cells(outR, DBCols.dbCategory).Value = cat
  .Cells(outR, DBCols.dbSubcategory).Value = subcat
  .Cells(outR, DBCols.dbAmount).Value = Round(bedrag, 2)
  .Cells(outR, DBCols.dbPayment).Value = betaal
  .Cells(outR, DBCols.dbType).Value = "Uitgave"
 End With
 MsgBox "Opgeslagen!", vbInformation
 Unload frm
 Exit Sub
ErrHandler:
 Utils.LogError4 "FormEvents", "HandleOpslaan_dyn", Err.Number, Err.Description
 MsgBox "Fout: " & Err.Description, vbCritical
End Sub

Option Explicit
Public Const DYN_PREFIX As String = "dyn_"
Public Sub FrmLayoutSafeAreaDyn(frm As Object)
    On Error GoTo ErrHandler

    MonthlySheet.EnsureConfigSheet
    Dim BG_MARGIN As Double: BG_MARGIN = MonthlySheet.ConfigGet("BG_MARGIN", 6)
    Dim ROW_SPACING As Double: ROW_SPACING = MonthlySheet.ConfigGet("ROW_SPACING", 34)
    Dim LABEL_WIDTH As Double: LABEL_WIDTH = MonthlySheet.ConfigGet("LABEL_WIDTH", 120)
    Dim GAP_LABEL_TO_INPUT As Double: GAP_LABEL_TO_INPUT = MonthlySheet.ConfigGet("GAP_LABEL_TO_INPUT", 12)
    Dim RIGHT_PADDING As Double: RIGHT_PADDING = MonthlySheet.ConfigGet("RIGHT_PADDING", 20)
    Dim INP_MAX As Double: INP_MAX = MonthlySheet.ConfigGet("INP_MAX", 360)
    Dim SAFE_LEFT_PCT As Double: SAFE_LEFT_PCT = MonthlySheet.ConfigGet("SAFE_LEFT_PCT", 0.12)
    Dim SAFE_RIGHT_PCT As Double: SAFE_RIGHT_PCT = MonthlySheet.ConfigGet("SAFE_RIGHT_PCT", 0.88)
    Dim SAFE_TOP_PCT As Double: SAFE_TOP_PCT = MonthlySheet.ConfigGet("SAFE_TOP_PCT", 0.2)
    Dim SAFE_BOTTOM_PCT As Double: SAFE_BOTTOM_PCT = MonthlySheet.ConfigGet("SAFE_BOTTOM_PCT", 0.8)

    Dim bgLeft As Single, bgTop As Single, bgW As Single, bgH As Single
    Dim bg As MSForms.Image: Set bg = TryGetImageDyn(frm, FormLayout.DYN_PREFIX & "imgBackground")
    If bg Is Nothing Then
        bgLeft = BG_MARGIN: bgTop = BG_MARGIN
        bgW = frm.InsideWidth - 2 * BG_MARGIN
        bgH = frm.InsideHeight - 2 * BG_MARGIN
    Else
        bgLeft = bg.Left: bgTop = bg.Top: bgW = bg.Width: bgH = bg.Height
    End If
    If bgW < 0 Then bgW = 0
    If bgH < 0 Then bgH = 0

    Dim safeLeft As Single, safeTop As Single, safeWidth As Single, safeHeight As Single
    safeLeft = bgLeft + (SAFE_LEFT_PCT * bgW)
    safeTop = bgTop + (SAFE_TOP_PCT * bgH)
    safeWidth = (SAFE_RIGHT_PCT - SAFE_LEFT_PCT) * bgW
    safeHeight = (SAFE_BOTTOM_PCT - SAFE_TOP_PCT) * bgH
    If safeWidth < 0 Then safeWidth = 0
    If safeHeight < 0 Then safeHeight = 0

    Dim colLbl As Single, colInp As Single, inpWidth As Single, t As Single
    colLbl = safeLeft
    colInp = safeLeft + LABEL_WIDTH + GAP_LABEL_TO_INPUT
    inpWidth = safeWidth - (colInp - safeLeft) - RIGHT_PADDING
    If inpWidth < 0 Then inpWidth = 0
    If inpWidth > INP_MAX Then
        Dim extraSpace As Single
        extraSpace = (inpWidth - INP_MAX) / 2
        colInp = colInp + extraSpace
        inpWidth = INP_MAX
    End If

    ' Lay-out toewijzing met guards
    t = safeTop
    Dim ctrl As Control
    Set ctrl = TryGetControl(frm, FormLayout.DYN_PREFIX & "lblDatumVandaag"): If Not ctrl Is Nothing Then ctrl.Left = colLbl: ctrl.Top = t
    t = t + ROW_SPACING
    Set ctrl = TryGetControl(frm, FormLayout.DYN_PREFIX & "lblCat"): If Not ctrl Is Nothing Then ctrl.Left = colLbl: ctrl.Top = t
    Set ctrl = TryGetControl(frm, FormLayout.DYN_PREFIX & "cmbCategorie"): If Not ctrl Is Nothing Then ctrl.Left = colInp: ctrl.Top = t: ctrl.Width = inpWidth
    t = t + ROW_SPACING
    Set ctrl = TryGetControl(frm, FormLayout.DYN_PREFIX & "lblSub"): If Not ctrl Is Nothing Then ctrl.Left = colLbl: ctrl.Top = t
    Dim wSub As Single: wSub = Application.Max(0, inpWidth - 36)
    Set ctrl = TryGetControl(frm, FormLayout.DYN_PREFIX & "cmbSubcategorie"): If Not ctrl Is Nothing Then ctrl.Left = colInp: ctrl.Top = t: ctrl.Width = wSub
    Set ctrl = TryGetControl(frm, FormLayout.DYN_PREFIX & "btnNieuwSub"): If Not ctrl Is Nothing Then ctrl.Left = colInp + wSub + 8: ctrl.Top = t: ctrl.Width = 28: ctrl.Height = 22
    t = t + ROW_SPACING
    Set ctrl = TryGetControl(frm, FormLayout.DYN_PREFIX & "lblBedrag"): If Not ctrl Is Nothing Then ctrl.Left = colLbl: ctrl.Top = t
    Set ctrl = TryGetControl(frm, FormLayout.DYN_PREFIX & "txtBedrag"): If Not ctrl Is Nothing Then ctrl.Left = colInp: ctrl.Top = t: ctrl.Width = inpWidth
    t = t + ROW_SPACING
    Set ctrl = TryGetControl(frm, FormLayout.DYN_PREFIX & "lblBetaal"): If Not ctrl Is Nothing Then ctrl.Left = colLbl: ctrl.Top = t
    Set ctrl = TryGetControl(frm, FormLayout.DYN_PREFIX & "cmbBetaalwijze"): If Not ctrl Is Nothing Then ctrl.Left = colInp: ctrl.Top = t: ctrl.Width = inpWidth

    Dim btnOps As Control, btnAnn As Control
    Set btnOps = TryGetControl(frm, FormLayout.DYN_PREFIX & "imgBtnOpslaan")
    Set btnAnn = TryGetControl(frm, FormLayout.DYN_PREFIX & "imgBtnAnnuleren")
    If Not btnOps Is Nothing And Not btnAnn Is Nothing Then
        Dim btnTotalWidth As Single: btnTotalWidth = btnOps.Width + 24 + btnAnn.Width
        Dim btnLeft As Single: btnLeft = safeLeft + Application.Max(0, (safeWidth - btnTotalWidth)) / 2
        Dim btnY As Single: btnY = safeTop + Application.Max(0, safeHeight - 60)
        btnOps.Left = btnLeft: btnOps.Top = btnY
        btnAnn.Left = btnLeft + btnOps.Width + 24: btnAnn.Top = btnY
    End If

    Exit Sub
ErrHandler:
    Utils.LogError4 "FormLayout", "FrmLayoutSafeAreaDyn", Err.Number, Err.Description
End Sub
Public Sub RemoveDynamicControls(frm As Object)
    On Error GoTo ErrHandler
    Dim i As Long, nm As String
    For i = frm.Controls.Count - 1 To 0 Step -1
        nm = frm.Controls.item(i).name
        If Left$(nm, Len(FormLayout.DYN_PREFIX)) = FormLayout.DYN_PREFIX Then
            On Error Resume Next
            frm.Controls.Remove nm
            On Error GoTo ErrHandler
        End If
    Next i
    Exit Sub
ErrHandler:
    Utils.LogError4 "FormLayout", "RemoveDynamicControls", Err.Number, Err.Description
End Sub

Public Sub AutoFitDropdownDyn(cmb As MSForms.ComboBox)
 On Error GoTo ErrHandler
 Dim i As Long, maxLen As Long
 For i = 0 To cmb.ListCount - 1
  If Len(CStr(cmb.List(i))) > maxLen Then maxLen = Len(CStr(cmb.List(i)))
 Next i
 Const MINW As Long = 120, MAXW As Long = 360

Option Explicit
Public Function GetLayoutSpec(ByVal keyName As String, ByVal fieldName As String) As Variant
 On Error GoTo ErrHandler
 Dim wsRef As Worksheet: Set wsRef = ThisWorkbook.Worksheets("Referenties")
 Dim lo As ListObject: Set lo = wsRef.ListObjects("tblLayoutSpecs")
 Dim hdr As Range, data As Range, r As Range, colIndex As Long
 Set hdr = lo.HeaderRowRange
 Set data = lo.DataBodyRange
 If data Is Nothing Then Exit Function
 Dim i As Long
 colIndex = 0
 For i = 1 To hdr.Columns.Count
    If StrComp(CStr(hdr.Cells(1, i).Value), fieldName, vbTextCompare) = 0 Then
        colIndex = i
        Exit For
    End If
 Next i
 For Each r In data.Rows
  If StrComp(CStr(r.Cells(1, 1).Value), keyName, vbTextCompare) = 0 Then
   GetLayoutSpec = r.Cells(1, colIndex).Value
   Exit Function
  End If
 Next r
 Exit Function
ErrHandler:
 Utils.LogError4 "LayoutSpecsHelpers", "GetLayoutSpec", Err.Number, Err.Description
End Function
Public Function GetShapePath(ByVal shapeName As String, Optional ByVal suffix As String = "") As String
    On Error GoTo ErrHandler

    Dim wsRef As Worksheet: Set wsRef = ThisWorkbook.Worksheets("Referenties")
    If wsRef Is Nothing Then Exit Function

    If wsRef.Visible = xlSheetVeryHidden Then wsRef.Visible = xlSheetVisible
    wsRef.Activate: DoEvents

    Dim shp As Shape
    On Error Resume Next
    Set shp = wsRef.Shapes(shapeName)
    On Error GoTo ErrHandler
    If shp Is Nothing Then Exit Function

    Dim tmp As String: tmp = Utils.GetTempFile(shapeName & suffix, ".jpg")
    Dim ch As ChartObject
    Set ch = wsRef.ChartObjects.Add(Left:=shp.Left, Top:=shp.Top, Width:=Application.Max(8, shp.Width), Height:=Application.Max(8, shp.Height))

    Dim i As Long, pasted As Boolean
    For i = 1 To 3
        On Error Resume Next
        shp.CopyPicture Appearance:=xlScreen, Format:=xlPicture
        ch.Activate
        ch.Chart.Paste
        On Error GoTo ErrHandler

        If ch.Chart.Shapes.Count > 0 Then pasted = True: Exit For
        DoEvents: Application.CutCopyMode = False
    Next i

    If Not pasted Then GoTo CleanUp

    ch.Chart.Export tmp, "JPG"
    If Dir(tmp) <> "" Then GetShapePath = tmp

CleanUp:
    On Error Resume Next
    ch.Delete: Application.CutCopyMode = False
    On Error GoTo 0
    Exit Function

ErrHandler:
    Utils.LogError4 "LayoutSpecsHelpers", "GetShapePath", Err.Number, Err.Description
    Resume CleanUp
End Function

Option Explicit

' Rapporttype voor leesbare status
Private Type tCheckReport
    OkCount As Long
    WarnCount As Long
    ErrCount As Long
    Details As String
End Type

' Publieke entry-point: voert alle sanity checks uit en geeft samenvatting terug
Public Function RunAssetSanity(Optional ByVal includeSpecsCheck As Boolean = True) As String
    On Error GoTo ErrHandler
    
    Dim rep As tCheckReport
    rep.Details = ""
    
    ' 1) Check generieke button shapes + hover
    AppendReport rep, CheckGenericButtonShapes()
    
    ' 2) Check background shape (statisch)
    AppendReport rep, CheckBackgroundShape("bg_frmInvoer_landscape")
    
    ' 3) Optioneel: check tblLayoutSpecs keys & ShapeName koppeling
    If includeSpecsCheck Then
        AppendReport rep, CheckLayoutSpecsKeys()
    End If
    
    Dim header As String
    header = "Sanity-rapport" & vbCrLf & String(40, "-") & vbCrLf & _
             "OK: " & rep.OkCount & " | Waarschuwingen: " & rep.WarnCount & " | Fouten: " & rep.ErrCount & vbCrLf & vbCrLf
    
    RunAssetSanity = header & rep.Details
    Exit Function
    
ErrHandler:
    RunAssetSanity = "FOUT in RunAssetSanity: [" & Err.Number & "] " & Err.Description
End Function

' ---- Helpers: per check produceer tCheckReport ----

Private Function CheckGenericButtonShapes() As tCheckReport
    On Error GoTo ErrHandler
    Dim rc As tCheckReport
    Dim baseNames As Variant
    baseNames = Array("btnShapeSmall", "btnShapeMedium", "btnShapeLarge")
    
    Dim wsRef As Worksheet: Set wsRef = ThisWorkbook.Worksheets("Referenties")
    Dim i As Long, nm As String, nmHover As String
    
    For i = LBound(baseNames) To UBound(baseNames)
        nm = CStr(baseNames(i))
        nmHover = nm & "_hover"
        
        If ShapeExistsOn(wsRef, nm) Then
            rc.OkCount = rc.OkCount + 1
            rc.Details = rc.Details & "OK  : Shape '" & nm & "'" & vbCrLf
        Else
            rc.ErrCount = rc.ErrCount + 1
            rc.Details = rc.Details & "FOUT: Shape '" & nm & "' ontbreekt op Referenties" & vbCrLf
        End If
        
        If ShapeExistsOn(wsRef, nmHover) Then
            rc.OkCount = rc.OkCount + 1
            rc.Details = rc.Details & "OK  : Shape '" & nmHover & "'" & vbCrLf
        Else
            rc.WarnCount = rc.WarnCount + 1
            rc.Details = rc.Details & "LET OP: Hover-shape '" & nmHover & "' ontbreekt (hover werkt dan niet)" & vbCrLf
        End If
    Next i
    
    CheckGenericButtonShapes = rc
    Exit Function
    
ErrHandler:
    CheckGenericButtonShapes.ErrCount = CheckGenericButtonShapes.ErrCount + 1
    CheckGenericButtonShapes.Details = CheckGenericButtonShapes.Details & "FOUT in CheckGenericButtonShapes: [" & Err.Number & "] " & Err.Description & vbCrLf
End Function

Private Function CheckBackgroundShape(ByVal bgShapeName As String) As tCheckReport
    On Error GoTo ErrHandler
    Dim rc As tCheckReport
    Dim wsRef As Worksheet: Set wsRef = ThisWorkbook.Worksheets("Referenties")
    
    If ShapeExistsOn(wsRef, bgShapeName) Then
        rc.OkCount = rc.OkCount + 1
        rc.Details = rc.Details & "OK  : Background-shape '" & bgShapeName & "'" & vbCrLf
    Else
        rc.WarnCount = rc.WarnCount + 1
        rc.Details = rc.Details & "LET OP: Background-shape '" & bgShapeName & "' niet gevonden (form laad zonder achtergrond)" & vbCrLf
    End If
    
    CheckBackgroundShape = rc
    Exit Function
    
ErrHandler:
    CheckBackgroundShape.ErrCount = CheckBackgroundShape.ErrCount + 1
    CheckBackgroundShape.Details = CheckBackgroundShape.Details & "FOUT in CheckBackgroundShape: [" & Err.Number & "] " & Err.Description & vbCrLf
End Function

Private Function CheckLayoutSpecsKeys() As tCheckReport
    On Error GoTo ErrHandler
    Dim rc As tCheckReport
    
    Dim wsRef As Worksheet: Set wsRef = ThisWorkbook.Worksheets("Referenties")
    Dim lo As ListObject
    On Error Resume Next
    Set lo = wsRef.ListObjects("tblLayoutSpecs")
    On Error GoTo ErrHandler
    
    If lo Is Nothing Then
        rc.WarnCount = rc.WarnCount + 1
        rc.Details = rc.Details & "LET OP: Tabel 'tblLayoutSpecs' niet gevonden op Referenties" & vbCrLf
        CheckLayoutSpecsKeys = rc
        Exit Function
    End If
    
    If lo.DataBodyRange Is Nothing Then
        rc.WarnCount = rc.WarnCount + 1
        rc.Details = rc.Details & "LET OP: 'tblLayoutSpecs' heeft geen rijen" & vbCrLf
        CheckLayoutSpecsKeys = rc
        Exit Function
    End If
    
    ' Zoek kolommen Key en ShapeName
    Dim colKey As Long, colShape As Long
    colKey = FindColumnIndex(lo, "Key")
    colShape = FindColumnIndex(lo, "ShapeName")
    
    If colKey = 0 Or colShape = 0 Then
        rc.ErrCount = rc.ErrCount + 1
        rc.Details = rc.Details & "FOUT: Kolom 'Key' of 'ShapeName' ontbreekt in tblLayoutSpecs" & vbCrLf
        CheckLayoutSpecsKeys = rc
        Exit Function
    End If
    
    ' Loop door rijen; check dat ShapeName bestaat als shape
    Dim r As Range, keyName As String, shapeName As String
    For Each r In lo.DataBodyRange.Rows
        keyName = CStr(Trim$(r.Cells(1, colKey).Value))
        shapeName = CStr(Trim$(r.Cells(1, colShape).Value))
        
        If Len(keyName) = 0 Then
            rc.WarnCount = rc.WarnCount + 1
            rc.Details = rc.Details & "LET OP: Lege Key in tblLayoutSpecs (rij " & r.Row & ")" & vbCrLf
        End If
        
        If Len(shapeName) = 0 Then
            rc.ErrCount = rc.ErrCount + 1
            rc.Details = rc.Details & "FOUT: Key '" & keyName & "' heeft lege ShapeName" & vbCrLf
        ElseIf Not ShapeExistsOn(wsRef, shapeName) Then
            rc.ErrCount = rc.ErrCount + 1
            rc.Details = rc.Details & "FOUT: Key '" & keyName & "' verwijst naar ontbrekende shape '" & shapeName & "'" & vbCrLf
        Else
            rc.OkCount = rc.OkCount + 1
            rc.Details = rc.Details & "OK  : Key '" & keyName & "' ? Shape '" & shapeName & "'" & vbCrLf
        End If
    Next r
    
    CheckLayoutSpecsKeys = rc
    Exit Function
    
ErrHandler:
    CheckLayoutSpecsKeys.ErrCount = CheckLayoutSpecsKeys.ErrCount + 1
    CheckLayoutSpecsKeys.Details = CheckLayoutSpecsKeys.Details & "FOUT in CheckLayoutSpecsKeys: [" & Err.Number & "] " & Err.Description & vbCrLf
End Function

' ---- Utility helpers ----

Private Function ShapeExistsOn(ws As Worksheet, ByVal shapeName As String) As Boolean
    On Error Resume Next
    Dim shp As Shape
    Set shp = ws.Shapes(shapeName)
    ShapeExistsOn = Not shp Is Nothing
    On Error GoTo 0
End Function

Private Function FindColumnIndex(lo As ListObject, ByVal headerName As String) As Long
    On Error Resume Next
    Dim i As Long
    For i = 1 To lo.HeaderRowRange.Columns.Count
        If StrComp(CStr(lo.HeaderRowRange.Cells(1, i).Value), headerName, vbTextCompare) = 0 Then
            FindColumnIndex = i
            Exit Function
        End If
    Next i
    FindColumnIndex = 0
End Function

Private Sub AppendReport(ByRef total As tCheckReport, ByRef part As tCheckReport)
    total.OkCount = total.OkCount + part.OkCount
    total.WarnCount = total.WarnCount + part.WarnCount
    total.ErrCount = total.ErrCount + part.ErrCount
    If Len(part.Details) > 0 Then
        total.Details = total.Details & part.Details
        If Right$(total.Details, 2) <> vbCrLf Then total.Details = total.Details & vbCrLf
    End If
End Sub

Public Sub BerekenWeekTotaal()
 On Error GoTo ErrHandler
 Dim ws As Worksheet: Set ws = ThisWorkbook.Worksheets("Database")
 Utils.EnsureDBHeaders
 Dim lr As Long: lr = Utils.LastRow(ws, DBCols.dbDate)
 If lr < 2 Then Exit Sub
 Dim dPrev As Date: dPrev = DateAdd("d", -7, Date)
 Dim tgtWeek As Long: tgtWeek = Utils.IsoWeek(dPrev)
 Dim tgtYear As Long: tgtYear = Year(dPrev)
 Dim i As Long, total As Double
 Dim catDict As Object: Set catDict = CreateObject("Scripting.Dictionary")
 For i = 2 To lr
  If CLng(Val(ws.Cells(i, DBCols.dbYear).Value)) = tgtYear And _
     CLng(Val(ws.Cells(i, DBCols.dbWeek).Value)) = tgtWeek And _
     CStr(ws.Cells(i, DBCols.dbType).Value) = "Uitgave" Then
     Dim amount As Double: amount = CDbl(Val(ws.Cells(i, DBCols.dbAmount).Value))
     total = total + amount
     Dim cat As String: cat = CStr(ws.Cells(i, DBCols.dbCategory).Value)
     If Len(Trim$(cat)) > 0 Then Utils.DictAddOrIncrement catDict, cat, 1
  End If
 Next i
 Dim mostCat As String, maxCount As Long, k As Variant
 For Each k In catDict.keys
  If catDict(k) > maxCount Then maxCount = catDict(k): mostCat = CStr(k)
 Next k
 If ws.Cells(1, SummaryCols.colWeekSummaryStart).Value = "" Then
  ws.Cells(1, SummaryCols.colWeekSummaryStart).Resize(1, 5).Value = Array("Jaar" & vbLf & "Week", "Jaar", "Weeknr", "Totaal", "Meest gebruikte categorie")
 End If

Public Sub BerekenMaandTotaal()
 On Error GoTo ErrHandler
 Dim ws As Worksheet: Set ws = ThisWorkbook.Worksheets("Database")
 Utils.EnsureDBHeaders
 Dim lr As Long: lr = Utils.LastRow(ws, DBCols.dbDate)
 If lr < 2 Then Exit Sub
 Dim prevMonthKey As String: prevMonthKey = Format(DateAdd("m", -1, Date), "YYYY-MM")
 Dim i As Long
 Dim totaal As Double, dagenCount As Long
 Dim weekDict As Object: Set weekDict = CreateObject("Scripting.Dictionary")
 Dim catDict As Object: Set catDict = CreateObject("Scripting.Dictionary")
 Dim catCountDict As Object: Set catCountDict = CreateObject("Scripting.Dictionary")
 Dim duursteDag As Double, duursteWeek As Double
 Dim maxCatValue As Double, minCatValue As Double
 Dim maxCatName As String, minCatName As String
 maxCatValue = 0: minCatValue = 1E+20
 For i = 2 To lr
  If CStr(ws.Cells(i, DBCols.dbMonthKey).Value) = prevMonthKey And CStr(ws.Cells(i, DBCols.dbType).Value) = "Uitgave" Then
   Dim bedrag As Double: bedrag = CDbl(Val(ws.Cells(i, DBCols.dbAmount).Value))
   totaal = totaal + bedrag
   dagenCount = dagenCount + 1
   If bedrag > duursteDag Then duursteDag = bedrag
   Dim weekKey As String: weekKey = CStr(ws.Cells(i, DBCols.dbYear).Value) & "-W" & CStr(ws.Cells(i, DBCols.dbWeek).Value)
   Utils.DictAddOrIncrement weekDict, weekKey, bedrag
   Dim categorie As String: categorie = CStr(ws.Cells(i, DBCols.dbCategory).Value)
   Utils.DictAddOrIncrement catDict, categorie, bedrag
   Utils.DictAddOrIncrement catCountDict, categorie, 1
  End If
 Next i
 Dim key As Variant
 For Each key In weekDict.keys: If weekDict(key) > duursteWeek Then duursteWeek = weekDict(key)
 Next key
 For Each key In catDict.keys
  If catDict(key) > maxCatValue Then maxCatValue = catDict(key): maxCatName = CStr(key)
  If catDict(key) < minCatValue Then minCatValue = catDict(key): minCatName = CStr(key)
 Next key
 Dim overzichtRow As Long: overzichtRow = Utils.LastRow(ws, SummaryCols.colMonthSummaryStart) + 1
 If ws.Cells(1, SummaryCols.colMonthSummaryStart).Value = "" Then
  ws.Cells(1, SummaryCols.colMonthSummaryStart).Resize(1, 7).Value = Array("mnd", "ttl", "gpd", "gpw", "dudg", "duwk", "m_u_a")
 End If
 ws.Cells(overzichtRow, SummaryCols.colMonthSummaryStart).Value = prevMonthKey
 ws.Cells(overzichtRow, SummaryCols.colMonthSummaryStart + 1).Value = totaal
 ws.Cells(overzichtRow, SummaryCols.colMonthSummaryStart + 2).Value = IIf(dagenCount > 0, totaal / dagenCount, 0)
 ws.Cells(overzichtRow, SummaryCols.colMonthSummaryStart + 3).Value = IIf(weekDict.Count > 0, totaal / weekDict.Count, 0)
 ws.Cells(overzichtRow, SummaryCols.colMonthSummaryStart + 4).Value = duursteDag
 ws.Cells(overzichtRow, SummaryCols.colMonthSummaryStart + 5).Value = duursteWeek
 ws.Cells(overzichtRow, SummaryCols.colMonthSummaryStart + 6).Value = maxCatName
 ' Categorie-overzicht append met maandkolom
 Dim catStartCol As Long: catStartCol = SummaryCols.colMonthSummaryStart + 8 ' X
 If ws.Cells(1, catStartCol - 1).Value = "" Then
  ws.Cells(1, catStartCol - 1).Resize(1, 4).Value = Array("Maand", "Categorie", "Aantal", "Totaal")
 End If
 Dim r As Long: r = Utils.LastRow(ws, catStartCol - 1) + 1
 For Each key In catDict.keys
  ws.Cells(r, catStartCol - 1).Value = prevMonthKey
  ws.Cells(r, catStartCol).Value = key
  ws.Cells(r, catStartCol + 1).Value = catCountDict(key)
  ws.Cells(r, catStartCol + 2).Value = catDict(key)
  r = r + 1
 Next key
 Exit Sub

Public Function ToMonthlyAmount(ByVal periode As String, ByVal bedrag As Double) As Double

  Case "kwartaal", "per kwartaal": ToMonthlyAmount = bedrag / 3

Public Sub ExportErrorLogCSV()
    On Error GoTo ErrHandler
    Dim ws As Worksheet
    Set ws = ThisWorkbook.Worksheets("ErrorLog")
    If ws Is Nothing Then
        MsgBox "ErrorLog sheet not found.", vbExclamation
        Exit Sub
    End If
    Dim tempPath As String
    tempPath = Environ("TEMP") & "\ErrorLog_" & Format(Now, "yyyymmdd_hhnnss") & ".csv"
    ws.Copy
    ActiveWorkbook.SaveAs Filename:=tempPath, FileFormat:=xlCSV
