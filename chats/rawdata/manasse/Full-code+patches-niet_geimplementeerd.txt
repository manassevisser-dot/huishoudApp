' PATCH 1: FormBackground.GetBackgroundPath
' FIX: Shape.Export met retry + ChartObject fallback
' ============================================================
Option Explicit

Public Function GetBackgroundPath() As String
    On Error GoTo ErrHandler
    Dim wsRef As Worksheet: Set wsRef = ThisWorkbook.Worksheets("Referenties")
    If wsRef Is Nothing Then
        Utils.LogError4 "FormBackground", "GetBackgroundPath", 0, "Referenties sheet niet gevonden"
        Exit Function
    End If

    Dim bgShapeName As String
    bgShapeName = LayoutSpecsHelpers.GetBackgroundShapeNameFromSpecs()
    If Len(bgShapeName) = 0 Then bgShapeName = "bg_frmInvoer_landscape"

    Dim shp As Shape
    On Error Resume Next
    Set shp = wsRef.Shapes(bgShapeName)
    On Error GoTo ErrHandler
    
    If shp Is Nothing Then
        Utils.LogError4 "FormBackground", "GetBackgroundPath", 0, "Shape niet gevonden: " & bgShapeName
        Exit Function
    End If

    Dim tmp As String
    tmp = Utils.GetTempFile("frmInvoer_bg", ".jpg")

    ' ===== NIEUWE METHODE: Shape.Export met retry =====
    Dim exportSuccess As Boolean: exportSuccess = False
    Dim attempt As Long
    
    For attempt = 1 To 3
        On Error Resume Next
        shp.Export tmp, "JPG"
        
        ' Wacht tot file echt bestaat + valideer grootte
        Dim waitStart As Double: waitStart = Timer
        Do While Timer - waitStart < 0.5
            DoEvents
            If Dir(tmp) <> "" Then
                If FileLen(tmp) > 100 Then  ' Min 100 bytes = valide JPG
                    exportSuccess = True
                    Exit For
                End If
            End If
        Loop
        
        If exportSuccess Then Exit For
        On Error GoTo ErrHandler
        
        ' Cleanup failed attempt
        If Dir(tmp) <> "" Then Kill tmp
    Next attempt

    If exportSuccess Then
        GetBackgroundPath = tmp
        Exit Function
    End If

    ' ===== FALLBACK: ChartObject methode (als Export faalt) =====
    Utils.LogError4 "FormBackground", "GetBackgroundPath", 0, "Shape.Export gefaald, probeer ChartObject fallback"
    
    Dim ch As ChartObject
    Dim safeLeft As Double, safeTop As Double, safeWidth As Double, safeHeight As Double
    
    ' FIX: Bounds checking voor negatieve coördinaten
    safeLeft = IIf(shp.Left < 0, 0, shp.Left)
    safeTop = IIf(shp.Top < 0, 0, shp.Top)
    safeWidth = IIf(shp.Width < 10, 10, shp.Width)
    safeHeight = IIf(shp.Height < 10, 10, shp.Height)
    
    On Error Resume Next
    Set ch = wsRef.ChartObjects.Add(safeLeft, safeTop, safeWidth, safeHeight)
    If ch Is Nothing Then
        Utils.LogError4 "FormBackground", "GetBackgroundPath", 0, "ChartObject.Add gefaald"
        Exit Function
    End If
    
    shp.CopyPicture Appearance:=xlScreen, Format:=xlPicture
    ch.Chart.Paste
    ch.Chart.Export tmp, "JPG"
    ch.Delete
    On Error GoTo ErrHandler
    
    If Dir(tmp) <> "" And FileLen(tmp) > 100 Then
        GetBackgroundPath = tmp
    Else
        Utils.LogError4 "FormBackground", "GetBackgroundPath", 0, "Export volledig gefaald (beide methoden)"
    End If
    Exit Function

ErrHandler:
    Utils.LogError4 "FormBackground", "GetBackgroundPath", Err.Number, Err.Description
    On Error Resume Next
    If Not ch Is Nothing Then ch.Delete
End Function

' PATCH 2: Utils.GetTempFile
' FIX: Path separator bug
' ============================================================

Public Function GetTempFile_FIXED(ByVal baseName As String, ByVal ext As String) As String
    On Error GoTo ErrHandler
    
    Dim tmp As String
    tmp = Environ$("TEMP")
    If Len(tmp) = 0 Then tmp = Environ$("TMP")
    If Len(tmp) = 0 Then tmp = ThisWorkbook.Path
    
    ' FIX: Correcte path separator check
    If Right$(tmp, 1) <> Application.PathSeparator Then
        tmp = tmp & Application.PathSeparator
    End If
    
    ' FIX: Sanitize baseName (verwijder invalide karakters)
    Dim safeName As String
    safeName = baseName
    safeName = Replace(safeName, "\", "_")
    safeName = Replace(safeName, "/", "_")
    safeName = Replace(safeName, ":", "_")
    safeName = Replace(safeName, "*", "_")
    safeName = Replace(safeName, "?", "_")
    safeName = Replace(safeName, """", "_")
    safeName = Replace(safeName, "<", "_")
    safeName = Replace(safeName, ">", "_")
    safeName = Replace(safeName, "|", "_")
    
    GetTempFile_FIXED = tmp & safeName & "_" & Format(Now, "yyyymmdd_hhnnss") & ext
    Exit Function
    
ErrHandler:
    ' Fallback naar workbook directory
    GetTempFile_FIXED = ThisWorkbook.Path & Application.PathSeparator & _
                        baseName & "_" & Format(Now, "yyyymmdd_hhnnss") & ext
    Utils.LogError4 "Utils", "GetTempFile_FIXED", Err.Number, Err.Description
End Function

#Else
    Private Declare Function GetForegroundWindow Lib "user32" () As Long
    Private Declare Function GetWindowText Lib "user32" Alias "GetWindowTextA" _
        (ByVal hwnd As Long, ByVal lpString As String, ByVal cch As Long) As Long
#End If

Private Function GetActiveFormSafe_FIXED() As Object
    On Error Resume Next
    
    ' METHODE 1: Zoek via foreground window handle (runtime)
    #If VBA7 Then
        Dim hwnd As LongPtr
    #Else
        Dim hwnd As Long
    #End If
    
    hwnd = GetForegroundWindow()
    
    If hwnd <> 0 Then
        Dim ufControl As Object
        For Each ufControl In VBA.UserForms
            ' Check of form window handle overeenkomt
            If ufControl.hwnd = hwnd Then
                Set GetActiveFormSafe_FIXED = ufControl
                Exit Function
            End If
        Next ufControl
    End If
    
    ' METHODE 2: Zoek vorm met Caption matching window title
    If hwnd <> 0 Then
        Dim winTitle As String: winTitle = Space$(255)
        Dim titleLen As Long
        titleLen = GetWindowText(hwnd, winTitle, 255)
        winTitle = Left$(winTitle, titleLen)
        
        If Len(winTitle) > 0 Then
            Dim uf As Object
            For Each uf In VBA.UserForms
                If StrComp(uf.Caption, winTitle, vbTextCompare) = 0 Then
                    Set GetActiveFormSafe_FIXED = uf
                    Exit Function
                End If
            Next uf
        End If
    End If
    
    ' METHODE 3: Laatste getoonde form (fallback)
    If VBA.UserForms.Count > 0 Then
        ' Laatste form in collectie is meestal laatst getoond
        Set GetActiveFormSafe_FIXED = VBA.UserForms(VBA.UserForms.Count - 1)
    End If
    
    On Error GoTo 0
End Function

' ============================================================
' PATCH 4: Utils.ParseAmount
' FIX: Locale-aware parsing met duizendtallen support
' ============================================================

Public Function ParseAmount_FIXED(ByVal s As String) As Double
    On Error GoTo ErrHandler
    
    Dim decSep As String: decSep = Application.International(xlDecimalSeparator)
    Dim thouSep As String: thouSep = Application.International(xlThousandsSeparator)
    
    s = Trim$(s)
    If s = "" Then
        ParseAmount_FIXED = 0
        Exit Function
    End If
    
    ' Verwijder currency symbolen
    s = Replace(s, "€", "")
    s = Replace(s, "$", "")
    s = Replace(s, "£", "")
    s = Trim$(s)
    
    ' Detecteer formaat: Europees (1.234,56) vs Amerikaans (1,234.56)
    Dim lastDot As Long: lastDot = InStrRev(s, ".")
    Dim lastComma As Long: lastComma = InStrRev(s, ",")
    
    If lastDot > lastComma Then
        ' Amerikaans formaat: 1,234.56
        s = Replace(s, ",", "")  ' Verwijder duizendtallen
        s = Replace(s, ".", decSep)  ' Converteer decimaal
    ElseIf lastComma > lastDot Then
        ' Europees formaat: 1.234,56
        s = Replace(s, ".", "")  ' Verwijder duizendtallen
        s = Replace(s, ",", decSep)  ' Converteer decimaal
    Else
        ' Geen beide separators, of allebei op zelfde positie
        ' Converteer naar system decimaal separator
        If InStr(s, ",") > 0 Then s = Replace(s, ",", decSep)
        If InStr(s, ".") > 0 Then s = Replace(s, ".", decSep)
    End If
    
    ' Final sanity check
    If Not IsNumeric(s) Then
        Utils.LogError4 "Utils", "ParseAmount_FIXED", 0, "Kon bedrag niet parsen: " & s
        ParseAmount_FIXED = 0
        Exit Function
    End If
    
    ParseAmount_FIXED = CDbl(s)
    Exit Function
    
ErrHandler:
    Utils.LogError4 "Utils", "ParseAmount_FIXED", Err.Number, Err.Description & " (input: " & s & ")"
    ParseAmount_FIXED = 0
End Function

' PATCH 5: LayoutSpecsHelpers.GetShapePath
' FIX: Betrouwbare paste detection met sleep
' ============================================================

Public Function GetShapePath_FIXED(ByVal shapeName As String, Optional ByVal suffix As String = "") As String
    On Error GoTo ErrHandler
    
    Dim wsRef As Worksheet: Set wsRef = ThisWorkbook.Worksheets("Referenties")
    If wsRef Is Nothing Then Exit Function
    
    ' Ensure sheet visible
    If wsRef.Visible = xlSheetVeryHidden Then wsRef.Visible = xlSheetVisible
    wsRef.Activate
    DoEvents
    
    Dim shp As Shape
    On Error Resume Next
    Set shp = wsRef.Shapes(shapeName)
    On Error GoTo ErrHandler
    
    If shp Is Nothing Then
        Utils.LogError4 "LayoutSpecsHelpers", "GetShapePath_FIXED", 1004, "Shape niet gevonden: " & shapeName
        Exit Function
    End If
    
    Dim w As Single, h As Single
    w = Application.Max(8, shp.Width)
    h = Application.Max(8, shp.Height)
    
    Dim tmp As String
    tmp = Utils.GetTempFile(shapeName & suffix, ".jpg")
    
    Dim ch As ChartObject
    Set ch = wsRef.ChartObjects.Add(Left:=shp.Left, Top:=shp.Top, Width:=w, Height:=h)
    
    ' FIX: Verbeterde paste logic met retry + validation
    Dim i As Long, pasted As Boolean
    pasted = False
    
    For i = 1 To 5  ' Verhoog van 3 naar 5 retries
        On Error Resume Next
        
        ' Clear previous clipboard
        Application.CutCopyMode = False
        DoEvents
        
        ' Copy shape
        shp.CopyPicture Appearance:=xlScreen, Format:=xlPicture
        
        ' Activate chart en paste
        ch.Activate
        DoEvents
        
        ' Clear chart eerst
        Dim tempShape As Shape
        For Each tempShape In ch.Chart.Shapes
            tempShape.Delete
        Next tempShape
        
        ch.Chart.Paste
        
        ' FIX: Wacht tot paste compleet is (max 1 sec)
        Dim waitStart As Double: waitStart = Timer
        Do While Timer - waitStart < 1
            DoEvents
            If ch.Chart.Shapes.Count > 0 Then
                ' Extra validatie: check of shape data heeft
                If ch.Chart.Shapes(1).Width > 0 And ch.Chart.Shapes(1).Height > 0 Then
                    pasted = True
                    Exit For
                End If
            End If
            ' Sleep 50ms tussen checks
            Application.Wait Now + TimeValue("00:00:00.05")
        Loop
        
        On Error GoTo ErrHandler
        
        If pasted Then Exit For
        
        ' Cleanup failed attempt
        Application.CutCopyMode = False
    Next i
    
    If Not pasted Then
        Utils.LogError4 "LayoutSpecsHelpers", "GetShapePath_FIXED", 0, "Paste gefaald na 5 pogingen: " & shapeName
        GoTo CleanUp
    End If
    
    ' Export chart
    On Error Resume Next
    ch.Chart.Export tmp, "JPG"
    On Error GoTo ErrHandler
    
    ' Validate export
    If Dir$(tmp) <> "" And FileLen(tmp) > 100 Then
        GetShapePath_FIXED = tmp
    Else
        Utils.LogError4 "LayoutSpecsHelpers", "GetShapePath_FIXED", 0, "Export gefaald of corrupt: " & tmp
    End If

CleanUp:
    On Error Resume Next
    ch.Delete
    Application.CutCopyMode = False
    On Error GoTo 0
    Exit Function

ErrHandler:
    Utils.LogError4 "LayoutSpecsHelpers", "GetShapePath_FIXED", Err.Number, Err.Description
    Resume CleanUp
End Function

Private Function SpecsColumnsAndRowsCheck() As String
    On Error GoTo ErrHandler
    Dim wsRef As Worksheet: Set wsRef = ThisWorkbook.Worksheets("Referenties")
    Dim lo As ListObject: Set lo = wsRef.ListObjects("tblLayoutSpecs")
    If lo Is Nothing Then
        SpecsColumnsAndRowsCheck = "FOUT: Tabel 'tblLayoutSpecs' niet gevonden."
        Exit Function
    End If

    Dim requiredCols As Variant
    requiredCols = Array("Key", "ShapeName", "Width", "Height", "X", "Y")
    Dim i As Long, missing As String
    For i = LBound(requiredCols) To UBound(requiredCols)
        If FindColumnIndex(lo, CStr(requiredCols(i))) = 0 Then
            missing = missing & "- Kolom ontbreekt: " & requiredCols(i) & vbCrLf
        End If
    Next i

    If Len(missing) > 0 Then
        SpecsColumnsAndRowsCheck = "FOUT (Specs):" & vbCrLf & missing
        Exit Function
    End If

    If lo.DataBodyRange Is Nothing Then
        SpecsColumnsAndRowsCheck = "LET OP: 'tblLayoutSpecs' heeft nog geen rijen."
        Exit Function
    End If

    Dim expectedKeys As Variant
    expectedKeys = Array("Prev", "Next", "Save", "Cancel", "RowAdd", "RowDel")

    Dim colKey As Long, colShape As Long, colW As Long, colH As Long, colX As Long, colY As Long
    colKey = FindColumnIndex(lo, "Key")
    colShape = FindColumnIndex(lo, "ShapeName")
    colW = FindColumnIndex(lo, "Width")
    colH = FindColumnIndex(lo, "Height")
    colX = FindColumnIndex(lo, "X")
    colY = FindColumnIndex(lo, "Y")

    Dim r As Range, k As Variant, found As Boolean, line As String
    For Each k In expectedKeys
        found = False
        For Each r In lo.DataBodyRange.Rows
            If Trim$(CStr(r.Cells(1, colKey).Value)) = CStr(k) Then
                found = True
                If Len(Trim$(CStr(r.Cells(1, colShape).Value))) = 0 _
                   Or Len(Trim$(CStr(r.Cells(1, colW).Value))) = 0 _
                   Or Len(Trim$(CStr(r.Cells(1, colH).Value))) = 0 _
                   Or Len(Trim$(CStr(r.Cells(1, colX).Value))) = 0 _
                   Or Len(Trim$(CStr(r.Cells(1, colY).Value))) = 0 Then
                    line = line & "- Key '" & k & "': heeft lege velden (ShapeName/Width/Height/X/Y)." & vbCrLf
                End If
                Exit For
            End If
        Next r
        If Not found Then line = line & "- Key ontbreekt in specs: " & k & vbCrLf
    Next k

    If Len(line) > 0 Then
        SpecsColumnsAndRowsCheck = "LET OP (Specs):" & vbCrLf & line
    Else
        SpecsColumnsAndRowsCheck = "OK  (Specs): Kolommen & rijen zijn compleet."
    End If
    Exit Function

ErrHandler:
    SpecsColumnsAndRowsCheck = "FOUT in SpecsColumnsAndRowsCheck: [" & Err.Number & "] " & Err.Description
End Function

Private Function ShapesExistenceCheck() As String
    On Error GoTo ErrHandler
    Dim wsRef As Worksheet: Set wsRef = ThisWorkbook.Worksheets("Referenties")
    Dim lo As ListObject: Set lo = wsRef.ListObjects("tblLayoutSpecs")
    If lo Is Nothing Or lo.DataBodyRange Is Nothing Then
        ShapesExistenceCheck = "LET OP: 'tblLayoutSpecs' ontbreekt of heeft geen rijen; skip shapes-check."
        Exit Function
    End If

    Dim colShape As Long, colKey As Long
    colShape = FindColumnIndex(lo, "ShapeName")
    colKey = FindColumnIndex(lo, "Key")

    Dim r As Range, shapeName As String, keyName As String, report As String
    For Each r In lo.DataBodyRange.Rows
        keyName = Trim$(CStr(r.Cells(1, colKey).Value))
        shapeName = Trim$(CStr(r.Cells(1, colShape).Value))
        If Len(shapeName) > 0 Then
            If Not ShapeExistsOn(wsRef, shapeName) Then
                report = report & "- Key '" & keyName & "': base-shape ontbreekt: " & shapeName & vbCrLf
            Else
                If LCase$(keyName) <> "background" Then
                    Dim hoverName As String: hoverName = shapeName & "_hover"
                    If Not ShapeExistsOn(wsRef, hoverName) Then
                        report = report & "LET OP: Key '" & keyName & "': hover-shape ontbreekt: " & hoverName & vbCrLf
                    End If
                End If
            End If
        Else
            report = report & "- Key '" & keyName & "': ShapeName is leeg." & vbCrLf
        End If
    Next r

    If Len(report) = 0 Then
        ShapesExistenceCheck = "OK  (Shapes): base/hover bestaan (hover niet vereist voor Background)."
    Else
        ShapesExistenceCheck = "LET OP (Shapes):" & vbCrLf & report
    End If
    Exit Function

ErrHandler:
    ShapesExistenceCheck = "FOUT in ShapesExistenceCheck: [" & Err.Number & "] " & Err.Description
End Function

Private Function FormImageTagCheck() As String
    On Error GoTo ErrHandler
    Dim frm As Object
    Set frm = GetActiveFormSafe()

    If frm Is Nothing Then
        FormImageTagCheck = "LET OP (Form): geen actief form gevonden; open een UserForm en run opnieuw."
        Exit Function
    End If

    Dim c As Control, issues As String
    For Each c In frm.Controls
        If TypeOf c Is MSForms.Image Then
            Dim tagText As String: tagText = CStr(c.Tag)
            Dim baseName As String: baseName = ParseTagLocal(tagText, "base")
            Dim hoverName As String: hoverName = ParseTagLocal(tagText, "hover")

            If Len(baseName) = 0 Then
                issues = issues & "- Image '" & c.Name & "': Tag mist 'base='." & vbCrLf
            Else
                If Not ShapeExistsOn(ThisWorkbook.Worksheets("Referenties"), baseName) Then
                    issues = issues & "- Image '" & c.Name & "': base-shape niet gevonden: " & baseName & vbCrLf
                End If
            End If

            If InStr(1, LCase$(c.Name), "imgbtn", vbTextCompare) > 0 Or _
               InStr(1, LCase$(c.Name), "imgaddrow", vbTextCompare) > 0 Or _
               InStr(1, LCase$(c.Name), "imgdelrow", vbTextCompare) > 0 Then
                If Len(hoverName) = 0 Then
                    issues = issues & "LET OP: Image '" & c.Name & "': Tag mist 'hover='." & vbCrLf
                ElseIf Not ShapeExistsOn(ThisWorkbook.Worksheets("Referenties"), hoverName) Then
                    issues = issues & "LET OP: Image '" & c.Name & "': hover-shape niet gevonden: " & hoverName & vbCrLf
                End If
            End If
        End If
    Next c

    If Len(issues) = 0 Then
        FormImageTagCheck = "OK  (Form): alle Image-tags valide (base/hover) en shapes gevonden."
    Else
        FormImageTagCheck = "LET OP (Form):" & vbCrLf & issues
    End If
    Exit Function

ErrHandler:
    FormImageTagCheck = "FOUT in FormImageTagCheck: [" & Err.Number & "] " & Err.Description
End Function

Private Function BackgroundExportCheck() As String
    On Error GoTo ErrHandler
    Dim p As String
    p = FormBackground.GetBackgroundPath()
    If Len(p) = 0 Then
        BackgroundExportCheck = "FOUT (Background): GetBackgroundPath gaf leeg pad."
    ElseIf Dir$(p) = "" Then
        BackgroundExportCheck = "FOUT (Background): bestand bestaat niet: " & p
    Else
        ' Valideer file size
        Dim fSize As Long: fSize = FileLen(p)
        If fSize < 100 Then
            BackgroundExportCheck = "FOUT (Background): bestand te klein (" & fSize & " bytes): " & p
        Else
            BackgroundExportCheck = "OK  (Background): export pad = " & p & " (" & fSize & " bytes)"
        End If
    End If
    Exit Function

ErrHandler:
    BackgroundExportCheck = "FOUT in BackgroundExportCheck: [" & Err.Number & "] " & Err.Description
End Function

Private Function FindColumnIndex(lo As ListObject, ByVal headerName As String) As Long
    On Error Resume Next
    Dim i As Long
    For i = 1 To lo.HeaderRowRange.Columns.Count
        If StrComp(CStr(lo.HeaderRowRange.Cells(1, i).Value), headerName, vbTextCompare) = 0 Then
            FindColumnIndex = i
            Exit Function
        End If
    Next i
    FindColumnIndex = 0
End Function

Private Function ShapeExistsOn(ws As Worksheet, ByVal shapeName As String) As Boolean
    On Error Resume Next
    Dim shp As Shape
    Set shp = ws.Shapes(shapeName)
    ShapeExistsOn = Not shp Is Nothing
    On Error GoTo 0
End Function

' PATCH 3: Verbeterde ParseTag met sanitization
Private Function ParseTagLocal(ByVal tagText As String, ByVal key As String) As String
    On Error GoTo ErrHandler
    If Len(tagText) = 0 Then Exit Function
    
    Dim parts() As String: parts = Split(tagText, vbLf)
    Dim i As Long, kv() As String, rawValue As String
    
    For i = LBound(parts) To UBound(parts)
        kv = Split(parts(i), "=")
        If UBound(kv) >= 1 Then  ' Support voor embedded =
            If StrComp(Trim$(kv(0)), key, vbTextCompare) = 0 Then
                ' Join alles na eerste =
                rawValue = Mid$(parts(i), InStr(parts(i), "=") + 1)
                rawValue = Trim$(rawValue)
                
                ' Valideer niet-leeg EN geen invalid karakters
                If Len(rawValue) > 0 And _
                   InStr(rawValue, vbNullChar) = 0 And _
                   InStr(rawValue, Chr(0)) = 0 Then
                    ParseTagLocal = rawValue
                    Exit Function
                End If
            End If
        End If
    Next i
    
    Exit Function
ErrHandler:
    ' Log maar retourneer lege string (graceful degradation)
    On Error Resume Next
    Utils.LogError4 "modLayoutValidator", "ParseTagLocal", Err.Number, _
                    "Tag parse error voor key '" & key & "': " & Err.Description
    ParseTagLocal = ""
End Function

' PATCH 2: Robuuste active form detection met hwnd lookup
Private Function GetActiveFormSafe() As Object
    On Error Resume Next
    
    ' METHODE 1: Probeer VBE ActiveWindow (design-time)
    Dim vbeForm As Object
    Set vbeForm = Application.VBE.ActiveWindow
    If Not vbeForm Is Nothing Then

        Dim hwnd As LongPtr

    #End If

    If hwnd <> 0 Then
        Dim ufControl As Object

            ' Check of form window handle overeenkomt

                Set GetActiveFormSafe = ufControl
                Exit Function
            End If
        Next ufControl
    End If
    
    ' METHODE 3: Laatste getoonde form (fallback)
    If VBA.UserForms.Count > 0 Then
        Set GetActiveFormSafe = VBA.UserForms(VBA.UserForms.Count - 1)
    End If
    
    On Error GoTo 0
End Function

' PATCH 1: Verbeterde Shape.Export met retry logic en file validation
Public Function GetBackgroundPath() As String
    On Error GoTo ErrHandler
    Dim wsRef As Worksheet: Set wsRef = ThisWorkbook.Worksheets("Referenties")
    If wsRef Is Nothing Then Exit Function

    Dim bgShapeName As String
    bgShapeName = LayoutSpecsHelpers.GetBackgroundShapeNameFromSpecs()
    If Len(bgShapeName) = 0 Then bgShapeName = "bg_frmInvoer_landscape"

    Dim shp As Shape
    On Error Resume Next
    Set shp = wsRef.Shapes(bgShapeName)
    On Error GoTo ErrHandler
    If shp Is Nothing Then
        Utils.LogError4 "FormBackground", "GetBackgroundPath", 0, "Shape niet gevonden: " & bgShapeName
        Exit Function
    End If

    Dim tmp As String
    tmp = Utils.GetTempFile("frmInvoer_bg", ".jpg")

    ' VERBETERD: Probeer Shape.Export met retry + file validation
    Dim exportSuccess As Boolean: exportSuccess = False
    Dim attempt As Long
    For attempt = 1 To 3
        On Error Resume Next
        shp.Export tmp, "JPG"
        
        ' Wacht tot file echt bestaat (max 500ms)
        Dim waitStart As Double: waitStart = Timer
        Do While Timer - waitStart < 0.5
            DoEvents
            If Dir(tmp) <> "" Then
                ' Controleer of file > 0 bytes (niet corrupt)
                If FileLen(tmp) > 100 Then
                    exportSuccess = True
                    Exit For
                End If
            End If
        Loop
        
        If exportSuccess Then Exit For
        On Error GoTo ErrHandler
    Next attempt

    If exportSuccess Then
        GetBackgroundPath = tmp
        Exit Function
    End If

    ' VERBETERD: Fallback met betere bounds checking
    Dim ch As ChartObject
    Dim safeLeft As Double, safeTop As Double, safeWidth As Double, safeHeight As Double
    
    safeLeft = IIf(shp.Left < 0, 0, shp.Left)
    safeTop = IIf(shp.Top < 0, 0, shp.Top)
    safeWidth = IIf(shp.Width < 10, 10, shp.Width)
    safeHeight = IIf(shp.Height < 10, 10, shp.Height)
    
    On Error Resume Next
    Set ch = wsRef.ChartObjects.Add(safeLeft, safeTop, safeWidth, safeHeight)
    If ch Is Nothing Then Exit Function
    
    shp.CopyPicture Appearance:=xlScreen, Format:=xlPicture
    ch.Chart.Paste
    ch.Chart.Export tmp, "JPG"
    ch.Delete
    On Error GoTo ErrHandler
    
    If Dir(tmp) <> "" And FileLen(tmp) > 100 Then
        GetBackgroundPath = tmp
    Else
        Utils.LogError4 "FormBackground", "GetBackgroundPath", 0, "Export gefaald na fallback"
    End If
    Exit Function

ErrHandler:
    Utils.LogError4 "FormBackground", "GetBackgroundPath", Err.Number, Err.Description
    On Error Resume Next
    If Not ch Is Nothing Then ch.Delete
End Function

Public Function GetNextForm(currentForm As Object) As String
    On Error GoTo ErrHandler
    If Not m_IsInitialized Then Initialize
    
    Dim currentSeq As FormSequence
    currentSeq = GetFormSequence(TypeName(currentForm))
    
    If currentSeq = 0 Then
        Utils.LogError4 "FormNavigator", "GetNextForm", 0, "Onbekend form: " & TypeName(currentForm)
        GetNextForm = ""
        Exit Function
    End If
    
    ' Volgende form (cyclisch)
    Dim nextSeq As FormSequence
    nextSeq = IIf(currentSeq = fsInvoer, fsHuishouden, currentSeq + 1)
    
    GetNextForm = GetFormName(nextSeq)
    Exit Function

    GetNextForm = ""

Private Function GetFormSequence(ByVal formName As String) As FormSequence
    Dim i As Long
    For i = LBound(m_Forms) To UBound(m_Forms)
        If StrComp(m_Forms(i).FormName, formName, vbTextCompare) = 0 Then
            GetFormSequence = m_Forms(i).SequenceIndex
            Exit Function
        End If
    Next i
    GetFormSequence = 0
End Function

Private Function GetFormName(ByVal seq As FormSequence) As String
    Dim i As Long
    For i = LBound(m_Forms) To UBound(m_Forms)
        If m_Forms(i).SequenceIndex = seq Then
            GetFormName = m_Forms(i).FormName
            Exit Function
        End If
    Next i
    GetFormName = ""
End Function

Public Const DYN_PREFIX As String = "dyn_"
Public Sub FrmLayoutSafeAreaDyn(frm As Object)
    On Error GoTo ErrHandler

    MonthlySheet.EnsureConfigSheet
    Dim BG_MARGIN As Double: BG_MARGIN = MonthlySheet.ConfigGet("BG_MARGIN", 6)
    Dim ROW_SPACING As Double: ROW_SPACING = MonthlySheet.ConfigGet("ROW_SPACING", 34)
    Dim LABEL_WIDTH As Double: LABEL_WIDTH = MonthlySheet.ConfigGet("LABEL_WIDTH", 120)
    Dim GAP_LABEL_TO_INPUT As Double: GAP_LABEL_TO_INPUT = MonthlySheet.ConfigGet("GAP_LABEL_TO_INPUT", 12)
    Dim RIGHT_PADDING As Double: RIGHT_PADDING = MonthlySheet.ConfigGet("RIGHT_PADDING", 20)
    Dim INP_MAX As Double: INP_MAX = MonthlySheet.ConfigGet("INP_MAX", 360)
    Dim SAFE_LEFT_PCT As Double: SAFE_LEFT_PCT = MonthlySheet.ConfigGet("SAFE_LEFT_PCT", 0.12)
    Dim SAFE_RIGHT_PCT As Double: SAFE_RIGHT_PCT = MonthlySheet.ConfigGet("SAFE_RIGHT_PCT", 0.88)
    Dim SAFE_TOP_PCT As Double: SAFE_TOP_PCT = MonthlySheet.ConfigGet("SAFE_TOP_PCT", 0.2)
    Dim SAFE_BOTTOM_PCT As Double: SAFE_BOTTOM_PCT = MonthlySheet.ConfigGet("SAFE_BOTTOM_PCT", 0.8)

    Dim bgLeft As Single, bgTop As Single, bgW As Single, bgH As Single
    Dim bg As MSForms.Image: Set bg = TryGetImageDyn(frm, FormLayout.DYN_PREFIX & "imgBackground")
    If bg Is Nothing Then
        bgLeft = BG_MARGIN: bgTop = BG_MARGIN
        bgW = frm.InsideWidth - 2 * BG_MARGIN
        bgH = frm.InsideHeight - 2 * BG_MARGIN
    Else
        bgLeft = bg.Left: bgTop = bg.Top: bgW = bg.Width: bgH = bg.Height
    End If
    If bgW < 0 Then bgW = 0
    If bgH < 0 Then bgH = 0

    Dim safeLeft As Single, safeTop As Single, safeWidth As Single, safeHeight As Single
    safeLeft = bgLeft + (SAFE_LEFT_PCT * bgW)
    safeTop = bgTop + (SAFE_TOP_PCT * bgH)
    safeWidth = (SAFE_RIGHT_PCT - SAFE_LEFT_PCT) * bgW
    safeHeight = (SAFE_BOTTOM_PCT - SAFE_TOP_PCT) * bgH
    If safeWidth < 0 Then safeWidth = 0
    If safeHeight < 0 Then safeHeight = 0

    Dim colLbl As Single, colInp As Single, inpWidth As Single, t As Single
    colLbl = safeLeft
    colInp = safeLeft + LABEL_WIDTH + GAP_LABEL_TO_INPUT
    inpWidth = safeWidth - (colInp - safeLeft) - RIGHT_PADDING
    If inpWidth < 0 Then inpWidth = 0
    If inpWidth > INP_MAX Then
        Dim extraSpace As Single
        extraSpace = (inpWidth - INP_MAX) / 2
        colInp = colInp + extraSpace
        inpWidth = INP_MAX
    End If

    ' Lay-out toewijzing met guards
    t = safeTop
    Dim ctrl As Control
    Set ctrl = TryGetControl(frm, FormLayout.DYN_PREFIX & "lblDatumVandaag"): If Not ctrl Is Nothing Then ctrl.Left = colLbl: ctrl.Top = t
    t = t + ROW_SPACING
    Set ctrl = TryGetControl(frm, FormLayout.DYN_PREFIX & "lblCat"): If Not ctrl Is Nothing Then ctrl.Left = colLbl: ctrl.Top = t
    Set ctrl = TryGetControl(frm, FormLayout.DYN_PREFIX & "cmbCategorie"): If Not ctrl Is Nothing Then ctrl.Left = colInp: ctrl.Top = t: ctrl.Width = inpWidth
    t = t + ROW_SPACING
    Set ctrl = TryGetControl(frm, FormLayout.DYN_PREFIX & "lblSub"): If Not ctrl Is Nothing Then ctrl.Left = colLbl: ctrl.Top = t
    Dim wSub As Single: wSub = Application.Max(0, inpWidth - 36)
    Set ctrl = TryGetControl(frm, FormLayout.DYN_PREFIX & "cmbSubcategorie"): If Not ctrl Is Nothing Then ctrl.Left = colInp: ctrl.Top = t: ctrl.Width = wSub
    Set ctrl = TryGetControl(frm, FormLayout.DYN_PREFIX & "btnNieuwSub"): If Not ctrl Is Nothing Then ctrl.Left = colInp + wSub + 8: ctrl.Top = t: ctrl.Width = 28: ctrl.Height = 22
    t = t + ROW_SPACING
    Set ctrl = TryGetControl(frm, FormLayout.DYN_PREFIX & "lblBedrag"): If Not ctrl Is Nothing Then ctrl.Left = colLbl: ctrl.Top = t
    Set ctrl = TryGetControl(frm, FormLayout.DYN_PREFIX & "txtBedrag"): If Not ctrl Is Nothing Then ctrl.Left = colInp: ctrl.Top = t: ctrl.Width = inpWidth
    t = t + ROW_SPACING
    Set ctrl = TryGetControl(frm, FormLayout.DYN_PREFIX & "lblBetaal"): If Not ctrl Is Nothing Then ctrl.Left = colLbl: ctrl.Top = t
    Set ctrl = TryGetControl(frm, FormLayout.DYN_PREFIX & "cmbBetaalwijze"): If Not ctrl Is Nothing Then ctrl.Left = colInp: ctrl.Top = t: ctrl.Width = inpWidth

    Dim btnOps As Control, btnAnn As Control
    Set btnOps = TryGetControl(frm, FormLayout.DYN_PREFIX & "imgBtnOpslaan")
    Set btnAnn = TryGetControl(frm, FormLayout.DYN_PREFIX & "imgBtnAnnuleren")
    If Not btnOps Is Nothing And Not btnAnn Is Nothing Then
        Dim btnTotalWidth As Single: btnTotalWidth = btnOps.Width + 24 + btnAnn.Width
        Dim btnLeft As Single: btnLeft = safeLeft + Application.Max(0, (safeWidth - btnTotalWidth)) / 2
        Dim btnY As Single: btnY = safeTop + Application.Max(0, safeHeight - 60)
        btnOps.Left = btnLeft: btnOps.Top = btnY
        btnAnn.Left = btnLeft + btnOps.Width + 24: btnAnn.Top = btnY
    End If

    Exit Sub
ErrHandler:
    Utils.LogError4 "FormLayout", "FrmLayoutSafeAreaDyn", Err.Number, Err.Description
End Sub

' === LayoutSpecsHelpers PATCH (robust kolom-lookup, veilige conversie, hover optioneel) ===
Option Explicit

Public Function GetLayoutSpec(ByVal keyName As String, ByVal fieldName As String) As Variant
  On Error GoTo ErrHandler
  Dim wsRef As Worksheet: Set wsRef = ThisWorkbook.Worksheets("Referenties")
  Dim lo As ListObject: Set lo = wsRef.ListObjects("tblLayoutSpecs")
  If lo Is Nothing Or lo.DataBodyRange Is Nothing Then Exit Function

  Dim hdr As Range: Set hdr = lo.HeaderRowRange
  Dim data As Range: Set data = lo.DataBodyRange
  Dim i As Long, colIndex As Long: colIndex = 0

  ' Zoek kolom case-insensitief, met Trim
  For i = 1 To hdr.Columns.Count
    If StrComp(Trim$(CStr(hdr.Cells(1, i).Value)), Trim$(fieldName), vbTextCompare) = 0 Then
      colIndex = i: Exit For
    End If
  Next i
  If colIndex = 0 Then
    Utils.LogError4 "LayoutSpecsHelpers", "GetLayoutSpec", 1001, "Kolom '" & fieldName & "' niet gevonden"
    Exit Function
  End If

  Dim r As Range
  For Each r In data.Rows
    If StrComp(Trim$(CStr(r.Cells(1, 1).Value)), Trim$(keyName), vbTextCompare) = 0 Then
      GetLayoutSpec = r.Cells(1, colIndex).Value
      Exit Function
    End If
  Next r
  ' Niet gevonden
  Utils.LogError4 "LayoutSpecsHelpers", "GetLayoutSpec", 1002, "Key '" & keyName & "' niet gevonden in specs"
  Exit Function

ErrHandler:
  Utils.LogError4 "LayoutSpecsHelpers", "GetLayoutSpec", Err.Number, Err.Description
End Function

Private Function NzDbl(v As Variant, ByVal def As Double) As Double
  If IsError(v) Then NzDbl = def: Exit Function
  If IsMissing(v) Then NzDbl = def: Exit Function
  If IsNull(v) Then NzDbl = def: Exit Function
  If VarType(v) = vbString And Trim$(CStr(v)) = "" Then NzDbl = def: Exit Function
  If Not IsNumeric(v) Then NzDbl = def Else NzDbl = CDbl(v)
End Function

Public Sub PlaceButtonFromSpec(frm As Object, ByVal controlName As String, ByVal keyName As String)
  On Error GoTo ErrHandler
  Dim baseShape As String
  baseShape = CStr(GetLayoutSpec(keyName, "ShapeName"))
  If Len(Trim$(baseShape)) = 0 Then
    Utils.LogError4 "LayoutSpecsHelpers", "PlaceButtonFromSpec", 1003, "Lege ShapeName voor key '" & keyName & "'"
    Exit Sub
  End If

  Dim X As Single, Y As Single, w As Single, h As Single
  X = CSng(NzDbl(GetLayoutSpec(keyName, "X"), 0))
  Y = CSng(NzDbl(GetLayoutSpec(keyName, "Y"), 0))
  w = CSng(NzDbl(GetLayoutSpec(keyName, "Width"), 80))   ' veilige default
  h = CSng(NzDbl(GetLayoutSpec(keyName, "Height"), 28))  ' veilige default
  If w < 0 Then w = 0
  If h < 0 Then h = 0

  Dim Img As MSForms.Image
  Set Img = FormLayout.TryGetImageDyn(frm, controlName)
  If Img Is Nothing Then Set Img = frm.Controls.Add("Forms.Image.1", controlName, True)

  Dim picPath As String: picPath = GetShapePath(baseShape)
  If Len(picPath) > 0 Then Img.Picture = LoadPicture(picPath)
  Img.Left = X: Img.Top = Y: Img.Width = w: Img.Height = h

  ' Hover alleen voor knoppen (achtergrond heeft geen hover)
  Dim hoverShape As String
  hoverShape = baseShape & "_hover"
  If InStr(1, LCase$(baseShape), "bg_", vbTextCompare) > 0 Then
    Img.Tag = "base=" & baseShape          ' geen hover
  Else
    Img.Tag = "base=" & baseShape & vbLf & "hover=" & hoverShape
  End If
  Exit Sub

ErrHandler:
  Utils.LogError4 "LayoutSpecsHelpers", "PlaceButtonFromSpec", Err.Number, Err.Description
End Sub

Private Function ParseTag(ByVal tagText As String, ByVal key As String) As String
  On Error GoTo ErrHandler
  Dim parts() As String: parts = Split(tagText, vbLf)
  Dim i As Long, kv()
  For i = LBound(parts) To UBound(parts)
    kv = Split(parts(i), "=")
    If UBound(kv) = 1 Then If StrComp(kv(0), key, vbTextCompare) = 0 Then ParseTag = kv(1): Exit Function
  Next i
  Exit Function
ErrHandler:
  Utils.LogError4 "LayoutSpecsHelpers", "ParseTag", Err.Number, Err.Description
End Function

Public Function GetShapePath(ByVal shapeName As String, Optional ByVal suffix As String = "") As String
  On Error GoTo ErrHandler
  Dim wsRef As Worksheet: Set wsRef = ThisWorkbook.Worksheets("Referenties")
  If wsRef Is Nothing Then Exit Function
  If wsRef.Visible = xlSheetVeryHidden Then wsRef.Visible = xlSheetVisible
  wsRef.Activate: DoEvents

  Dim shp As Shape: Set shp = wsRef.Shapes(shapeName)
  If shp Is Nothing Then
    Utils.LogError4 "LayoutSpecsHelpers", "GetShapePath", 1004, "Shape '" & shapeName & "' niet gevonden"
    Exit Function
  End If

  Dim w As Single, h As Single
  w = Application.Max(8, shp.Width)
  h = Application.Max(8, shp.Height)

  Dim tmp As String: tmp = Utils.GetTempFile(shapeName & suffix, ".jpg")
  Dim ch As ChartObject: Set ch = wsRef.ChartObjects.Add(Left:=shp.Left, Top:=shp.Top, Width:=w, Height:=h)

  Dim i As Long, pasted As Boolean
  For i = 1 To 3
    On Error Resume Next
    shp.CopyPicture Appearance:=xlScreen, Format:=xlPicture
    ch.Activate
    ch.Chart.Paste
    On Error GoTo ErrHandler
    If ch.Chart.Shapes.Count > 0 Then pasted = True: Exit For
    DoEvents: Application.CutCopyMode = False
  Next i

  If pasted Then ch.Chart.Export tmp, "JPG"
  If Dir$(tmp) <> "" Then GetShapePath = tmp

CleanUp:
  On Error Resume Next
  ch.Delete: Application.CutCopyMode = False
  On Error GoTo 0
  Exit Function

ErrHandler:
  Utils.LogError4 "LayoutSpecsHelpers", "GetShapePath", Err.Number, Err.Description
  Resume CleanUp
End Function

' ---- 1) Specs: verplichte kolommen en rijen gevuld ----
Private Function SpecsColumnsAndRowsCheck() As String
    On Error GoTo ErrHandler
    Dim wsRef As Worksheet: Set wsRef = ThisWorkbook.Worksheets("Referenties")
    Dim lo As ListObject: Set lo = wsRef.ListObjects("tblLayoutSpecs")
    If lo Is Nothing Then
        SpecsColumnsAndRowsCheck = "FOUT: Tabel 'tblLayoutSpecs' niet gevonden."
        Exit Function
    End If

    Dim requiredCols As Variant
    requiredCols = Array("Key", "ShapeName", "Width", "Height", "X", "Y")
    Dim i As Long, missing As String
    For i = LBound(requiredCols) To UBound(requiredCols)
        If FindColumnIndex(lo, CStr(requiredCols(i))) = 0 Then
            missing = missing & "- Kolom ontbreekt: " & requiredCols(i) & vbCrLf
        End If
    Next i

    If Len(missing) > 0 Then
        SpecsColumnsAndRowsCheck = "FOUT (Specs):" & vbCrLf & missing
        Exit Function
    End If

    If lo.DataBodyRange Is Nothing Then
        SpecsColumnsAndRowsCheck = "LET OP: 'tblLayoutSpecs' heeft nog geen rijen."
        Exit Function
    End If

    ' Keys die we minimaal verwachten
    Dim expectedKeys As Variant
    expectedKeys = Array("Prev", "Next", "Save", "Cancel", "RowAdd", "RowDel")

    Dim colKey As Long, colShape As Long, colW As Long, colH As Long, colX As Long, colY As Long
    colKey = FindColumnIndex(lo, "Key")
    colShape = FindColumnIndex(lo, "ShapeName")
    colW = FindColumnIndex(lo, "Width")
    colH = FindColumnIndex(lo, "Height")
    colX = FindColumnIndex(lo, "X")
    colY = FindColumnIndex(lo, "Y")

    Dim r As Range, k As Variant, found As Boolean, line As String
    For Each k In expectedKeys
        found = False
        For Each r In lo.DataBodyRange.Rows
            If Trim$(CStr(r.Cells(1, colKey).Value)) = CStr(k) Then
                found = True
                ' Check leegtes
                If Len(Trim$(CStr(r.Cells(1, colShape).Value))) = 0 _
                   Or Len(Trim$(CStr(r.Cells(1, colW).Value))) = 0 _
                   Or Len(Trim$(CStr(r.Cells(1, colH).Value))) = 0 _
                   Or Len(Trim$(CStr(r.Cells(1, colX).Value))) = 0 _
                   Or Len(Trim$(CStr(r.Cells(1, colY).Value))) = 0 Then
                    line = line & "- Key '" & k & "': heeft lege velden (ShapeName/Width/Height/X/Y)." & vbCrLf
                End If
                Exit For
            End If
        Next r
        If Not found Then line = line & "- Key ontbreekt in specs: " & k & vbCrLf
    Next k

    If Len(line) > 0 Then
        SpecsColumnsAndRowsCheck = "LET OP (Specs):" & vbCrLf & line
    Else
        SpecsColumnsAndRowsCheck = "OK  (Specs): Kolommen & rijen zijn compleet."
    End If
    Exit Function

ErrHandler:
    SpecsColumnsAndRowsCheck = "FOUT in SpecsColumnsAndRowsCheck: [" & Err.Number & "] " & Err.Description
End Function

' ---- 2) Shapes bestaan? (base + hover) ----
Private Function ShapesExistenceCheck() As String
    On Error GoTo ErrHandler
    Dim wsRef As Worksheet: Set wsRef = ThisWorkbook.Worksheets("Referenties")
    Dim lo As ListObject: Set lo = wsRef.ListObjects("tblLayoutSpecs")
    If lo Is Nothing Or lo.DataBodyRange Is Nothing Then
        ShapesExistenceCheck = "LET OP: 'tblLayoutSpecs' ontbreekt of heeft geen rijen; skip shapes-check."
        Exit Function
    End If

    Dim colShape As Long, colKey As Long
    colShape = FindColumnIndex(lo, "ShapeName")
    colKey = FindColumnIndex(lo, "Key")

    Dim r As Range, shapeName As String, keyName As String, report As String
    For Each r In lo.DataBodyRange.Rows
        keyName = Trim$(CStr(r.Cells(1, colKey).Value))
        shapeName = Trim$(CStr(r.Cells(1, colShape).Value))
        If Len(shapeName) > 0 Then
            If Not ShapeExistsOn(wsRef, shapeName) Then
                report = report & "- Key '" & keyName & "': base-shape ontbreekt: " & shapeName & vbCrLf
            Else
                ' Hover only for buttons (geen hover voor Background)
                If LCase$(keyName) <> "background" Then
                    Dim hoverName As String: hoverName = shapeName & "_hover"
                    If Not ShapeExistsOn(wsRef, hoverName) Then
                        report = report & "LET OP: Key '" & keyName & "': hover-shape ontbreekt: " & hoverName & vbCrLf
                    End If
                End If
            End If
        Else
            report = report & "- Key '" & keyName & "': ShapeName is leeg." & vbCrLf
        End If
    Next r

    If Len(report) = 0 Then
        ShapesExistenceCheck = "OK  (Shapes): base/hover bestaan (hover niet vereist voor Background)."
    Else
        ShapesExistenceCheck = "LET OP (Shapes):" & vbCrLf & report
    End If
    Exit Function

ErrHandler:
    ShapesExistenceCheck = "FOUT in ShapesExistenceCheck: [" & Err.Number & "] " & Err.Description
End Function

' ---- 3) Form controls: Tag-validatie & shapekloppen ----
Private Function FormImageTagCheck() As String
    On Error GoTo ErrHandler
    Dim frm As Object
    Set frm = GetActiveFormSafe()

    If frm Is Nothing Then
        FormImageTagCheck = "LET OP (Form): geen actief form gevonden; open een UserForm en run opnieuw."
        Exit Function
    End If

    Dim c As Control, issues As String
    For Each c In frm.Controls
        If TypeOf c Is MSForms.Image Then
            Dim tagText As String: tagText = CStr(c.Tag)
            Dim baseName As String: baseName = ParseTagLocal(tagText, "base")
            Dim hoverName As String: hoverName = ParseTagLocal(tagText, "hover")

            If Len(baseName) = 0 Then
                issues = issues & "- Image '" & c.name & "': Tag mist 'base='." & vbCrLf
            Else
                ' base-shape moet bestaan
                If Not ShapeExistsOn(ThisWorkbook.Worksheets("Referenties"), baseName) Then
                    issues = issues & "- Image '" & c.name & "': base-shape niet gevonden: " & baseName & vbCrLf
                End If
            End If

            ' Hover mag ontbreken op Background; voor buttons verwachten we hem wél
            If InStr(1, LCase$(c.name), "imgbtn", vbTextCompare) > 0 Or _
               InStr(1, LCase$(c.name), "imgaddrow", vbTextCompare) > 0 Or _
               InStr(1, LCase$(c.name), "imgdelrow", vbTextCompare) > 0 Then
                If Len(hoverName) = 0 Then
                    issues = issues & "LET OP: Image '" & c.name & "': Tag mist 'hover='." & vbCrLf
                ElseIf Not ShapeExistsOn(ThisWorkbook.Worksheets("Referenties"), hoverName) Then
                    issues = issues & "LET OP: Image '" & c.name & "': hover-shape niet gevonden: " & hoverName & vbCrLf
                End If
            End If
        End If
    Next c

    If Len(issues) = 0 Then
        FormImageTagCheck = "OK  (Form): alle Image-tags valide (base/hover) en shapes gevonden."
    Else
        FormImageTagCheck = "LET OP (Form):" & vbCrLf & issues
    End If
    Exit Function

ErrHandler:
    FormImageTagCheck = "FOUT in FormImageTagCheck: [" & Err.Number & "] " & Err.Description
End Function

' ---- 4) Achtergrond export check ----

ErrHandler:

End Function

' Trimsamere ParseTag (lokale kopie; laat jouw Private ParseTag in LayoutSpecsHelpers intact)
Private Function ParseTagLocal(ByVal tagText As String, ByVal key As String) As String
    On Error GoTo ErrHandler
    If Len(tagText) = 0 Then Exit Function
    Dim parts() As String: parts = Split(tagText, vbLf)
    Dim i As Long, kv()
    For i = LBound(parts) To UBound(parts)
        kv = Split(parts(i), "=")
        If UBound(kv) = 1 Then If StrComp(Trim$(kv(0)), key, vbTextCompare) = 0 Then ParseTagLocal = Trim$(kv(1)): Exit Function
    Next i
    Exit Function
ErrHandler:
    ParseTagLocal = ""
End Function

    Else
        Set activeForm = Application.ActiveWindow
    End If
    Set GetActiveFormSafe = activeForm
    On Error GoTo 0
End Function

' ---- Helpers: per check produceer tCheckReport ----

Private Function CheckGenericButtonShapes() As tCheckReport
    On Error GoTo ErrHandler
    Dim rc As tCheckReport
    Dim baseNames As Variant
    baseNames = Array("btnShapeSmall", "btnShapeMedium", "btnShapeLarge")
    
    Dim wsRef As Worksheet: Set wsRef = ThisWorkbook.Worksheets("Referenties")
    Dim i As Long, nm As String, nmHover As String
    
    For i = LBound(baseNames) To UBound(baseNames)
        nm = CStr(baseNames(i))
        nmHover = nm & "_hover"
        
        If ShapeExistsOn(wsRef, nm) Then
            rc.OkCount = rc.OkCount + 1
            rc.Details = rc.Details & "OK  : Shape '" & nm & "'" & vbCrLf
        Else
            rc.ErrCount = rc.ErrCount + 1
            rc.Details = rc.Details & "FOUT: Shape '" & nm & "' ontbreekt op Referenties" & vbCrLf
        End If
        
        If ShapeExistsOn(wsRef, nmHover) Then
            rc.OkCount = rc.OkCount + 1
            rc.Details = rc.Details & "OK  : Shape '" & nmHover & "'" & vbCrLf
        Else
            rc.WarnCount = rc.WarnCount + 1
            rc.Details = rc.Details & "LET OP: Hover-shape '" & nmHover & "' ontbreekt (hover werkt dan niet)" & vbCrLf
        End If
    Next i
    
    CheckGenericButtonShapes = rc
    Exit Function
    
ErrHandler:
    CheckGenericButtonShapes.ErrCount = CheckGenericButtonShapes.ErrCount + 1
    CheckGenericButtonShapes.Details = CheckGenericButtonShapes.Details & "FOUT in CheckGenericButtonShapes: [" & Err.Number & "] " & Err.Description & vbCrLf
End Function

Private Function CheckLayoutSpecsKeys() As tCheckReport
    On Error GoTo ErrHandler
    Dim rc As tCheckReport
    
    Dim wsRef As Worksheet: Set wsRef = ThisWorkbook.Worksheets("Referenties")
    Dim lo As ListObject
    On Error Resume Next
    Set lo = wsRef.ListObjects("tblLayoutSpecs")
    On Error GoTo ErrHandler
    
    If lo Is Nothing Then
        rc.WarnCount = rc.WarnCount + 1
        rc.Details = rc.Details & "LET OP: Tabel 'tblLayoutSpecs' niet gevonden op Referenties" & vbCrLf
        CheckLayoutSpecsKeys = rc
        Exit Function
    End If
    
    If lo.DataBodyRange Is Nothing Then
        rc.WarnCount = rc.WarnCount + 1
        rc.Details = rc.Details & "LET OP: 'tblLayoutSpecs' heeft geen rijen" & vbCrLf
        CheckLayoutSpecsKeys = rc
        Exit Function
    End If
    
    ' Zoek kolommen Key en ShapeName
    Dim colKey As Long, colShape As Long
    colKey = FindColumnIndex(lo, "Key")
    colShape = FindColumnIndex(lo, "ShapeName")
    
    If colKey = 0 Or colShape = 0 Then
        rc.ErrCount = rc.ErrCount + 1
        rc.Details = rc.Details & "FOUT: Kolom 'Key' of 'ShapeName' ontbreekt in tblLayoutSpecs" & vbCrLf
        CheckLayoutSpecsKeys = rc
        Exit Function
    End If
    
    ' Loop door rijen; check dat ShapeName bestaat als shape
    Dim r As Range, keyName As String, shapeName As String
    For Each r In lo.DataBodyRange.Rows
        keyName = CStr(Trim$(r.Cells(1, colKey).Value))
        shapeName = CStr(Trim$(r.Cells(1, colShape).Value))
        
        If Len(keyName) = 0 Then
            rc.WarnCount = rc.WarnCount + 1
            rc.Details = rc.Details & "LET OP: Lege Key in tblLayoutSpecs (rij " & r.Row & ")" & vbCrLf
        End If
        
        If Len(shapeName) = 0 Then
            rc.ErrCount = rc.ErrCount + 1
            rc.Details = rc.Details & "FOUT: Key '" & keyName & "' heeft lege ShapeName" & vbCrLf
        ElseIf Not ShapeExistsOn(wsRef, shapeName) Then
            rc.ErrCount = rc.ErrCount + 1
            rc.Details = rc.Details & "FOUT: Key '" & keyName & "' verwijst naar ontbrekende shape '" & shapeName & "'" & vbCrLf
        Else
            rc.OkCount = rc.OkCount + 1
            rc.Details = rc.Details & "OK  : Key '" & keyName & "' ? Shape '" & shapeName & "'" & vbCrLf
        End If

Public Sub NormalizeSpecsHeaders()
  On Error GoTo ErrHandler

  ' 1) Sheet 'Referenties' ophalen of melden
  Dim ws As Worksheet
  Set ws = GetWorksheetSafe("Referenties")
  If ws Is Nothing Then
    MsgBox "FOUT: Werkblad 'Referenties' niet gevonden.", vbCritical
    Exit Sub
  End If

  ' 2) Probeer bestaande tabel op te halen
  Dim lo As ListObject
  Set lo = TryGetListObjectSafe(ws, "tblLayoutSpecs")

  ' 3) Als tabel niet bestaat: laat gebruiker header-rij selecteren en maak de tabel
  If lo Is Nothing Then
    Dim hdr As Range
    MsgBox "Selecteer nu de HEADER-RIJ van tblLayoutSpecs (exact één rij met alle kolomnamen) en klik OK in de volgende prompt.", vbInformation
    On Error Resume Next
    Set hdr = Application.InputBox(Prompt:="Selecteer de header-rij (bijv. K1:S1)", Type:=8) ' Type 8 = Range
    On Error GoTo ErrHandler

    If hdr Is Nothing Then
      MsgBox "Actie geannuleerd. Geen header-rij geselecteerd.", vbExclamation
      Exit Sub
    End If
    If hdr.Rows.Count <> 1 Then
      MsgBox "Selecteer exact één rij met de kopteksten.", vbExclamation
      Exit Sub
    End If

    ' Maak ListObject vanaf de geselecteerde header-rij + CurrentRegion
    Set lo = ws.ListObjects.Add(xlSrcRange, hdr.CurrentRegion, , xlYes)
    lo.name = "tblLayoutSpecs"
  End If

  ' 4) Zet de exacte headers (9 kolommen, volgorde vast)
  Dim exactHeaders As Variant
  exactHeaders = Array("Key", "ShapeName", "Width", "Height", "X", "Y", "DefaultColor", "HoverColor", "Notes")

  Dim i As Long
  ' Controleren of aantal kolommen voldoende is, anders breidt de header-rij uit
  If lo.HeaderRowRange.Columns.Count < UBound(exactHeaders) - LBound(exactHeaders) + 1 Then
    ' Vergroot de header-range door naastliggende lege cellen op te nemen
    Dim needCols As Long
    needCols = (UBound(exactHeaders) - LBound(exactHeaders) + 1) - lo.HeaderRowRange.Columns.Count
    lo.HeaderRowRange.Resize(1, lo.HeaderRowRange.Columns.Count + needCols).Cells(1, 1).Select
    ' (Excel past de ListObject range zelf meestal aan bij CurrentRegion)
  End If

  For i = LBound(exactHeaders) To UBound(exactHeaders)
    lo.HeaderRowRange.Cells(1, i + 1).Value = exactHeaders(i)
  Next i

  ' 5) Trim spaties in DataBodyRange (als er rijen zijn)
  If Not lo.DataBodyRange Is Nothing Then
    Dim c As Range
    For Each c In lo.DataBodyRange
      If VarType(c.Value) = vbString Then c.Value = Trim$(CStr(c.Value))
    Next c
  End If

  ' 6) Feedback
  MsgBox "tblLayoutSpecs genormaliseerd:" & vbCrLf & _
         " - Sheet: " & ws.name & vbCrLf & _
         " - Tabelnaam: " & lo.name & vbCrLf & _
         " - Headers gezet (Key, ShapeName, Width, Height, X, Y, DefaultColor, HoverColor, Notes)", vbInformation
  Exit Sub

ErrHandler:
  MsgBox "FOUT in NormalizeSpecsHeaders: [" & Err.Number & "] " & Err.Description, vbCritical
End Sub

Public Sub BerekenWeekTotaal()
 On Error GoTo ErrHandler
 Dim ws As Worksheet: Set ws = ThisWorkbook.Worksheets("Database")
 Utils.EnsureDBHeaders
 Dim lr As Long: lr = Utils.LastRow(ws, DBCols.dbDate)
 If lr < 2 Then Exit Sub
 Dim dPrev As Date: dPrev = DateAdd("d", -7, Date)
 Dim tgtWeek As Long: tgtWeek = Utils.IsoWeek(dPrev)
 Dim tgtYear As Long: tgtYear = Year(dPrev)
 Dim i As Long, total As Double
 Dim catDict As Object: Set catDict = CreateObject("Scripting.Dictionary")
 For i = 2 To lr
  If CLng(Val(ws.Cells(i, DBCols.dbYear).Value)) = tgtYear And _
     CLng(Val(ws.Cells(i, DBCols.dbWeek).Value)) = tgtWeek And _
     CStr(ws.Cells(i, DBCols.dbType).Value) = "Uitgave" Then
     Dim amount As Double: amount = CDbl(Val(ws.Cells(i, DBCols.dbAmount).Value))
     total = total + amount
     Dim cat As String: cat = CStr(ws.Cells(i, DBCols.dbCategory).Value)
     If Len(Trim$(cat)) > 0 Then Utils.DictAddOrIncrement catDict, cat, 1
  End If
 Next i
 Dim mostCat As String, maxCount As Long, k As Variant
 For Each k In catDict.keys
  If catDict(k) > maxCount Then maxCount = catDict(k): mostCat = CStr(k)
 Next k
 If ws.Cells(1, SummaryCols.colWeekSummaryStart).Value = "" Then
  ws.Cells(1, SummaryCols.colWeekSummaryStart).Resize(1, 5).Value = Array("Jaar" & vbLf & "Week", "Jaar", "Weeknr", "Totaal", "Meest gebruikte categorie")
 End If
 Dim outRow As Long: outRow = Utils.LastRow(ws, SummaryCols.colWeekSummaryStart) + 1
 ws.Cells(outRow, SummaryCols.colWeekSummaryStart).Value = CStr(tgtYear) & vbLf & CStr(tgtWeek)
 ws.Cells(outRow, SummaryCols.colWeekSummaryStart + 1).Value = tgtYear
 ws.Cells(outRow, SummaryCols.colWeekSummaryStart + 2).Value = tgtWeek
 ws.Cells(outRow, SummaryCols.colWeekSummaryStart + 3).Value = total
 ws.Cells(outRow, SummaryCols.colWeekSummaryStart + 4).Value = mostCat
 Exit Sub
ErrHandler:
 Utils.LogError4 "Reporting", "BerekenWeekTotaal", Err.Number, Err.Description
End Sub
Public Sub BerekenMaandTotaal()
 On Error GoTo ErrHandler
 Dim ws As Worksheet: Set ws = ThisWorkbook.Worksheets("Database")
 Utils.EnsureDBHeaders
 Dim lr As Long: lr = Utils.LastRow(ws, DBCols.dbDate)
 If lr < 2 Then Exit Sub
 Dim prevMonthKey As String: prevMonthKey = Format(DateAdd("m", -1, Date), "YYYY-MM")
 Dim i As Long
 Dim totaal As Double, dagenCount As Long
 Dim weekDict As Object: Set weekDict = CreateObject("Scripting.Dictionary")
 Dim catDict As Object: Set catDict = CreateObject("Scripting.Dictionary")
 Dim catCountDict As Object: Set catCountDict = CreateObject("Scripting.Dictionary")
 Dim duursteDag As Double, duursteWeek As Double
 Dim maxCatValue As Double, minCatValue As Double
 Dim maxCatName As String, minCatName As String
 maxCatValue = 0: minCatValue = 1E+20
 For i = 2 To lr
  If CStr(ws.Cells(i, DBCols.dbMonthKey).Value) = prevMonthKey And CStr(ws.Cells(i, DBCols.dbType).Value) = "Uitgave" Then
   Dim bedrag As Double: bedrag = CDbl(Val(ws.Cells(i, DBCols.dbAmount).Value))
   totaal = totaal + bedrag
   dagenCount = dagenCount + 1
   If bedrag > duursteDag Then duursteDag = bedrag
   Dim weekKey As String: weekKey = CStr(ws.Cells(i, DBCols.dbYear).Value) & "-W" & CStr(ws.Cells(i, DBCols.dbWeek).Value)
   Utils.DictAddOrIncrement weekDict, weekKey, bedrag
   Dim categorie As String: categorie = CStr(ws.Cells(i, DBCols.dbCategory).Value)
   Utils.DictAddOrIncrement catDict, categorie, bedrag
   Utils.DictAddOrIncrement catCountDict, categorie, 1
  End If
 Next i
 Dim key As Variant
 For Each key In weekDict.keys: If weekDict(key) > duursteWeek Then duursteWeek = weekDict(key)
 Next key
 For Each key In catDict.keys
  If catDict(key) > maxCatValue Then maxCatValue = catDict(key): maxCatName = CStr(key)
  If catDict(key) < minCatValue Then minCatValue = catDict(key): minCatName = CStr(key)
 Next key
 Dim overzichtRow As Long: overzichtRow = Utils.LastRow(ws, SummaryCols.colMonthSummaryStart) + 1
 If ws.Cells(1, SummaryCols.colMonthSummaryStart).Value = "" Then
  ws.Cells(1, SummaryCols.colMonthSummaryStart).Resize(1, 7).Value = Array("mnd", "ttl", "gpd", "gpw", "dudg", "duwk", "m_u_a")
 End If
 ws.Cells(overzichtRow, SummaryCols.colMonthSummaryStart).Value = prevMonthKey
 ws.Cells(overzichtRow, SummaryCols.colMonthSummaryStart + 1).Value = totaal
 ws.Cells(overzichtRow, SummaryCols.colMonthSummaryStart + 2).Value = IIf(dagenCount > 0, totaal / dagenCount, 0)
 ws.Cells(overzichtRow, SummaryCols.colMonthSummaryStart + 3).Value = IIf(weekDict.Count > 0, totaal / weekDict.Count, 0)
 ws.Cells(overzichtRow, SummaryCols.colMonthSummaryStart + 4).Value = duursteDag
 ws.Cells(overzichtRow, SummaryCols.colMonthSummaryStart + 5).Value = duursteWeek
 ws.Cells(overzichtRow, SummaryCols.colMonthSummaryStart + 6).Value = maxCatName
 ' Categorie-overzicht append met maandkolom
 Dim catStartCol As Long: catStartCol = SummaryCols.colMonthSummaryStart + 8 ' X
 If ws.Cells(1, catStartCol - 1).Value = "" Then
  ws.Cells(1, catStartCol - 1).Resize(1, 4).Value = Array("Maand", "Categorie", "Aantal", "Totaal")
 End If
 Dim r As Long: r = Utils.LastRow(ws, catStartCol - 1) + 1
 For Each key In catDict.keys
  ws.Cells(r, catStartCol - 1).Value = prevMonthKey
  ws.Cells(r, catStartCol).Value = key
  ws.Cells(r, catStartCol + 1).Value = catCountDict(key)
  ws.Cells(r, catStartCol + 2).Value = catDict(key)
  r = r + 1
 Next key
 Exit Sub
ErrHandler:
 Utils.LogError4 "Reporting", "BerekenMaandTotaal", Err.Number, Err.Description
End Sub

Public Sub LogError(ByVal source As String, ByVal errDesc As String)

 Call LogError4(source, "", Err.Number, errDesc)
End Sub
Public Function SheetBestaat(ByVal sheetNaam As String) As Boolean
 On Error Resume Next
 Dim ws As Worksheet: Set ws = ThisWorkbook.Worksheets(sheetNaam)
 SheetBestaat = Not ws Is Nothing
 On Error GoTo 0
End Function
Public Function LastRow(ByVal ws As Worksheet, Optional ByVal col As Long = 1) As Long
 With ws
  If Application.WorksheetFunction.CountA(.Columns(col)) = 0 Then
    LastRow = 1
  Else
    LastRow = .Cells(.Rows.Count, col).End(xlUp).Row
  End If
 End With
End Function
Public Function IsoWeek(ByVal d As Date) As Long
 IsoWeek = DatePart("ww", d, vbMonday, vbFirstFourDays)
End Function
Public Function ParseAmount(ByVal s As String) As Double
 Dim decSep As String: decSep = Application.International(xlDecimalSeparator)
 s = Trim$(s)
 If s = "" Then ParseAmount = 0: Exit Function
 s = Replace(Replace(s, ".", decSep), ",", decSep)
 ParseAmount = CDbl(s)
End Function
Public Sub EnsureDBHeaders()
 Dim ws As Worksheet: Set ws = ThisWorkbook.Worksheets("Database")
 If ws Is Nothing Then
   Set ws = ThisWorkbook.Worksheets.Add
   ws.name = "Database"
 End If
 If ws.Cells(1, DBCols.dbDate).Value = "" Then
   ws.Cells(1, DBCols.dbDate).Resize(1, 11).Value = Array( _
    "Datum", "Jaar", "MaandKey", "WeekISO", "DagNaam", "Categorie", "Subcategorie", "Omschrijving", "Bedrag", "Betaalwijze", "Soort")
 End If
End Sub
Public Function DictAddOrIncrement(ByRef dict As Object, ByVal key As Variant, Optional ByVal inc As Double = 1) As Double
 If Not dict.exists(key) Then dict(key) = inc Else dict(key) = dict(key) + inc
 DictAddOrIncrement = dict(key)
End Function
Public Function ToMonthlyAmount(ByVal periode As String, ByVal bedrag As Double) As Double
 periode = LCase$(Trim$(periode))
 Select Case periode
  Case "jaar", "per jaar": ToMonthlyAmount = bedrag / 12
  Case "kwartaal", "per kwartaal": ToMonthlyAmount = bedrag / 3
  Case "maand", "per maand": ToMonthlyAmount = bedrag
  Case "4 weken": ToMonthlyAmount = bedrag * 13 / 12
  Case "week", "per week": ToMonthlyAmount = bedrag * 52 / 12
  Case "dag", "per dag": ToMonthlyAmount = bedrag * 365 / 12
  Case Else: ToMonthlyAmount = bedrag
 End Select
End Function
Public Function TryGetListObject(ByVal ws As Worksheet, ByVal tableName As String) As ListObject

End Function
Public Function TryGetNamedRange(ByVal nm As String) As Range
 On Error Resume Next
 Set TryGetNamedRange = ThisWorkbook.names(nm).RefersToRange
 On Error GoTo 0
End Function
Public Function EnsureTable(ByVal ws As Worksheet, ByVal tableName As String, ByVal headerRange As Range) As ListObject
 Dim lo As ListObject
 Set lo = TryGetListObject(ws, tableName)
 If lo Is Nothing Then
  Set lo = ws.ListObjects.Add(xlSrcRange, headerRange.CurrentRegion, , xlYes)
  lo.name = tableName
 End If
 Set EnsureTable = lo
End Function
Public Function MonthNameNL(ByVal m As Long) As String
 Dim arr
 arr = Array("Januari", "Februari", "Maart", "April", "Mei", "Juni", "Juli", "Augustus", "September", "Oktober", "November", "December")
 If m >= 1 And m <= 12 Then MonthNameNL = arr(m - 1) Else MonthNameNL = "Onbekend"
End Function
Public Function GetTempFile(ByVal baseName As String, ByVal ext As String) As String
 On Error GoTo ErrHandler
 Dim tmp As String
 tmp = Environ$("TEMP")
 If Len(tmp) = 0 Then tmp = Environ$("TMP")
 If Len(tmp) = 0 Then tmp = ThisWorkbook.Path
 If Right$(tmp, 1) <> "" Then tmp = tmp & ""
 GetTempFile = tmp & baseName & "_" & Format(Now, "yyyymmdd_hhnnss") & ext
 Exit Function
ErrHandler:
 GetTempFile = ThisWorkbook.Path & "" & baseName & "_" & Format(Now, "yyyymmdd_hhnnss") & ext
 LogError4 "Utils", "GetTempFile", Err.Number, Err.Description
End Function

'=========================
' CLASS: CTxtEvents
'=========================
' Class module name: CTxtEvents
Option Explicit
Public WithEvents Txt As MSForms.TextBox
Private Sub Txt_KeyPress(ByVal KeyAscii As MSForms.ReturnInteger)
 Dim dec As String: dec = Application.International(xlDecimalSeparator)
 Select Case KeyAscii
  Case 48 To 57 ' 0–9
  Case Asc(dec) ' één decimaal
   If InStr(Txt.Text, dec) > 0 Then KeyAscii = 0
  Case 8 ' Backspace
  Case Else: KeyAscii = 0
 End Select
End Sub
Private Sub Txt_Exit(ByVal Cancel As MSForms.ReturnBoolean)
 On Error Resume Next
 If Len(Txt.Text) = 0 Then Exit Sub
 Txt.Text = Format(CDbl(Txt.Text), "0.00")
End Sub
