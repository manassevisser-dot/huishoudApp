
Sub ExporteerDataNaarCSV()
    Dim ws As Worksheet
    Dim csvFile As String
    Dim lastRow As Long
    Dim i As Long
    Dim line As String
    Dim fso As Object, ts As Object

    ' Pad voor CSV-bestand
    csvFile = ThisWorkbook.Path & "\export_data.csv"

    ' Gebruik Database-sheet
    Set ws = ThisWorkbook.Sheets("Database")
    lastRow = ws.Cells(ws.Rows.Count, "A").End(xlUp).Row

    ' Maak CSV-bestand
    Set fso = CreateObject("Scripting.FileSystemObject")
    Set ts = fso.CreateTextFile(csvFile, True)

    ' Schrijf header
    ts.WriteLine "Datum,Maand,Wknr,Dag,Categorie,Bedrag"

    ' Schrijf data
    For i = 2 To lastRow
        line = ws.Cells(i, 1).Value & "," & ws.Cells(i, 2).Value & "," & ws.Cells(i, 3).Value & "," & ws.Cells(i, 4).Value & "," & ws.Cells(i, 5).Value & "," & ws.Cells(i, 6).Value
        ts.WriteLine line
    Next i

    ts.Close
    MsgBox "Data geëxporteerd naar " & csvFile, vbInformation
End Sub
