const FormContext = createContext();

const formReducer = (state, action) => {
  switch (action.type) {
    case 'SET_PAGE_DATA':
      return {
        ...state,
        [action.pageId]: { ...state[action.pageId], ...action.data }
      };
    case 'LOAD_SAVED_STATE':
      return { ...state, ...action.data };
    case 'SET_USER_ID':
      return { ...state, userId: action.userId };
    case 'RESET_STATE':
      return { userId: null };
    default:
      return state;
  }
};

const FormProvider = ({ children }) => {
  const [state, dispatch] = useReducer(formReducer, {});
  
  return (
    <FormContext.Provider value={{ state, dispatch }}>
      {children}
    </FormContext.Provider>
  );
};

const useFormContext = () => {
  const context = useContext(FormContext);
  if (!context) throw new Error('useFormContext must be used within FormProvider');
  return context;
};

const Storage = {
  store: {},
  
  async getItem(key) {
    return this.store[key] || null;
  },
  
  async setItem(key, value) {
    this.store[key] = value;
  },
  
  async removeItem(key) {
    delete this.store[key];
  },
  
  async clear() {
    this.store = {};
  }
};

const savePageData = async (pageId, data) => {
  try {
    await Storage.setItem(`setupData_${pageId}`, JSON.stringify(data));
    console.log(`✓ Saved ${pageId} data:`, data);
  } catch (error) {
    console.error(`✗ Error saving ${pageId}:`, error);
  }
};

const loadPageData = async (pageId) => {
  try {
    const data = await Storage.getItem(`setupData_${pageId}`);
    return data ? JSON.parse(data) : null;
  } catch (error) {
    console.error(`✗ Error loading ${pageId}:`, error);
    return null;
  }
};

const saveUserId = async (userId) => {
  await Storage.setItem('userId', userId);
  console.log('✓ User ID saved:', userId);
};

  return await Storage.getItem('userId');
};

// ============================================================================

const validation = {
  required: (value, fieldLabel) => {
    if (value === null || value === undefined || value === '') {
      return { valid: false, error: `${fieldLabel} is verplicht` };
    }
    return { valid: true, error: null };
  },
  
  min: (value, min, fieldLabel) => {
    if (value < min) {
      return { valid: false, error: `${fieldLabel} moet minimaal ${min} zijn` };
    }
    return { valid: true, error: null };
  },
  
  max: (value, max, fieldLabel) => {
    if (value > max) {
      return { valid: false, error: `${fieldLabel} mag maximaal ${max} zijn` };
    }
    return { valid: true, error: null };
  },
  
  postcode: (value, fieldLabel) => {
    if (!/^\d{4}$/.test(value)) {
      return { valid: false, error: `${fieldLabel} moet 4 cijfers zijn` };
    }
    return { valid: true, error: null };
  },
  
  numeric: (value, fieldLabel) => {
    if (isNaN(value)) {
      return { valid: false, error: `${fieldLabel} moet een getal zijn` };
    }
    return { valid: true, error: null };
  }
};

const validateField = (field, value) => {
  // Required check
  if (field.required) {
    const result = validation.required(value, field.label);
    if (!result.valid) return result;
  }
  
  // Skip other validations if empty and not required
  if (value === null || value === undefined || value === '') {
    return { valid: true, error: null };
  }
  
  // Type-specific validation
  if (field.type === 'numeric') {
    const result = validation.numeric(value, field.label);
    if (!result.valid) return result;
  }
  
  // Validation rules
  if (field.validation) {
    if (field.validation.min !== undefined) {
      const result = validation.min(value, field.validation.min, field.label);
      if (!result.valid) return result;
    }
    if (field.validation.max !== undefined) {
      const result = validation.max(value, field.validation.max, field.label);
      if (!result.valid) return result;
    }
    if (field.validation.postcode) {
      const result = validation.postcode(value, field.label);
      if (!result.valid) return result;
    }
  }
  
  return { valid: true, error: null };
};

// FormField: Universal field renderer
const FormField = ({ field, value, onChange, error, formData }) => {
  // Dynamic label for C10 wonen field
  let displayLabel = field.label;
  if (field.labelDynamic && field.id === 'wonen' && formData.C1?.woonsituatie) {
    const woonsituatie = formData.C1.woonsituatie;
    if (woonsituatie === 'Huurder') displayLabel = 'Huur (€)';
    else if (woonsituatie === 'Koper') displayLabel = 'Hypotheek (€)';
  }
  
  const handleChange = (newValue) => {
    onChange(field.id, newValue);
  };
  
  return (
    <View style={styles.fieldContainer}>
      <Text style={styles.fieldLabel}>{displayLabel}</Text>
      
      {field.type === 'text' && (
        <TextInput
          style={[styles.input, error && styles.inputError]}
          value={value || ''}
          onChangeText={handleChange}
          placeholder={field.placeholder}
          accessibilityLabel={displayLabel}
        />
      )}
      
      {field.type === 'numeric' && (
        <TextInput
          style={[styles.input, error && styles.inputError]}
          value={value?.toString() || ''}
          onChangeText={(text) => {
            const numValue = text === '' ? '' : parseFloat(text) || 0;
            handleChange(numValue);
          }}
          keyboardType="numeric"
          placeholder={field.placeholder}
          accessibilityLabel={displayLabel}
        />
      )}
      
      {field.type === 'select' && (
        <View style={[styles.input, styles.selectContainer, error && styles.inputError]}>
          <select
            style={styles.select}
            value={value || ''}
            onChange={(e) => handleChange(e.target.value)}
            aria-label={displayLabel}
          >
            {field.options.map(opt => (
              <option key={opt.value} value={opt.value}>{opt.label}</option>
            ))}
          </select>
        </View>
      )}
      
      {field.type === 'toggle' && (
        <View style={styles.toggleContainer}>
          <TouchableOpacity
            style={[styles.toggleButton, value === true && styles.toggleButtonActive]}
            onPress={() => handleChange(true)}
            accessibilityLabel={`${displayLabel} - Ja`}
          >
            <Text style={[styles.toggleText, value === true && styles.toggleTextActive]}>Ja</Text>
          </TouchableOpacity>
          <TouchableOpacity
            style={[styles.toggleButton, value === false && styles.toggleButtonActive]}
            onPress={() => handleChange(false)}
            accessibilityLabel={`${displayLabel} - Nee`}
          >
            <Text style={[styles.toggleText, value === false && styles.toggleTextActive]}>Nee</Text>
          </TouchableOpacity>
        </View>
      )}
      
      {error && <Text style={styles.errorText}>{error}</Text>}
    </View>
  );
};

// ConditionalField: Smart conditional renderer
const ConditionalField = ({ condition, formData, children }) => {
  const evaluateCondition = () => {
    // Parse field path (e.g., "C1.aantalKinderen")
    const fieldPath = condition.field.split('.');
    let value = formData;
    
    for (const key of fieldPath) {
      value = value?.[key];
      if (value === undefined) return false;
    }
    
    // Evaluate operator
    switch (condition.operator) {
      case '>':
        return value > condition.value;
      case '<':
        return value < condition.value;
      case '>=':
        return value >= condition.value;
      case '<=':
        return value <= condition.value;
      case '===':
        return value === condition.value;
      case '!==':
        return value !== condition.value;
      default:
        return false;
    }
  };
  
  const shouldRender = evaluateCondition();
  
  return shouldRender ? <>{children}</> : null;
};

  );
};

// WizardPage: Page wrapper with navigation logic
const WizardPage = ({ pageId, onNext, onBack, loading }) => {
  const { state, dispatch } = useFormContext();
  const config = formConfig[pageId];
  const [pageData, setPageData] = useState({});
  const [errors, setErrors] = useState({});
  const [touched, setTouched] = useState({});
  
  // Load saved data on mount
  useEffect(() => {
    const loadData = async () => {
      const saved = await loadPageData(pageId);
      if (saved) {
        setPageData(saved);
        dispatch({ type: 'SET_PAGE_DATA', pageId, data: saved });
      } else {
        // Set default values
        const defaults = {};
        config.fields.forEach(field => {
          if (field.defaultValue !== undefined) {
            defaults[field.id] = field.defaultValue;
          }
        });
        setPageData(defaults);
      }
    };
    loadData();
  }, [pageId]);
  
  const handleFieldChange = (fieldId, value) => {
    const newData = { ...pageData, [fieldId]: value };
    setPageData(newData);
    dispatch({ type: 'SET_PAGE_DATA', pageId, data: newData });
    
    // Clear error on change
    if (errors[fieldId]) {
      setErrors({ ...errors, [fieldId]: null });
    }
  };
  
  const validatePage = () => {
    const newErrors = {};
    let isValid = true;
    
    config.fields.forEach(field => {
      // Check if field should be rendered (conditional logic)
      if (field.conditional) {
        const shouldRender = evaluateCondition(field.conditional, state);
        if (!shouldRender) return; // Skip validation if not rendered
      }
      
      const result = validateField(field, pageData[field.id]);
      if (!result.valid) {
        newErrors[field.id] = result.error;
        isValid = false;
      }
    });
    
    setErrors(newErrors);
    setTouched(config.fields.reduce((acc, f) => ({ ...acc, [f.id]: true }), {}));
    return isValid;
  };
  
  const evaluateCondition = (condition, formData) => {
    const fieldPath = condition.field.split('.');
    let value = formData;
    
    for (const key of fieldPath) {
      value = value?.[key];
      if (value === undefined) return false;
    }
    
    switch (condition.operator) {
      case '>': return value > condition.value;
      case '<': return value < condition.value;
      case '>=': return value >= condition.value;
      case '<=': return value <= condition.value;
      case '===': return value === condition.value;
      case '!==': return value !== condition.value;
      default: return false;
    }
  };
  
  const handleNext = async () => {
    if (validatePage()) {
      await savePageData(pageId, pageData);
      onNext();
    }
  };
  
  const handleBack = async () => {
    await savePageData(pageId, pageData);
    onBack();
  };
  
  const isFormValid = () => {
    return config.fields.every(field => {
      // Check conditional
      if (field.conditional) {
        const shouldRender = evaluateCondition(field.conditional, state);
        if (!shouldRender) return true;
      }
      
      const result = validateField(field, pageData[field.id]);
      return result.valid;
    });
  };
  
  return (
    <ScrollView style={styles.container}>
      <View style={styles.content}>
        <Text style={styles.stepIndicator}>
          Stap {config.step} van {config.totalSteps}
        </Text>
        <Text style={styles.title}>{config.title}</Text>
        <Text style={styles.subtitle}>{config.subtitle}</Text>
        
        {loading && (
          <View style={styles.loadingContainer}>
            <ActivityIndicator size="large" color="#007AFF" />
            <Text style={styles.loadingText}>Gegevens worden verwerkt...</Text>
          </View>
        )}
        
        {!loading && (
          <>
            {/* Regular fields */}
            {config.fields.map(field => {
              // Check if field has conditional rendering
              if (field.conditional) {
                return (
                  <ConditionalField
                    key={field.id}
                    condition={field.conditional}
                    formData={state}
                  >
                    <FormField
                      field={field}
                      value={pageData[field.id]}
                      onChange={handleFieldChange}
                      error={touched[field.id] ? errors[field.id] : null}
                      formData={state}
                    />
                  </ConditionalField>
                );
              }
              
              return (
                <FormField
                  key={field.id}
                  field={field}
                  value={pageData[field.id]}
                  onChange={handleFieldChange}
                  error={touched[field.id] ? errors[field.id] : null}
                  formData={state}
                />
              );
            })}
            
            {/* Conditional displays */}
            {config.conditionals?.map(conditional => (
              <ConditionalField
                key={conditional.id}
                condition={conditional.condition}
                formData={state}
              >
                <View style={styles.conditionalDisplay}>
                  <Text style={styles.conditionalText}>
                    {conditional.template
                      .replace('{aantalKinderen}', state.C1?.aantalKinderen || 0)
                      .replace('{plural}', (state.C1?.aantalKinderen || 0) === 1 ? 'kind' : 'kinderen')}
                  </Text>
                </View>
              </ConditionalField>
            ))}
            
            {/* Dynamic sections */}
            {config.sections?.map(section => (
              <DynamicSection
                key={section.id}
                section={section}
                formData={state}
                pageData={pageData}
                onFieldChange={handleFieldChange}
                errors={errors}
              />
            ))}
          </>
        )}
        
        <View style={styles.buttonContainer}>
          <TouchableOpacity
            style={styles.buttonSecondary}
            onPress={handleBack}
            disabled={loading}
            accessibilityLabel="Ga terug"
          >
            <Text style={styles.buttonSecondaryText}>Terug</Text>
          </TouchableOpacity>
          
          <TouchableOpacity
            style={[styles.buttonPrimary, (!isFormValid() || loading) && styles.buttonDisabled]}
            onPress={handleNext}
            disabled={!isFormValid() || loading}
            accessibilityLabel="Ga verder"
          >
            <Text style={styles.buttonPrimaryText}>Verder</Text>
          </TouchableOpacity>
        </View>
      </View>
    </ScrollView>
  );
};

        setSummary(result.financialSummary);
        setLoading(false);
      } catch (error) {
        console.error('Error loading summary:', error);
        setLoading(false);
      }
    };
    
    loadSummary();
  }, []);
  
  if (loading) {
    return (
      <View style={styles.container}>
        <View style={styles.centerContent}>
          <ActivityIndicator size="large" color="#007AFF" />
          <Text style={styles.loadingText}>Dashboard wordt geladen...</Text>
        </View>
      </View>
    );
  }
  
  const cashflow = summary?.cashflowNetto || 0;
  const getMessage = () => {
    if (cashflow > 0) {
      return `Gefeliciteerd! U heeft maandelijks nog €${cashflow.toFixed(2)} beschikbaar.`;
    } else if (cashflow < 0) {
      return `Uw overzicht is klaar. U heeft maandelijks een tekort van €${Math.abs(cashflow).toFixed(2)}. Laten we samen kijken waar u kunt besparen.`;
    } else {
      return 'Uw inkomsten en vaste lasten zijn in balans. Laten we uw flexibele budget nu bepalen.';
    }
  };
  
  const getColor = () => {
    if (cashflow > 0) return '#34C759';
    if (cashflow < -100) return '#FF3B30';
    return '#FF9500';
  };
  
  return (
    <ScrollView style={styles.container}>
      <View style={styles.content}>
        <Text style={styles.title}>Uw Financieel Overzicht</Text>
        
        <View style={[styles.dashboardCard, { borderLeftColor: getColor() }]}>
          <Text style={styles.dashboardLabel}>Maandelijkse Cashflow</Text>
          <Text style={[styles.dashboardKPI, { color: getColor() }]}>
            €{cashflow.toFixed(2)}
          </Text>
          <Text style={styles.dashboardMessage}>{getMessage()}</Text>
        </View>
        
        <View style={styles.summarySection}>
          <Text style={styles.sectionTitle}>Samenvatting</Text>
          
          <View style={styles.summaryRow}>
            <Text style={styles.summaryLabel}>Totaal Inkomen:</Text>
            <Text style={styles.summaryValue}>€{summary?.inkomenTotaalMaand?.toFixed(2)}</Text>
          </View>
          
          <View style={styles.summaryRow}>
            <Text style={styles.summaryLabel}>Vaste Lasten:</Text>
            <Text style={styles.summaryValue}>€{summary?.lastenTotaalVast?.toFixed(2)}</Text>
          </View>
          
          <View style={[styles.summaryRow, styles.summaryRowTotal]}>
            <Text style={[styles.summaryLabel, styles.summaryLabelBold]}>Cashflow:</Text>
            <Text style={[styles.summaryValue, styles.summaryValueBold, { color: getColor() }]}>
              €{cashflow.toFixed(2)}
            </Text>
          </View>
        </View>
        
        <View style={styles.buttonContainer}>
          <TouchableOpacity
            style={styles.buttonSecondary}
            onPress={onBack}
            accessibilityLabel="Terug naar instellingen"
          >
            <Text style={styles.buttonSecondaryText}>Terug naar instellingen</Text>
          </TouchableOpacity>
        </View>
      </View>
    </ScrollView>
  );
};

// ============================================================================
// MAIN APP & NAVIGATION
// ============================================================================

const App = () => {
  const [currentScreen, setCurrentScreen] = useState('Landing');
  const [loading, setLoading] = useState(false);
  const { state, dispatch } = useFormContext();
  
  const handleNavigation = async (nextScreen, mockCall = null) => {
    if (mockCall) {
      setLoading(true);
      try {
        let result;
        if (mockCall === 'P3') {
          result = await mockN8n.callP3(state);
          await saveUserId(result.userId);
          dispatch({ type: 'SET_USER_ID', userId: result.userId });
        } else if (mockCall === 'P5') {
          result = await mockN8n.callP5(state);
        } else if (mockCall === 'P6') {
          result = await mockN8n.callP6(state);
        }
        console.log(`✓ ${mockCall} completed:`, result);
      } catch (error) {
        console.error(`✗ Error in ${mockCall}:`, error);
      }
      setLoading(false);
    }
    
    setCurrentScreen(nextScreen);
  };
  
  const renderScreen = () => {
    switch (currentScreen) {
      case 'Landing':
        return <LandingScreen onStart={(screen) => setCurrentScreen(screen)} />;
      
      case 'C1':
        return (
          <WizardPage
            pageId="C1"
            onNext={() => setCurrentScreen('C4')}
            onBack={() => setCurrentScreen('Landing')}
            loading={loading}
          />
        );
      
      case 'C4':
        return (
          <WizardPage
            pageId="C4"
            onNext={() => handleNavigation('C7', 'P3')}
            onBack={() => setCurrentScreen('C1')}
            loading={loading}
          />
        );
      
      case 'C7':
        return (
          <WizardPage
            pageId="C7"
            onNext={() => handleNavigation('C10', 'P5')}
            onBack={() => setCurrentScreen('C4')}
            loading={loading}
          />
        );
      
      case 'C10':
        return (
          <WizardPage
            pageId="C10"
            onNext={() => handleNavigation('Dashboard', 'P6')}
            onBack={() => setCurrentScreen('C7')}
            loading={loading}
          />
        );
      
      case 'Dashboard':
        return <DashboardScreen onBack={() => setCurrentScreen('C1')} />;
      
      default:
        return <LandingScreen onStart={(screen) => setCurrentScreen(screen)} />;
    }
  };
  
  return (
    <View style={styles.app}>
      {renderScreen()}
    </View>
  );
};
