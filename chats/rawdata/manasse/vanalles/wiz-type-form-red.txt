// src/app/context/formReducer.ts
import { FormState, FormAction } from '../../shared-types/form';

export function formReducer(state: FormState, action: FormAction): FormState {
  const metaUpdate = { ...state.meta, lastModified: new Date().toISOString() };

  switch (action.type) {
    case 'UPDATE_FIELD': {
      const { section, field, value } = action.payload;
      return {
        ...state,
        data: {
          ...state.data,
          [section]: {
            ...(state.data as any)[section],
            [field]: value,
          },
        },
        meta: metaUpdate,
      };
    }
    case 'SET_STEP':
      return { ...state, activeStep: action.payload, meta: metaUpdate };
    case 'RESET_APP':
      return { ...state, data: { ...state.data }, meta: metaUpdate }; // Versimpeld voor voorbeeld
    default:
      return state;
  }
}

// src/shared-types/types.ts
export type DeepPartial<T> = {
  [P in keyof T]?: T[P] extends (infer U)[]
    ? DeepPartial<U>[]                     // arrays: maak elementtype diep-partial
    : T[P] extends Map<infer K, infer V>
    ? Map<K, DeepPartial<V>>               // Maps (optioneel; alleen als je ze gebruikt)
    : T[P] extends Set<infer S>
    ? Set<DeepPartial<S>>                  // Sets (optioneel)
    : T[P] extends object
    ? DeepPartial<T[P]>                    // objecten: recursief
    : T[P];                                // primitives: laat zoals-is
};

import type { FormState, DataSection } from './wizard';

export type FieldType =
  | 'counter'
  | 'text'
  | 'radio-chips'
  | 'toggle'
  | 'repeater'
  | 'date'
  | 'select'
  | 'section'
  | 'collapsible-section'
  | 'derived-label';

export interface FieldConfig {
  fieldId: string;
  type: FieldType;

  /** In welke lade van FormState.data dit veld schrijft (override per veld). */
  section?: DataSection; // Aanrader i.p.v. 'targetSection' voor consistente naamgeving

  /** Labeling */
  labelToken?: string;
  label?: string;

  /** Keuze-opties voor select/radio */
  options?: Array<{ label: string; value: unknown }>;

  /** Geneste velden (section/collapsible-section/repeater) */
  fields?: FieldConfig[];

  /** Zichtbaarheid/logica */
  visibleIf?: string | ((state: FormState) => boolean);
  valueGetter?: (state: FormState) => unknown;
  maxGetter?: (state: FormState) => number;
  countGetter?: (state: FormState) => number;
  filter?: (state: FormState) => unknown[];

  /** Validatie */
  validation?: {
    min?: number;
    max?: number;
    postcode?: boolean;
    pattern?: string;
    lengthEqualsTo?: string | number;
  };

  /** Verplicht veld */
  required?: boolean;

  /** Conditioneel veld */
  dependentField?: FieldConfig;

  /** Defaultwaarde voor initialisatie */
  defaultValue?: unknown;
}

import type { DeepPartial } from './types';
import type { FieldConfig } from './fields';

export type DataSection = 'setup' | 'household' | 'finance';

export interface FormState {
  schemaVersion: string;
  activeStep: string;
  currentPageId: string;
  isValid: boolean;
  data: {
    setup: {
      aantalMensen: number;
      aantalVolwassen: number;
      autoCount: 'Nee' | 'Een' | 'Twee';
      heeftHuisdieren?: boolean;
      [key: string]: unknown;
    };
    household: { members: Array<Record<string, unknown>> };
    finance: {
      income: { items: Array<Record<string, unknown>>; totalAmount?: number };
      expenses: { items: Array<Record<string, unknown>>; totalAmount?: number };
    };
  };
  meta: { lastModified: string; version: number };
}

export interface WizardPageConfig {
  pageId: string;
  titleToken?: string;
  title?: string;
  section?: DataSection;
  componentName?: 'WizardPage' | 'SummaryPage' | 'CustomPage' | string;
  fields: FieldConfig[];
  onNext?: (state: FormState) => string;
}

export type FormAction =
  | { type: 'SET_STEP'; payload: string }
  | { type: 'UPDATE_DATA'; payload: DeepPartial<FormState['data']> }
  | { type: 'SET_FIELD'; payload: { section: DataSection; field: string; value: any } }
  | { type: 'SYNC_MEMBERS'; payload: Array<Record<string, unknown>> }
  | { type: 'RESET_APP' }
  | { type: 'NEXT_STEP' }
  | { type: 'PREV_STEP' };
