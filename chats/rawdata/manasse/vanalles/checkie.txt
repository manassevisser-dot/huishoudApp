const fs = require('fs');
const path = require('path');

// Gebruik een prefix underscore voor variabelen die we uit de env halen 
// maar in deze specifieke run misschien niet direct muteren
const _isDryRun = process.env.DRY_RUN === 'true';
const isVerbose = process.env.VERBOSE === 'true';

function getTrinityData() {
    const auditScore = 85; 
    let coverageScore = 0;
    let riskPenalty = 0;

    try {
        const coveragePath = path.join(process.cwd(), 'coverage/coverage-summary.json');
        if (fs.existsSync(coveragePath)) {
            const summary = JSON.parse(fs.readFileSync(coveragePath, 'utf8'));
            coverageScore = Math.round(summary.total.branches.pct);
            const uncoveredLines = summary.total.lines.total - summary.total.lines.covered;
            riskPenalty = Math.min(Math.round(uncoveredLines / 10), 20);
        }
    } catch { 
        // GEFIXED: De (e) of (_e) is volledig verwijderd. 
        // In modern JavaScript (ES2019+) mag een catch blok zonder variabele.
        if (isVerbose) console.warn("вҡ пёҸ Kon coverage data niet inlezen.");
    }

    const stabilityScore = Math.max(0, coverageScore - riskPenalty);
    const masterScore = Math.round((auditScore + stabilityScore) / 2);

    const getGrade = (s) => {
        if (s >= 90) return 'S';
        if (s >= 75) return 'A';
        if (s >= 60) return 'B';
        return 'C';
    };

    return {
        audit: auditScore,
        coverage: coverageScore,
        stability: stabilityScore,
        master: getGrade(masterScore)
    };
}

const scores = getTrinityData();

// We gebruiken _isDryRun hier om de linting error te voorkomen
if (isVerbose && _isDryRun) {
    console.log("[DRY-RUN MODE] Berekeningen worden uitgevoerd zonder schrijven.");
}

// Output de data-string voor de bash dispatcher
console.log(`TRINITY_DATA|AUDIT:${scores.audit}|COV:${scores.coverage}|STAB:${scores.stability}|MASTER:${scores.master}`);

#!/usr/bin/env bash

# Parser voor opties
parse_args() {
    while [[ $# -gt 0 ]]; do
        case "$1" in
            -d|--dry-run) DRY_RUN=true; shift ;;
            -v|--verbose) VERBOSE=true; shift ;;
            *) break ;; # Stop bij het eerste commando (full, all, etc.)
        esac
    done
    export DRY_RUN VERBOSE
}

check_dependencies() {
  local deps=(node npm bash grep awk sed git)
  for cmd in "${deps[@]}"; do
    command -v "$cmd" >/dev/null || { log_err "Missing dependency: $cmd"; exit 40; }
  done
}

# Helper voor dry-runs
run_cmd() {
    local msg=$1
    local cmd=$2
    if [ "$DRY_RUN" = true ]; then
        log_warn "[DRY-RUN] Zou uitvoeren: $msg"
    else
        log_info "$msg"
        eval "$cmd"
    fi
}

cmd_audit() {
    log_info "рҹ”Һ Audit uitvoeren..."
    node scripts/maintenance/audit-orchestrator.js
}

cmd_test() {
    log_info "рҹ§Ә Test Suite draaien..."
    local test_cmd="npm test -- --watchAll=false"
    [ "$VERBOSE" = false ] && test_cmd="$test_cmd --silent"
    eval "$test_cmd"
}

cmd_all() {
    log_info "рҹҡҖ Volledige inspectie (Audit + Tests)..."
    cmd_audit || true
    cmd_test
}

cmd_full() {
    log_info "рҹҢ… Starten Phoenix End-of-Day routine..."
    
    cmd_audit || true
    
    run_cmd "рҹ§№ Linting & Formatting" "npm run lint -- --fix"
    
    log_info "рҹ“Ғ Root directory opschonen..."
    if [ "$DRY_RUN" = false ]; then
        mkdir -p scripts/temp
        find . -maxdepth 1 -name "*.sh" ! -name "phoenix.sh" -exec mv {} scripts/temp/ \;
    else
        log_warn "[DRY-RUN] Zou .sh bestanden verplaatsen naar scripts/temp/"
    fi
    
    cmd_test || { log_err "Tests falen! EOD afgebroken."; return 1; }
    
    if [ "$DRY_RUN" = false ]; then
        log_info "рҹ“Ұ Wijzigingen veiligstellen in Git..."
        git add .
        local grade; grade=$(grep -o 'Grade: .' reports/phoenix-audit.html 2>/dev/null | cut -d' ' -f2 || echo 'A')
        git commit -m "Phoenix EOD: $(date +'%Y-%m-%d %H:%M') - Grade $grade"
    else
        log_warn "[DRY-RUN] Zou Git commit uitvoeren met Grade info."
    fi
    
    log_ok "вң… Werkdag succesvol afgesloten."
}

cmd_watch() {
    log_info "рҹ‘Җ Watch-modus: Dashboard ververst elke 30 seconden (Ctrl+C om te stoppen)"

        sleep 30
    done

Hier zijn de voorgestelde verbeteringen:

### 1. Toevoegen van de 'Einde van de dag echo'

Je hebt gelijk dat de Trinity-versie momenteel direct naar de commit gaat zonder een duidelijke samenvatting te printen. In de oude versie was de "echo" duidelijker. Je kunt de `cmd_full` in het Trinity-script uitbreiden met een visuele afsluiting:

```bash
echo -e "рҹҸҶ Behaalde Master Grade: $master"
echo -e "рҹ“Ҡ Stats: $stats"
echo -e "==============================================="
log_ok "вң… Werkdag succesvol afgesloten en veiliggesteld in Git."

```

### 2. Verbetering van de Idempotency Tests

Hoewel de documentatie idempotency noemt, zie ik in de huidige testsuites nog geen expliciete checks die verifiГ«ren of de output van een tweede run identiek is aan de eerste. Dit is cruciaal voor migraties.

* 
**Verbetering:** Voeg een testcase toe die `migrate(input)` twee keer achter elkaar draait en de resultaten diepgaand vergelijkt (`toStrictEqual`).



### 3. Verscherping van de Coverage Thresholds per Component

In de `testing.txt` staan algemene doelen van 80% , maar de tabel voor "Core Logic" vraagt om 95%.

* **Verbetering:** Gebruik de `jest.config.js` om specifieke drempels per map af te dwingen. Dit voorkomt dat "makkelijke" tests de lage coverage van "kritieke" onderdelen verbloemen:



```javascript
coverageThreshold: {
  "./src/domain/logic/": { "branches": 95, "functions": 95 },
  "./src/utils/": { "branches": 80 }
}

```

### 4. Consistentie in de 'Fuzzy' PII checks

In de `privacyHelpers.test.ts` zijn de fuzzy checks voor PII aangescherpt voor velden als `achternaamPartner`.

* 
**Verbetering:** Zorg dat de `forbiddenKeys` lijst in de broncode (die we hier niet volledig zien) synchroon loopt met deze nieuwe test-scenarios om valse negatieven te voorkomen.



### 5. Robuustheid van de 'Clean-up' routine

De EOD routine verplaatst alle `.sh` bestanden naar `scripts/temp/`, behalve `phoenix.sh`.

* 
**Verbetering:** Voeg een check toe die controleert of `scripts/temp/` niet per ongeluk wordt mee-gecommit naar productie, of zorg dat deze map in de `.gitignore` staat, aangezien dit "vervuiling" van de repository kan veroorzaken bij dagelijks gebruik.



### 6. Integratie van de Audit-Orchestrator in de Watch-modus

De huidige `cmd_watch` in de Trinity-editie herhaalt alleen `cmd_all` elke 30 seconden.

* **Verbetering:** Voeg een notificatie-trigger toe (bijv. een systeem-beep of terminal-kleurverandering) wanneer de Master Grade zakt onder een bepaald niveau tijdens het programmeren.

# рҹ§© Deel 1 вҖ” **Clean Commit Policy (Phoenix-proof)**

## рҹҺҜ Doel

Commits mogen alleen binnen als:

1пёҸвғЈ Code bouwt
2пёҸвғЈ Tests slagen
3пёҸвғЈ Basiskwaliteit op orde is
4пёҸвғЈ Trinity-scores gecheckt zijn

## рҹ“ң **Commit Message Standaard**

**Format:**

```
<type>(<scope>): <summary>

[optional body]

[optional BREAKING CHANGE]
```

### Types

| Type     | Gebruik                      |
| -------- | ---------------------------- |
| feat     | nieuwe functionaliteit       |
| fix      | bug fix                      |
| refactor | codewijziging zonder feature |
| test     | testtoevoeging               |
| chore    | onderhoud / scripts          |
| docs     | documentatie                 |
| perf     | performance-verbetering      |
| security | kwetsbaarheid / privacy fix  |

### Scope

Bijv:

```
feat(finance): parse Rabobank CSV
fix(piiguard): ignore extra IBAN patterns
```

### MUST for Trinity commits

Voeg **Master Grade toe aan commit body**:

```
Phoenix Grade: A (Audit 92 | Coverage 88 | Stability 95)
```

вҡ пёҸ **Nooit faken вҖ” alleen door script laten invullen**

### Advies op verbeteringen

*   **Parsing:** Stap af van `grep` en `cut` voor het parsen van audit-resultaten. Gebruik JSON output (`--json`) en parse dit met `jq` of een klein Node.js scriptje. Dit is veel robuuster.
*   **Modulariteit:** Splits de scripts op in kleinere, herbruikbare modules (bijv. `audit.sh`, `test.sh`, `commit.sh`). Dit maakt het makkelijker om individuele onderdelen te testen en te onderhouden.
*   **Configuratie:** Maak de thresholds (drempelwaarden) configureerbaar via een bestand (bijv. `.phoenixrc`), zodat je ze per project of omgeving kunt aanpassen.
*   **Error Handling:** Wees explicieter in de foutafhandeling. In plaats van `|| true`, vang fouten op en log ze duidelijk, eventueel met een suggestie voor een fix.
*   **Documentatie:** Voeg commentaar toe aan de scripts om de logica uit te leggen, vooral bij complexe parsing of berekeningen.

#### 3. **Quality Gates Strategie** вӯҗвӯҗвӯҗвӯҗвӯҗ

De voorgestelde hard/soft gates zijn **perfect gebalanceerd**:

| Gate Type | Threshold | Gedrag |
|-----------|-----------|--------|
| **Hard** | Coverage < 70% | вқҢ Block commit |
| **Hard** | Master Grade < B | вқҢ Block commit |
| **Hard** | Tests fail | вқҢ Block commit |
| **Soft** | Coverage 70-80% | вҡ пёҸ Warn but allow |
| **Soft** | Stability < 80% | вҡ пёҸ Suggest review |

**Implementatie:**
```bash
check_quality_gates() {
    local json=$1
    local master=$(echo "$json" | jq -r '.master')
    local coverage=$(echo "$json" | jq -r '.coverage')
    
    # Hard gates
    if [[ "$coverage" -lt 70 ]]; then
        log_err "вқҢ HARD GATE: Coverage below 70% ($coverage%)"
        return 1
    fi
    
    if [[ "$master" < "B" ]]; then
        log_err "вқҢ HARD GATE: Master grade below B ($master)"
        return 1
    fi
    
    # Soft gates (warnings)
    if [[ "$coverage" -lt 80 ]]; then
        log_warn "вҡ пёҸ  Coverage is ${coverage}% (target: 80%+)"
    fi
    
    return 0
}
```

### вң… 1. **State Machine voor Trinity Scores**

In plaats van direct scores berekenen, model ze als state:

```javascript
// audit-orchestrator.js
class TrinityState {
  constructor() {
    this.audit = 0;
    this.coverage = 0;
    this.stability = 0;
    this.master = 'U'; // Unknown
    this.errors = [];
    this.warnings = [];
  }
  
  computeMaster() {
    const avg = (this.audit + this.stability) / 2;
    if (avg >= 90) return 'S';
    if (avg >= 75) return 'A';
    if (avg >= 60) return 'B';
    return 'C';
  }
  
  toJSON() {
    return {
      audit: this.audit,
      coverage: this.coverage,
      stability: this.stability,
      master: this.computeMaster(),
      meta: {
        timestamp: new Date().toISOString(),
        errors: this.errors,
        warnings: this.warnings
      }
    };
  }
}
```

**Voordeel:** Testbaar, voorspelbaar, uitbreidbaar.
