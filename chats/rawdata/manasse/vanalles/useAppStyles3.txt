### 6. src/styles/useAppStyles.ts
The aggregator file.

```typescript
// src/styles/useAppStyles.ts
import { StyleSheet } from 'react-native';
import { useTheme } from '@app/context/ThemeContext';
import { Colors, Theme } from './Colors';
import { Space, Type, Radius } from './Tokens';

import { getLayoutStyles, getHeaderStyles } from './modules/Layout';
import { getFormStyles, getComplexInputStyles } from './modules/Form';
import { getButtonStyles, getCardStyles, getChipStyles } from './modules/Components';
import { getDashboardStyles, getSummaryStyles } from './modules/Dashboard';

// Cache per theme
const styleCache: Partial<Record<Theme, any>> = {};

export function getAppStyles(theme: Theme) {
  if (styleCache[theme]) return styleCache[theme];
  const c = Colors[theme];

  const styles = StyleSheet.create({
    // Merge all module styles
    ...getLayoutStyles(c),
    ...getHeaderStyles(c),
    ...getFormStyles(c),
    ...getComplexInputStyles(c),
    ...getButtonStyles(c),
    ...getCardStyles(c),
    ...getChipStyles(c),
    ...getDashboardStyles(c),
    ...getSummaryStyles(c),

    // Misc / Legacy / One-offs (Keep specific small styles here if they don't fit modules)
    navigationHint: {
      fontSize: Type.sm, color: c.textTertiary, textAlign: 'right',
      marginTop: Space.sm, marginRight: Space.sm, fontStyle: 'italic',
    },
    hintOverlayBottomRight: { position: 'absolute', bottom: 6, right: 12 },
    warningTextOrange: { color: c.warning, fontSize: Type.sm, marginTop: Space.xs, marginLeft: Space.xs },
    warningTextRed:    { color: c.error,   fontSize: Type.sm, marginTop: Space.xs, marginLeft: Space.xs, fontWeight: '600' },
    debugBox: { backgroundColor: c.secondary, padding: Space.sm, borderRadius: 6, marginTop: Space.sm },
    debugText: { fontSize: Type.xs, color: c.textSecondary },
  });

  styleCache[theme] = styles;
  return styles;
}

export type AppStyles = ReturnType<typeof getAppStyles>;

export function useAppStyles() {
  const { theme } = useTheme();
  return { styles: getAppStyles(theme), colors: Colors[theme], space: Space, type: Type, radius: Radius };
}
```
