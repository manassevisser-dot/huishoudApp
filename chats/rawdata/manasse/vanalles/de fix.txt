import * as React from 'react';
import { View, Text, TouchableOpacity, ScrollView, ActivityIndicator } from 'react-native';
import { TransactionService } from '@services/transactionService';

interface Transaction {
  id: string;
  amount: number;
  description: string;
}

export const UndoScreen: React.FC = () => {
  const [items, setItems] = React.useState<Transaction[]>([]);
  const [loading, setLoading] = React.useState(true);
  const [error, setError] = React.useState<string | null>(null);

  React.useEffect(() => {
    const fetchTransactions = async () => {
      try {
        setLoading(true);
        setError(null);
        const data = await TransactionService.getAllTransactions();
        setItems(data || []);
      } catch (err) {
        console.error('Failed to fetch transactions:', err);
        setError('Kon transacties niet laden');
        setItems([]);
      } finally {
        setLoading(false);
      }
    };
    fetchTransactions();
  }, []);

  const handleClearAll = async () => {
    try {
      await TransactionService.clearAll?.();
      setItems([]);
      setError(null);
    } catch (err) {
      console.error('Failed to clear transactions:', err);
      setError('Kon transacties niet verwijderen');
    }
  };

  if (loading) {
    return (
      <View style={{ padding: 20, alignItems: 'center', justifyContent: 'center' }}>
        <ActivityIndicator size="large" />
        <Text style={{ marginTop: 10 }}>Laden...</Text>
      </View>
    );
  }

  return (
    <View style={{ padding: 20 }}>
      <Text style={{ fontSize: 18, fontWeight: 'bold' }}>
        Laatste transacties: {items.length}
      </Text>

      {error && (
        <View style={{ 
          marginTop: 8, 
          padding: 10, 
          backgroundColor: '#ffebee', 
          borderRadius: 5 
        }}>
          <Text style={{ color: '#c62828' }}>{error}</Text>
        </View>
      )}

      {items.length === 0 && !error ? (
        <Text style={{ marginTop: 8, color: '#666' }}>
          Geen recente transacties
        </Text>
      ) : (
        <ScrollView 
          style={{ marginTop: 8, maxHeight: 400 }}
          testID="transaction-list"
        >
          {items.map((item) => (
            <View 
              key={item.id} 
              style={{ 
                paddingVertical: 8,
                paddingHorizontal: 12,
                marginBottom: 4,
                backgroundColor: '#f5f5f5',
                borderRadius: 4
              }}
            >
              <Text style={{ fontSize: 16 }}>
                {item.description}: € {item.amount.toFixed(2)}
              </Text>
            </View>
          ))}
        </ScrollView>
      )}

      <TouchableOpacity
        onPress={handleClearAll}
        accessibilityRole="button"
        accessibilityLabel="Verwijder alle transacties"
        disabled={items.length === 0}
        style={{ 
          marginTop: 20, 
          backgroundColor: items.length === 0 ? '#ccc' : '#d32f2f', 
          padding: 12, 
          borderRadius: 5 
        }}
      >
        <Text style={{ 
          color: 'white', 
          textAlign: 'center',
          fontWeight: 'bold'
        }}>
          Verwijder Alles {items.length > 0 && `(${items.length})`}
        </Text>
      </TouchableOpacity>
    </View>
  );
};

export default UndoScreen;

/**
 * Extraheert text uit een React Native node
 */
function extractText(node: any): string {
  if (typeof node === 'string') {
    return node;
  }
  
  if (node?.props?.children) {
    if (Array.isArray(node.props.children)) {
      return node.props.children.map(extractText).join('');
    }
    return extractText(node.props.children);
  }
  
  return '';
}

/**
 * Wacht tot een text fragment verschijnt
 */
export async function waitForTextContaining(
  searchText: string,
  options?: { timeout?: number }
): Promise<void> {
  await waitFor(() => {
    const allNodes = screen.queryAllByText(() => true);
    const found = allNodes.some(node => {
      const text = extractText(node);
      return text.includes(searchText);
    });
    expect(found).toBe(true);
  }, options);
}

# 🧪 React Native Testing Guide - UndoScreen

## ❌ Het Probleem

```typescript
// ❌ DIT WERKT NIET
const firstItem = await screen.findByText((content: string, element: any) => {
  return element?.type === 'Text' && content.includes('Boodschappen');
});
// Error: matcher.test is not a function
```

**Waarom niet?**
`findByText` accepteert alleen:
- String: `findByText('Boodschappen')`
- Regex: `findByText(/Boodschappen/i)`

**GEEN** custom matcher functions!

**Wanneer gebruiken:** Altijd, tenzij je exact match nodig hebt.

**Pro's:**
- Simpel

- Ondersteunt whitespace patterns

**Con's:**

## 🐛 Common Pitfalls

### ❌ Pitfall 1: Act Warnings
```typescript
// ❌ FOUT
render(<UndoScreen />);
expect(screen.getByText(/Boodschappen/i)).toBeTruthy();
// Warning: An update was not wrapped in act(...)

// ✅ GOED
render(<UndoScreen />);
await screen.findByText(/Boodschappen/i);
// of
await waitFor(() => {
  expect(screen.getByText(/Boodschappen/i)).toBeTruthy();
});
```

---

### ❌ Pitfall 2: Query Timing
```typescript
// ❌ FOUT - getBy faalt als element er niet is
const button = screen.getByRole('button');

// ✅ GOED - findBy wacht tot element er is
const button = await screen.findByRole('button');

// ✅ GOED - queryBy returnt null
const button = screen.queryByRole('button');
expect(button).toBeNull();
```

---

### ❌ Pitfall 3: Mock Timing
```typescript
// ❌ FOUT - mock wordt niet gereset
test('test 1', () => {
  mockedService.getData.mockResolvedValueOnce(['A']);
});

test('test 2', () => {
  // Mock van test 1 is nog actief!
});

// ✅ GOED
beforeEach(() => {
  jest.clearAllMocks();
});
```

## 📚 Query Cheatsheet

| Method | Wanneer gebruiken | Throws? | Waits? |
|--------|-------------------|---------|---------|
| `getBy` | Element moet er zijn | ✅ | ❌ |
| `queryBy` | Element mag er niet zijn | ❌ | ❌ |
| `findBy` | Async loading | ✅ | ✅ |
| `getAllBy` | Meerdere matches | ✅ | ❌ |
| `queryAllBy` | 0+ matches | ❌ | ❌ |
| `findAllBy` | Async meerdere | ✅ | ✅ |

## 🎨 Best Practices

### ✅ DO's

1. **Wacht op async updates**
   ```typescript
   await screen.findByText(/text/i);
   ```

2. **Gebruik testID voor complexe queries**
   ```typescript
   screen.getByTestId('transaction-list');
   ```

3. **Mock proper cleanup**
   ```typescript
   beforeEach(() => jest.clearAllMocks());
   afterEach(cleanup);
   ```

4. **Test user perspective**
   ```typescript
   // ✅ Test wat gebruiker ziet
   expect(screen.getByText(/Boodschappen/i)).toBeTruthy();
   
   // ❌ Test niet implementatie details
   expect(component.state.items.length).toBe(1);
   ```

5. **Use meaningful test names**
   ```typescript
   it('moet error tonen bij network failure', ...)
   ```

---

### ❌ DON'Ts

1. **Niet testen zonder waitFor**
   ```typescript
   // ❌
   render(<Component />);
   expect(screen.getByText(/loaded/i)).toBeTruthy();
   ```

2. **Niet testen op implementatie**
   ```typescript
   // ❌
   expect(component.instance().handleClick).toHaveBeenCalled();
   ```

3. **Niet custom matchers in queries**
   ```typescript
   // ❌
   screen.findByText((text, element) => ...)
   ```

4. **Niet snapshots van loading states**
   ```typescript
   // ❌
   const { toJSON } = render(<Component />);
   expect(toJSON()).toMatchSnapshot(); // Capturing loading state!
   
   // ✅
   await screen.findByText(/loaded/i);
   expect(toJSON()).toMatchSnapshot();
   ```
