/**
 * Dynamic style factory - creates theme-aware StyleSheet
 * Optimized for performance with caching and domain separation
 */
export function getAppStyles(theme: Theme) {
  if (styleCache[theme]) {
    return styleCache[theme]!;
  }

  const c = Colors[theme];

  const styles = StyleSheet.create({
    ...getLayoutStyles(c),
    ...getTypographyStyles(c),
    ...getHeaderStyles(c),
    ...getFormStyles(c),
    ...getMoneyStyles(c),
    ...getActionStyles(c),
    ...getToggleStyles(c),
    ...getCardStyles(c),
    ...getFeedbackStyles(c),
    ...getDashboardStyles(c),
  });

  styleCache[theme] = styles;
  return styles;
}

export type AppStyles = ReturnType<typeof getAppStyles>;

export function useAppStyles() {
  const { theme } = useTheme();
  return {
    styles: getAppStyles(theme),
    colors: Colors[theme],
  };
}
