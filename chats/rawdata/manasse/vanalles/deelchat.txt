/**

 * Extraheert text uit een React Native node

 */

function extractText(node: any): string {

  if (typeof node === 'string') {

    return node;

  }

  

  if (node?.props?.children) {

    if (Array.isArray(node.props.children)) {

      return node.props.children.map(extractText).join('');

    }

    return extractText(node.props.children);

  }

  

  return '';

}




/**

 * Wacht tot een text fragment verschijnt

 */

export async function waitForTextContaining(

  searchText: string,

  options?: { timeout?: number }

): Promise<void> {

  await waitFor(() => {

    const allNodes = screen.queryAllByText(() => true);

    const found = allNodes.some(node => {

      const text = extractText(node);

      return text.includes(searchText);

    });

    expect(found).toBe(true);

  }, options);

}




Wat kunnen we hiermee?

Met deze helper-functies heb je een krachtig wapen in handen om de meest hardnekkige testfouten in React Native op te lossen. Het lost specifiek het probleem op waarbij TypeScript klaagt over complexe matchers en waarbij de test elementen niet kan vinden door tekstfragmentatie.

1. Waarom is dit de oplossing?

In React Native worden teksten vaak opgesplitst in de Virtual DOM, bijvoorbeeld wanneer je variabelen gebruikt: {description}: € {amount}. Een standaard getByText zoekt vaak naar een exacte match in één enkele node en faalt dan.

extractText: Deze functie graaft door alle geneste "children" van een component heen om alle tekststukjes aan elkaar te plakken tot één leesbare string.

waitForTextContaining: Deze helper gebruikt een "broad search" (queryAllByText(() => true)) om elk element in de boom te controleren met de extractText logica.

2. Hoe implementeer je dit in je falende test?

Je kunt nu je UndoScreen.test.tsx opschonen en de TypeScript-fouten (code 2345) elimineren.

Gebruiksvriendelijk: Je test nu op wat de gebruiker daadwerkelijk ziet, in plaats van hoe de code technisch is opgebouwd.
