Python
import sys
from bs4 import BeautifulSoup
from pathlib import Path

def extract_text(input_html: Path) -> str:
    html_content = input_html.read_text(encoding="utf-8", errors="ignore")
    soup = BeautifulSoup(html_content, "html.parser")
    # Verwijder niet-zichtbare delen
    for tag in soup(["script", "style", "noscript", "template"]):
        tag.decompose()
    # Pak zichtbare tekst, met nette separators
    text = soup.get_text(separator="\n")
    # Opschonen: lege regels en overbodige spaties verwijderen
    lines = [line.strip() for line in text.splitlines() if line.strip()]
    return "\n".join(lines)

def main():
    if len(sys.argv) < 2:
        print("Gebruik: python extract_html_text.py <bron.html> [uitvoer.txt]")
        print("Voorbeeld: python extract_html_text.py pagina.html pagina.txt")
        sys.exit(1)

    input_path = Path(sys.argv[1])
    if not input_path.exists():
        print(f"❌ Bestand niet gevonden: {input_path}")
        sys.exit(2)

    output_path = Path(sys.argv[2]) if len(sys.argv) >= 3 else input_path.with_suffix(".txt")

    clean_text = extract_text(input_path)
    output_path.write_text(clean_text, encoding="utf-8")
    print(f"✅ Klaar! Tekst opgeslagen in: {output_path}")

if __name__ == "__main__":
    main()

Fixen van ESLint-fouten
Dank je voor het lint‑rapport, Manasse. Dit zijn veelvoorkomende ESLint‑meldingen in een ESM‑script (.mjs). Hieronder leg ik uit waarom ze ontstaan en geef ik je kant‑en‑klare fixes (inclusief patches) die je meteen kunt toepassen.

1) no-undef: 'console' is not defined. / 'process' is not defined.
Waarom gebeurt dit?
ESLint weet niet automatisch in welke runtime jouw code draait. In Node.js horen console en process tot de globale variabelen. Als ESLint denkt dat het een browser of een “generic” omgeving is, markeert het ze als undefined.

Oplossing A (snelste): zet de Node‑omgeving aan in je script
Plaats bovenaan je .mjs bestand:




JavaScript
/* eslint-env node */
Dit vertelt ESLint dat je file in Node.js draait, waardoor console en process als geldige globals worden herkend.

Oplossing B: configureer het in .eslintrc.*
Open je ESLint config (bijv. .eslintrc.json) en zet de env:




JSON
{
  "env": {
    "node": true,
    "es2021": true
  },
  "parserOptions": {
    "sourceType": "module"
  }
}
node: true geeft toegang tot Node‑globals; sourceType: "module" past bij .mjs/ESM.

Oplossing C: als je bewust in de browser runt
Wil je het script in de browser draaien, dan bestaan process en soms console niet (of anders). Gebruik dan een polyfill of switch naar een bundler die process.env injecteert (zoals Vite/Webpack via DefinePlugin). Maar gezien je .mjs pad en context, is Node hier het meest logisch.

2) no-useless-escape: “Unnecessary escape character: -”
Waarom gebeurt dit?
Je hebt ergens een regex met een ge‑escape‑te - (backslash‑minus). ESLint zegt dat het escape‑teken overbodig is — of andersom: je gebruikt - in een character class ([...]) op een plek waar het gezien wordt als range. Dan is ofwel de escape niet nodig, ofwel moet je de - verplaatsen.

Snelle regel:
Buiten een character class ([...]), dus in een gewone regex zoals /-/, hoef je - niet te escapen.
Binnen een character class, als je een letterlijke - wilt, zet je hem:
helemaal vooraan of achteraan: [-A-Z] of [A-Z-], of
escapen: [A-Z\-].
Voorbeelden van correcte vormen
Goed (buiten char class):



JavaScript
const r1 = /-/;           // '-' letterlijk, geen escape nodig
Goed (binnen char class, achteraan):



JavaScript
const r2 = /[A-Za-z0-9-]/; // '-' achteraan, geen escape nodig
Goed (binnen char class, met escape):



JavaScript
const r3 = /[A-Za-z0-9\-]/; // '-' geëscapet, expliciet
Wat jij moet aanpassen op de gemelde regels (40 en 42)
Zonder je exacte regex is de veilige fix meestal: verplaats de - naar het einde van je character class of haal de backslash weg als je niet in een character class zit.

Voorbeeld patch (typisch scenario):




Diff
- const pattern = /[A-Za-z0-9\-.]/;   // \- triggert 'no-useless-escape'
+ const pattern = /[A-Za-z0-9.-]/;    // '-' achteraan, escape niet nodig

- const other = /[\-_.]/;             // \- triggert 'no-useless-escape'
+ const other = /[-_.]/;               // '-' vooraan, escape niet nodig
Als jouw regex juist een range bedoelde (bijv. [A-Z\-] waar je alleen een letterlijke - wilde), laat dan de escape staan en zet de - helemaal vooraan/achteraan. Het doel is dat ESLint niet meer denkt dat \- overbodig is én dat je intentie (letterlijke minus) behouden blijft.

(() => {
  // 1) Probeer de hoofdcontainer(s)
  const chatRoot =
    document.querySelector('chat-app#app-root') ||
    document.querySelector('chat-app') ||
    document.querySelector('main.chat-app') ||
    document.body;

  if (!chatRoot) {
    console.warn('Geen chatcontainer gevonden. Controleer selector zoals chat-app of main.chat-app.');
    return;
  }

  // 2) Haal zichtbare tekst op (innerText respecteert CSS zichtbaarheid)
  let rawText = chatRoot.innerText || '';

  // 3) Optioneel: extra opschoning / normalisatie van whitespace
  //    - vervang meerdere lege regels door één
  //    - trim spaties aan lijn-einde en begin
  //    - normaliseer Windows/Mac/Unix line endings
  const cleanedText = rawText
    .replace(/\r\n/g, '\n')         // CRLF -> LF
    .replace(/\u00A0/g, ' ')        // non-breaking space -> gewone spatie
    .split('\n')
    .map(line => line.trimEnd())
    .join('\n')
    .replace(/\n{3,}/g, '\n\n')     // >2 lege regels terugbrengen naar 1 lege regel
    .trim();

  if (!cleanedText) {
    console.warn('Er werd geen tekst gevonden. Scroll eerst door het volledige gesprek (lazy-loading), en voer de snippet opnieuw uit.');
    return;
  }

  // 4) Maak een downloadbaar .txt-bestand
  const fileName = 'gemini-conversatie.txt';
  const blob = new Blob([cleanedText], { type: 'text/plain;charset=utf-8' });

  // 5) Forceer download via een tijdelijk <a> element
  const a = document.createElement('a');
  const url = URL.createObjectURL(blob);
  a.href = url;
  a.download = fileName;
  a.style.display = 'none';
  document.body.appendChild(a);
  a.click();

  // 6) Opruimen
  setTimeout(() => {
    URL.revokeObjectURL(url);
    a.remove();
    console.info(`✅ Gedownload: ${fileName} (regels: ${cleanedText.split('\n').length})`);
  }, 100);
})();

  ].join(',');

  // === 1) Vind root & kandidaat-berichten ===================================
  const root = getRoot();
  if (!root) { console.warn('Geen chat-root gevonden.'); return; }

  // Kandidaten in DOM-volgorde
  let candidates = Array.from(root.querySelectorAll(MESSAGE_SELECTOR))
    .filter(el => textOf(el).length > 0);

  // Alleen top-level messages (geen nested duplicaten)
  const topLevel = [];
  for (const el of candidates) {
    const containedInKept = topLevel.some(parent => parent !== el && parent.contains(el));
    if (!containedInKept) topLevel.push(el);
  }

  // === 2) Fallback als we te weinig nodes herkennen ==========================
  let items = [];
  if (topLevel.length >= 2) {
    items = topLevel.map((el, i) => ({ el, raw: textOf(el), idx: i }));
  } else {
    // Fallback: splits hele chat-tekst op blokken (2+ lege regels)
    const rawAll = textOf(root);
    const blocks = rawAll.split(/\n{2,}/).map(b => b.trim()).filter(b => b.length > 0);
    items = blocks.map((raw, i) => ({ el: null, raw, idx: i }));
  }

  if (!items.length) { console.warn('Geen berichten gevonden. Scroll eerst alles in beeld en probeer opnieuw.'); return; }

  // === 3) Label per bericht bepalen =========================================
  const labeled = [];
  let prev = 'User';
  for (const it of items) {
    const speaker = detectSpeaker(it.el, it.idx, prev);
    labeled.push({ speaker, text: it.raw });
    prev = speaker;
  }

  // === 4) Minimal cleanup / samenstellen output ==============================
  const lines = [];
  labeled.forEach(({ speaker, text }) => {
    // Optioneel: kort trimmen van extra lege regels aan begin/einde
    const block = norm(text);
    if (block) {
      lines.push(`${speaker}:`);
      lines.push(block);
      lines.push(''); // lege regel tussen berichten
    }
  });
  const output = lines.join('\n');

  if (!output.trim()) { console.warn('Transcript leeg.'); return; }

  a.href = url; a.download = FILENAME; a.style.display='none';
  document.body.appendChild(a); a.click();
  setTimeout(() => { URL.revokeObjectURL(url); a.remove(); }, 100);

(function () {
  // ---------- helpers ----------
  const pad = n => String(n).padStart(2, '0');
  const stamp = () => {
    const d = new Date();
    return `${d.getFullYear()}${pad(d.getMonth()+1)}${pad(d.getDate())}-${pad(d.getHours())}${pad(d.getMinutes())}`;
  };
  const norm = (s='') => (s||'')
    .replace(/\r\n/g, '\n')
    .replace(/\u00A0/g, ' ')
    .split('\n').map(l => l.trimEnd()).join('\n')
    .replace(/\n{3,}/g, '\n\n')
    .trim();
  const textOf = el => norm(el?.innerText || '');
  const getRoot = () =>
    document.querySelector('chat-app#app-root') ||
    document.querySelector('chat-app') ||
    document.querySelector('main.chat-app') ||
    document.querySelector('[role="feed"]') ||
    document.body;

  const MESSAGE_SELECTOR = [
    '[role="listitem"]',
    'article',
    'li',
    'div[class*="message"]',
    'div[class*="msg"]',
    'div[class*="bubble"]',
    'div[class*="prompt"]',
    'div[class*="response"]',
    'chat-message',
    'bard-message',
    'gemini-message'
  ].join(',');

  const detectSpeaker = (el, i, prev='User') => {
    const val = (...xs) => xs.filter(Boolean).map(x => String(x).toLowerCase()).join(' | ');
    const cls = el?.className ? String(el.className).toLowerCase() : '';
    const role = el?.getAttribute?.('role');
    const who  = val(
      role,
      el?.getAttribute?.('data-author'),
      el?.getAttribute?.('data-actor'),
      el?.getAttribute?.('data-sender'),
      el?.getAttribute?.('aria-label'),
      el?.getAttribute?.('aria-labelledby'),
      cls
    );
    const userHints = /(user|you|me|self|owner|author|prompt|request|from-user)/i;
    const aiHints   = /(assistant|bot|ai|model|gemini|bard|response|reply|answer|assistant-message)/i;

    if (userHints.test(who)) return 'User';
    if (aiHints.test(who))   return 'AI';
    if (role === 'user')      return 'User';
    if (role === 'assistant') return 'AI';
    if (i === 0) return 'User';
    return prev === 'User' ? 'AI' : 'User';
  };

    });

        lines.push(block);
        lines.push('');
      }
    });

  }

🧩 Optie B — Tampermonkey “live” logger (pakt elk nieuw bericht automatisch)
Als je gesprekken voortdurend wilt vastleggen — inclusief “aha‑momenten” zodra ze ontstaan — dan is een MutationObserver ideaal. Deze userscript:

zet een zwevende knop: “Export → Labeled TXT/JSON”
logt live elk nieuw message‑element met {speaker, text, timestamp} in window.__chatLog
exporteert (TXT of JSON) wanneer jij klikt
Installatie: Tampermonkey → “Nieuw script” → plak onderstaande → Opslaan → open Gemini.




JavaScript
// ==UserScript==
// @name         Gemini Live Chat Logger (User/AI + timestamps)
// @namespace    manasse.tools
// @version      1.0.0
// @description  Logt elk nieuw bericht (User/AI) met tijdstempel; exporteer naar TXT/JSON
// @match        https://gemini.google.com/*
// @grant        none
// ==/UserScript==
(function () {
  // ---------- helpers ----------
  const pad = n => String(n).padStart(2,'0');
  const stamp = () => {
    const d = new Date();
    return `${d.getFullYear()}${pad(d.getMonth()+1)}${pad(d.getDate())}-${pad(d.getHours())}${pad(d.getMinutes())}`;
  };
  const tsISO = () => new Date().toISOString();
  const norm = (s='') => (s||'')
    .replace(/\r\n/g, '\n')
    .replace(/\u00A0/g, ' ')
    .split('\n').map(l => l.trimEnd()).join('\n')
    .replace(/\n{3,}/g, '\n\n')
    .trim();
  const textOf = el => norm(el?.innerText || '');
  const getRoot = () =>
    document.querySelector('chat-app#app-root') ||
    document.querySelector('chat-app') ||
    document.querySelector('main.chat-app') ||
    document.querySelector('[role="feed"]') ||
    document.body;

  const MSG_SELECTOR = [
    '[role="listitem"]',
    'article', 'li',
    'div[class*="message"]', 'div[class*="msg"]',
    'div[class*="bubble"]',
    'div[class*="prompt"]',
    'div[class*="response"]',
    'chat-message', 'bard-message', 'gemini-message'
  ].join(',');

  // Spreker-detectie (zoals eerder)
  const detectSpeaker = (el, index, prev='User') => {
    const val = (...xs) => xs.filter(Boolean).map(x => String(x).toLowerCase()).join(' | ');
    const cls = el?.className ? String(el.className).toLowerCase() : '';
    const role = el?.getAttribute?.('role');
    const who  = val(
      role,
      el?.getAttribute?.('data-author'),
      el?.getAttribute?.('data-actor'),
      el?.getAttribute?.('data-sender'),
      el?.getAttribute?.('aria-label'),
      el?.getAttribute?.('aria-labelledby'),
      cls
    );
    const userHints = /(user|you|me|self|owner|author|prompt|request|from-user)/i;
    const aiHints   = /(assistant|bot|ai|model|gemini|bard|response|reply|answer|assistant-message)/i;

    if (userHints.test(who)) return 'User';
    if (aiHints.test(who))   return 'AI';
    if (role === 'user')      return 'User';
    if (role === 'assistant') return 'AI';
    if (index === 0) return 'User';
    return prev === 'User' ? 'AI' : 'User';
  };

  // Globale logbuffer
  window.__chatLog = window.__chatLog || [];

  // Observeer nieuwe berichten
  function startObserver() {
    const root = getRoot();
    if (!root) return;
    let prev = window.__chatLog.length ? window.__chatLog[window.__chatLog.length-1].speaker : 'User';

    const observer = new MutationObserver(muts => {
      muts.forEach(m => {
        const added = Array.from(m.addedNodes || []).filter(n => n.nodeType === 1);
        added.forEach(node => {
          // Vind mogelijke message elementen binnen de nieuw toegevoegde node
          const messages = node.matches?.(MSG_SELECTOR) ? [node] : Array.from(node.querySelectorAll?.(MSG_SELECTOR) || []);
          messages.forEach((el, idx) => {
            const txt = textOf(el);
            if (!txt) return;
            const speaker = detectSpeaker(el, window.__chatLog.length, prev);
            prev = speaker;
            window.__chatLog.push({ timestamp: tsISO(), speaker, text: txt });
          });
        });
      });
    });

    observer.observe(root, { childList: true, subtree: true });
    console.info('✅ Live logger gestart');
  }

  // Export-knoppen
  function exportTXT() {
    const lines = [];
    window.__chatLog.forEach(({ timestamp, speaker, text }) => {
      lines.push(`[${timestamp}] ${speaker}:`);
      lines.push(text);
      lines.push('');
    });
    const out = lines.join('\n');
    const blob = new Blob([out], { type: 'text/plain;charset=utf-8' });
    const url  = URL.createObjectURL(blob);
    const a    = document.createElement('a');
    a.href = url; a.download = `gemini-chat-labeled-${stamp()}.txt`;
    document.body.appendChild(a); a.click();
    setTimeout(() => { URL.revokeObjectURL(url); a.remove(); }, 100);
  }
  function exportJSON() {
    const out = JSON.stringify(window.__chatLog, null, 2);
    const blob = new Blob([out], { type: 'application/json;charset=utf-8' });
    const url  = URL.createObjectURL(blob);
    const a    = document.createElement('a');
    a.href = url; a.download = `gemini-chat-labeled-${stamp()}.json`;
    document.body.appendChild(a); a.click();
    setTimeout(() => { URL.revokeObjectURL(url); a.remove(); }, 100);
  }

  // UI
  function addButtons() {
    if (document.getElementById('mc-gemini-export-txt')) return;
    const wrap = document.createElement('div');
    Object.assign(wrap.style, {
      position:'fixed', right:'16px', bottom:'16px', zIndex: 999999,
      display:'flex', gap:'8px'
    });

    const btnTxt = document.createElement('button');
    btnTxt.id = 'mc-gemini-export-txt';
    btnTxt.textContent = 'Export → Labeled TXT';
    Object.assign(btnTxt.style, {
      background:'#1a73e8', color:'#fff', border:'none', borderRadius:'10px',
      padding:'10px 14px', fontSize:'14px', cursor:'pointer',
      boxShadow:'0 6px 18px rgba(0,0,0,.25)', fontFamily:'system-ui,-apple-system,Segoe UI,Roboto'
    });
    btnTxt.onclick = exportTXT;

    const btnJson = document.createElement('button');
    btnJson.id = 'mc-gemini-export-json';
    btnJson.textContent = 'Export → JSON';
    Object.assign(btnJson.style, {
      background:'#34a853', color:'#fff', border:'none', borderRadius:'10px',
      padding:'10px 14px', fontSize:'14px', cursor:'pointer',
      boxShadow:'0 6px 18px rgba(0,0,0,.25)', fontFamily:'system-ui,-apple-system,Segoe UI,Roboto'
    });
    btnJson.onclick = exportJSON;

    wrap.appendChild(btnTxt);
    wrap.appendChild(btnJson);
    document.body.appendChild(wrap);
  }

  // Init
  const run = () => { startObserver(); addButtons(); };
  if (document.readyState === 'loading') {
    document.addEventListener('DOMContentLoaded', run);
  } else {
    run();
  }
})();

import re
from pathlib import Path
import json

raw = Path("chattie.txt").read_text(encoding="utf-8", errors="ignore")

# 1) Split op ARTICLE-blokken
article_re = re.compile(r'(?m)^localName\s*:\s*"article"\s*$[\s\S]*?(?=^localName\s*:|^$)')
chunks = article_re.findall(raw)

def pick(pattern, text, group=1):
    m = re.search(pattern, text, flags=re.M|re.S)
    return m.group(group).strip() if m else None

items = []
for i, ch in enumerate(chunks):
    role = pick(r"dataset\s*:\s*DOMStringMap\s*\{[^}]*\bturn:\s*'([^']+)'", ch)
    testid = pick(r"dataset\s*:\s*DOMStringMap\s*\{[^}]*\btestid:\s*'conversation-turn-(\d+)'", ch)
    turn_id = pick(r"dataset\s*:\s*DOMStringMap\s*\{[^}]*\bturnId:\s*'([^']+)'", ch)

    # Prefer outerText > innerText > textContent
    text = (pick(r'(?ms)outerText\s*:\s*"([^"]+)"', ch) or
            pick(r'(?ms)innerText\s*:\s*"([^"]+)"', ch) or
            pick(r'(?ms)textContent\s*:\s*"([^"]+)"', ch) or "")

    # Basic cleanup
    # 1) verwijder SR-labelregel
    text = re.sub(r'^(Je zei:|ChatGPT zei:)\s*\n?', '', text)
    # 2) verwijder 'Code kopiëren' noise
    text = re.sub(r'\bCode kopiëren\b', '', text)
    # 3) herstel dubbele backslashes-artefacten
    text = text.replace('\\\\n', '\n').replace('\\n', '\n')

    # index voor sorteren
    idx = int(testid) if testid and testid.isdigit() else None

    items.append({
        "index": idx if idx is not None else 10_000 + i,  # fallback-stabiel
        "role": role or "unknown",
        "turnId": turn_id,
        "text": text.strip()
    })

# 2) sorteer en exporteer
items.sort(key=lambda x: x["index"])

# Markdown-transcript
md_lines = []
for it in items:
    speaker = "Je" if it["role"] == "user" else ("ChatGPT" if it["role"] == "assistant" else "Onbekend")
    md_lines.append(f"**{speaker}:**\n\n{it['text']}\n")

Path("transcript.md").write_text("\n---\n".join(md_lines), encoding="utf-8")
Path("transcript.json").write_text(json.dumps(items, ensure_ascii=False, indent=2), encoding="utf-8")

print("✅ transcript.md & transcript.json gegenereerd.")

function toMarkdownFromArticle(a){
  const blocks = [];
  for (const node of a.querySelectorAll(':scope > *')) {
    if (node.matches('pre code')) {
      // taal (als aanwezig via class of data-lbl)
      const lang = (node.className.match(/language-([a-z0-9]+)/i)||[])[1] || '';
      blocks.push('```'+lang+'\n'+node.textContent.replace(/\u00A0/g,' ')+'\n```');
    } else if (node.matches('ul,ol')) {
      blocks.push([...node.querySelectorAll('li')].map(li => '- '+li.textContent).join('\n'));
    } else if (node.matches('table')) {
      // eenvoudige tabel -> Markdown
      const rows=[...node.rows].map(r=>[...r.cells].map(c=>c.textContent.trim()));
      if (rows.length) {
        const header = rows[0].join(' | '), sep = rows[0].map(()=> '---').join(' | ');
        blocks.push(header+'\n'+sep+'\n'+rows.slice(1).map(r=>r.join(' | ')).join('\n'));
      }
    } else {
      blocks.push(node.textContent);
    }
  }
  return blocks.join('\n\n');
}

import fs from 'fs';
import os from 'os';
import path from 'path';
import { chromium } from 'playwright';

// --- 1) Edge-profiel bepalen (pas aan als nodig) ---
function edgeUserDataDir() {
  const p = process.platform;
  if (p === 'win32') {
    const home = process.env.USERPROFILE || os.homedir();
    return path.join(home, 'AppData', 'Local', 'Microsoft', 'Edge', 'User Data');
  } else if (p === 'darwin') {
    return path.join(os.homedir(), 'Library', 'Application Support', 'Microsoft Edge');
  } else {
    // Linux
    return path.join(os.homedir(), '.config', 'microsoft-edge');
  }
}

// --- 2) CLI-URL lezen ---
const argUrl = process.argv.find(a => a.startsWith('--url='))
  ? process.argv.find(a => a.startsWith('--url=')).split('=')[1]
  : process.argv[process.argv.indexOf('--url') + 1];

if (!argUrl) {
  console.error('❌ Gebruik: node export-chatgpt.js --url "<chat-url>"');
  process.exit(1);
}

// --- 3) Helpers voor sanitizing en wachten ---
const sleep = (ms) => new Promise(r => setTimeout(r, ms));
const safe = (s) => (s || 'export').replace(/[\\/:*?"<>|]+/g, '_').slice(0, 70);

const pageExtractor = `
(async () => {
  // Vind scrollcontainer (probeer meerdere bekende containers)
  function findScroller() {
    const candidates = [
      document.querySelector('[data-testid="conversation"]'),
      document.querySelector('main'),
      document.scrollingElement,
      document.body
    ].filter(Boolean);

    // Kies degene met grootste scrollHeight
    let best = candidates[0], bestH = 0;
    for (const el of candidates) {
      const sh = el.scrollHeight || 0;
      if (sh >= bestH) { best = el; bestH = sh; }
    }
    return best || document.body;
  }

  async function autoLoadAll(maxMs = 45000) {
    const scroller = findScroller();
    const start = Date.now();
    let prevCount = 0, stableHits = 0;

    function artCount() {
      return document.querySelectorAll('article[data-testid^="conversation-turn"], article[data-turn], article').length;
    }

    // Zig-zag scroll: boven, beneden, midden — totdat tel niet meer groeit
    while ((Date.now() - start) < maxMs) {
      const before = artCount();
      scroller.scrollTo({ top: 0, behavior: 'instant' });
      await new Promise(r => setTimeout(r, 900));
      scroller.scrollTo({ top: scroller.scrollHeight, behavior: 'instant' });
      await new Promise(r => setTimeout(r, 900));
      scroller.scrollTo({ top: Math.floor(scroller.scrollHeight / 2), behavior: 'instant' });
      await new Promise(r => setTimeout(r, 600));
      const after = artCount();

      if (after > prevCount) {
        prevCount = after;
        stableHits = 0;
      } else {
        stableHits++;
        if (stableHits >= 3) break; // 3 cycli zonder groei -> stop
      }
    }
  }

  function toMarkdownFromArticle(a) {
    const blocks = [];

    // verwijder SR-label (spreker)
    const sr = a.querySelector('h6.sr-only'); if (sr) sr.remove();

    const children = a.children.length ? Array.from(a.children) : [a];
    for (const node of children) {
      if (node.matches?.('pre, pre code')) {
        const codeEl = node.matches('code') ? node : node.querySelector('code');
        const langMatch = (codeEl?.className || '').match(/language-([a-z0-9]+)/i);
        const lang = langMatch ? langMatch[1] : '';
        const code = (codeEl?.textContent || node.textContent || '').replace(/\u00A0/g, ' ');
        blocks.push('```' + lang + '\\n' + code + '\\n```');
      } else if (node.matches?.('table')) {
        const rows = Array.from(node.rows).map(r => Array.from(r.cells).map(c => c.textContent.replace(/\\s+/g, ' ').trim()));
        if (rows.length) {
          const header = rows[0].join(' | ');
          const sep = rows[0].map(() => '---').join(' | ');
          const body = rows.slice(1).map(r => r.join(' | ')).join('\\n');
          blocks.push(header + '\\n' + sep + '\\n' + body);
        }
      } else if (node.matches?.('ul,ol')) {
        const list = Array.from(node.querySelectorAll('li')).map(li => '- ' + (li.textContent || '').trim()).join('\\n');
        if (list.trim()) blocks.push(list);
      } else {
        const txt = (node.textContent || '').replace(/\\bCode kopiëren\\b/gi, '').trim();
        if (txt) blocks.push(txt);
      }
    }
    return blocks.join('\\n\\n').trim();
  }

  function roleOf(a) {
    let role = a.getAttribute('data-turn') || a.dataset?.turn || '';
    if (!role) {
      const lbl = (a.querySelector('h6.sr-only')?.innerText || '').toLowerCase();
      role = (/je zei|you said/.test(lbl)) ? 'user' : 'assistant';
    }
    return role;
  }

  function indexOf(a, i) {
    const t = a.getAttribute('data-testid') || '';
    const m = t.match(/conversation-turn-(\\d+)/);
    return m ? parseInt(m[1], 10) : i;
  }

  await autoLoadAll(45000);

  const arts = Array.from(document.querySelectorAll('article[data-testid^="conversation-turn"], article[data-turn]'));
  const items = arts.map((a, i) => ({
    index: indexOf(a, i),
    role: roleOf(a),
    text: toMarkdownFromArticle(a)
  })).filter(x => x.text);

  items.sort((a, b) => a.index - b.index);

  const title = document.title || 'export';
  return { items, title };
})();
`;

  } catch (e) {
    console.error('💥 Fout:', e.message);
  } finally {
    await ctx.close();
  }
})();

JavaScript
// ==UserScript==
// @name         ChatGPT Conversation Exporter (MD/JSON)
// @namespace    manasse.tools.chatgpt
// @version      1.2.0
// @description  Exporteer ChatGPT-gesprekken naar Markdown/JSON met lazy-load en juiste scrollcontainer
// @author       Manasse + Copilot
// @match        https://chatgpt.com/*
// @match        https://chat.openai.com/*
// @run-at       document-idle
// @grant        none
// ==/UserScript==

(function () {
  "use strict";

  // ---------- Kleine helpers ----------
  const sleep = (ms) => new Promise((r) => setTimeout(r, ms));
  const safe = (s) => (s || "export").replace(/[\\/:*?"<>|]+/g, "_").slice(0, 70);

  function el(tag, cssText = "", text = "") {
    const n = document.createElement(tag);
    if (cssText) n.style.cssText = cssText;
    if (text) n.textContent = text;
    return n;
  }

  // ---------- UI toevoeging ----------
  function ensureButton() {
    if (document.getElementById("cgpt-exporter-btn")) return;

    const box = el(
      "div",
      "position:fixed;top:10px;right:10px;z-index:2147483647;display:flex;gap:8px;" +
      "background:#0e0e0e;border:1px solid #3a3a3a;border-radius:10px;padding:8px;" +
      "font:12px/1 system-ui;color:#eee;box-shadow:0 4px 18px rgba(0,0,0,.35)"
    );
    box.id = "cgpt-exporter-btn";

    const mdBtn = el(
      "button",
      "background:#1e90ff;color:#fff;border:none;border-radius:6px;padding:7px 10px;cursor:pointer;"
      + "font-weight:600"
    , "Export Chat (MD)");
    const jsonBtn = el(
      "button",
      "background:#2aa84a;color:#fff;border:none;border-radius:6px;padding:7px 10px;cursor:pointer;"
      + "font-weight:600"
    , "Export Chat (JSON)");

    const status = el(
      "div",
      "margin-left:4px;color:#ccc;min-width:180px;font-weight:500"
    );

    mdBtn.addEventListener("click", async () => {
      status.textContent = "Lazy-load & exporteren (MD)…";
      try {
        const data = await collectConversation(status);
        const md = toMarkdown(data.items);
        downloadBlob(md, "text/markdown;charset=utf-8", `chatgpt_conversation_${safe(data.title)}.md`);
        status.textContent = "✅ MD klaar";
      } catch (e) {
        console.error(e);
        status.textContent = "❌ Fout (MD): " + e.message;
      }
    });

    jsonBtn.addEventListener("click", async () => {
      status.textContent = "Lazy-load & exporteren (JSON)…";
      try {
        const data = await collectConversation(status);
        downloadBlob(JSON.stringify(data.items, null, 2), "application/json;charset=utf-8",
          `chatgpt_conversation_${safe(data.title)}.json`);
        status.textContent = "✅ JSON klaar";
      } catch (e) {
        console.error(e);
        status.textContent = "❌ Fout (JSON): " + e.message;
      }
    });

    box.append(mdBtn, jsonBtn, status);
    document.body.appendChild(box);
  }

  function downloadBlob(content, type, filename) {
    const blob = new Blob([content], { type });
    const a = document.createElement("a");
    a.href = URL.createObjectURL(blob);
    a.download = filename;
    document.body.appendChild(a);
    a.click();
    setTimeout(() => URL.revokeObjectURL(a.href), 1500);
    a.remove();
  }

  // ---------- Core scraping ----------
  function findScroller() {
    const candidates = [
      document.querySelector('[data-testid="conversation"]'),
      document.querySelector('main'),
      document.scrollingElement,
      document.body
    ].filter(Boolean);

    // Grootste scrollHeight = meest waarschijnlijke container
    return candidates.sort((a, b) => (b?.scrollHeight || 0) - (a?.scrollHeight || 0))[0] || document.body;
  }

  function countArticles() {
    return document.querySelectorAll('article[data-testid^="conversation-turn"], article[data-turn], article').length;
  }

  async function lazyLoadAll(statusEl, maxMs = 45000) {
    const scroller = findScroller();
    const start = Date.now();
    let prev = 0, stable = 0;

    // Zig-zag scroll (boven -> onder -> midden) totdat aantal artikelen niet meer groeit
    while ((Date.now() - start) < maxMs) {
      const before = countArticles();
      scroller.scrollTo({ top: 0, behavior: "instant" });         await sleep(900);
      scroller.scrollTo({ top: scroller.scrollHeight, behavior: "instant" }); await sleep(900);
      scroller.scrollTo({ top: Math.floor(scroller.scrollHeight / 2), behavior: "instant" }); await sleep(600);
      const after = countArticles();
      if (statusEl) statusEl.textContent = `🔄 Laden… artikelen: ${after}`;

      if (after > prev) { prev = after; stable = 0; }
      else if (++stable >= 3) break; // 3 cycli zonder groei -> klaar
    }
    if (statusEl) statusEl.textContent = `✅ Lazy-load klaar (${countArticles()} artikelen)`;
  }

  function roleOf(a) {
    let role = a.getAttribute("data-turn") || a.dataset?.turn || "";
    if (!role) {
      const lbl = (a.querySelector("h6.sr-only")?.innerText || "").toLowerCase();
      role = (/je zei|you said/.test(lbl)) ? "user" : "assistant";
    }
    return role;
  }

  function indexOf(a, i) {
    const t = a.getAttribute("data-testid") || "";
    const m = t.match(/conversation-turn-(\d+)/);
    return m ? parseInt(m[1], 10) : i;
  }

  function toMarkdownFromArticle(a) {
    const blocks = [];
    // verwijder SR-label (spreker)
    const sr = a.querySelector("h6.sr-only");
    if (sr) sr.remove();

    // Neem directe kinderen; zo niet, neem het element zelf
    const children = a.children.length ? Array.from(a.children) : [a];

    const push = (t) => { if (t && t.trim()) blocks.push(t.trim()); };

    for (const node of children) {
      if (node.matches?.("pre, pre code")) {
        const codeEl = node.matches("code") ? node : node.querySelector("code");
        const langMatch = (codeEl?.className || "").match(/language-([a-z0-9]+)/i);
        const lang = langMatch ? langMatch[1] : "";
        const code = (codeEl?.textContent || node.textContent || "").replace(/\u00A0/g, " ");
        push("```" + lang + "\n" + code + "\n```");
      } else if (node.matches?.("table")) {
        const rows = Array.from(node.rows).map(r =>
          Array.from(r.cells).map(c => c.textContent.replace(/\s+/g, " ").trim())
        );
        if (rows.length) {
          const header = rows[0].join(" | ");
          const sep = rows[0].map(() => "---").join(" | ");
          const body = rows.slice(1).map(r => r.join(" | ")).join("\n");
          push(header + "\n" + sep + "\n" + body);
        }
      } else if (node.matches?.("ul,ol")) {
        const list = Array.from(node.querySelectorAll("li"))
          .map(li => "- " + (li.textContent || "").trim()).join("\n");
        push(list);
      } else {
        push((node.textContent || "").replace(/\bCode kopiëren\b/gi, ""));
      }
    }
    return blocks.join("\n\n").trim();
  }

  async function collectConversation(statusEl) {
    if (!/chatgpt\.com|chat\.openai\.com/.test(location.hostname)) {
      throw new Error("Open dit op chatgpt.com of chat.openai.com");
    }

    // 1) Lazy-load volledige chat
    await lazyLoadAll(statusEl, 45000);

    // 2) Selecteer artikelen
    let arts = [...document.querySelectorAll('article[data-testid^="conversation-turn"], article[data-turn]')];
    if (!arts.length) arts = [...document.querySelectorAll("article")];
    if (!arts.length) throw new Error("Geen beurten gevonden. Scroll even en probeer opnieuw.");

    // 3) Items + sorteren
    const items = arts.map((a, i) => {
      const idx = indexOf(a, i);
      const role = roleOf(a);
      const text = toMarkdownFromArticle(a);
      return { index: idx, role, text };
    }).filter(x => x.text).sort((a, b) => a.index - b.index);

    const title = document.title || "export";
    return { items, title };
  }

  function toMarkdown(items) {
    return items
      .map(it => `**${it.role === "user" ? "User" : "ChatGPT"}:**\n\n${it.text}`)
      .join("\n\n---\n\n");
  }

  // ---------- Observer om knop terug te plaatsen bij navigatie ----------
  const observer = new MutationObserver(() => {
    // Zet de knop pas als er content is
    if (document.body && !document.getElementById("cgpt-exporter-btn")) {
      ensureButton();
    }
  });
  observer.observe(document.documentElement, { childList: true, subtree: true });

  // Eerste run
  ensureButton();
})();

⚠️ Gaps & risico’s (wat ik bij scherp nalezen zie)
IBAN‑regex “ruw”
Regex \b[A-Z]{2}\d{2}[A-Z0-9]{4}\d{7,}\b is ruim (goed als generieke mask), maar:

Soms maskeer je meer dan strikt nodig; dat is privacy‑veilig, maar kan ook niet‑IBAN strings raken.
➜ Aanbeveling: maskeer pattern‑match, maar laat laatste 4 zichtbaar (NL**[REDACTED]-1234) voor traceability in interne debugging.
toCents & dubbele conversie
In de CSV‑hybride wordt Number(value) gegeven aan toCents. Als jouw toCents string verwacht, let op dat je niet dubbel converteert.
➜ Aanbeveling: definieer één contract: toCents(euros: number|string) → cents:number en test strings (met locale) én numbers.

Mutatie‑labels varianten
Nu gedekt: 'af', 'debit', '-', 'd'.
➜ Uitbreiden met: 'bij', 'credit', '+', 'CR'. (Met fallback: signed amount wint altijd → teken is leading.)

Datum‑fallback
Fallback '1970-01-01' is deterministisch, maar misschien wil je liever null of expliciet 'UNKNOWN' zodat downstream logica het niet als echte datum ziet.
➜ Aanbeveling: kies één beleid en documenteer het; UI kan “datum ontbreekt” tonen.

Categorisatie via .includes
Simpel en werkt; maar false positives zijn denkbaar (“woon” in “woonkamerlamp”).
➜ Aanbeveling: gebruik woordgrenzen/regex (bv. \bwoon\b) of een lemmatiseringslijst later.

Test‑mock injectie
Je test faalde omdat mockDispatch niet echt in de context kwam of omdat events niet met testID worden getriggerd.
➜ Check: in FormProvider, dispatch = mockDispatch ?? dispatch; en voeg testID in LandingScreen.

JavaScript
// ==UserScript==
// @name         ChatGPT Exporter (Clean Markdown)
// @namespace    manasse.tools.chatgpt
// @version      1.0.0
// @description  Exporteer ChatGPT-gesprekken naar schone Markdown (zonder artefacten, met fenced codeblocks)
// @author       Manasse + Copilot
// @match        https://chatgpt.com/*
// @match        https://chat.openai.com/*
// @run-at       document-idle
// @grant        none
// ==/UserScript==

(function () {
  "use strict";

  const sleep = (ms) => new Promise((r) => setTimeout(r, ms));
  const safe = (s) => (s || "export").replace(/[\\/:*?"<>|]+/g, "_").slice(0, 70);

  function el(tag, cssText = "", text = "") {
    const n = document.createElement(tag);
    if (cssText) n.style.cssText = cssText;
    if (text) n.textContent = text;
    return n;
  }

  function ensureButton() {
    if (document.getElementById("cgpt-exporter-btn")) return;

    const box = el(
      "div",
      "position:fixed;top:10px;right:10px;z-index:999999;display:flex;gap:8px;" +
        "background:#0e0e0e;border:1px solid #3a3a3a;border-radius:10px;padding:8px;" +
        "font:12px system-ui;color:#eee;box-shadow:0 4px 18px rgba(0,0,0,.35)"
    );
    box.id = "cgpt-exporter-btn";

    const mdBtn = el(
      "button",
      "background:#1e90ff;color:#fff;border:none;border-radius:6px;padding:7px 10px;cursor:pointer;font-weight:600",
      "Export Chat (MD)"
    );
    const jsonBtn = el(
      "button",
      "background:#2aa84a;color:#fff;border:none;border-radius:6px;padding:7px 10px;cursor:pointer;font-weight:600",
      "Export Chat (JSON)"
    );
    const status = el("div", "margin-left:4px;color:#ccc;min-width:180px;font-weight:500");

    mdBtn.addEventListener("click", async () => {
      status.textContent = "Lazy-load & exporteren (MD)…";
      try {
        const data = await collectConversation(status);
        const md = toMarkdown(data.items);
        downloadBlob(md, "text/markdown;charset=utf-8", `chatgpt_conversation_${safe(data.title)}.md`);
        status.textContent = "✅ MD klaar";
      } catch (e) {
        console.error(e);
        status.textContent = "❌ Fout (MD): " + e.message;
      }
    });

    jsonBtn.addEventListener("click", async () => {
      status.textContent = "Lazy-load & exporteren (JSON)…";
      try {
        const data = await collectConversation(status);
        downloadBlob(JSON.stringify(data.items, null, 2), "application/json;charset=utf-8",
          `chatgpt_conversation_${safe(data.title)}.json`);
        status.textContent = "✅ JSON klaar";
      } catch (e) {
        console.error(e);
        status.textContent = "❌ Fout (JSON): " + e.message;
      }
    });

    box.append(mdBtn, jsonBtn, status);
    document.body.appendChild(box);
  }

  function downloadBlob(content, type, filename) {
    const blob = new Blob([content], { type });
    const a = document.createElement("a");
    a.href = URL.createObjectURL(blob);
    a.download = filename;
    document.body.appendChild(a);
    a.click();
    setTimeout(() => URL.revokeObjectURL(a.href), 1500);
    a.remove();
  }

  function findScroller() {
    const candidates = [
      document.querySelector('[data-testid="conversation"]'),
      document.querySelector("main"),
      document.scrollingElement,
      document.body,
    ].filter(Boolean);
    return candidates.sort((a, b) => (b.scrollHeight || 0) - (a.scrollHeight || 0))[0] || document.body;
  }

  function countArticles() {
    return document.querySelectorAll('article[data-testid^="conversation-turn"], article[data-turn], article').length;
  }

  async function lazyLoadAll(statusEl, maxMs = 45000) {
    const scroller = findScroller();
    const start = Date.now();
    let prev = 0, stable = 0;

    while ((Date.now() - start) < maxMs) {
      const before = countArticles();
      scroller.scrollTo({ top: 0, behavior: "instant" }); await sleep(900);
      scroller.scrollTo({ top: scroller.scrollHeight, behavior: "instant" }); await sleep(900);
      scroller.scrollTo({ top: Math.floor(scroller.scrollHeight / 2), behavior: "instant" }); await sleep(600);
      const after = countArticles();
      if (statusEl) statusEl.textContent = `🔄 Laden… artikelen: ${after}`;

      if (after > prev) { prev = after; stable = 0; }
      else if (++stable >= 3) break;
    }
    if (statusEl) statusEl.textContent = `✅ Lazy-load klaar (${countArticles()} artikelen)`;
  }

  function roleOf(a) {
    let role = a.getAttribute("data-turn") || a.dataset?.turn || "";
    if (!role) {
      const lbl = (a.querySelector("h6.sr-only")?.innerText || "").toLowerCase();
      role = (/je zei|you said/.test(lbl)) ? "user" : "assistant";
    }
    return role;
  }

  function indexOf(a, i) {
    const t = a.getAttribute("data-testid") || "";
    const m = t.match(/conversation-turn-(\d+)/);
    return m ? parseInt(m[1], 10) : i;
  }

  function toMarkdownFromArticle(a) {
    const blocks = [];
    const sr = a.querySelector("h6.sr-only"); if (sr) sr.remove();
    const children = a.children.length ? Array.from(a.children) : [a];
    const push = (t) => { if (t && t.trim()) blocks.push(t.trim()); };

    for (const node of children) {
      if (node.matches?.("pre, pre code")) {
        const codeEl = node.matches("code") ? node : node.querySelector("code");
        const langMatch = (codeEl?.className || "").match(/language-([a-z0-9]+)/i);
        const lang = langMatch ? langMatch[1] : "";
        const code = (codeEl?.textContent || node.textContent || "").replace(/\u00A0/g, " ");
        push("```" + lang + "\n" + code + "\n```");
      } else if (node.matches?.("table")) {
        const rows = Array.from(node.rows).map(r =>
          Array.from(r.cells).map(c => c.textContent.replace(/\s+/g, " ").trim())
        );
        if (rows.length) {
          const header = rows[0].join(" | ");
          const sep = rows[0].map(() => "---").join(" | ");
          const body = rows.slice(1).map(r => r.join(" | ")).join("\n");
          push(header + "\n" + sep + "\n" + body);
        }
      } else if (node.matches?.("ul,ol")) {
        const list = Array.from(node.querySelectorAll("li"))
          .map(li => "- " + (li.textContent || "").trim()).join("\n");
        push(list);
      } else {
        push((node.textContent || "").replace(/\bCode kopiëren\b/gi, ""));
      }
    }
    return blocks.join("\n\n").trim();
  }

  async function collectConversation(statusEl) {
    if (!/chatgpt\.com|chat\.openai\.com/.test(location.hostname)) {
      throw new Error("Open dit op chatgpt.com of chat.openai.com");
    }
    await lazyLoadAll(statusEl, 45000);
    let arts = [...document.querySelectorAll('article[data-testid^="conversation-turn"], article[data-turn]')];
    if (!arts.length) arts = [...document.querySelectorAll("article")];
    if (!arts.length) throw new Error("Geen beurten gevonden.");

    const items = arts.map((a, i) => {
      const idx = indexOf(a, i);
      const role = roleOf(a);
      const text = toMarkdownFromArticle(a);
      return { index: idx, role, text };
    }).filter(x => x.text).sort((a, b) => a.index - b.index);

    const title = document.title || "export";
    return { items, title };
  }

  function toMarkdown(items) {
    return items.map(it => `**${it.role === "user" ? "User" : "ChatGPT"}:**\n\n${it.text}`).join("\n\n---\n\n");
  }

  const observer = new MutationObserver(() => {
    if (document.body && !document.getElementById("cgpt-exporter-btn")) {
      ensureButton();
    }
  });
  observer.observe(document.documentElement, { childList: true, subtree: true });
  ensureButton();
})();

// ==UserScript==
// @name         Claude Exporter (Clean Markdown/JSON)
// @namespace    manasse.tools.claude
// @version      1.0.0
// @description  Exporteer Claude-chats naar schone Markdown/JSON (filters aria-hidden/overlays; tables/code netjes)
// @author       Manasse + Copilot
// @match        https://claude.ai/*
// @run-at       document-idle
// @grant        none
// ==/UserScript==

(function () {
  "use strict";

  // ---------- Helpers ----------
  const sleep = (ms) => new Promise((r) => setTimeout(r, ms));
  const safe = (s) => (s || "export").replace(/[\\/:*?"<>|]+/g, "_").slice(0, 70);
  const isOnChat = () => /\/chat\/[0-9a-f-]+$/i.test(location.pathname);

  function el(tag, cssText = "", text = "") {
    const n = document.createElement(tag);
    if (cssText) n.style.cssText = cssText;
    if (text) n.textContent = text;
    return n;
  }

  // ---------- UI ----------
  function ensureButton() {
    if (!isOnChat()) return; // alleen op chatpagina
    if (document.getElementById("claude-exporter-btn")) return;

    const box = el(
      "div",
      "position:fixed;top:10px;right:10px;z-index:2147483647;display:flex;gap:8px;" +
      "background:#0e0e0e;border:1px solid #3a3a3a;border-radius:10px;padding:8px;" +
      "font:12px system-ui;color:#eee;box-shadow:0 4px 18px rgba(0,0,0,.35)"
    );
    box.id = "claude-exporter-btn";

    const mdBtn = el(
      "button",
      "background:#1e90ff;color:#fff;border:none;border-radius:6px;padding:7px 10px;cursor:pointer;font-weight:600",
      "Export Chat (MD)"
    );
    const jsonBtn = el(
      "button",
      "background:#2aa84a;color:#fff;border:none;border-radius:6px;padding:7px 10px;cursor:pointer;font-weight:600",
      "Export Chat (JSON)"
    );
    const status = el("div", "margin-left:4px;color:#ccc;min-width:220px;font-weight:500");

    mdBtn.addEventListener("click", async () => {
      status.textContent = "Lazy-load & exporteren (MD)…";
      try {
        const data = await collectClaude(status);
        const md = toMarkdown(data.items, data.title);
        downloadBlob(md, "text/markdown;charset=utf-8", `claude_conversation_${safe(data.title)}.md`);
        status.textContent = "✅ MD klaar";
      } catch (e) {
        console.error(e);
        status.textContent = "❌ Fout (MD): " + (e?.message || e);
      }
    });

    jsonBtn.addEventListener("click", async () => {
      status.textContent = "Lazy-load & exporteren (JSON)…";
      try {
        const data = await collectClaude(status);
        downloadBlob(JSON.stringify(data.items, null, 2), "application/json;charset=utf-8",
          `claude_conversation_${safe(data.title)}.json`);
        status.textContent = "✅ JSON klaar";
      } catch (e) {
        console.error(e);
        status.textContent = "❌ Fout (JSON): " + (e?.message || e);
      }
    });

    box.append(mdBtn, jsonBtn, status);
    document.body.appendChild(box);
  }

  function downloadBlob(content, type, filename) {
    const blob = new Blob([content], { type });
    const a = document.createElement("a");
    a.href = URL.createObjectURL(blob);
    a.download = filename;
    document.body.appendChild(a);
    a.click();
    setTimeout(() => URL.revokeObjectURL(a.href), 1500);
    a.remove();
  }

  // ---------- Claude-specifiek ----------
  // Vind scrollcontainer met grootste scrollHeight (Claude gebruikt vaak interne containers)
  function findScroller() {
    const candidates = [
      document.querySelector("main"),
      document.querySelector('[role="main"]'),
      document.scrollingElement,
      document.body,
    ].filter(Boolean);
    return candidates.sort((a, b) => (b.scrollHeight || 0) - (a.scrollHeight || 0))[0] || document.body;
  }

  function isHiddenOrOverlay(node) {
    const st = window.getComputedStyle(node);
    const ariaHidden = node.getAttribute?.("aria-hidden") === "true";
    const peNone = st.pointerEvents === "none";
    const vis = st.visibility === "hidden" || st.display === "none";
    // overlays op Claude hebben vaak pointer-events:none en aria-hidden:true
    return ariaHidden || peNone || vis;
  }

  function countContentBlocks() {
    // Claude zet de beurtcontent vaak in article/divs binnen main; filter overlays
    const all = [...document.querySelectorAll("main article, main div")];
    return all.filter(n => !isHiddenOrOverlay(n) && (n.innerText || "").trim()).length;
  }

  async function lazyLoadAll(statusEl, maxMs = 45000) {
    const scroller = findScroller();
    const start = Date.now();
    let prev = 0, stable = 0;
    while ((Date.now() - start) < maxMs) {
      const before = countContentBlocks();
      scroller.scrollTo({ top: 0, behavior: "instant" }); await sleep(900);
      scroller.scrollTo({ top: scroller.scrollHeight, behavior: "instant" }); await sleep(900);
      scroller.scrollTo({ top: Math.floor(scroller.scrollHeight / 2), behavior: "instant" }); await sleep(600);
      const after = countContentBlocks();
      if (statusEl) statusEl.textContent = `🔄 Laden… contentblokken: ${after}`;
      if (after > prev) { prev = after; stable = 0; }
      else if (++stable >= 3) break;
    }
    if (statusEl) statusEl.textContent = `✅ Lazy-load klaar (${countContentBlocks()} blokken)`;
  }

  // Heuristiek: vind beurten (user/assistant) in main; filter aria-hidden/overlays; pak zichtbare tekst
  function collectTurns() {
    const main = document.querySelector("main") || document.body;
    // Claude markup varieert; we verzamelen zichtbare 'message'-containers met tekst
    const nodes = [...main.querySelectorAll("article, section, div")].filter(n => {
      const txt = (n.innerText || "").trim();
      return txt && !isHiddenOrOverlay(n);
    });

    // Grove heuristiek: splitsen op blokken die headers/titels/promptkenmerken en content bevatten
    // We dedupliceren door text-signature en minimaliseren lege/noise nodes.
    const seenSig = new Set();
    const items = [];
    let idx = 0;

    for (const n of nodes) {
      const txt = sanitizeText(n);
      const sig = txt.slice(0, 120); // signature
      if (!txt) continue;
      if (seenSig.has(sig)) continue; // dupe filter
      seenSig.add(sig);

      // rol-detectie: simpele heuristiek op basis van prefixen
      const role = detectRole(txt);
      const md = toMDFromNode(n, txt);
      items.push({ index: idx++, role, text: md });
    }

    // post-filter: verwijder pure navigatie/recents (veel links bovenin)
    const filtered = items.filter(it => !/^\s*(New chat|Chats|Projects|Artifacts|Code)\s*$/i.test(it.text));
    return filtered;
  }

  function detectRole(txt) {
    // Claude toont niet altijd expliciet "Je zei" of "Claude"; heuristiek:
    const lower = txt.toLowerCase();
    if (lower.startsWith("vraag ") || lower.startsWith("je bent") || lower.startsWith("kunnen we")) return "user";
    if (lower.includes("###") || lower.includes("ik adviseer") || lower.includes("n8n") || lower.includes("definitieve")) return "assistant";
    return "assistant";
  }

  function sanitizeText(node) {
    let t = (node.innerText || "").trim();
    // verwijder typische UI kopjes / waarschuwingen die geen inhoud zijn
    t = t.replace(/\b(Claude is AI and can make mistakes\.)\b.*$/i, "").trim();
    return t;
  }

  function toMDFromNode(node, fallbackText) {
    // Prioriteit: structurele children naar Markdown; anders fallback innerText.
    const blocks = [];
    const push = (s) => { if (s && s.trim()) blocks.push(s.trim()); };

    // 1) Tabellarische inhoud → Markdown table
    node.querySelectorAll("table").forEach(tbl => {
      const rows = [...tbl.rows].map(r => [...r.cells].map(c => c.textContent.replace(/\s+/g, " ").trim()));
      if (!rows.length) return;
      const header = rows[0].join(" | ");
      const sep = rows[0].map(() => "---").join(" | ");
      const body = rows.slice(1).map(r => r.join(" | ")).join("\n");
      push(header + "\n" + sep + "\n" + body);
    });

    // 2) Codeblokken (Claude gebruikt vaak <code>/<pre> of backticks in rich-text)
    const preBlocks = [
      ...node.querySelectorAll("pre, code")
    ];
    if (preBlocks.length) {
      preBlocks.forEach(pb => {
        const lang = guessLanguage(pb.className || "", pb.textContent || "");
        const code = (pb.textContent || "").replace(/\u00A0/g, " ");
        push("```" + lang + "\n" + code + "\n```");
      });
    }

    // 3) Lijsten
    node.querySelectorAll("ul,ol").forEach(list => {
      const li = [...list.querySelectorAll("li")].map(x => "- " + (x.textContent || "").trim()).join("\n");
      push(li);
    });

    // 4) Koppen/paragraphs
    node.querySelectorAll("h1,h2,h3,h4,h5,h6,p,blockquote").forEach(el => {
      const txt = (el.textContent || "").trim();
      if (!txt) return;
      if (/^\s*(New chat|Chats|Projects|Artifacts|Code)\s*$/.test(txt)) return;
      push(txt);
    });

    // fallback: gemiddeld is fallbackText al de complete innerText zonder UI-ruis
    if (!blocks.length) push(fallbackText);

    return blocks
      .join("\n\n")
      .replace(/\bCode kopiëren\b/gi, "")
      .replace(/\u00A0/g, " ")
      .trim();
  }

  function guessLanguage(cls, code) {
    const c = cls.toLowerCase();
    if (/language-(ts|typescript)/.test(c) || /\bexport\s+type\b/.test(code)) return "ts";
    if (/language-(js|javascript)/.test(c) || /\bfunction\b/.test(code)) return "js";
    if (/language-(py|python)/.test(c) || /\bdef\s+\w+\(/.test(code)) return "python";
    if (/language-(yaml|yml)/.test(c) || /^[\s-]*\w+:\s/.test(code)) return "yaml";
    if (/language-(json)/.test(c) || /^\s*\{[\s\S]*\}\s*$/.test(code)) return "json";
    if (/language-(html)/.test(c) || /<\/?[a-z][\s\S]*>/i.test(code)) return "html";
    if (/language-(csv)/.test(c) || /[,;]\s*[^,;]+/.test(code)) return "csv";
    return "";
  }

  async function collectClaude(statusEl) {
    if (!isOnChat()) throw new Error("Open een specifieke chatpagina (https://claude.ai/chat/<id>).");
    await lazyLoadAll(statusEl, 45000);

    const title = (document.title || "").replace(/\s*\|\s*Claude\s*$/i, "").trim() || "export";
    const items = collectTurns();
    if (!items.length) throw new Error("Geen inhoudelijke beurten gevonden. Scroll even en probeer opnieuw.");

    // ordenen op index
    items.sort((a, b) => a.index - b.index);
    return { title, items };
  }

  function toMarkdown(items, title) {
    const header = `# ${title}\n`;
    const body = items.map(it => {
      const who = it.role === "user" ? "User" : "Claude";
      return `**${who}:**\n\n${it.text}`;
    }).join("\n\n---\n\n");
    return header + "\n" + body + "\n";
  }

  // ---------- Observe navigatie ----------
  const observer = new MutationObserver(() => {
    if (document.body && isOnChat() && !document.getElementById("claude-exporter-btn")) {
      ensureButton();
    }
  });
  observer.observe(document.documentElement, { childList: true, subtree: true });

  // Eerste run
  ensureButton();
})();
