#!/usr/bin/env bash
# .phoenix/core.sh - Core Logic, Locking & Config (V3.2)
# Hardened: Race-condition free locking, safe defaults.

# --- Global Defaults ---
VERBOSE=${VERBOSE:-false}
PARALLEL=${PARALLEL:-false}
DRY_RUN=${DRY_RUN:-false}
SOFT_FAIL=${SOFT_FAIL:-false}
GENERATE_REPORT=${GENERATE_REPORT:-false}

# Cache Config
export CACHE_DIR="/tmp/phoenix-audit-cache"

# --- Styling & Icons ---
ICON_OK="✅"
ICON_FAIL="❌"
ICON_WARN="⚠️"

BOLD='\033[1m'
DIM='\033[2m'
RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[0;33m'
BLUE='\033[0;34m'
NC='\033[0m'

# --- Init ---
init_counters() {
    mkdir -p "$CACHE_DIR"
    echo "0" > "$CACHE_DIR/total"
    echo "0" > "$CACHE_DIR/passed"
    echo "0" > "$CACHE_DIR/failed"
    : > "$CACHE_DIR/failed_items.txt"
}


# --- Atomic Counters (mkdir-lock Based — flock-free) ---
ensure_file() {
  local file="$CACHE_DIR/$1"
  [[ ! -f "$file" ]] && echo "0" > "$file"
}


# Atomic symlink lock (portable, flock-free)
with_lock() {
  local lockfile="$1"; shift
  local stamp="$lockfile.$$.$RANDOM"
  local retries=500
  local delay=0.01

  # Probeer atomisch: ln -s <stamp> <lockfile>
  while ! ln -s "$stamp" "$lockfile" 2>/dev/null; do
    sleep "$delay"
    (( delay < 0.1 )) && delay=$(awk "BEGIN{printf(\"%.3f\", $delay*1.5)}")
    retries=$((retries-1))
    [[ $retries -le 0 ]] && {
      echo "Lock timeout on $lockfile" >&2
      return 1
    }
  done

  # Zorg dat lock altijd opgeruimd wordt
  trap 'rm -f "'"$lockfile"'" "'"$stamp"'"' RETURN
  "$@"
}



atomic_inc() {
  local name="$1"
  local file="$CACHE_DIR/$name"
  local lock="$CACHE_DIR/$name.lock"
  [[ ! -f "$file" ]] && echo "0" > "$file"

  with_lock "$lock" sh -c "
    val=\$(cat '$file' 2>/dev/null | tr -dc '0-9')
    echo \$(( \${val:-0} + 1 )) > '$file'
  "
}

read_counter() {
  cat "$CACHE_DIR/$1" 2>/dev/null || echo 0
}

record_failure() {
  local _id="${1:-unknown}"
  local _desc="${2:-No description}"
  atomic_inc "failed"
  local lock="$CACHE_DIR/failed_items.lock"
  local file="$CACHE_DIR/failed_items.txt"

  with_lock "$lock" sh -c "printf '%s %s\n' '$_id' '$_desc' >> '$file'"
}



log_section() {
    echo -e "\n${BLUE}${BOLD}=== $1 ===${NC}"
}

log_verbose() {
    [[ "$VERBOSE" == "true" ]] && echo -e "${DIM}  [DEBUG] $1${NC}"
}

# Config Loader
load_config() {
    local conf_file=".phoenix-audit.conf"
    if [[ -f "$conf_file" ]]; then
        source "$conf_file"
    fi
    # Defaults als config mist
    export max_household_size=${max_household_size:-5}
    declare -gA THRESHOLDS
    THRESHOLDS[max_household_size]=$max_household_size
}
# Direct laden bij sourcen
load_config
