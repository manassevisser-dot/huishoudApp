on_error() {
    echo "❌ Update failed at line $1" >&2
    echo "   Rolling back if backups exist..." >&2
    # Restore backups if they exist
    for backup in .phoenix/*.sh.bak.*; do
        [[ -f "$backup" ]] && {
            original="${backup%.bak.*}"
            echo "   Restoring $original"
            mv "$backup" "$original"
        }
    done
    exit 1
}

# Portable sed replacement (works on macOS/Linux)
portable_sed_replace() {
    local pattern="$1"
    local file="$2"
    local tmp
    
    if [[ ! -f "$file" ]]; then
        log_warning "File not found: $file (skipping sed patch)"
        return 0
    fi
    
    tmp="$(mktemp)"
    sed "$pattern" "$file" > "$tmp"
    mv "$tmp" "$file"
    log_success "Patched: $file"
}

# Atomic file writer with validation and backup
write_atomic() {
    local dst="$1"
    local label="$2"
    local content="$3"
    local tmp
    
    [[ "$DRY_RUN" == "true" ]] && {
        log_warning "DRY RUN: Would write $label → $dst"
        return 0
    }
    
    # Create temp file
    tmp="$(mktemp)"
    trap "rm -f '$tmp'" RETURN
    
    # Write content
    printf '%s\n' "$content" > "$tmp"
    
    # Validate shell scripts
    if [[ "$dst" == *.sh ]]; then
        if ! bash -n "$tmp" 2>&1; then
            log_error "Syntax error in $label"
            return 1
        fi
        chmod +x "$tmp"
    fi
    
    # Create backup if file exists and differs
    if [[ -f "$dst" ]] && ! cmp -s "$dst" "$tmp"; then
        local backup="${dst}.bak.${TIMESTAMP}"
        cp -p "$dst" "$backup"
        log_info "Backup created: $backup"
    fi
    
    # Atomic move
    mv "$tmp" "$dst"
    log_success "Written: $label → $dst"
}

# --- Universal Pattern Checker ---
check_pattern() {
    local _pat="${1:-}"
    local _target="${2:-}"
    local _desc="${3:-Pattern}"
    local _min="${4:-1}"
    local _exclude="${5:-}"
    local _id="${6:-}"
    
    atomic_inc "total"
    log_verbose "Check: ${_desc} op ${_target}"

    if [[ ! -e "${_target}" ]]; then
        echo -e "${RED}${ICON_FAIL} ${_desc} ${BOLD}(target niet gevonden: ${_target})${NC}"
        record_failure "${_desc}"
        return 1
    fi

    local _matches=""
    # Try ripgrep first (fastest, respects .gitignore)
    if command -v rg >/dev/null 2>&1; then
        if [[ -n "${_exclude}" ]]; then
            _matches="$(rg -i -n --type-add 'ts:*.{ts,tsx}' --type ts -e "${_pat}" "${_target}" 2>/dev/null | grep -vE "${_exclude}" || true)"
        else
            _matches="$(rg -i -n --type-add 'ts:*.{ts,tsx}' --type ts -e "${_pat}" "${_target}" 2>/dev/null || true)"
        fi
    else
        # Fallback to grep with proper exclusions
        local _exclude_dirs="--exclude-dir={node_modules,.venv,.phoenix,.git,__pycache__,dist,build}"
        local _includes="--include=*.ts --include=*.tsx --include=*.js --include=*.jsx --include=*.json"
        
        if [[ -n "${_exclude}" ]]; then
            _matches="$(grep -rni ${_exclude_dirs} ${_includes} -E "${_pat}" "${_target}" 2>/dev/null | grep -vE "${_exclude}" || true)"
        else
            _matches="$(grep -rni ${_exclude_dirs} ${_includes} -E "${_pat}" "${_target}" 2>/dev/null || true)"
        fi
    fi

    local _count="0"
    [[ -n "${_matches}" ]] && _count="$(printf '%s\n' "${_matches}" | grep -c '^')"

    if [[ "${_count}" -ge "${_min}" ]]; then
        echo -e "${GREEN}${ICON_OK} ${_desc} ${BOLD}(${_count})${NC}"
        atomic_inc "passed"
        [[ "${VERBOSE}" == "true" && -n "${_matches}" ]] && printf '%s\n' "${_matches}" | head -3 | sed 's/^/   /'
        return 0
    else
        echo -e "${RED}${ICON_FAIL} ${_desc} ${BOLD}(${_count} < ${_min})${NC}"
        record_failure "${_id}" "${_desc}"
        [[ -n "${_id}" ]] && suggest_fix "${_id}" "print"
        return 1
    fi
}

# --- Anti-pattern Checker (Forbidden elements) ---
check_antipattern() {
    local _pat="${1:-}"
    local _target="${2:-}"
    local _desc="${3:-Anti-pattern}"
    local _exclude="${4:-}"
    local _id="${5:-}"

    atomic_inc "total"
    
    if [[ ! -e "${_target}" ]]; then
        echo -e "${YELLOW}${ICON_WARN} ${_desc} ${BOLD}(overslag: doel niet gevonden)${NC}"
        atomic_inc "passed"
        return 0
    fi

    local _matches=""
    if command -v rg >/dev/null 2>&1; then
        if [[ -n "${_exclude}" ]]; then
            _matches="$(rg -i -n --type-add 'ts:*.{ts,tsx}' --type ts -e "${_pat}" "${_target}" 2>/dev/null | grep -vE "${_exclude}" || true)"
        else
            _matches="$(rg -i -n --type-add 'ts:*.{ts,tsx}' --type ts -e "${_pat}" "${_target}" 2>/dev/null || true)"
        fi
    else
        # Fallback with proper directory exclusions
        local _exclude_dirs="--exclude-dir={node_modules,.venv,.phoenix,.git,__pycache__,dist,build}"
        local _includes="--include=*.ts --include=*.tsx --include=*.js --include=*.jsx --include=*.json"
        
        if [[ -n "${_exclude}" ]]; then
            _matches="$(grep -rni ${_exclude_dirs} ${_includes} -E "${_pat}" "${_target}" 2>/dev/null | grep -vE "${_exclude}" || true)"
        else
            _matches="$(grep -rni ${_exclude_dirs} ${_includes} -E "${_pat}" "${_target}" 2>/dev/null || true)"
        fi
    fi

    local _count="0"
    [[ -n "${_matches}" ]] && _count="$(printf '%s\n' "${_matches}" | grep -c '^')"

    if [[ "${_count}" -eq 0 ]]; then
        echo -e "${GREEN}${ICON_OK} ${_desc} ${BOLD}(geen verboden elementen)${NC}"
        atomic_inc "passed"
        return 0
    else
        echo -e "${RED}${ICON_FAIL} ${_desc} ${BOLD}(${_count} verboden elementen gevonden!)${NC}"
        record_failure "${_id}" "${_desc}"
        
        # Always show first few matches for forbidden elements
        printf '%s\n' "${_matches}" | head -5 | sed 's/^/   /'
        [[ -n "${_id}" ]] && suggest_fix "${_id}" "print"
        return 1
    fi
}

# --- Command Runner ---
run_cmd() {
    local cmd="$1" 
    local desc="$2" 
    local allow="${3:-false}"
    atomic_inc "total"
    
    if [[ "${DRY_RUN:-false}" == "true" ]]; then
        echo -e "${YELLOW}${ICON_WARN} $desc [DRY RUN - SKIPPED]${NC}"
        atomic_inc "skipped"
        return 0
    fi

    local out
    if out=$(eval "$cmd" 2>&1); then
        echo -e "${GREEN}${ICON_OK} $desc${NC}"
        atomic_inc "passed"
        return 0
    else
        if [[ "$allow" == "true" ]]; then
            echo -e "${YELLOW}${ICON_WARN} $desc (toegestane fout)${NC}"
            atomic_inc "passed"
            return 0
        else
            echo -e "${RED}${ICON_FAIL} $desc${NC}"
            record_failure "cmd_fail" "$desc"
            
            if [[ "${VERBOSE:-false}" == "true" ]]; then
                echo -e "${DIM}Foutmelding details:${NC}"
                echo "$out" | tail -10 | sed 's/^/    /'
            fi
            return 1
        fi
    fi
}
