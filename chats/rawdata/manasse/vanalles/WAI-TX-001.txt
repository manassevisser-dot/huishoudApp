        fail "ADR-16 VIOLATION: State variables found in adapter"
        ((violations++))
    else
        pass "ADR-16 COMPLIANT: Adapter is stateless"
    fi
    
    # ADR-10: SchemaVersion tagging
    log_check "ADR-10: SchemaVersion tagging on adapter contracts"
    schema_count=$(grep -r "schemaVersion" src/adapters/undo/types/ | wc -l)
    if [ "$schema_count" -eq 0 ]; then
        fail "ADR-10 VIOLATION: No schemaVersion tagging found"
        ((violations++))
    else
        pass "ADR-10 COMPLIANT: $schema_count schemaVersion tags found"
    fi
    
    # ADR-11: Offline snapshot usage in tests
    log_check "ADR-11: Tests use offline snapshots (no live kernel)"
    snapshot_count=$(grep -r "loadSnapshot\|import.*snapshot" tests/adapters/undo/ | wc -l)
    live_kernel_count=$(grep -r "FinanceKernel\." tests/adapters/undo/ | grep -v "snapshot\|mock" | wc -l)
    
    if [ "$snapshot_count" -eq 0 ]; then
        fail "ADR-11 VIOLATION: No offline snapshots used in tests"
        ((violations++))
    elif [ "$live_kernel_count" -gt 0 ]; then
        fail "ADR-11 VIOLATION: Tests call live Finance Kernel ($live_kernel_count calls)"
        ((violations++))
    else
        pass "ADR-11 COMPLIANT: $snapshot_count offline snapshots, 0 live kernel calls"
    fi
    
    # ADR-11: Test coverage requirement (80%+)
    log_check "ADR-11: Adapter test coverage > 80%"
    if command -v npm &> /dev/null; then
        coverage=$(npm test -- --coverage --silent 2>&1 | grep "adapters/undo" | awk '{print $4}' | tr -d '%')
        if [ -z "$coverage" ]; then
            warn "ADR-11: Could not measure coverage (tests may not exist)"
            ((warnings++))
        elif [ "$coverage" -lt 80 ]; then
            fail "ADR-11 VIOLATION: Coverage ${coverage}% (required: 80%+)"
            ((violations++))
        else
            pass "ADR-11 COMPLIANT: Coverage ${coverage}%"
        fi
    else
        warn "ADR-11: npm not found, skipping coverage check"
        ((warnings++))
    fi
    
    # ADR-12: Audit logging implementation
    log_check "ADR-12: Audit logging present in adapter"
    log_count=$(grep -r "logAudit\|auditLog\|logger\.info.*adr" src/adapters/undo/ | wc -l)
    if [ "$log_count" -eq 0 ]; then
        fail "ADR-12 VIOLATION: No audit logging found"
        ((violations++))
    else
        pass "ADR-12 COMPLIANT: $log_count audit log calls found"
    fi
    
    # ADR-12: Changelog references WAI-TX-001
    log_check "ADR-12: Changelog contains WAI-TX-001 reference"
    if grep -q "WAI-TX-001" CHANGELOG.md 2>/dev/null; then
        pass "ADR-12 COMPLIANT: WAI-TX-001 documented in changelog"
    else
        fail "ADR-12 VIOLATION: WAI-TX-001 not in changelog"
        ((violations++))
    fi
    
    # ADR-05: No float operations in adapter (minor-units preserved)
    log_check "ADR-05: No float operations in adapter (minor-units)"
    if grep -r "parseFloat\|toFixed\|Number\\(" src/adapters/undo/ | grep -v "// ADR-05 exception"; then
        fail "ADR-05 VIOLATION: Float operations found (minor-units compromised)"
        ((violations++))
    else
        pass "ADR-05 COMPLIANT: No float operations, minor-units preserved"
    fi
    
    # ADR-16: Idempotency test exists
    log_check "ADR-16: Idempotency tests present"
    if grep -r "idempoten\|pure function\|same input.*same output" tests/adapters/undo/ | wc -l; then
        pass "ADR-16 COMPLIANT: Idempotency tests found"
    else
        warn "ADR-16: No explicit idempotency tests found"
        ((warnings++))
    fi
    
    # Calculate WAI-TX-001 specific grade
    local wai_violations=$violations
    local wai_warnings=$warnings
    
    if [ "$wai_violations" -eq 0 ] && [ "$wai_warnings" -eq 0 ]; then
        log_result "🏆 WAI-TX-001 Grade: A+ (Perfect compliance)"
    elif [ "$wai_violations" -eq 0 ]; then
        log_result "✅ WAI-TX-001 Grade: A ($wai_warnings warnings)"
    elif [ "$wai_violations" -le 2 ]; then
        log_result "⚠️ WAI-TX-001 Grade: B ($wai_violations violations)"
    else
        log_result "❌ WAI-TX-001 Grade: C/D ($wai_violations violations)"
    fi
}
