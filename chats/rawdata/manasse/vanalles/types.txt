/**
 * DeepPartial Helper: Zorgt ervoor dat alle geneste eigenschappen optioneel zijn.
 * Dit voorkomt TS-errors bij het updaten van specifieke velden in de state.
 */
type DeepPartial<T> = {
  [P in keyof T]?: T[P] extends (infer U)[]
    ? DeepPartial<U>[]
    : T[P] extends object
    ? DeepPartial<T[P]>
    : T[P];
};

/** * Secties binnen de data van FormState. 
 */
export type DataSection = 'setup' | 'household' | 'finance';

export interface FormState {
  schemaVersion: string;
  activeStep: string;
  currentPageId: string;
  isValid: boolean;
  data: {
    setup: {
      aantalMensen: number;
      aantalVolwassen: number;
      autoCount: 'Nee' | 'Een' | 'Twee';
      heeftHuisdieren?: boolean;
      [key: string]: unknown;
    };
    household: {
      members: Array<Record<string, unknown>>;
    };
    finance: {
      income: {
        items: Array<Record<string, unknown>>;
        totalAmount?: number;
      };
      expenses: {
        items: Array<Record<string, unknown>>;
        totalAmount?: number;
      };
    };
  };
  meta: {
    lastModified: string;
    version: number;
  };
}

/**
 * FormAction: Bevat alle acties voor de Phoenix 2025 architectuur.
 */
export type FormAction =
  | { type: 'SET_STEP'; payload: string }
  | { type: 'UPDATE_DATA'; payload: DeepPartial<FormState['data']> } // ✅ Nu met DeepPartial
  | { type: 'SET_FIELD'; payload: { section: DataSection; field: string; value: any } } 
  | { type: 'SYNC_MEMBERS'; payload: Array<Record<string, unknown>> }
  | { type: 'RESET_APP' }
  | { type: 'NEXT_STEP' } 
  | { type: 'PREV_STEP' };

/**
 * Wizard en Field configuratie types
 */
export interface FieldConfig {
  fieldId: string;
  type: 'counter' | 'text' | 'radio-chips' | 'toggle' | 'repeater' | 'date' | 'select' | 'section' | 'collapsible-section';
  labelToken?: string;
  label?: string;      // Toevoegen: voor directe tekst in configs
  options?: any[];    // Toevoegen: voor radio/select opties
  fields?: FieldConfig[];
  required?: boolean;
  visibleIf?: string | ((state: FormState) => boolean);
  condition?: (state: FormState) => boolean;
  dependentField?: FieldConfig & { id?: string };
  validation?: {
    min?: number;
    max?: number;
    postcode?: boolean;
    lengthEqualsTo?: string | number;
  };
}

export interface WizardPageConfig {
  pageId: string;
  titleToken?: string;
  componentName: string;
  fields: FieldConfig[]; // Nu gekoppeld aan FieldConfig
  onNext?: (state: FormState) => string;
}

//FORMREDUCER:
import { FormState, FormAction } from '../../shared-types/form';
import { initialState } from './FormContext';
// Importeer de deepMerge utility uit je factory
import { deepMerge } from '@utils/objects'

export function formReducer(state: FormState, action: FormAction): FormState {
  const metaUpdate = { ...state.meta, lastModified: new Date().toISOString() };

  switch (action.type) {
    case 'SET_STEP':
      return { 
        ...state, 
        activeStep: action.payload, 
        meta: metaUpdate 
      };

    case 'UPDATE_DATA':
      // ✅ Gebruik Deep Merge conform Phoenix 2025 SSOT.
      // Dit voorkomt dat 'household' of 'finance' verdwijnen bij een partial update.
      return {
        ...state,
        data: deepMerge(state.data, action.payload),
        meta: metaUpdate
      };

      case 'SET_FIELD':
        const { section, field, value } = action.payload; // Destructure 'field'
        return {
          ...state,
          data: {
            ...state.data,
            [section]: {
              ...(state.data as any)[section],
              [field]: value // Gebruik 'field' als key in de state
            }
          },
          meta: metaUpdate
        };

    case 'SYNC_MEMBERS':
      return {
        ...state,
        data: {
          ...state.data,
          household: { 
            ...state.data.household, 
            members: action.payload 
          }
        },
        meta: metaUpdate
      };

    case 'RESET_APP':
      return {
        ...initialState,
        meta: { ...initialState.meta, lastModified: new Date().toISOString() }
      };

    default:
      return state;
  }
}
