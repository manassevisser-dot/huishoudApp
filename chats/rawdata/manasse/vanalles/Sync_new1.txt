#!/usr/bin/env node
/**
 * Sync path aliases from tsconfig.json → babel.config.js + jest.config.js + jsconfig.json
 * 
 * Usage:
 *   node scripts/sync-aliases.js
 */

const fs = require('fs');
const path = require('path');
const ts = require('typescript');

/* ============================================================================
 * Configuration
 * ========================================================================== */

const ROOT = process.cwd();
const TSCONFIG = path.join(ROOT, 'tsconfig.json');
const JSCONFIG = path.join(ROOT, 'jsconfig.json');
const BABEL = path.join(ROOT, 'babel.config.js');
const JEST = path.join(ROOT, 'jest.config.js');

const FLAGS = {
  dryRun: process.argv.includes('--dry-run'),
  backup: process.argv.includes('--backup'),
  verbose: process.argv.includes('--verbose'),
  help: process.argv.includes('--help') || process.argv.includes('-h'),
  restore: process.argv.includes('--restore'),
};

// Reserved alias prefixes (npm/node conventions)
const RESERVED_PREFIXES = [
  '@types', '@react', '@expo', '@jest', '@babel', 
  '@node', '@native', '@test', '@testing-library',
];

// Marker comments for safe updates
const MARKERS = {
  babel: {
    start: '/* AUTO-SYNC-START: Generated by sync-aliases.js - Do not edit manually */',
    end: '/* AUTO-SYNC-END */',
  },
  jest: {
    start: '// AUTO-SYNC-START: Generated by sync-aliases.js - Do not edit manually',
    end: '// AUTO-SYNC-END',
  },
};

/* ============================================================================
 * Utilities
 * ========================================================================== */

function die(msg, code = 1) {
  console.error(`❌ ${msg}`);
  process.exit(code);
}

function info(msg) {
  console.log(`ℹ️  ${msg}`);
}

function ok(msg) {
  console.log(`✅ ${msg}`);
}

function warn(msg) {
  console.log(`⚠️  ${msg}`);
}

function verbose(msg) {
  if (FLAGS.verbose) {
    console.log(`🔍 ${msg}`);
  }
}

function showHelp() {
  console.log(`
📋 Sync Path Aliases

Syncs TypeScript path aliases to Babel, Jest, and jsconfig.json using
safe marker blocks. Only updates content between AUTO-SYNC markers.

Usage:
  node scripts/sync-aliases.js [options]

Options:
  --dry-run    Show what would change without writing files
  --backup     Create .bak files before modifying
  --verbose    Show detailed processing information
  --restore    Restore from most recent .bak files
  --help, -h   Show this help message
  `);
  process.exit(0);
}

/**
 * Create backup of file
 */
function createBackup(filePath) {
  const backupPath = `${filePath}.bak.${Date.now()}`;
  fs.copyFileSync(filePath, backupPath);
  verbose(`Created backup: ${path.basename(backupPath)}`);
  return backupPath;
}

/**
 * Check if directory exists
 */
function directoryExists(dirPath) {
  try {
    return fs.statSync(dirPath).isDirectory();
  } catch {
    return false;
  }
}

/**
 * Update content between marker comments
 */
function updateBetweenMarkers(content, startMarker, endMarker, newContent, indentLevel = 4) {
  const startPattern = escapeRegex(startMarker);
  const endPattern = escapeRegex(endMarker);
  
  // Match everything between markers (non-greedy)
  const regex = new RegExp(
    `(${startPattern})[\\s\\S]*?(${endPattern})`,
    'g'
  );
  
  if (!regex.test(content)) {
    return null; // Markers not found
  }
  
  // Create proper indentation
  const indent = ' '.repeat(indentLevel);
  
  // Reset regex after test
  regex.lastIndex = 0;
  
  return content.replace(regex, `$1\n${newContent}\n${indent}$2`);
}

/**
 * Escape string for use in regex
 */
function escapeRegex(str) {
  return str.replace(/[.*+?^${}()|[\]\\]/g, '\\$&');
}

/**
 * Find most recent backup for a file
 */
function findLatestBackup(filePath) {
  const dir = path.dirname(filePath);
  const basename = path.basename(filePath);
  
  try {
    const files = fs.readdirSync(dir);
    const backups = files
      .filter(f => f.startsWith(`${basename}.bak.`))
      .map(f => ({
        name: f,
        path: path.join(dir, f),
        time: parseInt(f.split('.bak.')[1]) || 0,
      }))
      .sort((a, b) => b.time - a.time);
    
    return backups.length > 0 ? backups[0].path : null;
  } catch {
    return null;
  }
}

/**
 * Restore file from backup
 */
function restoreFromBackup(filePath) {
  const backup = findLatestBackup(filePath);
  
  if (!backup) {
    warn(`No backup found for ${path.basename(filePath)}`);
    return false;
  }
  
  try {
    fs.copyFileSync(backup, filePath);
    ok(`Restored ${path.basename(filePath)} from ${path.basename(backup)}`);
    return true;
  } catch (err) {
    warn(`Failed to restore ${path.basename(filePath)}: ${err.message}`);
    return false;
  }
}

/**
 * Restore all config files from backups
 */
function restoreAll() {
  info('🔄 Restoring from backups...\n');
  
  let restored = 0;
  [BABEL, JEST, JSCONFIG].forEach(file => {
    if (fs.existsSync(file)) {
      if (restoreFromBackup(file)) restored++;
    }
  });
  
  console.log();
  if (restored > 0) {
    ok(`✨ Restored ${restored} file(s)`);
  } else {
    warn('No files were restored');
  }
  
  process.exit(0);
}

/* ============================================================================
 * Config File Handlers
 * ========================================================================== */

/**
 * Update babel.config.js with marker-based approach
 */
function updateBabel(aliases) {
  if (!fs.existsSync(BABEL)) {
    warn('babel.config.js not found (skipping)');
    return;
  }

  verbose('Processing babel.config.js');
  let babelSrc = fs.readFileSync(BABEL, 'utf8');

  // Check for markers
  if (!babelSrc.includes(MARKERS.babel.start) || !babelSrc.includes(MARKERS.babel.end)) {
    warn('babel.config.js: Missing AUTO-SYNC markers');
    info('Add markers inside the alias block:');
    info('  alias: {');
    info('    /* AUTO-SYNC-START: Generated by sync-aliases.js - Do not edit manually */');
    info('    /* AUTO-SYNC-END */');
    info('  }');
    return;
  }

  // Generate new alias content
  const babelAliases = aliases
    .map(a => `        '${a.name}': './${a.target}',`)
    .join('\n');

  // Update between markers
  const updated = updateBetweenMarkers(
    babelSrc,
    MARKERS.babel.start,
    MARKERS.babel.end,
    babelAliases,
    10 // Indent level for babel aliases
  );

  if (!updated) {
    warn('babel.config.js: Failed to update (regex issue)');
    return;
  }

  if (FLAGS.dryRun) {
    verbose('[DRY RUN] Would update babel.config.js');
    if (FLAGS.verbose) {
      console.log('\nPreview (babel.config.js):');
      console.log(babelAliases);
    }
  } else {
    if (FLAGS.backup) createBackup(BABEL);
    fs.writeFileSync(BABEL, updated, 'utf8');
    ok('Updated babel.config.js');
  }
}

/**
 * Update jest.config.js with marker-based approach
 */
function updateJest(aliases) {
  if (!fs.existsSync(JEST)) {
    warn('jest.config.js not found (skipping)');
    return;
  }

  verbose('Processing jest.config.js');
  let jestSrc = fs.readFileSync(JEST, 'utf8');

  // Check for markers
  if (!jestSrc.includes(MARKERS.jest.start) || !jestSrc.includes(MARKERS.jest.end)) {
    warn('jest.config.js: Missing AUTO-SYNC markers');
    info('Add markers inside the moduleNameMapper block:');
    info('  moduleNameMapper: {');
    info('    // AUTO-SYNC-START: Generated by sync-aliases.js - Do not edit manually');
    info('    // AUTO-SYNC-END');
    info('  }');
    return;
  }

  // Generate new mapper content
  const jestMappers = aliases
    .map(a => {
      // Escape @ and other special regex chars for Jest pattern
      const escapedName = a.name.replace(/[@]/g, '\\@');
      return `    '^${escapedName}/(.*)$': '<rootDir>/${a.target}/$1',`;
    })
    .join('\n');

  // Update between markers
  const updated = updateBetweenMarkers(
    jestSrc,
    MARKERS.jest.start,
    MARKERS.jest.end,
    jestMappers,
    4 // Indent level for jest mappers
  );

  if (!updated) {
    warn('jest.config.js: Failed to update (regex issue)');
    return;
  }

  if (FLAGS.dryRun) {
    verbose('[DRY RUN] Would update jest.config.js');
    if (FLAGS.verbose) {
      console.log('\nPreview (jest.config.js):');
      console.log(jestMappers);
    }
  } else {
    if (FLAGS.backup) createBackup(JEST);
    fs.writeFileSync(JEST, updated, 'utf8');
    ok('Updated jest.config.js');
  }
}

/**
 * Update jsconfig.json - Complete replacement strategy
 * Uses metadata to track sync state
 */
function updateJsconfig(aliases) {
  if (!fs.existsSync(JSCONFIG)) {
    verbose('jsconfig.json not found - creating new one');
    
    const newConfig = {
      "//": "AUTO-GENERATED by sync-aliases.js - Do not edit paths manually",
      "compilerOptions": {
        "baseUrl": ".",
        "paths": {}
      },
      "_sync_meta": {
        "generated_by": "sync-aliases.js",
        "last_sync": new Date().toISOString(),
        "source": "tsconfig.json"
      },
      "exclude": ["node_modules", "dist", "build"]
    };
    
    // Add aliases
    aliases.forEach(a => {
      newConfig.compilerOptions.paths[`${a.name}/*`] = [`${a.target}/*`];
    });
    
    if (FLAGS.dryRun) {
      verbose('[DRY RUN] Would create jsconfig.json');
      if (FLAGS.verbose) {
        console.log('\nPreview (jsconfig.json):');
        console.log(JSON.stringify(newConfig, null, 2));
      }
    } else {
      fs.writeFileSync(JSCONFIG, JSON.stringify(newConfig, null, 2) + '\n', 'utf8');
      ok('Created jsconfig.json');
    }
    return;
  }

  verbose('Processing jsconfig.json');
  
  let jsconfig;
  try {
    const rawContent = fs.readFileSync(JSCONFIG, 'utf8');
    const result = ts.parseConfigFileTextToJson(JSCONFIG, rawContent);
    
    if (result.error) {
      warn(`jsconfig.json: Parse error (${result.error.messageText})`);
      return;
    }
    
    jsconfig = result.config;
  } catch (err) {
    warn(`jsconfig.json: Failed to read (${err.message})`);
    return;
  }

  // Ensure structure exists
  if (!jsconfig.compilerOptions) {
    jsconfig.compilerOptions = {};
  }
  
  // Initialize or preserve baseUrl
  if (!jsconfig.compilerOptions.baseUrl) {
    jsconfig.compilerOptions.baseUrl = ".";
  }

  // Build new paths object (complete replacement)
  const newPaths = {};
  
  // Add comment marker (if using _comment strategy)
  if (jsconfig.compilerOptions.paths?._comment !== undefined) {
    newPaths._comment = "Generated by sync-aliases.js - Do not edit manually";
  }
  
  // Add all aliases from tsconfig
  aliases.forEach(a => {
    newPaths[`${a.name}/*`] = [`${a.target}/*`];
  });
  
  // Preserve any manual paths that don't conflict
  const existingPaths = jsconfig.compilerOptions.paths || {};
  for (const [key, value] of Object.entries(existingPaths)) {
    // Skip meta keys
    if (key.startsWith('_') || key === '//' || key.startsWith('//')) {
      continue;
    }
    
    // Skip if we're managing this alias
    const aliasName = key.replace(/\/\*$/, '');
    const isManaged = aliases.some(a => a.name === aliasName);
    
    if (!isManaged) {
      verbose(`Preserving manual path: ${key}`);
      newPaths[key] = value;
    }
  }

  // Update paths
  jsconfig.compilerOptions.paths = newPaths;
  
  // Add/update metadata
  if (!jsconfig._sync_meta) {
    jsconfig._sync_meta = {};
  }
  jsconfig._sync_meta.generated_by = "sync-aliases.js";
  jsconfig._sync_meta.last_sync = new Date().toISOString();
  jsconfig._sync_meta.alias_count = aliases.length;
  
  // Add top-level comment if not present
  if (!jsconfig['//']) {
    jsconfig['//'] = "AUTO-GENERATED by sync-aliases.js - Do not edit paths manually";
  }

  if (FLAGS.dryRun) {
    verbose('[DRY RUN] Would update jsconfig.json');
    if (FLAGS.verbose) {
      console.log('\nPreview (jsconfig.json paths):');
      console.log(JSON.stringify(newPaths, null, 2));
    }
  } else {
    if (FLAGS.backup) createBackup(JSCONFIG);
    
    // Write with proper formatting
    fs.writeFileSync(JSCONFIG, JSON.stringify(jsconfig, null, 2) + '\n', 'utf8');
    ok('Updated jsconfig.json');
  }
}

/* ============================================================================
 * Main Logic
 * ========================================================================== */

function main() {
  if (FLAGS.help) {
    showHelp();
  }

  if (FLAGS.restore) {
    restoreAll();
  }

  info('🚀 Starting alias sync...\n');

  /* ---------- Step 1: Read & Parse tsconfig.json ---------- */
  
  if (!fs.existsSync(TSCONFIG)) {
    die('tsconfig.json not found in project root');
  }

  verbose(`Reading ${TSCONFIG}`);

  let tsconfig;
  try {
    const rawContent = fs.readFileSync(TSCONFIG, 'utf8');
    const result = ts.parseConfigFileTextToJson(TSCONFIG, rawContent);

    if (result.error) {
      const message = ts.flattenDiagnosticMessageText(result.error.messageText, '\n');
      die(`Failed to parse tsconfig.json: ${message}`);
    }

    tsconfig = result.config;
    verbose('Successfully parsed tsconfig.json');

  } catch (err) {
    die(`Failed to read tsconfig.json: ${err.message}`);
  }

  const paths = tsconfig?.compilerOptions?.paths;
  if (!paths || typeof paths !== 'object') {
    die('No compilerOptions.paths found in tsconfig.json');
  }

  /* ---------- Step 2: Extract & Validate Aliases ---------- */

  const aliases = [];
  const seenNames = new Set();
  const seenTargets = new Set();

  for (const [key, value] of Object.entries(paths)) {
    // Skip all meta/comment keys
    if (key.startsWith('_') || key === '//' || key.startsWith('//')) {
      verbose(`Skipping meta key: ${key}`);
      continue;
    }

    // Validate value is array
    if (!Array.isArray(value) || value.length === 0) {
      warn(`Skipping "${key}": value is not a non-empty array`);
      continue;
    }

    // Extract alias name (remove trailing /*)
    const name = key.replace(/\/\*$/, '');
    const target = value[0].replace(/\/\*$/, '');

    // Check for reserved names
    const isReserved = RESERVED_PREFIXES.some(prefix => 
      name.startsWith(prefix)
    );
    if (isReserved) {
      verbose(`Skipping "${name}": reserved prefix`);
      continue;
    }

    // Check for duplicates
    if (seenNames.has(name)) {
      warn(`Duplicate alias name: "${name}" (skipping)`);
      continue;
    }
    if (seenTargets.has(target)) {
      warn(`Duplicate target: "${target}" (used by "${name}")`);
    }

    // Validate target exists
    const targetPath = path.join(ROOT, target);
    if (!directoryExists(targetPath)) {
      warn(`Target directory does not exist: ${target}`);
    }

    aliases.push({ name, target });
    seenNames.add(name);
    seenTargets.add(target);
  }

  if (aliases.length === 0) {
    die('No valid aliases found in tsconfig.json');
  }

  ok(`Found ${aliases.length} valid alias(es):\n`);
  aliases.forEach(a => info(`  ${a.name} → ${a.target}`));
  console.log();

  /* ---------- Step 3: Update Config Files ---------- */

  updateBabel(aliases);
  updateJest(aliases);
  updateJsconfig(aliases);

  /* ---------- Step 4: Summary ---------- */

  console.log();
  if (FLAGS.dryRun) {
    ok('✨ Dry-run complete (no files modified)');
    info('Run without --dry-run to apply changes');
  } else {
    ok('✨ Alias sync complete!');
    info('\nNext steps:');
    info('  1. Clear Metro cache: npx react-native start --reset-cache');
    info('  2. Restart TypeScript server in your IDE');
    info('  3. Test imports: import { X } from \'@domain/Y\'');
  }
}

/* ============================================================================
 * Entry Point
 * ========================================================================== */

try {
  main();
} catch (err) {
  die(`Unexpected error: ${err.message}\n${err.stack}`);
}
