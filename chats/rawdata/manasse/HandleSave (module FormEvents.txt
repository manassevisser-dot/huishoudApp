Public Sub HandleSave(frm As Object)
On Error GoTo ErrHandler

    Select Case TypeName(frm)
        Case "frmHuishouden": SaveHuishouden frm
        Case "frmVast": SaveVast frm
        Case "frmInvoer": SaveDagelijks frm
        Case Else
            MsgBox "Onbekend formulier-type: " & TypeName(frm), vbExclamation
    End Select

    MsgBox "De gegevens zijn opgeslagen.", vbInformation
Exit Sub
ErrHandler:
    MsgBox "Fout tijdens opslaan: " & Err.Description, vbCritical
End Sub

Public Sub SaveHuishouden(frm As Object)
On Error GoTo ErrHandler
    Dim ws As Worksheet: Set ws = EnsureTableSheet("tblHuishouden")

    Dim lo As ListObject: Set lo = ws.ListObjects("tblHuishouden")
    Dim r As ListRow: Set r = lo.ListRows.Add

    r.Range(1, 1).Value = Now
    r.Range(1, 2).Value = frm.Controls("dyn_txtTotalPersons").Text
    r.Range(1, 3).Value = "Volwassenen: dynamisch toevoegen implementeren"
    r.Range(1, 4).Value = "Kinderen: idem"
    r.Range(1, 5).Value = "Huisdieren: idem"
Exit Sub
ErrHandler:
    MsgBox "SaveHuishouden fout: " & Err.Description, vbCritical
End Sub

Public Sub SaveVast(frm As Object)
On Error GoTo ErrHandler
    Dim mp As MultiPage: Set mp = frm.Controls("dyn_mpVast")

    SaveVasteRows(frm, mp.Pages(0), "tblVasteInkomsten", True)
    SaveVasteRows(frm, mp.Pages(1), "tblVasteUitgaven", False)

Exit Sub
ErrHandler:
    MsgBox "SaveVast fout: " & Err.Description, vbCritical
End Sub

Public Sub SaveVasteRows(frm As Object, pg As Page, tblName As String, inkomsten As Boolean)
On Error GoTo ErrHandler
    Dim ws As Worksheet: Set ws = EnsureTableSheet(tblName)
    Dim lo As ListObject: Set lo = ws.ListObjects(tblName)

    Dim lblRows As MSForms.Label
    Set lblRows = pg.Controls("dyn_lblRows_" & IIf(inkomsten, "inkomsten", "uitgaven"))

    Dim maxN As Long: maxN = CLng(lblRows.Caption)
    Dim i As Long

    For i = 1 To maxN
        Dim bron As String, per As String, bedrag As Double, ontv As String

        bron = pg.Controls("dyn_txtBron_" & i).Text
        per = pg.Controls("dyn_cmbPeriode_" & i).Text
        bedrag = CDbl(Replace(pg.Controls("dyn_txtBedrag_" & i).Text, ",", "."))
        If inkomsten Then
            ontv = pg.Controls("dyn_cmbOntvanger_" & i).Text
        Else
            ontv = ""
        End If

        If Len(bron) > 0 And bedrag > 0 Then
            Dim r As ListRow: Set r = lo.ListRows.Add
            r.Range(1, 1).Value = Now
            r.Range(1, 2).Value = bron
            r.Range(1, 3).Value = per
            r.Range(1, 4).Value = bedrag
            r.Range(1, 5).Value = UtilsToMonthlyAmount(bedrag, per)
            If inkomsten Then r.Range(1, 6).Value = ontv
        End If
    Next i
Exit Sub
ErrHandler:
    MsgBox "SaveVasteRows fout: " & Err.Description, vbCritical
End Sub

Public Sub SaveDagelijks(frm As Object)
On Error GoTo ErrHandler

    Dim ws As Worksheet: Set ws = EnsureTableSheet("tblDagelijks")
    Dim lo As ListObject: Set lo = ws.ListObjects("tblDagelijks")

    Dim r As ListRow: Set r = lo.ListRows.Add

    r.Range(1, 1).Value = Now
    r.Range(1, 2).Value = frm.Controls("dyn_lblDatumVandaag").Caption
    r.Range(1, 3).Value = frm.Controls("dyn_cmbCategorie").Text
    r.Range(1, 4).Value = frm.Controls("dyn_cmbSubcategorie").Text
    r.Range(1, 5).Value = CDbl(Replace(frm.Controls("dyn_txtBedrag").Text, ",", "."))
    r.Range(1, 6).Value = frm.Controls("dyn_cmbBetaalwijze").Text

Exit Sub
ErrHandler:
    MsgBox "SaveDagelijks fout: " & Err.Description, vbCritical
End Sub

Public Function EnsureTableSheet(tblName As String) As Worksheet
On Error Resume Next
    Dim ws As Worksheet: Set ws = ThisWorkbook.Worksheets(tblName)
    On Error GoTo 0

    If ws Is Nothing Then
        Set ws = ThisWorkbook.Worksheets.Add
        ws.Name = tblName
        Dim lo As ListObject
        Set lo = ws.ListObjects.Add(xlSrcRange, ws.Range("A1:F1"), , xlYes)
        lo.Name = tblName

        Dim headers As Variant
        Select Case tblName
            Case "tblHuishouden": headers = Array("Tijd", "Totaal Personen", "Volwassenen", "Kinderen", "Huisdieren")
            Case "tblVasteInkomsten": headers = Array("Tijd", "Bron", "Periode", "Bedrag", "Maandbedrag", "Ontvanger")
            Case "tblVasteUitgaven": headers = Array("Tijd", "Categorie", "Periode", "Bedrag", "Maandbedrag")
            Case "tblDagelijks": headers = Array("Tijd", "Datum", "Categorie", "Subcategorie", "Bedrag", "Betaalwijze")
        End Select

        Dim i As Long
        For i = LBound(headers) To UBound(headers)
            ws.Cells(1, i + 1).Value = headers(i)
        Next i
    End If

    Set EnsureTableSheet = ws
End Function
